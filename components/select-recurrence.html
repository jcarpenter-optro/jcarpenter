/**
 * Compares rule values ignoring termination (COUNT/UNTIL)
 */
function eqRule(value: string, rule: string | null): boolean {
	const valueString = value?.match(RRULE_REGEX)?.[0]?.slice(6) || '';
	const ruleString = rule || '';

	let valueWithoutTermination = valueString
		.split(';')
		.filter((item) => {
			return !(item.startsWith('COUNT') || item.startsWith('UNTIL'));
		})
		.join(';');

	if (!valueWithoutTermination) {
		if (!ruleString) {
			return true;
		}
		return false;
	}

	valueWithoutTermination = `RRULE:${valueWithoutTermination}`;

	return ruleString === valueWithoutTermination;
}

/**
 * Builder component for configuring recurrence rules.
 */
export function SelectRecurrenceBuilder({
	value,
	presets,
	minFrequency,
	singleWeeklyDay,
	allowCustom,
	onChange,
	onCancel,
	intl: intlProp,
}: SelectRecurrenceBuilderProps) {
	const intl = useMemo(() => intlProp ?? createDefaultIntl(), [intlProp]);

	// Initialize rule from value
	const [rule, setRule] = useState(() => new RRule(value, intl));

	// Determine if showing custom view
	const [isCustom, setIsCustom] = useState(() => {
		const hasValue = value || value === '';
		const matchesPreset = PRESETS.find((preset) => eqRule(value, preset.value));
		return (hasValue && !matchesPreset) || false;
	});

	// Cache whether current rule is custom for checkmark display
	const isCustomRule = useMemo(() => {
		const hasValue = value || value === '';
		const matchesPreset = PRESETS.find((preset) => eqRule(value, preset.value));
		return (hasValue && !matchesPreset) || false;
	}, [value]);

	// Update rule when value changes
	useEffect(() => {
		setRule(new RRule(value, intl));
	}, [value, intl]);

	// Filter frequency options based on minFrequency
	const freqOptions = useMemo(() => {
		if (!minFrequency) {
			return rule.freqOptions;
		}
		const index = Object.keys(FREQ).indexOf(minFrequency);
		return rule.freqOptions.slice(index);
	}, [rule, minFrequency]);

	/**
	 * Clones the current rule state into a fresh RRule so accumulated
	 * (uncommitted) edits are preserved across field changes.
	 */
	const cloneRule = useCallback(() => new RRule(rule.string, intl), [rule, intl]);

	/**
	 * Extracts the raw value from a Select onChange payload.
	 * The Select component wraps options as SelectOption objects, so onChange
	 * delivers `{ value, label }` rather than the bare `T` value.
	 */
	const extractValue = (val: unknown): string | null => {
		if (val === null || val === undefined) return null;
		if (typeof val === 'object' && 'value' in (val as Record<string, unknown>)) {
			return (val as { value: string }).value;
		}
		return typeof val === 'string' ? val : null;
	};

	const handleSaveRule = useCallback(async () => {
		onChange(rule.string);
	}, [rule, onChange]);

	const handleSwitchToCustom = useCallback(() => {
		setIsCustom(true);
	}, []);

	const handleSelectPreset = useCallback(
		(preset: Preset) => {
			const newRule = new RRule(value, intl);
			newRule.resetRecurrence();
			newRule.parseRule(preset.value);
			onChange(newRule.string);
		},
		[value, intl, onChange],
	);

	const handleFrequencyChange = useCallback(
		(val: unknown) => {
			const freqValue = extractValue(val);
			const newRule = cloneRule();
			newRule.freq = freqValue;
			newRule.byDay = null;
			newRule.byMonthDay = null;
			setRule(newRule);
		},
		[cloneRule],
	);

	const handleIntervalChange = useCallback(
		(intervalValue: string) => {
			const newRule = cloneRule();
			newRule.interval = intervalValue ? Number(intervalValue) : null;
			setRule(newRule);
		},
		[cloneRule],
	);

	const handleByDayToggle = useCallback(
		(dayValue: string) => {
			const newRule = cloneRule();
			newRule.toggleByDayValue(dayValue as 'MO' | 'TU' | 'WE' | 'TH' | 'FR' | 'SA' | 'SU');
			setRule(newRule);
		},
		[cloneRule],
	);

	const handleByDaySelect = useCallback(
		(val: unknown) => {
			const dayValue = extractValue(val) as 'MO' | 'TU' | 'WE' | 'TH' | 'FR' | 'SA' | 'SU' | null;
			const newRule = cloneRule();
			if (newRule.freq === FREQ.weekly) {
				newRule.byDay = dayValue;
			} else if (newRule.freq === FREQ.monthly) {
				if (dayValue) {
					newRule.setByDayValueAsOption({ value: dayValue, label: '' });
				} else {
					newRule.setByDayValueAsOption(null);
				}
			}
			setRule(newRule);
		},
		[cloneRule],
	);

	const handleTimeOfMonthChange = useCallback(
		(val: unknown) => {
			const rawValue = extractValue(val) as 'DATE' | 'DAY' | 'LAST' | null;
			if (rawValue) {
				const newRule = cloneRule();
				const option = newRule.timeOfMonthOptions.find((o) => o.value === rawValue);
				if (option) {
					newRule.updateTimeOfMonth(option);
				}
				setRule(newRule);
			}
		},
		[cloneRule],
	);

	const handleByMonthDayChange = useCallback(
		(val: unknown) => {
			const rawValue = extractValue(val);
			const newRule = cloneRule();
			newRule.byMonthDay = rawValue;
			setRule(newRule);
		},
		[cloneRule],
	);

	const handleSelectionChange = useCallback(
		(val: unknown) => {
			const rawValue = extractValue(val) as '1' | '2' | '3' | '4' | '-1' | null;
			if (rawValue) {
				const newRule = cloneRule();
				const option = newRule.selectionOptions.find((o) => o.value === rawValue);
				if (option) {
					newRule.setSelectionValue(option);
				}
				setRule(newRule);
			}
		},
		[cloneRule],
	);

	const stringIncludesText = (byDay: string | null, dayValue: string): boolean => {
		return byDay?.includes(dayValue) ?? false;
	};

	// Custom builder view
	if (isCustom) {
		return (
			<div className={styles.Builder} data-qid-select-recurrence-builder>
				<YStack spacing="md">
					<XStack spacing="md">
						<IconButton
							size="sm"
							variant="gray"
							tooltip={intl.t('input.selectRecurrence.custom.back')}
							icon="arrow-left"
							data-qid-recurrence-back
							onClick={() => setIsCustom(false)}
						/>
						<Text font="200" weight="strong" color="gray-900">
							{intl.t('input.selectRecurrence.customize')}
						</Text>
					</XStack>

					<Form onSubmit={handleSaveRule}>
						{() => (
							<YStack spacing="md">
								<XStack spacing="md" width="full">
									<Field
										className={styles.every}
										label={intl.t('input.selectRecurrence.field.every')}
										input={({ id }) => (
											<Input
												id={id}
												type="number"
												value={rule.interval?.toString() ?? ''}
												onChange={handleIntervalChange}
												requirements={{ min: 1 }}
												placeholder="1"
												min={1}
												data-qid-recurrence-interval
											/>
										)}
									/>
									<Field
										className={styles.frequency}
										label={intl.t('input.selectRecurrence.field.frequency')}
										input={({ id }) => (
											<Select
												id={id}
												value={rule.freq}
												options={freqOptions}
												onChange={handleFrequencyChange}
												requirements={{ required: true }}
												searchKeys={['label']}
												data-qid-recurrence-frequency
											/>
										)}
									/>
								</XStack>

								{rule.isWeekly && (
									<Field
										className={styles.frequency}
										label={intl.t('input.selectRecurrence.field.repeat-on')}
										input={({ id }) =>
											singleWeeklyDay ? (
												<Select
													id={id}
													value={rule.byDayValueAsOption?.value ?? null}
													options={rule.byDayOptions}
													onChange={handleByDaySelect}
													requirements={{ required: true }}
													searchKeys={['label']}
													data-qid-recurrence-by-day-of-week
												/>
											) : (
												<div className={styles.checkboxList}>
													{rule.byDayOptions.map((option) => (
														<label
															key={option.value}
															className={styles.checkbox}
															data-selected={stringIncludesText(rule.byDay, option.value) ? '' : undefined}
														>
															{option.label}

															{stringIncludesText(rule.byDay, option.value) ? (
																<div className={styles.iconWrapper}>
																	<Icon icon="check" size="sm" />
																</div>
															) : (
																<CheckboxElement size="sm" value={false} className={styles.checkboxElement} />
															)}

															<input
																className={styles.checkboxInput}
																value={String(stringIncludesText(rule.byDay, option.value))}
																data-qid-recurrence-by-day-of-week={option.value}
																type="checkbox"
																onChange={() => handleByDayToggle(option.value)}
															/>
														</label>
													))}
												</div>
											)
										}
									/>
								)}

								{rule.isMonthly && (

										<Field
											className={styles.timeOfMonth}
											label={intl.t('input.selectRecurrence.field.time-of-month')}

													label={intl.t('input.selectRecurrence.field.selection')}
													input={({ id }) => (

															value={rule.selectionValue?.value ?? null}

													className={styles.dayOfWeek}
													label={intl.t('input.selectRecurrence.field.day')}
													input={({ id }) => (
														<Select
															id={id}
															value={rule.byDayValueAsOption?.value ?? null}

								{rule.willSkipMonths && (
									<Notice variant="orange" data-qid-recurrence-skip-months-warning>
										{intl.t('input.selectRecurrence.skip-months', { count: rule.byMonthDay ?? rule.dtStartDay })}
									</Notice>
								)}

								<XStack spacing="md" width="full" className={styles.actions}>
									<div style={{ flex: 1 }} />
									<Button variant="transparent" data-qid-recurrence-cancel onClick={onCancel}>
										{intl.t('input.selectRecurrence.cancel')}
									</Button>
									<Button type="submit" variant="blue" data-qid-recurrence-save>
										{intl.t('input.selectRecurrence.apply')}
									</Button>
								</XStack>
							</YStack>
						)}
					</Form>
				</YStack>
			</div>
		);
	}

	// Preset list view
	return (
		<Menu data-qid-select-recurrence-presets>
			{presets.map((preset) => (
				<Menu.Option
					key={preset.id}
					selected={eqRule(value, preset.value)}
					data-qid-select-recurrence-preset={preset.id}
					onClick={() => handleSelectPreset(preset)}
				>
					{intl.t(preset.labelKey)}
				</Menu.Option>
			))}

			{allowCustom !== false && (
				<>
					<Divider className={styles.divider} tag="div" />

					<Menu.Option
						selected={isCustomRule}
						data-qid-select-recurrence-preset="custom"
						onClick={handleSwitchToCustom}
					>
						{intl.t('input.selectRecurrence.presets.custom')}
					</Menu.Option>
				</>
			)}
		</Menu>
	);
}

export default SelectRecurrenceBuilder;

let nextInputId = 0;

/**
 * A component for selecting recurrence patterns.
 */
export function SelectRecurrence(props: SelectRecurrenceProps) {
	const {
		ref,
		value,
		onChange,
		onCancel,
		active,
		placeholder,
		size = 'md',
		variant = 'white',
		disabled = false,
		id: providedId,
		requirements,
		name,
		placement = 'bottom',
		minFrequency,
		presetIds,
		allowCustom,
		singleWeeklyDay,
		serverErrors,
		ariaLabel,
		ariaLabelledBy,
		ariaDescribedBy,
		didAutosave,
		className,
		intl: intlProp,
		iconSlot,
		children,
	} = props;
	const generatedId = useId();
	const id = providedId ?? `luna-select-recurrence-${nextInputId++}`;
	const popupId = `${id}--popup`;

	const formContext = useFormContext();
	const triggerRef = useRef<HTMLButtonElement>(null);
	const intl = useMemo(() => intlProp ?? createDefaultIntl(), [intlProp]);

	// Parse the RRule when value changes
	const [rule, setRule] = useState(() => new RRule(value, intl));

	// Track popup visibility
	const [showPopup, setShowPopup] = useState(false);

	// Track internal selection
	const [selection, setSelection] = useState<string>(value);

	const valueIsUnchangedRef = useRef(true);

	// Update rule when value prop changes
	useEffect(() => {
		setRule(new RRule(value, intl));
	}, [value, intl]);

	// Compute presets based on requirements, presetIds, and minFrequency
	const presets = useMemo(() => {
		let filteredPresets = [...PRESETS];

		// Remove "none" option if required
		if (requirements?.required) {
			filteredPresets = filteredPresets.filter((p) => p.id !== 'none');
		}

		// Filter by explicit presetIds (overrides minFrequency)
		if (presetIds) {
			filteredPresets = filteredPresets.filter((preset) => presetIds.includes(preset.id));
		} else if (minFrequency) {
			filteredPresets = filteredPresets.filter((preset) => preset.minFrequency.includes(minFrequency));
		}

		return filteredPresets;
	}, [requirements?.required, presetIds, minFrequency]);

	// Input manager for form integration
	const manager = useInputManager({
		mode: 'change',
		name,
		valueIsUnchanged: valueIsUnchangedRef.current,
		didAutosave: useCallback(() => {
			valueIsUnchangedRef.current = true;
			didAutosave?.();
		}, [didAutosave]),
		ariaDescribedBy: ariaDescribedBy ?? null,
		value,
		isDisabled: disabled,
		requirements,
		serverErrors,
		formContext,
		selectorIsVisible: showPopup,
	} as InputManagerOptions);

	// Computed
	const triggerIsActive = active ?? showPopup;

	const formattedSelection = useMemo(() => {
		if (!value) return null;
		return rule.localeRecurrenceString || null;
	}, [value, rule]);

	// Handlers
	const handleFocusIn = useCallback(() => {
		manager.focusIn();
	}, [manager]);

	const handleOpen = useCallback(() => {
		if (disabled) return;
		setSelection(value);
		valueIsUnchangedRef.current = true;
		setShowPopup(true);
	}, [disabled, value]);

	const handleWillClose = useCallback(
		async (source: OnCloseSource, event: unknown) => {
			if (source === 'click-outside' && event instanceof Event) {
				const target = (event as MouseEvent).target as HTMLElement | null;
				if (target?.closest?.('[role="listbox"], [role="option"], [data-qid-component-type="Luna::Select"]')) {
					return false;
				}
			}

			if (valueIsUnchangedRef.current) {
				onCancel?.(source);
				return;
			}

			await onChange(selection);
			manager.onChange();
		},
		[selection, onCancel, onChange, manager],
	);

	const handleClose = useCallback(() => {
		setShowPopup(false);
	}, []);

	const handleBuilderChange = useCallback(
		(newValue: string) => {
			valueIsUnchangedRef.current = false;
			setSelection(newValue);
			// Close popup and apply change
			setShowPopup(false);
			onChange(newValue);
			manager.setValueIsUnchanged(false);
			manager.onChange();
		},
		[onChange, manager],
	);

	const handleBuilderCancel = useCallback(() => {
		setShowPopup(false);
	}, []);

	// Get computed aria-labelledby
	const computedAriaLabelledBy = useMemo(() => {
		if (ariaLabelledBy) return ariaLabelledBy;
		if (!ariaLabel) {
			return generatedId ? `${generatedId}-label` : undefined;
		}
		return undefined;
	}, [ariaLabelledBy, ariaLabel, generatedId]);

	return (
		<div
			ref={ref}
			className={cn(styles.SelectRecurrence, className)}
			data-size={size}
			data-disabled={disabled ? 'true' : undefined}
			data-qid-component-type="Luna::SelectRecurrence"
			data-focused={manager.inputHasFocus ? 'true' : undefined}
			data-errored={manager.showErrorMessage ? 'true' : undefined}
		>
			<Button
				ref={triggerRef}
				id={id}
				variant={variant}
				size={size}
				width="full"
				errored={manager.showErrorMessage}
				disabled={disabled}
				active={triggerIsActive}
				role="combobox"
				aria-controls={popupId}
				aria-expanded={showPopup}
				aria-owns={popupId}
				aria-haspopup="dialog"
				aria-required={requirements?.required ? 'true' : 'false'}
				aria-label={ariaLabel}
				aria-labelledby={computedAriaLabelledBy}
				aria-describedby={manager.ariaDescribedBy}
				aria-invalid={manager.showErrorMessage ? 'true' : undefined}
				onFocus={handleFocusIn}
				onClick={handleOpen}
			>
				{iconSlot ? <div className={styles.preIcon}>{iconSlot}</div> : null}

				<div className={styles.label}>
					{children
						? children(value)
						: (formattedSelection ?? placeholder ?? intl.t('input.selectRecurrence.presets.none'))}
				</div>

				{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
					<Spinner size="sm" transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined} />
				)}

				<Icon className={styles.icon} icon="chevron-down" size="sm" />
			</Button>

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
				{manager.errorMessageToDisplay}
			</InputError>

			{showPopup && (
				<Popup
					className={styles.popup}
					id={popupId}
					show={showPopup}
					relativeTo={triggerRef}
					placement={placement}
					trapFocus
					role="dialog"
					aria-label={!placeholder && !value ? intl.t('input.selectRecurrence.presets.none') : undefined}
					willClose={handleWillClose}
					onClose={handleClose}
				>
					<SelectRecurrenceBuilder
						value={value}
						presets={presets}
						minFrequency={minFrequency}
						singleWeeklyDay={singleWeeklyDay}
						allowCustom={allowCustom}
						onChange={handleBuilderChange}
						onCancel={handleBuilderCancel}
						intl={intl}
					/>
				</Popup>
			)}
		</div>
	);
}

export default SelectRecurrence;
