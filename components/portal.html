portal.gts
/* eslint-disable ember/no-runloop */
// NOTE: this does not export default because loose mode / hbs is not supported.
import { assert } from '@ember/debug';
import type Owner from '@ember/owner';
import { schedule } from '@ember/runloop';
import { buildWaiter } from '@ember/test-waiters';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';

const waiter = buildWaiter('luna-core:portal');

interface Signature {
	Element: null;
	Args: {
		/**
		 * A query string or Element to portal the default block contents in to
		 */
		to: string | Element;
	};
	Blocks: {
		/**
		 * The content to portal to the target element found via @to
		 */
		default: [];
	};
}

/**
 * A wrapper around the built-in `{{in-element}}` that queries the DOM for us,
 * even when an element might not yet be rendered.
 *
 * This works just like in element, where the default block's contents are all rendered in to
 * another location:
 *
 * ```gjs
 * import { Portal } from './portal.gts';
 *
 * <template>
 *   <Portal @to=".some-known-location">
 *      content here will render in to .some-known-location, if it exists.
 *   </Portal>
 * </template>
 * ```
 *
 * An error will be thrown if the element can't be found.
 */
export class Portal extends Component<Signature> {
	<template>
		{{#if this.portalTarget}}
			{{#in-element this.portalTarget insertBefore=null}}
				{{yield}}
			{{/in-element}}
		{{/if}}
	</template>

	constructor(owner: Owner, args: Signature['Args']) {
		super(owner, args);

		const to = this.args.to;

		this._eagerTarget = to instanceof Element ? to : document.getElementById(to);

		if (!this._eagerTarget) {
			const token = waiter.beginAsync();

			schedule('afterRender', () => {
				waiter.endAsync(token);
				if (this.isDestroyed || this.isDestroying) return;

				// This will assert if the element still isn't found
				this._target = queryElement(to);
			});
		}
	}

	/**
	 * We have to delay finding the portal target element, because we allow the element to dynamically be "anywhere"
	 * on the page, which means when we render portalled content it's possible that the portal target/element doesn't exist yet.
	 * This is common in tests, especially.
	 *
	 * Even for main app usage, because LunaSetup / the plain portal-target div is root/template are rendered *after* the {{outlet}}, we need to delay.
	 * However, the LunaSetup component may probably want move to the top of the application, rather than the bottom, so that
	 * we can portalled content can render in a single render pass, rather than waiting until the next render.
	 */
	_eagerTarget: Element | null = null;
	@tracked _target: Element | null = null;

	get portalTarget() {
		return this._eagerTarget ?? this._target;
	}
}

/**
 * Utility for finding or returning an element for use with in-element.
 * Will hard-error if the passed arg is ever falsey.
 *
 * If a falsey value has been passed, the caller has incorrect data.
 *
 * Examples:
 * ```gjs
 * <template>
 *  {{#in-element (queryElement "some-id")}}
 *
 *  {{/in-element}}
 *
 *  {{#in-element (queryElement "[query-selector]")}}
 *
 *  {{/in-element}}
 *
 *  {{#in-element document.body}}
 *
 *  {{/in-element}}
 * </template>
 * ```
 *
 */
function queryElement(query: string | null | undefined | Element) {
	assert(`Expected query/element to be truthy.`, query);

	if (query instanceof Element) {
		return query;
	}

	let found = document.getElementById(query);

	found ??= document.querySelector(query);

	assert(`Could not find element with id/selector ${query}`, found);

	return found;
}
