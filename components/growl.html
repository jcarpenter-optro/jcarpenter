Both libraries have a growl/notification component. In Ember it is exported as SetupGrowl from setup/growl.gts. In React it lives at components/growl/. Same concept, different naming convention.

import { service } from '@ember/service';
import Component from '@glimmer/component';

import type GrowlService from '../../services/growl.ts';
import type { Growl as GrowlComponent } from '../../services/growl.ts';

interface Signature {
	Element: HTMLElement;
	Args: NonNullable<unknown>;
}

export default class Growl extends Component<Signature> {
	@service('luna/growl') declare growl: GrowlService;

	get growls(): GrowlComponent[] {
		return this.growl.allGrowls;
	}

	<template>
		<div class="Growls" id="luna-growls">
			{{#each this.growls as |growl|}}
				<growl.component @data={{growl.data}} />
			{{/each}}
		</div>
	</template>
}
● React (TSX)
growl-context.tsx
import { createContext, type ReactNode, useCallback, useContext, useMemo } from 'react';

import { Toast } from '@base-ui-components/react/toast';

import { useAnnouncerOptional } from '../a11y-announcer';

/**
 * The type/variant of a growl notification
 */
export type GrowlType = 'success' | 'error' | 'warning' | 'info' | 'broadcast';

/**
 * Options for creating a growl notification
 */
export interface GrowlOptions {
	/** Optional header text */
	header?: string;
	/** Whether the growl should stay until manually dismissed (default: auto-dismiss after 7 seconds) */
	sticky?: boolean;
	/** Unique identifier for the growl - if provided, will replace any existing growl with the same uuid */
	uuid?: string;
}

/**
 * Data passed to the custom growl renderer
 */
export interface GrowlData {
	/** Function to close/dismiss this growl */
	close: () => void;
	/** The header text */
	header: string;
	/** Whether the message contains trusted HTML (SafeString) */
	isTrusted: boolean;
	/** The message content */
	message: string;
	/** Whether this growl persists until manually dismissed */
	sticky: boolean;
	/** Optional unique identifier */
	uuid?: string;
	/** The growl type/variant */
	type: GrowlType;
}

/**
 * A growl component that renders the notification UI
 */
export type GrowlComponent = React.ComponentType<{ data: GrowlData }>;

/**
 * Context value provided by GrowlProvider
 */
export interface GrowlContextValue {
	/** Show a success growl */
	success: (Component: GrowlComponent, message: string, options?: GrowlOptions) => void;
	/** Show an error growl */
	error: (Component: GrowlComponent, message: string, options?: GrowlOptions) => void;
	/** Show a warning growl */
	warning: (Component: GrowlComponent, message: string, options?: GrowlOptions) => void;
	/** Show an info growl */
	info: (Component: GrowlComponent, message: string, options?: GrowlOptions) => void;
	/** Show a broadcast growl */
	broadcast: (Component: GrowlComponent, message: string, options?: GrowlOptions) => void;
	/** The underlying toast manager for advanced usage */
	toastManager: ReturnType<typeof Toast.useToastManager>;
}

const GrowlContext = createContext<GrowlContextValue | null>(null);

/** Default timeout for auto-dismissing growls (7 seconds) */
const DEFAULT_TIMEOUT = 7000;

export interface GrowlProviderProps {
	/** Child components */
	children: ReactNode;
	/** Maximum number of growls to show at once (default: 5) */
	limit?: number;
	/** Custom timeout for auto-dismissing growls in ms (default: 7000) */
	timeout?: number;
}

/**
 * Provider that manages growl notifications.
 * Must wrap any components using the useGrowl hook.
 *
 * @example
 * ```tsx
 * <GrowlProvider>
 *   <App />
 *   <GrowlViewport />
 * </GrowlProvider>
 * ```
 */
export function GrowlProvider({ children, limit = 5, timeout = DEFAULT_TIMEOUT }: GrowlProviderProps) {
	return (
		<Toast.Provider limit={limit} timeout={timeout}>
			<GrowlProviderInner>{children}</GrowlProviderInner>
		</Toast.Provider>
	);
}

/**
 * Inner provider that has access to the toast manager context
 */
function GrowlProviderInner({ children }: { children: ReactNode }) {
	const toastManager = Toast.useToastManager();
	const announcer = useAnnouncerOptional();

	const showGrowl = useCallback(
		(
			type: GrowlType,
			Component: GrowlComponent,
			message: string,
			options: GrowlOptions = {},
			urgency: 'polite' | 'assertive' = 'polite',
		) => {
			const { header = '', sticky = false, uuid } = options;

			// If uuid exists, close any existing growl with the same uuid
			if (uuid) {
				const existingToast = toastManager.toasts.find((t) => (t.data as GrowlData | undefined)?.uuid === uuid);
				if (existingToast) {
					toastManager.close(existingToast.id);
				}
			}

			const growlData: GrowlData = {
				close: () => {
					// This will be updated once we have the toast ID
				},
				header,
				isTrusted: false, // React handles XSS protection
				message,
				sticky,
				uuid,
				type,
			};

			const toastId = toastManager.add({
				// Use sticky timeout (0 means no auto-dismiss)
				timeout: sticky ? 0 : undefined,
				type,
				data: {
					...growlData,
					Component,
				},
			});

			// Update the close function with the actual toast ID
			growlData.close = () => toastManager.close(toastId);

			// Announce to screen readers
			announcer?.announce(message, { urgency });
		},
		[toastManager, announcer],
	);

	const success = useCallback(
		(Component: GrowlComponent, message: string, options?: GrowlOptions) => {
			showGrowl('success', Component, message, options, 'polite');
		},
		[showGrowl],
	);

	const error = useCallback(
		(Component: GrowlComponent, message: string, options?: GrowlOptions) => {
			showGrowl('error', Component, message, options, 'assertive');
		},
		[showGrowl],
	);

	const warning = useCallback(
		(Component: GrowlComponent, message: string, options?: GrowlOptions) => {
			showGrowl('warning', Component, message, options, 'polite');
		},
		[showGrowl],
	);

	const info = useCallback(
		(Component: GrowlComponent, message: string, options?: GrowlOptions) => {
			showGrowl('info', Component, message, options, 'polite');
		},
		[showGrowl],
	);

	const broadcast = useCallback(
		(Component: GrowlComponent, message: string, options?: GrowlOptions) => {
			showGrowl('broadcast', Component, message, options, 'polite');
		},
		[showGrowl],
	);

	const contextValue = useMemo<GrowlContextValue>(
		() => ({
			success,
			error,
			warning,
			info,
			broadcast,
			toastManager,
		}),
		[success, error, warning, info, broadcast, toastManager],
	);

	return <GrowlContext.Provider value={contextValue}>{children}</GrowlContext.Provider>;
}

/**
 * Hook to access the growl notification system.
 *
 * @throws Error if used outside of GrowlProvider
 *
 * @example
 * ```tsx
 * function MyComponent() {
 *   const growl = useGrowl();
 *
 *   const handleSuccess = () => {
 *     growl.success(SuccessGrowl, 'Operation completed!', { header: 'Success' });
 *   };
 *
 *   return <button onClick={handleSuccess}>Do Something</button>;
 * }
 * ```
 */
export function useGrowl(): GrowlContextValue {
	const context = useContext(GrowlContext);
	if (!context) {
		throw new Error('useGrowl must be used within a GrowlProvider');
	}
	return context;
}

/**
 * Hook to optionally access the growl notification system.
 * Returns null if not within a GrowlProvider.
 */
export function useGrowlOptional(): GrowlContextValue | null {
	return useContext(GrowlContext);
}
growl-item.tsx
import { Toast } from '@base-ui-components/react/toast';

import { cn } from '../../utils/cn';
import { Heading } from '../heading';
import { Icon } from '../icon';
import { IconButton } from '../icon-button';
import { Text } from '../text';

import type { GrowlData, GrowlType } from './growl-context';

import styles from './growl.module.css';

export interface GrowlItemProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** The growl data passed from the GrowlViewport */
	data: GrowlData;
	/** Additional CSS class */
	className?: string;
	/** Children content (additional body content) */
	children?: React.ReactNode;
}

/** Icons for each growl type - using filled icons to match Ember */
const TYPE_ICONS: Record<GrowlType, string> = {
	success: 'circle-check-fill',
	error: 'circle-important-fill',
	warning: 'warning-fill',
	info: 'circle-info-fill',
	broadcast: 'circle-info-fill',
};

/**
 * Default growl item component that displays a notification with an icon,
 * header, message, and close button.
 *
 * This is a base implementation that can be used directly or as a reference
 * for creating custom growl components.
 *
 * @example
 * ```tsx
 * // Using the default GrowlItem
 * growl.success(GrowlItem, 'Operation completed!', { header: 'Success' });
 *
 * // Creating a custom growl component
 * function MyCustomGrowl({ data }: { data: GrowlData }) {
 *   return (
 *     <div className="custom-growl">
 *       <span>{data.message}</span>
 *       <button onClick={data.close}>Dismiss</button>
 *     </div>
 *   );
 * }
 * ```
 */
export function GrowlItem(props: GrowlItemProps) {
	const { ref, data, className, children } = props;
	const { close, header, message, type, uuid } = data;
	const icon = TYPE_ICONS[type];

	return (
		<div
			ref={ref}
			className={cn(styles.GrowlItem, className)}
			data-type={type}
			data-qid-component-type="Luna::GrowlItem"
			data-qid={uuid}
		>
			<div className={styles.IconWrapper} data-type={type}>
				<Icon icon={icon} size="md" />
			</div>

			<div className={styles.Body}>
				{header && <Toast.Title render={<Heading level={3} className={styles.Header} />}>{header}</Toast.Title>}

				<Toast.Description render={<Text className={styles.Message} />}>{message}</Toast.Description>

				{children}
			</div>

			<Toast.Close
				render={
					<IconButton
						icon="close"
						tooltip="Dismiss"
						variant="transparent"
						size="sm"
						onClick={close}
						className={styles.CloseButton}
						data-testid="growl-close-button"
					/>
				}
			/>
		</div>
	);
}

export default GrowlItem;
