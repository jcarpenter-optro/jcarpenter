import type { TOC } from '@ember/component/template-only';
import { htmlSafe } from '@ember/template';
import Component from '@glimmer/component';

import type { LunaSize } from '../types.ts';
import Badge from './badge.gts';
import Link from './link.gts';
import Text from './text.gts';

/**
 * Signature for the KeyValue sub-component yielded within DataCard's `:fields` block.
 * Each KeyValue renders a label/value pair in the card's fields row.
 */
interface KeyValueSignature {
	Element: HTMLDivElement;
	Args: {
		/** The label identifying a specific data point */
		key: string;
		/** The corresponding detail for its associated key */
		value: string | null | undefined;
	};
}

const KeyValue: TOC<KeyValueSignature> = <template>
	<div class="DataCard__key-value" role="listitem" ...attributes>
		<span class="DataCard__key" role="term" data-qid-data-card-key>{{@key}}</span>
		<span class="DataCard__value" role="definition" data-qid-data-card-value>
			{{#if @value}}
				{{@value}}
			{{else}}
				<span data-qid-data-card-value-placeholder>
					<Text @color="gray-300">—</Text>
				</span>
			{{/if}}
		</span>
	</div>
</template>;

/**
 * Signature for the DataCard component.
 *
 * DataCard serves as an entry point to more detailed information,
 * providing a scannable overview of essential data at a glance.
 */
export interface DataCardSignature {
	Element: HTMLElement;
	Args: {
		/** A unique identifier referencing the primary source associated with the card */
		id?: string;
		/** The primary headline of the card */
		title: string;
		/** When provided, renders the title as a link to this URL */
		href?: string;
		/** Supporting text that provides additional context or explanation */
		description?: string;
		/** Max width using a Luna size token value (e.g., 64, 80, 96). Omit for fluid width. */
		width?: LunaSize;
	};
	Blocks: {
		/** A high-level status indicator. Yields a pre-configured `Badge` component */
		status: [typeof Badge];
		/** Key/value pairs displayed in a row. Yields a `KeyValue` sub-component */
		fields: [typeof KeyValue];
	};
}

/**
 * A data card component that provides a scannable overview of essential data.
 *
 * @example
 * ```gts
 * <DataCard
 *   @id="CTRL-001"
 *   @title="Control Name"
 *   @href="/controls/1"
 *   @description="A brief description of the control"
 * >
 *   <:status as |Badge|>
 *     <Badge>Default</Badge>
 *   </:status>
 *   <:fields as |KeyValue|>
 *     <KeyValue @key="Owner" @value="Jane Doe" />
 *     <KeyValue @key="Due Date" @value="2026-03-15" />
 *   </:fields>
 * </DataCard>
 * ```
 */
export default class DataCard extends Component<DataCardSignature> {
	/** Computes max-width from a Luna size token when @width is provided */
	get widthStyle() {
		const { width } = this.args;
		if (width !== undefined) {
			return htmlSafe(`max-width: var(--luna-size-${width})`);
		}
		return undefined;
	}

	<template>
		<article class="DataCard" style={{this.widthStyle}} data-qid-component-type="Luna::DataCard" ...attributes>
			<div class="DataCard__header">
				{{#if @id}}
					<div class="DataCard__id" data-qid-data-card-id>
						{{@id}}
					</div>
				{{/if}}
				{{#if (has-block "status")}}
					<div class="DataCard__status" data-qid-data-card-status>
						{{yield Badge to="status"}}
					</div>
				{{/if}}
			</div>

			<div class="DataCard__title" data-qid-data-card-title>
				{{#if @href}}
					<Link @href={{@href}}>{{@title}}</Link>
				{{else}}
					{{@title}}
				{{/if}}
			</div>

			{{#if @description}}
				<div class="DataCard__description" data-qid-data-card-description>
					{{@description}}
				</div>
			{{/if}}

			{{#if (has-block "fields")}}
				<div class="DataCard__fields" role="list" data-qid-data-card-fields>
					{{yield KeyValue to="fields"}}
				</div>
			{{/if}}
		</article>
	</template>
}

export { KeyValue as DataCardKeyValue };

import { cn } from '../../utils/cn';
import type { ComponentProps, ComponentPropsWithRef } from '../../utils/component-types';
import { Badge, type BadgeColor } from '../badge/badge';
import { Frame } from '../frame/frame';
import { Link } from '../link/link';

import styles from './data-card.module.css';

// ---------------------------------------------------------------------------
// DataCardStatus
// ---------------------------------------------------------------------------

export type DataCardStatusType = 'success' | 'warning' | 'error' | 'info' | 'neutral';

const STATUS_COLORS: Record<DataCardStatusType, BadgeColor> = {
	success: 'green',
	warning: 'orange',
	error: 'red',
	info: 'blue',
	neutral: 'gray',
};

export interface DataCardStatusOwnProps {
	/** Semantic status type that maps to a Badge color. */
	type: DataCardStatusType;
	/** Text displayed inside the Badge. */
	label: string;
}

export type DataCardStatusProps = ComponentProps<DataCardStatusOwnProps, HTMLSpanElement>;

/**
 * Status badge rendered inside a DataCard header.
 */
export function DataCardStatus({ type, label, className }: DataCardStatusProps) {
	const color: BadgeColor = STATUS_COLORS[type] ?? 'gray';

	return (
		<Badge color={color} size="md" weight="weak" className={className} data-qid-component-type="Luna::DataCard::Status">
			{label}
		</Badge>
	);
}

// ---------------------------------------------------------------------------
// DataCardKeyValue
// ---------------------------------------------------------------------------

export interface DataCardKeyValueOwnProps {
	/** Label identifying the data point. */
	dataKey: string;
	/** Value content. */
	children?: React.ReactNode;
}

export type DataCardKeyValueProps = ComponentProps<DataCardKeyValueOwnProps, HTMLDivElement>;

/**
 * Renders a key/value pair inside a DataCard's `data` section.
 */
export function DataCardKeyValue({ dataKey, children, className, ...rest }: DataCardKeyValueProps) {
	return (
		<div
			className={cn(styles['DataCard__keyValue'], className)}
			data-qid-component-type="Luna::DataCard::KeyValue"
			{...rest}
		>
			<dt className={styles.DataCard__key} data-qid-data-card-key>
				{dataKey}
			</dt>
			<dd className={styles.DataCard__value} data-qid-data-card-value>
				{children != null ? (
					children
				) : (
					<span className={styles.DataCard__valuePlaceholder} data-qid-data-card-value-placeholder>
						—
					</span>
				)}
			</dd>
		</div>
	);
}

// DataCard

type LunaSize = string;

interface DataCardOwnProps {
	/** Unique identifier referencing the primary source (e.g., "RISK-42"). */
	id?: string;
	/** Primary headline of the card. */
	title?: string;
	/** When provided, the title renders as a link to this URL. */
	titleHref?: string;
	/** Supporting text providing additional context. */
	description?: string;
	/** Optional fixed width using a Luna size token value (e.g., "96", "80"). Omit for fluid (100%) width. */
	width?: LunaSize;
	/** Status badge slot. */
	status?: React.ReactNode;
	/** Key-value data section content. Use `<DataCard.KeyValue>` here. */
	data?: React.ReactNode;
	/** Children content */
	children?: React.ReactNode;
}

export type DataCardProps = ComponentPropsWithRef<DataCardOwnProps, HTMLDivElement>;

/**
 * A container that displays information on a single topic into a concise,
 * scannable snapshot, providing a quick overview of key content.
 */
export function DataCard(props: DataCardProps) {
	const { ref, id: cardId, title, titleHref, description, width, status, data, children, className, ...rest } = props;

	const hasHeader = status != null || cardId != null;

	return (
		<Frame
			ref={ref}
			border
			radius="md"
			padding="xl"
			className={cn(styles.DataCard, width && styles['DataCard--sized'], className)}
			{...(width ? { style: { '--data-card-max-width': `var(--luna-size-${width})` } as React.CSSProperties } : {})}
			data-qid-component-type="Luna::DataCard"
			{...rest}
		>
			{hasHeader && (
				<div className={styles.DataCard__header}>
					{cardId != null && (
						<span className={styles.DataCard__id} data-qid-data-card-id>
							{cardId}
						</span>
					)}
					{status != null && (
						<div className={styles.DataCard__status} data-qid-data-card-status>
							{status}
						</div>
					)}
				</div>
			)}

			{title != null && (
				<div className={styles.DataCard__title} data-qid-data-card-title>
					{titleHref ? <Link href={titleHref}>{title}</Link> : title}
				</div>
			)}

			{description != null && (
				<p className={styles.DataCard__description} data-qid-data-card-description>
					{description}
				</p>
			)}

			{data != null && (
				<dl className={styles.DataCard__data} data-qid-data-card-fields>
					{data}
				</dl>
			)}

			{children}
		</Frame>
	);
}

DataCard.Status = DataCardStatus;
DataCard.KeyValue = DataCardKeyValue;

export default DataCard;
