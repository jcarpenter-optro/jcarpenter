● Ember (Glimmer)
format-markdown.gts
import { service } from '@ember/service';
import type { SafeString } from '@ember/template';
import { htmlSafe, isHTMLSafe } from '@ember/template';
import Component from '@glimmer/component';
import { cached } from '@glimmer/tracking';

import { or } from 'ember-truth-helpers';

import type MarkdownService from '../services/markdown.ts';

interface FormatMarkdownSignature {
	Element: HTMLUListElement;
	Args: {
		text: string | SafeString | null;
		enabled: boolean;
		inline?: boolean;
	};
	Blocks: {
		default: [text: string | SafeString];
	};
}

/**
 * @deprecated
 * Do not use markdown for any new features or enhancements.
 *
 * New rich text content should be created using Luna RichTextRender from luna
 */
export default class FormatMarkdown extends Component<FormatMarkdownSignature> {
	// eslint-disable-next-line @typescript-eslint/no-deprecated
	@service('luna/markdown') declare markdown: MarkdownService;

	@cached
	get html() {
		if (!this.args.text) {
			return '';
		}
		// eslint-disable-next-line @typescript-eslint/no-deprecated
		return this.markdown.render(this.args.text, this.args.inline);
	}

	get htmlSafeHtml() {
		if (isHTMLSafe(this.html)) {
			return this.html;
		}

		return htmlSafe(this.html);
	}

	<template>
		{{#if @enabled}}
			<div class="markdown" data-qid-component-type="Luna::FormatMarkdown">
				{{#if (has-block)}}
					{{yield this.html}}
				{{else}}
					{{this.htmlSafeHtml}}
				{{/if}}
			</div>
		{{else}}
			{{#if (has-block)}}
				{{yield (or @text "")}}
			{{else}}
				<p>
					{{@text}}
				</p>
			{{/if}}
		{{/if}}
	</template>
}
