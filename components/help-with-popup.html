import { isTesting } from '@embroider/macros';

import { concat, fn } from '@ember/helper';
import { on } from '@ember/modifier';
import { action } from '@ember/object';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';

import { timeout } from 'ember-concurrency';
import t from 'ember-intl/helpers/t';
import setHelper from 'ember-set-helper/helpers/set';

import Block from './block.gts';
import Button from './button.gts';
import Trigger from './help/trigger.gts';
import Icon from './icon.gts';
import type { LunaPopupSignature } from './popup.gts';
import Popup from './popup.gts';

interface HelpWithPopupSignature {
	Element: HTMLDivElement;
	Args: {
		padding?: string;
		width?: string;
	} & Pick<LunaPopupSignature['Args'], 'placement' | 'enableArrow'>;
	// We do not enable variant for popup as dark popups don't support mouse events. See popup.css
	Blocks: {
		trigger?: [];
		default: [];
	};
}

let i = 1;

/**
 * A help used to show a popup with multiple types of content.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/help-with-popup.md)
 */
export default class HelpWithPopup extends Component<HelpWithPopupSignature> {
	targetId = `target--${i++}`;
	popupId = `popup--${i++}`;

	@tracked showPopup = false;

	@action
	async openPopup() {
		this.showPopup = true;

		await timeout(isTesting() ? 0 : 100);
	}

	get placement() {
		return this.args.placement ?? 'bottom-start';
	}

	get padding() {
		return this.args.padding ?? 'md';
	}

	<template>
		{{#if (has-block "trigger")}}
			<button
				class="trigger"
				type="button"
				aria-haspopup="dialog"
				aria-expanded={{this.showPopup}}
				aria-controls={{this.popupId}}
				id={{this.targetId}}
				data-qid-trigger
				data-qid-component-type="Luna::HelpWithPopup"
				{{on "click" this.openPopup}}
			>
				{{yield to="trigger"}}
			</button>
		{{else}}
			<Trigger
				@cursor="pointer"
				aria-haspopup="dialog"
				aria-expanded={{this.showPopup}}
				aria-controls={{this.popupId}}
				id={{this.targetId}}
				data-qid-trigger
				{{on "click" this.openPopup}}
			/>
		{{/if}}
		<Popup
			class="Popup"
			role="dialog"
			@id={{this.popupId}}
			@show={{this.showPopup}}
			@relativeTo={{concat "#" this.targetId}}
			@placement={{this.placement}}
			@enableArrow={{@enableArrow}}
			@trapFocus={{true}}
			@onClose={{setHelper this "showPopup" false}}
			...attributes
			as |p|
		>
			<Button
				{{on "click" (fn p.closeTask.perform "manual")}}
				@variant="transparent"
				@size="xs"
				aria-label={{t "command.close"}}
				data-test-close-popup
				class="close-button"
			>
				<Icon @icon="close" />
			</Button>
			<Block @padding={{this.padding}} class="block" data-width={{@width}}>
				{{yield}}
			</Block>
		</Popup>
	</template>
}

help-with-popup.tsx
import {
	Children,
	cloneElement,
	isValidElement,
	type ReactElement,
	type ReactNode,
	useCallback,
	useId,
	useRef,
	useState,
} from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { t } from '../../utils/i18n';
import { Block, type BlockPadding } from '../block';
import { Button } from '../button';
import { Icon } from '../icon';
import { Popup, type PopupPlacement } from '../popup';

import styles from './help-with-popup.module.css';

export type HelpWithPopupWidth = 'auto' | 'sm' | 'md' | 'lg';

/** Symbol used to identify Trigger slot when extracting children (avoids string keys like $$). */
export const HWP_TRIGGER = Symbol('HelpWithPopup.Trigger');
/** Symbol used to identify Content slot when extracting children. */
export const HWP_CONTENT = Symbol('HelpWithPopup.Content');

// ============================================================================
// Trigger Subcomponent
// ============================================================================

export interface TriggerProps {
	/** Children content for custom trigger */
	children?: ReactNode;
	/** Additional CSS class */
	className?: string;
}

/**
 * Custom trigger element for HelpWithPopup.
 * Replaces the default question mark icon.
 * Must be a valid React element that accepts onClick and ref props.
 */
export function Trigger({ children }: TriggerProps) {
	// Implementation is handled by parent component
	return <>{children}</>;
}

Trigger.displayName = 'HelpWithPopup.Trigger';
(Trigger as unknown as Record<symbol, boolean>)[HWP_TRIGGER] = true;

// ============================================================================
// Content Subcomponent
// ============================================================================

export interface ContentProps {
	/** Popup content */
	children?: ReactNode;
}

/**
 * Content to display inside the help popup.
 */
export function Content({ children }: ContentProps) {
	// Implementation is handled by parent component
	return <>{children}</>;
}

Content.displayName = 'HelpWithPopup.Content';
(Content as unknown as Record<symbol, boolean>)[HWP_CONTENT] = true;

// ============================================================================
// Main Component
// ============================================================================

interface HelpWithPopupBaseProps {
	/** Padding for the popup content block */
	'padding'?: BlockPadding;
	/** Width of the popup content */
	'width'?: HelpWithPopupWidth;
	/** Placement of the popup relative to the trigger */
	'placement'?: PopupPlacement;
	/** Whether to show an arrow on the popup */
	'enableArrow'?: boolean;
	/** Children components - use HelpWithPopup.Trigger and HelpWithPopup.Content */
	'children': ReactNode;
	/** Additional CSS class for the popup */
	'className'?: string;
	/** ARIA label for the popup */
	'aria-label'?: string;
}

export type HelpWithPopupProps = ComponentPropsWithRef<HelpWithPopupBaseProps, HTMLDivElement>;

/**
 * Extract trigger and content slots from children
 */
function extractHelpWithPopupSlots(children: ReactNode): {
	triggerSlot: ReactElement | null;
	contentSlot: ReactElement | null;
	otherChildren: ReactNode[];
} {
	let triggerSlot: ReactElement | null = null;
	let contentSlot: ReactElement | null = null;
	const otherChildren: ReactNode[] = [];

	Children.forEach(children, (child) => {
		if (!child || typeof child !== 'object' || !('type' in child)) {
			otherChildren.push(child);
			return;
		}

		const element = child as ReactElement & {
			type: Record<symbol, boolean>;
		};

		if (element.type?.[HWP_TRIGGER]) {
			triggerSlot = element;
		} else if (element.type?.[HWP_CONTENT]) {
			contentSlot = element;
		} else {
			otherChildren.push(child);
		}
	});

	return { triggerSlot, contentSlot, otherChildren };
}

interface DefaultTriggerProps {
	triggerRef: React.RefObject<HTMLButtonElement | null>;
	triggerProps: React.ButtonHTMLAttributes<HTMLButtonElement>;
}

function DefaultTrigger({ triggerRef, triggerProps }: DefaultTriggerProps) {
	return (
		<button
			ref={triggerRef}
			className={cn(styles.trigger, styles.defaultTrigger)}
			type="button"
			aria-label={t('help-with-popup.ariaHelp')}
			{...triggerProps}
		>
			<div className={styles.indicator}>
				<svg className={styles.icon} xmlns="http://www.w3.org/2000/svg" viewBox="0 0 12 12">
					<g fill="none" fillRule="evenodd" stroke="currentColor" strokeLinecap="round" strokeLinejoin="round">
						<path
							strokeWidth="1.5"
							d="M3.75 4S4 1.75 6 1.75 8.25 3 8.25 4c0 .751-.423 1.503-1.27 1.83C6.465 6.029 6 6.448 6 7v.25"
						/>
						<path d="M6.5 10a.5.5 0 11-1 0 .5.5 0 011 0z" />
					</g>
				</svg>
			</div>
		</button>
	);
}

interface CustomTriggerProps {
	triggerSlot: ReactElement<TriggerProps>;
	triggerRef: React.RefObject<HTMLButtonElement | null>;
	triggerProps: React.ButtonHTMLAttributes<HTMLButtonElement>;
}

function CustomTrigger({ triggerSlot, triggerRef, triggerProps }: CustomTriggerProps) {
	if (!isValidElement(triggerSlot)) {
		return null;
	}
	const triggerChildren = triggerSlot.props.children;
	if (!isValidElement(triggerChildren)) {
		return null;
	}
	const element = triggerChildren as React.ReactElement<{ className?: string }>;
	return cloneElement(element, {
		...triggerProps,
		ref: triggerRef,
		className: cn(styles.customTrigger, element.props.className),
	} as React.HTMLAttributes<HTMLElement> & { ref: React.RefObject<HTMLButtonElement> });
}

/**
 * A help component that displays a popup with content when triggered.
 * By default, shows a question mark icon as the trigger.
 *
 * Matches the Luna::HelpWithPopup Ember component API and styling.
 *
 * @example
 * ```tsx
 * // With default trigger (question mark icon)
 * <HelpWithPopup>
 *   <HelpWithPopup.Content>
 *     <p>Help content here</p>
 *   </HelpWithPopup.Content>
 * </HelpWithPopup>
 *
 * // With custom trigger
 * <HelpWithPopup>
 *   <HelpWithPopup.Trigger>
 *     <span>Custom trigger</span>
 *   </HelpWithPopup.Trigger>
 *   <HelpWithPopup.Content>
 *     <p>Help content here</p>
 *   </HelpWithPopup.Content>
 * </HelpWithPopup>
 * ```
 */
function HelpWithPopupComponent(props: HelpWithPopupProps) {
	const {
		ref,
		padding = 'md',
		width,
		placement = 'bottom-start',
		enableArrow = false,
		children,
		className,
		'aria-label': ariaLabel,
		...restProps
	} = props;
	const [showPopup, setShowPopup] = useState(false);
	const generatedId = useId();
	const targetId = `target--${generatedId}`;
	const popupId = `popup--${generatedId}`;
	const triggerRef = useRef<HTMLButtonElement>(null);

	const { triggerSlot, contentSlot } = extractHelpWithPopupSlots(children);

	const handleOpen = useCallback(() => {
		setShowPopup(true);
	}, []);

	const handleClose = useCallback(() => {
		setShowPopup(false);
	}, []);

	// Common props for the trigger element
	const triggerProps = {
		'aria-haspopup': 'dialog' as const,
		'aria-expanded': showPopup,
		'aria-controls': popupId,
		'id': targetId,
		'data-qid-trigger': true,
		'data-qid-component-type': 'Luna::HelpWithPopup',
		'onClick': handleOpen,
	};

	return (
		<>
			{triggerSlot ? (
				<CustomTrigger
					triggerSlot={triggerSlot as ReactElement<TriggerProps>}
					triggerRef={triggerRef}
					triggerProps={triggerProps}
				/>
			) : (
				<DefaultTrigger triggerRef={triggerRef} triggerProps={triggerProps} />
			)}
			<Popup
				ref={ref}
				id={popupId}
				show={showPopup}
				relativeTo={triggerRef}
				placement={placement}
				enableArrow={enableArrow}
				trapFocus={true}
				allowOverflow={true}
				onClose={handleClose}
				className={className}
				aria-label={ariaLabel}
				data-testid="popup"
				{...restProps}
			>
				<Button
					variant="transparent"
					size="xs"
					aria-label={t('help-with-popup.ariaClose')}
					className={styles.closeButton}
					onClick={handleClose}
					data-test-close-popup
				>
					<Icon icon="close" />
				</Button>
				<Block padding={padding} className={styles.block} data-width={width}>
					{contentSlot}
				</Block>
			</Popup>
		</>
	);
}

// Attach subcomponents
const HelpWithPopup = Object.assign(HelpWithPopupComponent, {
	Trigger,
	Content,
});

export { HelpWithPopup };

export default HelpWithPopup;
