import { assert } from '@ember/debug';
import { registerDestructor } from '@ember/destroyable';
import { on } from '@ember/modifier';
import type Owner from '@ember/owner';
import Component from '@glimmer/component';

import { timeout } from 'ember-concurrency';
import { consume } from 'ember-provide-consume-context';
import { eq } from 'ember-truth-helpers';
import or from 'ember-truth-helpers/helpers/or';

import { focusById } from '../utils/focus.ts';
import type { InputManagerOptions } from '../utils/input-manager.ts';
import InputManager from '../utils/input-manager.ts';
import type { LunaFormBlockValue } from './form.ts';
import Spinner from './spinner.gts';

let inputId = 0;

export interface LunaRadioButtonSignature {
	Element: HTMLButtonElement;
	Args: {
		ariaLabel?: string;
		ariaDescribedBy?: string;
		getOptionId?: (value: string) => string;
		id?: string;
		isDisabled?: boolean;
		label?: string;
		selectedValue?: string | null;
		size?: 'sm' | 'md' | 'lg';
		onSelect?: (value: string) => unknown;
		value?: string;
	};
	Blocks: {
		default?: [];
	};
}

/**
 * A radio button. Used for selecting one option from a group of options.
 * Should generally only be used by the RadioGroup component.
 */
export default class RadioButton extends Component<LunaRadioButtonSignature> {
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;
	@consume('luna-input-id') inputId!: string;

	manager: InputManager;

	constructor(owner: Owner, args: LunaRadioButtonSignature['Args']) {
		super(owner, args);

		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;

		const managerOptions: InputManagerOptions = {
			mode: 'change',
			name: that.id,
			focus: this.focusInput.bind(this),
			ariaDescribedBy: args.ariaDescribedBy ?? null,

			get value(): string {
				assert('value is required', args.value !== undefined);
				return args.value;
			},

			get isDisabled(): boolean {
				return args.isDisabled ?? false;
			},

			get formContext() {
				return that.formContext;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	get id() {
		return this.args.id ?? this.args.getOptionId?.(this.value) ?? this.inputId ?? `luna-radio-button-${inputId++}`;
	}

	get isSelected() {
		return this.args.selectedValue === this.value ? 'true' : 'false';
	}

	get value() {
		assert('value is required', this.args.value !== undefined);
		return this.args.value;
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	select = async () => {
		await this.args.onSelect?.(this.value);
		this.manager.onChange();
	};

	focusInput = () => {
		focusById(this.args.id);
	};

	<template>
		<button
			...attributes
			data-focused={{if this.manager.inputHasFocus "true" "false"}}
			aria-checked={{this.isSelected}}
			aria-describedby={{this.manager.ariaDescribedBy}}
			class="RadioButton"
			data-qid-luna-radio-button={{this.id}}
			data-size={{or @size "md"}}
			disabled={{@isDisabled}}
			id={{this.id}}
			role="radio"
			tabindex={{if (eq @selectedValue this.value) 0 -1}}
			type="button"
			{{on "click" this.select}}
		>
			<div class="radio-button-element"></div>
			<div data-qid-luna-radio-button-label>
				{{#if (has-block)}}
					{{yield}}
				{{else}}
					{{@label}}
				{{/if}}
			</div>
			{{#if (or this.manager.isAutosaving this.manager.didAutosaveSuccessfully)}}
				<div class="spinner">
					<Spinner role="presentation" @size="sm" @transitionTo={{if this.manager.didAutosaveSuccessfully "success"}} />
				</div>
			{{/if}}
		</button>
	</template>
}

import { type MutableRefObject, useCallback, useEffect, useId, useRef, useState } from 'react';

import { Radio as BaseRadio } from '@base-ui-components/react/radio';
import type { Requirements } from '@soxhub/luna-utils/validators';

import { cn } from '../../utils/cn';
import { useInputManager } from '../../utils/use-input-manager';
import { useFormContext } from '../form/form';
import { InputError } from '../input-error';
import { Spinner } from '../spinner';

import styles from './radio-button.module.css';

export type RadioButtonSize = 'sm' | 'md' | 'lg';

interface RadioButtonBaseProps {
	/** Forwarded ref to the underlying button element */
	'ref'?: React.Ref<HTMLButtonElement>;
	/** The value of this radio button */
	'value': string;
	/** The currently selected value (for controlled usage) */
	'selectedValue'?: string | null;
	/** Callback fired when radio button is selected */
	'onSelect'?: (value: string) => void;
	/** Radio button size */
	'size'?: RadioButtonSize;
	/** Whether radio button is disabled */
	'isDisabled'?: boolean;
	/** Radio button ID (auto-generated if not provided) */
	'id'?: string;
	/** Radio button name */
	'name'?: string;
	/** ARIA label (for non-label scenarios) */
	'ariaLabel'?: string;
	/** Label text (if not using children) */
	'label'?: string;
	/** ARIA described by */
	'aria-describedby'?: string;
	/** Additional CSS class */
	'className'?: string;
	/** Children (label content, overrides label prop) */
	'children'?: React.ReactNode;
	/** Validation requirements */
	'requirements'?: Requirements;
	/** @deprecated Server errors (use externalErrorsFor on Form instead) */
	'serverErrors'?: Array<{ message?: string } | string>;
	/** Callback fired when autosave completes successfully */
	'didAutosave'?: () => void;
}

type RadioButtonProps = RadioButtonBaseProps &
	Omit<React.ButtonHTMLAttributes<HTMLButtonElement>, keyof RadioButtonBaseProps | 'onSelect'>;

/**
 * A radio button. Used for selecting one option from a group of options.
 *
 * Matches the Luna::RadioButton Ember component API and styling.
 * Uses Base UI Radio as the foundation.
 */
export function RadioButton(props: RadioButtonProps) {
	const {
		ref,
		value,
		selectedValue,
		onSelect,
		size = 'md',
		isDisabled = false,
		'id': providedId,
		name,
		label,
		'aria-describedby': providedAriaDescribedBy,
		className,
		children,
		requirements,
		serverErrors,
		didAutosave,
		...rest
	} = props;

	const generatedId = useId();
	const id = providedId ?? generatedId;
	const formContext = useFormContext();
	const radioButtonRef = useRef<HTMLButtonElement>(null);
	const valueIsUnchangedRef = useRef(true);

	const focusInput = useCallback(() => {
		if (radioButtonRef.current) {
			radioButtonRef.current.focus();
		} else if (ref && typeof ref === 'object' && ref.current) {
			ref.current.focus();
		}
	}, [ref]);

	const manager = useInputManager({
		mode: 'change',
		valueIsUnchanged: valueIsUnchangedRef.current,
		serverErrors,
		requirements,
		name,
		value: selectedValue === value ? value : undefined,
		isDisabled,
		formContext,
		ariaDescribedBy: providedAriaDescribedBy ?? null,
		focus: focusInput,
		didAutosave: () => {
			valueIsUnchangedRef.current = true;
			didAutosave?.();
		},
	});

	// Register/unregister with form
	const managerRef = useRef(manager);
	managerRef.current = manager;

	useEffect(() => {
		if (formContext) {
			formContext.actions.registerInputManager(managerRef.current);
			return () => {
				formContext.actions.unregisterInputManager(managerRef.current);
			};
		}
	}, [formContext]);

	const [focused, setFocused] = useState(false);
	const focusedState = focused || manager.inputHasFocus;

	const isSelected = selectedValue === value;

	const handleSelect = useCallback(async () => {
		if (!isDisabled) {
			valueIsUnchangedRef.current = false;
			manager.setValueIsUnchanged(false);
			await onSelect?.(value);
			manager.onChange();
		}
	}, [onSelect, value, isDisabled, manager]);

	const handleFocus = useCallback(() => {
		setFocused(true);
		manager.focusIn();
	}, [manager]);

	const handleBlur = useCallback(() => {
		setFocused(false);
		manager.focusOutAndTrySave(new Event('blur'));
	}, [manager]);

	const radioButtonClassName = cn(styles.RadioButton, className);

	const labelContent = children ?? label;

	return (
		<div>
			<BaseRadio.Root
				value={value}
				disabled={isDisabled}
				render={(radioProps) => (
					<button
						{...radioProps}
						{...rest}
						ref={(node) => {
							radioButtonRef.current = node;
							if (typeof ref === 'function') {
								ref(node);
							} else if (ref) {
								(ref as MutableRefObject<HTMLButtonElement | null>).current = node;
							}
						}}
						className={radioButtonClassName}
						data-focused={focusedState ? 'true' : 'false'}
						aria-checked={isSelected ? 'true' : 'false'}
						aria-describedby={manager.ariaDescribedBy ?? providedAriaDescribedBy}
						aria-invalid={manager.showErrorMessage ? 'true' : undefined}
						data-qid-luna-radio-button={id}
						data-size={size}
						disabled={isDisabled}
						id={id}
						role="radio"
						tabIndex={isSelected ? 0 : -1}
						type="button"
						onClick={handleSelect}
						onFocus={handleFocus}
						onBlur={handleBlur}
					>
						<div className={styles.radioButtonElement}></div>
						{labelContent && (
							<div data-qid-luna-radio-button-label className={styles.label}>
								{labelContent}
								{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
									<Spinner
										size="sm"
										transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined}
										className={styles.spinner}
									/>
								)}
							</div>
						)}
					</button>
				)}
			/>

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage} align="left">
				{manager.errorMessageToDisplay}
			</InputError>
		</div>
	);
}

export default RadioButton;
