export interface LunaTextArgs {
	font?: LunaFont;
	color?: LunaColor;
	weight?: 'weak' | 'normal' | 'medium' | 'strong';
	clamp?: 1 | 2 | 3 | 4 | 5;
	align?: LunaAlign;
}

export interface LunaTextSignature<TagName extends OptionalTagName = undefined> {
	Element: ElementFromTagName<TagName, HTMLSpanElement>;
	Args: LunaTextArgs & OptionalTagAttr<TagName>;
	Blocks: {
		default: [];
	};
}

	get tagName(): keyof HTMLElementTagNameMap {
		return this.args.tag ?? 'span';

● React (TSX)
text.tsx
import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './text.module.css';

export type LunaFont = 'inherit' | '25' | '50' | '100' | '150' | '200' | '300' | '400' | '450' | '500' | '600';
export type LunaColor = string; // TODO: Import from types when available
export type LunaAlign = 'left' | 'center' | 'right';
export type TextWeight = 'weak' | 'normal' | 'medium' | 'strong';
export type TextClamp = 1 | 2 | 3 | 4 | 5;

type TextTag = keyof React.JSX.IntrinsicElements;

interface TextBaseProps {
	/** Font size from typography scale */
	font?: LunaFont;
	/** Text color */
	color?: LunaColor;
	/** Font weight */
	weight?: TextWeight;
	/** Line clamp (truncate after N lines) */
	clamp?: TextClamp;
	/** Text alignment */
	align?: LunaAlign;
	/** HTML tag to render as */
	tag?: TextTag;
	/** Children content */
	children?: React.ReactNode;
}

type TextProps = ComponentPropsWithRef<TextBaseProps, HTMLElement>;

/**
 * Shows text styled using the typography scale.
 *
 * Matches the Luna::Text Ember component API and styling.
 */
export function Text(props: TextProps & { ref?: React.Ref<HTMLElement> }) {
	const {
		ref,
		font = 'inherit',
		color,
		weight = 'normal',
		clamp,
		align = 'left',
		tag = 'span',
		children,
		className,
		...rest
	} = props;

	const Tag = tag as React.ElementType;

	const textClassName = cn(styles.Text, className);

	return (
		<Tag
			ref={ref}
			className={textClassName}
			data-font={font}
			data-color={color}
			data-weight={weight}
			data-clamp={clamp}
			data-align={align}
			data-qid-component-type="Luna::Text"
			{...rest}
		>
			{children}
		</Tag>
	);
}

export default Text;
