import type { WithBoundArgs } from '@glint/template';

import { concat, hash } from '@ember/helper';
import { action } from '@ember/object';
import { guidFor } from '@ember/object/internals';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';

import setHelper from 'ember-set-helper/helpers/set';

import MenuButton from './button-with-menu/button.gts';
import Menu from './button-with-menu/menu.gts';

interface ButtonWithMenuSignature {
	Blocks: {
		default: [
			{
				Button: WithBoundArgs<typeof MenuButton, 'id' | 'popupId' | 'isOpen' | 'showMenu'>;
				Popup: WithBoundArgs<typeof Menu, 'popupId' | 'isOpen' | 'relativeTo' | 'hideMenu'>;
				popupIsOpen: boolean;
				showMenu: (event: Event) => void;
				hideMenu: () => void;
			},
		];
	};
}

/**
 * A button used to show a menu of options to the user. The menu closes once the user selects an option.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/button-with-menu.md)
 */
export default class ButtonWithMenu extends Component<ButtonWithMenuSignature> {
	targetId = `${guidFor(this)}--target`;
	popupId = `${guidFor(this)}--popup`;

	@tracked showMenu = false;

	@action
	openMenu(event: Event) {
		/**
		 * This shouldn't be necessary, but in our app we unfortunately nest
		 * clickable elements right now so clicking the button would also
		 * trigger another action we don't want to.
		 */
		event.stopPropagation();

		this.showMenu = true;
	}

	<template>
		{{yield
			(hash
				Button=(component MenuButton id=this.targetId popupId=this.popupId isOpen=this.showMenu showMenu=this.openMenu)
				Popup=(component
					Menu
					popupId=this.popupId
					isOpen=this.showMenu
					relativeTo=(concat "#" this.targetId)
					hideMenu=(setHelper this "showMenu" false)
				)
				popupIsOpen=this.showMenu
				showMenu=this.openMenu
				hideMenu=(setHelper this "showMenu" false)
			)
		}}
	</template>
}

	useContext,
	useId,
	useRef,
	useState,

import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Button, type ButtonProps } from '../button';
import { Menu } from '../menu';
import { Option, type OptionProps } from '../option';
import { OptionHeading } from '../option-heading';
import { Popup, type PopupPlacement, usePopupContext } from '../popup';

import styles from './button-with-menu.module.css';

/**
 * Context for sharing menu state between ButtonWithMenu and its subcomponents
 */
interface ButtonWithMenuContextValue {
	/** Whether the menu is currently open */
	isOpen: boolean;
	/** Open the menu */
	showMenu: (event?: React.MouseEvent) => void;
	/** Close the menu */
	hideMenu: () => void;
	/** ID of the trigger button */
	buttonId: string;
	/** ID of the popup menu */
	popupId: string;
	/** Ref to the trigger button element */
	buttonRef: RefObject<HTMLButtonElement | null>;
}

const ButtonWithMenuContext = createContext<ButtonWithMenuContextValue | null>(null);

function useButtonWithMenuContext() {
	const context = useContext(ButtonWithMenuContext);
	if (!context) {
		throw new Error('ButtonWithMenu subcomponents must be used within ButtonWithMenu');
	}
	return context;
}

/**
 * Hook to access the ButtonWithMenu control functions and state from within menu content or button children
 */
export function useButtonWithMenuHandle(): {
	showMenu: (event?: React.MouseEvent) => void;
	hideMenu: () => void;
	popupIsOpen: boolean;
} {
	const context = useButtonWithMenuContext();
	return { showMenu: context.showMenu, hideMenu: context.hideMenu, popupIsOpen: context.isOpen };
}

type ButtonOnlyProps = Extract<ButtonProps, { href?: undefined }>;

type MenuButtonBaseProps = Omit<ButtonOnlyProps, 'active' | 'aria-haspopup' | 'aria-expanded' | 'aria-controls'>;

export type MenuButtonProps = ComponentPropsWithRef<MenuButtonBaseProps, HTMLButtonElement>;

/**
 * The trigger button for ButtonWithMenu.
 * Automatically handles ARIA attributes and active state.
 */
export function MenuButton(props: MenuButtonProps) {
	const { ref, children, onClick, ...buttonProps } = props;
	const { isOpen, showMenu, buttonId, popupId, buttonRef } = useButtonWithMenuContext();

	type ButtonClickEvent = Parameters<NonNullable<MenuButtonProps['onClick']>>[0];

	const handleClick = useCallback(
		(event: ButtonClickEvent) => {
			event.stopPropagation();
			showMenu(event);
			onClick?.(event);
		},
		[showMenu, onClick],
	);

	// Merge refs
	const setRef = useCallback(
		(el: HTMLButtonElement | null) => {
			(buttonRef as MutableRefObject<HTMLButtonElement | null>).current = el;
			if (typeof ref === 'function') {
				ref(el);
			} else if (ref) {
				(ref as MutableRefObject<HTMLButtonElement | null>).current = el;
			}
		},
		[buttonRef, ref],
	);

	return (
		<Button
			ref={setRef}
			id={buttonId}
			active={isOpen}
			aria-haspopup="menu"
			aria-expanded={isOpen}
			aria-controls={popupId}
			data-qid-luna-button-with-menu-button
			onClick={handleClick}
			{...buttonProps}
		>
			{children}
		</Button>
	);
}

MenuButton.displayName = 'ButtonWithMenu.Button';
MenuButton.$$lunaBwmButton = true as const;

/**
 * An option within the ButtonWithMenu popup.
 * Automatically closes the menu when clicked.
 */
export function MenuOption(props: MenuOptionProps) {
	const { ref, onClick, children, ...optionProps } = props;
	const popupContext = usePopupContext();

	const handleClick = useCallback(() => {
		onClick?.();
		// Close the popup after the option action
		popupContext?.close('manual');
	}, [onClick, popupContext]);

	return (
		<Option ref={ref} onClick={handleClick} {...optionProps}>
			{children}
		</Option>
	);
}

/**
 * Extract button and menu slots from children
 */
function extractButtonWithMenuSlots(children: ReactNode): {
	buttonSlot: ReactElement | null;
	menuSlot: ReactElement | null;
	otherChildren: ReactNode[];
} {
	let buttonSlot: ReactElement | null = null;
	let menuSlot: ReactElement | null = null;
	const otherChildren: ReactNode[] = [];

	Children.forEach(children, (child) => {
		if (!child || typeof child !== 'object' || !('type' in child)) {
			otherChildren.push(child);
			return;
		}

		const element = child as ReactElement & {
			type: { $$lunaBwmButton?: boolean; $$lunaBwmMenu?: boolean };
		};

		if (element.type?.$$lunaBwmButton) {
			buttonSlot = element;
		} else if (element.type?.$$lunaBwmMenu) {
			menuSlot = element;
		} else {
			otherChildren.push(child);
		}
	});

	return { buttonSlot, menuSlot, otherChildren };
}

export interface ButtonWithMenuProps {
	/** Children components - use ButtonWithMenu.Button and ButtonWithMenu.Menu */
	children: ReactNode;
}

/**
 * A button used to show a menu of options to the user.
 * The menu closes once the user selects an option.
 *
 * @example
 * ```tsx
 * <ButtonWithMenu>
 *   <ButtonWithMenu.Button>Options</ButtonWithMenu.Button>
 *   <ButtonWithMenu.Menu>
 *     <ButtonWithMenu.Option onClick={() => console.log('Edit')}>Edit</ButtonWithMenu.Option>
 *     <ButtonWithMenu.Option onClick={() => console.log('Delete')}>Delete</ButtonWithMenu.Option>
 *   </ButtonWithMenu.Menu>
 * </ButtonWithMenu>
 * ```
 */
function ButtonWithMenuComponent({ children }: ButtonWithMenuProps) {
	const id = useId();
	const buttonId = `${id}--target`;
	const popupId = `${id}--popup`;
	const buttonRef = useRef<HTMLButtonElement | null>(null);

	const [isOpen, setIsOpen] = useState(false);

	const showMenu = useCallback((event?: React.MouseEvent) => {
		event?.stopPropagation();
		setIsOpen(true);
	}, []);

	const hideMenu = useCallback(() => {
		setIsOpen(false);
	}, []);

	const contextValue: ButtonWithMenuContextValue = {
		isOpen,
		showMenu,
		hideMenu,
		buttonId,
		popupId,
		buttonRef,
	};

	const { buttonSlot, menuSlot } = extractButtonWithMenuSlots(children);

	return (
		<ButtonWithMenuContext.Provider value={contextValue}>
			{buttonSlot}
			{menuSlot}
		</ButtonWithMenuContext.Provider>
	);
}

// Attach subcomponents
const ButtonWithMenu = Object.assign(ButtonWithMenuComponent, {
	Button: MenuButton,
	Menu: MenuPopup,
	Option: MenuOption,
	Heading: OptionHeading,
});

export { ButtonWithMenu };

export default ButtonWithMenu;
