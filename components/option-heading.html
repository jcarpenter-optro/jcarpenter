● React (TSX)
option-heading.tsx
import { cn } from '../../utils/cn';

import styles from './option-heading.module.css';

export interface OptionHeadingProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** Children content */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

/**
 * A heading/separator for grouping menu options.
 *
 * Matches the Luna::OptionHeading Ember component API and styling.
 * Note: Base UI Menu.Separator requires Menu.Popup context, but Luna Menu is standalone,
 * so we render a simple separator div instead.
 */
export function OptionHeading(props: OptionHeadingProps) {
	const { ref, children, className, ...rest } = props;

	const headingClassName = cn(styles.OptionHeading, className);

	return (
		<div ref={ref} className={headingClassName} role="separator" {...rest}>
			{children}
		</div>
	);
}

export default OptionHeading;
