bubble-menu.gts
import type { Editor, SingleCommands } from '@tiptap/core';

import type { TOC } from '@ember/component/template-only';
import { fn } from '@ember/helper';
import { on } from '@ember/modifier';

import { attachBubbleMenu, editorStateIsActive, runCommand } from '@auditboard/rich-text/lib/bubble-menu';

import IconButton from '../icon-button.gts';
import XStack from '../x-stack.gts';
import { Highlight } from './bubble-menu/highlight.gts';

export interface BubbleMenuSignature {
	Args: {
		editor: Editor;
	};
}

type SupportedCommandKeys = keyof SingleCommands | 'toggleHighlight' | 'toggleBulletList' | 'toggleOrderedList';

function hasCommand(editor: Editor, commandKey: string): commandKey is keyof SingleCommands {
	return editor.commands[commandKey as keyof SingleCommands] !== undefined;
}

const RunButton: TOC<{
	Args: {
		command: SupportedCommandKeys;
		editor: Editor;
		icon: string;
		tooltip: string;
		stateName: string;
	};
}> = <template>
	{{#if (hasCommand @editor @command)}}
		{{#let (editorStateIsActive @editor @stateName) as |active|}}
			<IconButton
				@icon={{@icon}}
				@tooltip={{@tooltip}}
				@variant={{if active "blue" "transparent"}}
				@size="xs"
				{{on "click" (fn runCommand @editor @command)}}
				data-qid-rich-text-bubble-menu-button={{@command}}
				data-active={{String active}}
			/>
		{{/let}}
	{{/if}}
</template>;

export const BubbleMenu: TOC<BubbleMenuSignature> = <template>
	<div class="bubble-menu" data-qid-component-type="Luna::RichText::BubbleMenu" {{attachBubbleMenu @editor}}>
		<XStack @spacing="sm">
			{{! @glint-expect-error incorrect command type }}
			<RunButton @editor={{@editor}} @tooltip="Bold" @command="toggleBold" @icon="bold" @stateName="bold" />
			{{! @glint-expect-error incorrect command type }}
			<RunButton @editor={{@editor}} @tooltip="Italic" @command="toggleItalic" @icon="italic" @stateName="italic" />
			<RunButton
				@editor={{@editor}}
				@tooltip="Underline"
				{{! @glint-expect-error incorrect command type }}
				@command="toggleUnderline"
				@icon="underline"
				@stateName="underline"
			/>
			<RunButton
				@editor={{@editor}}
				@tooltip="Bulleted List"
				@command="toggleBulletList"
				@icon="list-ul"
				@stateName="bulletList"
			/>
			<RunButton
				@editor={{@editor}}
				@tooltip="Ordered List"
				@command="toggleOrderedList"
				@icon="list-ol"
				@stateName="orderedList"
			/>

			{{#if (hasCommand @editor "toggleHighlight")}}
				<Highlight @editor={{@editor}} />
			{{/if}}
		</XStack>
	</div>
</template>;
