import { assert } from '@ember/debug';
import Component from '@glimmer/component';

import { consume } from 'ember-provide-consume-context';
import { and, not, or } from 'ember-truth-helpers';

import { RENDERING_IN_SCHEMA_TABLE_CONTEXT } from '@auditboard/consts/schema-table';

import EmptyStateAnnouncer from './empty-state-announcer.gts';
import Heading from './heading.gts';
import Icon from './icon.gts';
import type { IllustrationName } from './illustration.gts';
import LunaIllustration from './illustration.gts';
import Text from './text.gts';

interface EmptyStateSignature {
	Element: HTMLElement;
	Args: {
		/**
		 * Sets the url of the image.
		 */
		imageUrl?: string;
		/**
		 * Name of luna icon to display.
		 */
		icon?: string;
		/**
		 * Sets the heading message.
		 */
		heading?: string;
		/**
		 * Sets the description message.
		 */
		description?: string;
		/**
		 * Sets the illustration to display
		 */
		illustration?: IllustrationName;
		/**
		 * Sets the heading level (1-6). Defaults to 4.
		 */
		headingLevel?: 1 | 2 | 3 | 4 | 5 | 6;
	};
	Blocks: {
		'heading': [];
		'description': [];
		'primary-actions': [];
		'secondary-actions': [];
	};
}

// remove when we upgrade to ember-source@5.2, and replace with built-in uniqueId
let emptyStateCount = 0;
function uniqueId(): string {
	return `empty-state-${emptyStateCount++}`;
}

/**
 * A temporary placeholder for empty, error, or success states that provides an explanation and guidance on next steps the user can take.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/empty-state.md)
 */
class EmptyState extends Component<EmptyStateSignature> {
	@consume(RENDERING_IN_SCHEMA_TABLE_CONTEXT) isInSchemaTable: boolean | undefined;

	<template>
		{{#let (uniqueId) as |emptyStateId|}}
			<aside
				class="EmptyState"
				data-align="center"
				data-qid-component-type="Luna::EmptyState"
				aria-labelledby={{emptyStateId}}
				...attributes
			>
				{{#if @illustration}}
					<div class="graphic illustration">
						<LunaIllustration @illustration={{@illustration}} @size="24" />
					</div>
				{{else if @imageUrl}}
					<div class="graphic" data-qid-empty-state-image>
						<img src={{@imageUrl}} aria-hidden="true" />
					</div>
				{{else if @icon}}
					<div class="graphic icon">
						<Icon @icon={{@icon}} @size="3xl" />
					</div>
				{{/if}}

				<Heading
					id={{emptyStateId}}
					@level={{or @headingLevel 4}}
					@font="400"
					@weight="strong"
					data-qid-empty-state-heading
				>
					{{#if (has-block "heading")}}
						{{yield to="heading"}}
					{{else}}
						{{assert
							"You must pass a @heading arg or use the <:heading> block when using the <EmptyState> component"
							@heading
						}}
						{{#if this.isInSchemaTable}}
							<EmptyStateAnnouncer @text={{@heading}} />
						{{else}}
							{{@heading}}
						{{/if}}
					{{/if}}
				</Heading>

				<Text @tag="p" @clamp={{3}} @font="200" @color="gray-500" @align="center" data-qid-empty-state-description>
					{{#if (has-block "description")}}
						{{yield to="description"}}
					{{else}}
						{{assert
							"You must pass a @description arg or use the <:description> block when using the <EmptyState> component"
							@description
						}}
						{{! In schema table without a heading, announce the description }}
						{{#if (and this.isInSchemaTable (not (or @heading (has-block "heading"))))}}
							<EmptyStateAnnouncer @text={{@description}} />
						{{else}}
							{{@description}}
						{{/if}}
					{{/if}}
				</Text>

				{{#if (has-block "primary-actions")}}
					<div class="primary-actions" data-qid-empty-state-primary-actions>
						{{yield to="primary-actions"}}
					</div>
				{{/if}}

				{{#if (has-block "secondary-actions")}}
					<Text @tag="div" @font="200" @weight="strong" data-qid-empty-state-secondary-actions>
						{{yield to="secondary-actions"}}
					</Text>
				{{/if}}
			</aside>
		{{/let}}
	</template>
}

export default EmptyState;

import { useId } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { extractCompoundSlots } from '../../utils/extract-compound-slots';
import { Heading } from '../heading';
import { Icon } from '../icon';
import { Illustration, type IllustrationName } from '../illustration';
import { Text } from '../text';

import styles from './empty-state.module.css';

const EMPTY_STATE_SLOT_MAP = {
	'EmptyState.Heading': 'heading',
	'EmptyState.Description': 'description',
	'EmptyState.PrimaryActions': 'primaryActions',
	'EmptyState.SecondaryActions': 'secondaryActions',
} as const;

type EmptyStateSlotKey = (typeof EMPTY_STATE_SLOT_MAP)[keyof typeof EMPTY_STATE_SLOT_MAP];

interface EmptyStateBaseProps {
	/** URL of the image to display */
	imageUrl?: string;
	/** Name of luna icon to display */
	icon?: string;
	/** Sets the illustration to display */
	illustration?: IllustrationName;
	/** Sets the heading level (1-6). Defaults to 2. */
	headingLevel?: 1 | 2 | 3 | 4 | 5 | 6;
	/** Heading text (use when not using EmptyState.Heading or headingSlot) */
	heading?: string;
	/** Description text (use when not using EmptyState.Description or descriptionSlot) */
	description?: string;
	/** Custom heading content (use when not using heading prop or EmptyState.Heading) */
	headingSlot?: React.ReactNode;
	/** Custom description content (use when not using description prop or EmptyState.Description) */
	descriptionSlot?: React.ReactNode;
	/** Primary action buttons */
	primaryActionsSlot?: React.ReactNode;
	/** Secondary action links */
	secondaryActionsSlot?: React.ReactNode;
	/** Compound component children (EmptyState.Heading, EmptyState.Description, etc.) */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type EmptyStateProps = ComponentPropsWithRef<EmptyStateBaseProps, HTMLElement>;

interface EmptyStateSubcomponentProps {
	children: React.ReactNode;
}

/**
 * A temporary placeholder for empty, error, or success states that provides an explanation
 * and guidance on next steps the user can take.
 *
 * Use compound components to compose the empty state:
 * - `EmptyState.Heading` - Required heading text
 * - `EmptyState.Description` - Required description text
 * - `EmptyState.PrimaryActions` - Optional primary action buttons
 * - `EmptyState.SecondaryActions` - Optional secondary action links
 *
 * Children can be passed in any order - they will always render in the correct position.
 *
 * @example
 * ```tsx
 * <EmptyState illustration="upload">
 *   <EmptyState.Heading>No files uploaded</EmptyState.Heading>
 *   <EmptyState.Description>Drag and drop files here to upload.</EmptyState.Description>
 *   <EmptyState.PrimaryActions>
 *     <Button>Upload Files</Button>
 *   </EmptyState.PrimaryActions>
 * </EmptyState>
 * ```
 */
function EmptyStateRoot(props: EmptyStateProps) {
	const {
		ref,
		imageUrl,
		icon,
		illustration,
		headingLevel,
		heading,
		description,
		headingSlot,
		descriptionSlot,
		primaryActionsSlot,
		secondaryActionsSlot,
		children,
		className,
		...rest
	} = props;

	const emptyStateId = useId();
	const emptyStateClassName = cn(styles.EmptyState, className);

	// Extract slot content from children (order-independent)
	const compoundSlots = extractCompoundSlots(children ?? null, EMPTY_STATE_SLOT_MAP) as Partial<
		Record<EmptyStateSlotKey, React.ReactNode>
	>;

	// Props/slots take precedence over compound children
	const slots = {
		heading: headingSlot ?? compoundSlots.heading ?? (heading != null ? heading : undefined),
		description: descriptionSlot ?? compoundSlots.description ?? (description != null ? description : undefined),
		primaryActions: primaryActionsSlot ?? compoundSlots.primaryActions,
		secondaryActions: secondaryActionsSlot ?? compoundSlots.secondaryActions,
	};

	return (
		<aside
			ref={ref}
			className={emptyStateClassName}
			data-align="center"
			data-qid-component-type="Luna::EmptyState"
			aria-labelledby={emptyStateId}
			{...rest}
		>
			{/* Graphic: illustration, image, or icon */}
			{illustration && (
				<div className={cn(styles.graphic, styles.illustration)}>
					<Illustration illustration={illustration} size="24" />
				</div>
			)}
			{!illustration && imageUrl && (
				<div className={styles.graphic} data-qid-empty-state-image>
					<img src={imageUrl} aria-hidden="true" alt="" />
				</div>
			)}
			{!illustration && !imageUrl && icon && (
				<div className={cn(styles.graphic, styles.icon)}>
					<Icon icon={icon} size="3xl" />
				</div>
			)}

			{/* Heading */}
			{slots.heading != null && (
				<Heading id={emptyStateId} level={headingLevel} font="400" weight="strong" data-qid-empty-state-heading>
					{slots.heading}
				</Heading>
			)}

			{/* Description */}
			{slots.description != null && (
				<Text tag="p" clamp={3} font="200" color="gray-500" align="center" data-qid-empty-state-description>
					{slots.description}
				</Text>
			)}

			{/* Primary actions */}
			{slots.primaryActions && (
				<div className={styles.primaryActions} data-qid-empty-state-primary-actions>
					{slots.primaryActions}
				</div>
			)}

			{/* Secondary actions */}
			{slots.secondaryActions && (
				<Text tag="div" font="200" weight="strong" data-qid-empty-state-secondary-actions>
					{slots.secondaryActions}
				</Text>
			)}
		</aside>
	);
}

EmptyStateRoot.displayName = 'EmptyState';

// Subcomponents - these don't render themselves, they're extracted by the parent
const EmptyStateHeading = ({ children }: EmptyStateSubcomponentProps) => children;
EmptyStateHeading.displayName = 'EmptyState.Heading';

const EmptyStateDescription = ({ children }: EmptyStateSubcomponentProps) => children;
EmptyStateDescription.displayName = 'EmptyState.Description';

const EmptyStatePrimaryActions = ({ children }: EmptyStateSubcomponentProps) => children;
EmptyStatePrimaryActions.displayName = 'EmptyState.PrimaryActions';

const EmptyStateSecondaryActions = ({ children }: EmptyStateSubcomponentProps) => children;
EmptyStateSecondaryActions.displayName = 'EmptyState.SecondaryActions';

/**
 * EmptyState compound component with namespaced subcomponents.
 *
 * Subcomponents can be passed in any order - they will render in the correct position.
 */
export const EmptyState = Object.assign(EmptyStateRoot, {
	Heading: EmptyStateHeading,
	Description: EmptyStateDescription,
	PrimaryActions: EmptyStatePrimaryActions,
	SecondaryActions: EmptyStateSecondaryActions,
});

export default EmptyState;
