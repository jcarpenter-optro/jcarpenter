import type { Editor, Extensions, JSONContent } from '@tiptap/core';

import type { TOC } from '@ember/component/template-only';
import { registerDestructor } from '@ember/destroyable';
import type Owner from '@ember/owner';
import { service } from '@ember/service';
import Component from '@glimmer/component';

import { modifier } from 'ember-modifier';

import { TIPTAP_EXTENSION_GROUPS } from '@auditboard/rich-text/extension-groups';
import { attachContent } from '@auditboard/rich-text/lib/attach-tiptap';
import { EmberEditor } from '@auditboard/rich-text/lib/ember-editor';

import type AsyncService from '../services/async.ts';
import { BubbleMenu } from './rich-text/bubble-menu.gts';

interface RichTextEditorArgs {
	value?: string | JSONContent;
	onEditorCreated?: (editor: Editor) => void;
	onEditorBlur?: () => void;
	onChange: (s: JSONContent) => void;
	extensions?: Extensions;
	readonly?: boolean;
}

/**
 * @deprecated
 *
 * This is not ready for use *quite* yet. It should not be used in production until CoreX approval
 * Known issues:
 * 1. Documentation is not complete
 * 2. Confirm UAT around keyboard shortcuts and things like typing "*text*" will bold things
 * 3. Enable Link extension
 * 4. Enable Highlight and fix bubble menu popup
 */
export class RichTextEditor extends Component<{ Element: HTMLDivElement; Args: RichTextEditorArgs }> {
	@service('luna/async') declare asyncService: AsyncService;
	editor: Editor;

	constructor(owner: Owner, args: RichTextEditorArgs) {
		super(owner, args);

		this.editor = new EmberEditor({
			extensions: this.args.extensions ?? TIPTAP_EXTENSION_GROUPS.ALL,
			content: this.args.value ?? '',

			// This allows the editor to auto format with MD style syntax when typing or pasting
			enableInputRules: true,
			enablePasteRules: true,
			editable: !this.args.readonly,
			onBlur: () => {
				this.args.onEditorBlur?.();
			},
		});

		this.editor.on('update', ({ editor, transaction }) => {
			this.asyncService.requestAnimationFrame(() => {
				if (transaction.docChanged) {
					this.args.onChange(editor.getJSON());
				}
			});
		});

		this.args.onEditorCreated?.(this.editor);

		registerDestructor(this, () => {
			if (!this.editor.isDestroyed) {
				this.editor.destroy();
			}
		});
	}

	/**
	 * Triggers an update to the content of the editor.
	 *
	 * If we have rendering components that are Ember components, this may need to be revisited
	 * to ensure that the content is updated correctly without unnecessary re-renders or render loops
	 *
	 * @TODO this could be refactored to use reactiveweb effect when we get to 1.5+
	 */
	updateContent = modifier(() => {
		const value = this.args.value;
		const { editor } = this;

		this.asyncService.requestAnimationFrame(() => {
			if (!value || typeof value === 'string') {
				editor.commands.setContent(value ?? '', { emitUpdate: false });
				return;
			}

			if (!editor.schema.nodeFromJSON(value).eq(editor.state.doc)) {
				editor.commands.setContent(value ?? '', { emitUpdate: false });
			}
		});
	});

	updateIsEditable = modifier(() => {
		if (this.isEditable === !this.args.readonly) {
			return;
		}

		this.editor.setEditable(!this.args.readonly);
	});

	get isEditable() {
		return this.editor.isEditable;
	}

	<template>
		<div class="editor-container" {{this.updateContent @value}} {{this.updateIsEditable @readonly}}>
			<div
				class="editor-content {{if this.isEditable 'editable'}}"
				...attributes
				data-editor
				{{attachContent this.editor}}
			></div>
			<BubbleMenu @editor={{this.editor}} />
		</div>
	</template>
}

import { useCallback, useEffect, useImperativeHandle, useRef } from 'react';

import type { Editor, Extensions, JSONContent } from '@tiptap/core';
import Bold from '@tiptap/extension-bold';
import BulletList from '@tiptap/extension-bullet-list';
import Document from '@tiptap/extension-document';
import Gapcursor from '@tiptap/extension-gapcursor';
import History from '@tiptap/extension-history';
import Italic from '@tiptap/extension-italic';
import ListItem from '@tiptap/extension-list-item';
import OrderedList from '@tiptap/extension-ordered-list';
import Paragraph from '@tiptap/extension-paragraph';
import Text from '@tiptap/extension-text';
import Underline from '@tiptap/extension-underline';
import { EditorContent, useEditor } from '@tiptap/react';
import { BubbleMenu } from '@tiptap/react/menus';

import { cn } from '../../utils/cn';
import { IconButton } from '../icon-button';
import { XStack } from '../x-stack';

import styles from './rich-text-editor.module.css';

// TODO: Extract TipTap functionality (extensions, BubbleMenu config) into a shared package
// so luna-core (Ember) and luna-react can both use it.

/**
 * Basic TipTap extensions for the rich text editor
 */
const BASIC_EXTENSIONS: Extensions = [
	Document,
	Paragraph,
	Text,
	Bold,
	Italic,
	Underline,
	BulletList,
	OrderedList,
	ListItem,
	History,
	Gapcursor,
];

export interface RichTextEditorProps {
	ref?: React.Ref<RichTextEditorHandle>;
	/** The value of the editor (string or TipTap JSONContent) */
	value?: string | JSONContent;
	/** Callback when content changes */
	onChange?: (content: JSONContent) => void;
	/** Custom TipTap extensions (defaults to basic formatting) */
	extensions?: Extensions;
	/** Whether the editor is read-only */
	readonly?: boolean;
	/** Whether to show the bubble menu toolbar */
	showBubbleMenu?: boolean;
	/** Placeholder text when editor is empty */
	placeholder?: string;
	/** Additional CSS class */
	className?: string;
	/** Minimum height for the editor */
	minHeight?: string;
}

export interface RichTextEditorHandle {
	/** The TipTap editor instance */
	editor: Editor | null;
	/** Focus the editor */
	focus: () => void;
	/** Clear the editor content */
	clear: () => void;
}

interface BubbleMenuButtonConfig {
	command: (editor: Editor) => void;
	icon: string;
	tooltip: string;
	activeCheck: string;
}

const BUBBLE_MENU_BUTTONS: BubbleMenuButtonConfig[] = [
	{
		command: (editor) => editor.chain().focus().toggleBold().run(),
		icon: 'bold',
		tooltip: 'Bold',
		activeCheck: 'bold',
	},
	{
		command: (editor) => editor.chain().focus().toggleItalic().run(),
		icon: 'italic',
		tooltip: 'Italic',
		activeCheck: 'italic',
	},
	{
		command: (editor) => editor.chain().focus().toggleUnderline().run(),
		icon: 'underline',
		tooltip: 'Underline',
		activeCheck: 'underline',
	},
	{
		command: (editor) => editor.chain().focus().toggleBulletList().run(),
		icon: 'list-ul',
		tooltip: 'Bulleted List',
		activeCheck: 'bulletList',
	},
	{
		command: (editor) => editor.chain().focus().toggleOrderedList().run(),
		icon: 'list-ol',
		tooltip: 'Ordered List',
		activeCheck: 'orderedList',
	},
];

interface BubbleMenuButtonProps {
	editor: Editor;
	config: BubbleMenuButtonConfig;
	isActive: boolean;
}

function BubbleMenuButton({ editor, config, isActive }: BubbleMenuButtonProps) {
	const handleClick = useCallback(() => {
		config.command(editor);
	}, [editor, config]);

	return (
		<IconButton
			icon={config.icon}
			tooltip={config.tooltip}
			variant={isActive ? 'blue' : 'transparent'}
			size="xs"
			onClick={handleClick}
			data-qid-rich-text-bubble-menu-button={config.tooltip}
			data-active={String(isActive)}
		/>
	);
}

/**
 * A rich text editor built on TipTap.
 *
 * Supports basic formatting (bold, italic, underline, lists) with a bubble menu toolbar.
 */
export function RichTextEditor({
	value,
	onChange,
	extensions = BASIC_EXTENSIONS,
	readonly = false,
	showBubbleMenu = true,
	placeholder: _placeholder,
	className,
	minHeight = '5rem',
	ref,
	...props
}: RichTextEditorProps): React.ReactElement {
	const lastValueRef = useRef<string | JSONContent | undefined>(value);

	const editor = useEditor({
		extensions,
		content: value ?? '',
		editable: !readonly,
		enableInputRules: false,
		enablePasteRules: false,
		onUpdate: ({ editor }) => {
			const json = editor.getJSON();
			onChange?.(json);
		},
	});

	// Update content when value prop changes externally
	useEffect(() => {
		if (!editor) return;

		// Skip if value hasn't changed (avoid loops)
		if (lastValueRef.current === value) return;
		lastValueRef.current = value;

		// Handle string content
		if (!value || typeof value === 'string') {
			const currentText = editor.getText();
			if (currentText !== (value ?? '')) {
				editor.commands.setContent(value ?? '', { emitUpdate: false });
			}
			return;
		}

		// Handle JSON content - only update if different
		try {
			const currentDoc = editor.state.doc;
			const newDoc = editor.schema.nodeFromJSON(value);
			if (!currentDoc.eq(newDoc)) {
				editor.commands.setContent(value, { emitUpdate: false });
			}
		} catch {
			// If JSON parsing fails, set content anyway
			editor.commands.setContent(value, { emitUpdate: false });
		}
	}, [editor, value]);

	// Update editable state when readonly prop changes
	useEffect(() => {
		if (editor && editor.isEditable === readonly) {
			editor.setEditable(!readonly);
		}
	}, [editor, readonly]);

	// Expose editor methods via ref
	useImperativeHandle(
		ref,
		() => ({
			editor,
			focus: () => editor?.chain().focus().run(),
			clear: () => editor?.commands.clearContent(),
		}),
		[editor],
	);

	const isEditable = editor?.isEditable ?? !readonly;

	return (
		<div className={cn(styles.editorContainer, className)} data-qid-component-type="Luna::RichTextEditor" {...props}>
			<div
				className={cn(styles.editorContent, isEditable && styles.editable)}
				style={{ minHeight }}
				data-editor
				data-qid-text-editor
			>
				<EditorContent editor={editor} />
			</div>

			{editor && showBubbleMenu && isEditable && (
				<BubbleMenu editor={editor} className={styles.bubbleMenu} data-qid-component-type="Luna::RichText::BubbleMenu">
					<XStack spacing="sm">
						{BUBBLE_MENU_BUTTONS.map((config) => (
							<BubbleMenuButton
								key={config.tooltip}
								editor={editor}
								config={config}
								isActive={editor.isActive(config.activeCheck)}
							/>
						))}
					</XStack>
				</BubbleMenu>
			)}
		</div>
	);
}

/**
 * A read-only rich text renderer.
 * Convenience component that renders RichTextEditor in readonly mode.
 */
export interface RichTextRenderProps {
	ref?: React.Ref<RichTextEditorHandle>;
	/** The value to render (string or TipTap JSONContent) */
	value?: string | JSONContent;
	/** Additional CSS class */
	className?: string;
}

export function RichTextRender({ value, className, ref, ...props }: RichTextRenderProps): React.ReactElement {
	return <RichTextEditor ref={ref} value={value} readonly showBubbleMenu={false} className={className} {...props} />;
}

export default RichTextEditor;
