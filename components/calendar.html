const ADJUSTMENTS = {
	PREVIOUS_DAY: -1,
	NEXT_DAY: 1,
	PREVIOUS_WEEK: -7,
	NEXT_WEEK: 7,
	PREVIOUS_MONTH: -1,
	NEXT_MONTH: 1,
	PREVIOUS_YEAR: -1,
	NEXT_YEAR: 1,
};

function noOp() {
	// No-op

export interface LunaCalendarSignature {
	Element: HTMLDivElement;
	Args: {
		firstDayOfWeek?: DayOfWeekValue;
		// TODO: selectionType and value should be linked
		selectionType?: 'range' | 'single';
		value: string | { start?: string; end?: string } | null | undefined;
		min?: string;
		max?: string;
		disableDate?: (date: Date) => boolean;
		onChange?: (value: string | { start?: string; end?: string } | null) => void;
	};
}

/**
 * The core component to choose a single date or date range. The Calendar is used most commonly within the [Select Date](https://docs.auditboardteam.com/Luna/component/components/select-date) component.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/calendar.md)
 */
export default class LunaCalendar extends Component<LunaCalendarSignature> {
	// Instead of using bang we should defined the dependent properties in a constructor
	@service intl!: IntlService;

	monthTableId = `${guidFor(this)}--table`;

	today = new Date();

	@tracked currentMonth = this.startDate ?? this.today;
	private _lastSyncedStartDate: Date | undefined = this.startDate;
	private _lastSyncedEndDate: Date | undefined = this.endDate;

	/**
	 * Modifier that syncs the displayed month when the value changes externally
	 * to a different month.
	 */
	syncMonthToValue = modifier(() => {
		const startDate = this.startDate;
		const endDate = this.endDate;

		// Check if start date changed to a different month
		const startDateMonthChanged = startDate && !isEqualMonth(startDate, this._lastSyncedStartDate);
		// Check if end date changed to a different month
		const endDateMonthChanged = endDate && !isEqualMonth(endDate, this._lastSyncedEndDate);

		// Prefer syncing to the date that changed; if both changed, prefer end date (most recent edit)
		if (endDateMonthChanged) {
			this._lastSyncedStartDate = startDate;
			this._lastSyncedEndDate = endDate;
			this.adjustMonthTo(endDate);
		} else if (startDateMonthChanged) {
			this._lastSyncedStartDate = startDate;
			this._lastSyncedEndDate = endDate;
			this.adjustMonthTo(startDate);
		}
	});

	firstDayOfWeek = this.args.firstDayOfWeek ?? firstDayOfWeek(this.intl.primaryLocale);
	@tracked monthView = getViewOfMonth(this.currentMonth, this.firstDayOfWeek);
	headerView = getDaysOfWeek(this.firstDayOfWeek);

	@tracked focusableDate = this.startDate ?? this.today;

	@tracked announcementText = '';

	get minDate() {
		const now = this.today;
		// Default to 20 years ago
		const defaultDate = startOfYear(setYear(now, now.getFullYear() - 20));

		return createDate(this.args.min) ?? defaultDate;
	}

	get maxDate() {
		const now = this.today;
		// Default to 10 years from now
		const defaultDate = endOfYear(setYear(now, now.getFullYear() + 10));

		return createDate(this.args.max) ?? defaultDate;
	}

	/**
	 * Whether dates should be clamped to the min and max values.
	 * Redeclared and bound in the instance initializer of the consuming app.
	 * See `client/app/initializers/luna.ts` for example implementation.
	 */
	// eslint-disable-next-line @typescript-eslint/class-literal-property-style
	get shouldClampDates() {
		return false;
	}

	/**
	 * Clamps a date to the min/max bounds. If the date is outside the bounds,
	 * returns the nearest boundary date.
	 */
	private clampToMinOrMax(date: Date): Date {
		if (date < this.minDate) {
			return this.minDate;
		}
		if (date > this.maxDate) {
			return this.maxDate;
		}
		return date;
	}

	get months() {
		const { minDate, maxDate } = this;
		const months = [];

		for (let i = 0; i < 12; i++) {
			const startOfMonth = new Date(this.currentMonth.getFullYear(), i, 1);
			const endOfMonthDate = endOfMonth(startOfMonth);
			const disabled = endOfMonthDate < minDate || startOfMonth > maxDate;
			months.push({
				index: i,
				label: this.intl.formatDate(new Date(2020, i, 15), { month: 'long' }),
				disabled,
			});
		}
		return months;
	}

	get years() {
		const years = [];
		const minYear = this.minDate.getFullYear();
		const maxYear = this.maxDate.getFullYear();

		for (let step = minYear; step <= maxYear; step++) {
			years.push(step);
		}
		return years;
	}

	get currentMonthIndex() {
		return this.currentMonth.getMonth();
	}

	get currentMonthYear() {
		return this.currentMonth.getFullYear();
	}

	get weekView() {
		const days = this.monthView;
		const chunks = [];
		let index = 0;

		while (index < days.length) {
			chunks.push(days.slice(index, 7 + index));
			index += 7;
		}
		return chunks;
	}

	get isRangeSelection() {
		return this.args.selectionType === 'range';
	}

	get startDate() {
		let value;
		if (this.isRangeSelection) {
			// Handle case where value might be a string in range mode
			if (typeof this.args.value === 'string') {
				value = this.args.value;
			} else {
				value = this.args.value?.start;
			}
		} else if (typeof this.args.value === 'object' && this.args.value !== null) {
			// Handle case where value might be an object in single mode
			value = (this.args.value as { start?: string }).start;
		} else {
			value = this.args.value;
		}

		const date = createDate(value);

		if (date && this.shouldClampDates) {
			return this.clampToMinOrMax(date);
		}

		return date;
	}

	get endDate() {
		let value;
		if (typeof this.args.value === 'object' && this.args.value !== null) {
			value = this.args.value.end;
		} else {
			value = undefined;
		}
		const date = createDate(value);

		if (date && this.shouldClampDates) {
			return this.clampToMinOrMax(date);
		}

		return date;
	}

	get isPreviousMonthDisabled() {
		const previousMonth = addMonths(this.currentMonth, -1);
		return previousMonth < this.minDate;
	}

	get isNextMonthDisabled() {
		const nextMonth = addMonths(this.currentMonth, 1);
		return nextMonth > this.maxDate;
	}

	/**
	 * Adjusts the focusable date by a number of days
	 *
	 * @param adjustment Amount of days to adjust the current date by
	 * @param event Key event that triggered this adjustment
	 */
	adjustFocusableDay(adjustment: number, kind = 'day', event: KeyboardEvent) {
		event.preventDefault();

		if (kind === 'day') {
			this.focusableDate = addDays(this.focusableDate, adjustment);

			const isInCurrentMonthView = this.monthView.find((day) => isEqual(day, this.focusableDate));

			if (!isInCurrentMonthView) {
				this.adjustMonthTo(this.focusableDate);
			}

			void this.synchronizeFocus();
			return;
		}

		if (kind === 'month') {
			this.focusableDate = addMonths(this.focusableDate, adjustment);
			this.adjustMonthTo(this.focusableDate);
			void this.synchronizeFocus();
		}

		if (kind === 'year') {
			this.focusableDate = addMonths(this.focusableDate, adjustment * 12);
			this.adjustMonthTo(this.focusableDate);
			void this.synchronizeFocus();
		}
	}

	/**
	 * Set the current month being viewed to a different one
	 */
	adjustMonthTo(date: Date) {
		this.currentMonth = date;
		this.monthView = getViewOfMonth(date, this.firstDayOfWeek);
	}

	/**
	 * Ensures that the current month view has a date that can be focused. If
	 * there is none then we update the focusableDate to a day inside the
	 * current view.
	 */
	ensureFocusableDateExists() {
		if (this.monthView.find((day) => isEqual(day, this.focusableDate))) {
			return;
		}

		const selectedDate = this.monthView.find((day) => isEqual(day, this.startDate));

		if (selectedDate) {
			this.focusableDate = selectedDate;
			void this.synchronizeFocus();
			return;
		}

		const theFirst = this.monthView.find((day) => day.getDate() === 1);
		assert('Expected the first of the month to exist', theFirst);

		this.focusableDate = theFirst;
		void this.synchronizeFocus();
	}

	@action
	isDisabled(day: Date) {
		const isOutOfRange = day < this.minDate || day > this.maxDate;
		return this.args.disableDate?.(day) ?? isOutOfRange;
	}

	@action
	onKeyDown(event: KeyboardEvent) {
		switch (event.key) {
			case KEYBOARD_KEYS.LEFT_ARROW:
				this.adjustFocusableDay(ADJUSTMENTS.PREVIOUS_DAY, 'day', event);
				break;

			case KEYBOARD_KEYS.RIGHT_ARROW:
				this.adjustFocusableDay(ADJUSTMENTS.NEXT_DAY, 'day', event);
				break;

			case KEYBOARD_KEYS.UP_ARROW:
				this.adjustFocusableDay(ADJUSTMENTS.PREVIOUS_WEEK, 'day', event);
				break;

			case KEYBOARD_KEYS.DOWN_ARROW:
				this.adjustFocusableDay(ADJUSTMENTS.NEXT_WEEK, 'day', event);
				break;

			case KEYBOARD_KEYS.PAGE_UP:
				if (event.shiftKey) {
					this.adjustFocusableDay(ADJUSTMENTS.PREVIOUS_YEAR, 'year', event);
				} else {
					this.adjustFocusableDay(ADJUSTMENTS.PREVIOUS_MONTH, 'month', event);
				}
				break;

			case KEYBOARD_KEYS.PAGE_DOWN:
				if (event.shiftKey) {
					this.adjustFocusableDay(ADJUSTMENTS.NEXT_YEAR, 'year', event);
				} else {
					this.adjustFocusableDay(ADJUSTMENTS.NEXT_MONTH, 'month', event);
				}
				break;
		}
	}

	tableElement: HTMLTableElement | undefined;
	registerTable: ModifierLike<{ Element: HTMLTableElement }> = modifier((element: HTMLTableElement) => {
		this.tableElement = element;
	});

	/**
	 * Ensures the focusableElement has the focus as the user navigates
	 * around with the keyboard.
	 */
	@action
	async synchronizeFocus() {
		await /**
		 * TODO: don't use did-update
		 *
		 * This causes set-during-render situation with {{on 'focusin' announceEntry}}
		 * (because focus events are synchronous)
		 * announceEntry (reads focusableDate)
		 * -> adjustFocusableDay
		 *  -> focusableDate = someDate
		 */
		// @ts-expect-error - we are using an instance-safe version of next()
		// eslint-disable-next-line ember/no-runloop
		await new Promise((resolve) => next(() => resolve()));

		if (isDestroyed(this) || isDestroying(this)) {
			return;
		}

		if (!this.tableElement) {
			return;
		}

		const element = this.tableElement;
		/**
		 * If the user changes the month the view re-renders and the focus gets
		 * forced back onto the body as the focused element gets removed. In
		 * that case we force it back to the view.
		 *
		 * Otherwise we only force it if the focus is already inside the month view.
		 */
		if (document.body === document.activeElement || element.contains(document.activeElement)) {
			const button: Element | null = element.querySelector('button[tabindex="0"]');
			assert('Expected a button', button instanceof HTMLButtonElement);
			button.focus();
		}
	}

	@action
	onChangeSelection(value: Date) {
		this.focusableDate = value;
		void this.synchronizeFocus();

		let newValue: { start: string; end: string } | string;

		// Range selection
		if (this.isRangeSelection) {
			let startDate;
			let endDate;

			startDate = this.startDate ?? value;
			endDate = this.startDate ? value : null;

			if (this.startDate && this.endDate) {
				startDate = value;
				endDate = null;
			} else if (this.startDate) {
				endDate = value;
			}

			const newValueTmp = {
				start: startDate,
				end: endDate,
			};

			// Ensure startDate is always before endDate
			if (endDate && endDate < startDate) {
				newValueTmp.start = endDate;
				newValueTmp.end = startDate;
			}

			newValue = {
				start: printISODate(newValueTmp.start),
				end: printISODate(newValueTmp.end),
			};
		} else {
			// Single selection
			if (!isEqualMonth(value, this.currentMonth)) {
				this.adjustMonthTo(value);
			}

			newValue = printISODate(value);
		}

		// Single date selection
		this.args.onChange?.(newValue);

		if (this.selectionString) {
			void this.announce.perform(this.selectionString);
		}
	}

	@action
	nextMonth() {
		if (this.isNextMonthDisabled) {
			return;
		}
		this.adjustMonthTo(addMonths(this.currentMonth, 1));
		this.ensureFocusableDateExists();
		this.announceMonthChange();
	}

	@action
	previousMonth() {
		if (this.isPreviousMonthDisabled) {
			return;
		}
		this.adjustMonthTo(addMonths(this.currentMonth, -1));
		this.ensureFocusableDateExists();
		this.announceMonthChange();
	}

	@action
	onChangeMonth(select: Event) {
		assert('Expected a select element', select.target instanceof HTMLSelectElement);
		const newMonth = setMonth(this.currentMonth, Number(select.target.value));
		this.adjustMonthTo(newMonth);
		this.ensureFocusableDateExists();
		this.announceMonthChange();
	}

	@action
	onChangeYear(select: Event) {
		assert('Expected a select element', select.target instanceof HTMLSelectElement);
		const newMonth = setYear(this.currentMonth, Number(select.target.value));
		this.adjustMonthTo(newMonth);
		this.ensureFocusableDateExists();
		this.announceMonthChange();
	}

	/**
	 * Announces changes via the screen reader.
	 *
	 * TODO: Convert this to a service that can be used by any component
	 */
	announce = task({ keepLatest: true }, async (text: string) => {
		this.announcementText = text;
		await timeout(isTesting() ? 0 : 200);
		this.announcementText = '';
	});

	/**
	 * Announcement that happens when the user focuses inside the calendar view
	 * providing them with instructions on how to navigate it.
	 */
	@action
	async announceEntry(event: FocusEvent) {
		await Promise.resolve();
		void this.synchronizeFocus();

		if (
			event.relatedTarget instanceof Node &&
			document.querySelector(`#${this.monthTableId}`)?.contains(event.relatedTarget)
		) {
			return;
		}

		const selectableDate = this.intl.formatDate(this.focusableDate, { day: 'numeric', month: 'long' });
		const instructions = this.intl.t('input.calendar.keys');
		const currentFocus = this.intl.t('input.calendar.focus', { date: selectableDate });

		void this.announce.perform(`${this.selectionString} ${currentFocus} ${instructions}`);
	}

	/**
	 * Announcement that happens when the user uses the year or month selects to
	 * jump around the month view.
	 */
	announceMonthChange() {
		const currentMonth = this.intl.formatDate(this.currentMonth, { month: 'long', year: 'numeric' });
		void this.announce.perform(currentMonth);
	}

	/**
	 * Whether the currently selected date(s) are out of range.
	 * When true, shows an out-of-range message instead of the calendar.
	 */
	get dateIsOutOfRange() {
		if (this.shouldClampDates) {
			return false;
		}

		return (
			(this.startDate && (this.startDate < this.minDate || this.startDate > this.maxDate)) ||
			(this.endDate && (this.endDate < this.minDate || this.endDate > this.maxDate))
		);
	}

	/**
	 * Calculates the text that best describes the user's current selected.
	 */
	get selectionString() {
		const startDate = this.startDate && this.intl.formatDate(this.startDate, { day: 'numeric', month: 'long' });
		const endDate = this.endDate && this.intl.formatDate(this.endDate, { day: 'numeric', month: 'long' });

		let selectedValue;

		if (this.isRangeSelection) {
			if (startDate && endDate) {
				selectedValue = this.intl.t('input.calendar.selection.range', { start: startDate, end: endDate });
			} else if (startDate) {
				selectedValue = this.intl.t('input.calendar.selection.range-start-only', { start: startDate });
			} else {
				selectedValue = this.intl.t('input.calendar.selection.range-none');
			}
		} else if (startDate) {
			selectedValue = this.intl.t('input.calendar.selection.single', { date: startDate });
		} else {
			selectedValue = this.intl.t('input.calendar.selection.single-none');
		}

		return selectedValue;
	}

import { useCallback, useMemo, useState } from 'react';
import { type DateRange, DayPicker } from 'react-day-picker';

import {
	addYears,
	endOfYear,
	format,
	setMonth as dateFnsSetMonth,
	setYear as dateFnsSetYear,
	startOfDay,
	startOfYear,
	subYears,
} from 'date-fns';

import { cn } from '../../utils/cn';

import styles from './calendar.module.css';

// Custom formatter for 3-letter weekday names to match Ember
const formatWeekdayName = (date: Date): string => {
	return format(date, 'EEE').toUpperCase();
};

export type SelectionType = 'single' | 'range';

// Value can be a string (ISO date), Date, or range object
export type CalendarSingleValue = string | Date | null | undefined;
export type CalendarRangeValue = { start?: string; end?: string } | DateRange | null | undefined;
export type CalendarValue = CalendarSingleValue | CalendarRangeValue;

export interface CalendarProps {
	/** Optional ref for the root element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** First day of the week (0 = Sunday, 1 = Monday, etc.) */
	firstDayOfWeek?: 0 | 1 | 2 | 3 | 4 | 5 | 6;
	/** Selection type: single date or date range */
	selectionType?: SelectionType;
	/** Currently selected value */
	value?: CalendarValue;
	/** Minimum selectable date (ISO string) */
	min?: string;
	/** Maximum selectable date (ISO string) */
	max?: string;
	/** Function to disable specific dates */
	disableDate?: (date: Date) => boolean;
	/** Whether to highlight today's date (default: true) */
	showToday?: boolean;
	/** Called when the selection changes */
	onChange?: (value: string | { start: string; end: string } | null) => void;
	/** Additional CSS class */
	className?: string;
	/** Test ID */
	testId?: string;
}

/**
 * Parses a date value to a Date object
 */
function parseDate(value: string | Date | undefined | null): Date | undefined {
	if (!value) return undefined;
	if (value instanceof Date) return value;
	// Parse ISO date string as local date (not UTC)
	// ISO date strings like "2024-01-15" are parsed as UTC by new Date(),
	// which causes timezone issues (selecting wrong day). Parse components manually.
	const match = value.match(/^(\d{4})-(\d{2})-(\d{2})/);
	if (match) {
		const [, year, month, day] = match;
		return new Date(Number(year), Number(month) - 1, Number(day));
	}
	const date = new Date(value);
	return isNaN(date.getTime()) ? undefined : date;
}

/**
 * Formats a Date to ISO string (YYYY-MM-DD)
 */
function toISODateString(date: Date): string {
	return format(date, 'yyyy-MM-dd');
}

/**
 * Calendar component for selecting dates.
 * Used by SelectDate and can be used standalone.
 */
export function Calendar({
	ref,
	firstDayOfWeek = 0,
	selectionType = 'single',
	value,
	min,
	max,
	disableDate,
	showToday = true,
	onChange,
	className,
	testId,
}: CalendarProps) {
	const today = useMemo(() => startOfDay(new Date()), []);

	// Parse min/max dates with defaults
	const minDate = useMemo(() => {
		return parseDate(min) ?? startOfYear(subYears(today, 20));
	}, [min, today]);

	const maxDate = useMemo(() => {
		return parseDate(max) ?? endOfYear(addYears(today, 10));
	}, [max, today]);

	// Parse current value
	const { startDate, endDate } = useMemo(() => {
		if (!value) return { startDate: undefined, endDate: undefined };

		if (typeof value === 'string' || value instanceof Date) {
			return { startDate: parseDate(value), endDate: undefined };
		}

		// Range value
		if ('from' in value) {
			// DateRange format from react-day-picker
			return { startDate: value.from, endDate: value.to };
		}

		// Our custom format { start, end }
		return {
			startDate: parseDate(value.start),
			endDate: parseDate(value.end),
		};
	}, [value]);

	// Track the currently displayed month
	const [month, setMonth] = useState<Date>(() => startDate ?? today);

	// Generate month options
	const months = useMemo(() => {
		const result = [];
		for (let i = 0; i < 12; i++) {
			const monthDate = new Date(month.getFullYear(), i, 1);
			const disabled = monthDate < minDate || monthDate > maxDate;
			result.push({
				value: i,
				label: format(new Date(2020, i, 15), 'MMMM'),
				disabled,
			});
		}
		return result;
	}, [month, minDate, maxDate]);

	// Generate year options
	const years = useMemo(() => {
		const minYear = minDate.getFullYear();
		const maxYear = maxDate.getFullYear();
		const result = [];
		for (let year = minYear; year <= maxYear; year++) {
			result.push(year);
		}
		return result;
	}, [minDate, maxDate]);

	// Handle month select change
	const handleMonthChange = useCallback(
		(e: React.ChangeEvent<HTMLSelectElement>) => {
			const newMonth = dateFnsSetMonth(month, Number(e.target.value));
			setMonth(newMonth);
		},
		[month],
	);

	// Handle year select change
	const handleYearChange = useCallback(
		(e: React.ChangeEvent<HTMLSelectElement>) => {
			const newMonth = dateFnsSetYear(month, Number(e.target.value));
			setMonth(newMonth);
		},
		[month],
	);

	// Disable dates outside range or custom disabled
	const isDateDisabled = useCallback(
		(date: Date): boolean => {
			if (date < minDate || date > maxDate) return true;
			return disableDate?.(date) ?? false;
		},
		[minDate, maxDate, disableDate],
	);

	// Handle single date selection
	const handleSingleSelect = useCallback(
		(date: Date | undefined) => {
			if (date && !isDateDisabled(date)) {
				onChange?.(toISODateString(date));
			}
		},
		[onChange, isDateDisabled],
	);

	// Handle range selection
	const handleRangeSelect = useCallback(
		(range: DateRange | undefined) => {
			if (!range) {
				onChange?.(null);
				return;
			}

			// Build the range value
			const result: { start: string; end: string } = {
				start: range.from ? toISODateString(range.from) : '',
				end: range.to ? toISODateString(range.to) : '',
			};

			// Only emit if at least start is set
			if (result.start) {
				onChange?.(result);
			}
		},
		[onChange],
	);

	// Navigation disabled states
	const isPrevMonthDisabled = useMemo(() => {
		const prevMonth = new Date(month.getFullYear(), month.getMonth() - 1, 1);
		return prevMonth < minDate;
	}, [month, minDate]);

	const isNextMonthDisabled = useMemo(() => {
		const nextMonth = new Date(month.getFullYear(), month.getMonth() + 1, 1);
		return nextMonth > maxDate;
	}, [month, maxDate]);

	// Common DayPicker props - match Ember: variable row count (no extra grey rows)
	const commonProps = {
		month,
		onMonthChange: setMonth,
		weekStartsOn: firstDayOfWeek,
		disabled: isDateDisabled,
		showOutsideDays: true,
		fixedWeeks: false,
		hideNavigation: true,
		formatters: {
			formatWeekdayName,
		},
		classNames: {
			root: styles.dayPicker,
			months: styles.months,
			month: styles.month,
			month_caption: styles.monthCaption,
			weekdays: styles.weekdays,
			weekday: styles.weekday,
			weeks: styles.weeks,
			week: styles.week,
			day: styles.day,
			day_button: styles.dayButton,
			outside: styles.outside,
			disabled: styles.disabled,
			today: showToday ? styles.today : undefined,
			selected: styles.selected,
			range_start: styles.rangeStart,
			range_end: styles.rangeEnd,
			range_middle: styles.rangeMiddle,
		},
	};

	return (
		<div
			ref={ref}
			className={cn(styles.Calendar, className)}
			data-qid-component-type="Luna::Calendar"
			data-testid={testId}
		>
			{/* Custom navigation header */}
			<div className={styles.header}>
				<div className={styles.selects}>
					{/* Month select */}
					<div className={styles.selectWrapper}>
						<div className={styles.selectLabel} aria-hidden="true">
							{format(month, 'MMM')}
						</div>
						<div className={styles.selectIcon}>
							<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16">
								<path
									fill="currentColor"
									d="M8 10.15l5.27-5.27 1.48 1.49-5.27 5.26L8 13.12l-1.48-1.49-5.27-5.26 1.48-1.49z"
								/>
							</svg>
						</div>
						<select
							className={styles.selectInput}
							value={month.getMonth()}
							onChange={handleMonthChange}
							aria-label="Month"
							data-test-select="month"
						>
							{months.map((m) => (
								<option key={m.value} value={m.value} disabled={m.disabled}>
									{m.label}
								</option>
							))}
						</select>
					</div>

					{/* Year select */}
					<div className={styles.selectWrapper}>
						<div className={styles.selectLabel} aria-hidden="true">
							{format(month, 'yyyy')}
						</div>
						<div className={styles.selectIcon}>
							<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16">
								<path
									fill="currentColor"
									d="M8 10.15l5.27-5.27 1.48 1.49-5.27 5.26L8 13.12l-1.48-1.49-5.27-5.26 1.48-1.49z"
								/>
							</svg>
						</div>
						<select
							className={styles.selectInput}
							value={month.getFullYear()}
							onChange={handleYearChange}
							aria-label="Year"
							data-test-select="year"
						>
							{years.map((year) => (
								<option key={year} value={year}>
									{year}
								</option>
							))}
						</select>
					</div>
				</div>

				<div className={styles.spacer} />

				{/* Navigation buttons */}
				<button
					type="button"
					className={styles.navButton}
					onClick={() => setMonth((m) => new Date(m.getFullYear(), m.getMonth() - 1, 1))}
					disabled={isPrevMonthDisabled}
					aria-label="Previous month"
					data-test-previous-month
				>
					<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16">
						<path fill="currentColor" d="M6.85 8l5.27 5.27-1.49 1.48-5.26-5.27L3.88 8l1.49-1.48 5.26-5.27 1.49 1.48z" />
					</svg>
				</button>

				<button
					type="button"
					className={styles.navButton}
					onClick={() => setMonth((m) => new Date(m.getFullYear(), m.getMonth() + 1, 1))}
					disabled={isNextMonthDisabled}
					aria-label="Next month"
					data-test-next-month
				>
					<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16">
						<path
							fill="currentColor"
							d="M3.88 2.73l1.49-1.48 5.26 5.27L12.12 8l-1.49 1.48-5.26 5.27-1.49-1.48L9.15 8z"
						/>
					</svg>
				</button>
			</div>

			{/* DayPicker calendar - conditionally render based on selection type */}
			{selectionType === 'single' ? (
				<DayPicker mode="single" selected={startDate} onSelect={handleSingleSelect} {...commonProps} />
			) : (
				<DayPicker
					mode="range"
					selected={{ from: startDate, to: endDate }}
					onSelect={handleRangeSelect}
					{...commonProps}
				/>
			)}
		</div>
	);
}

export default Calendar;
