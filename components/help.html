import { fn, hash } from '@ember/helper';
import { action } from '@ember/object';
import { service } from '@ember/service';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';

import { modifier } from 'ember-modifier';
import { and, or } from 'ember-truth-helpers';

import type { TooltipPlacement } from '../modifiers/tooltip.ts';
import tooltip from '../modifiers/tooltip.ts';
import type LunaAnnouncerService from '../services/announcer.ts';
import type AsyncService from '../services/async.ts';
import type TooltipService from '../services/tooltip.ts';
import announceOnInit from './a11y-announcer/announce-on-init.ts';
import Trigger from './help/trigger.gts';
import { Portal } from './portal.gts';

export interface LunaHelpSignature {
	Element: HTMLDivElement;
	Args: {
		tooltip?: string | null;
		placement?: TooltipPlacement;
		disabled?: boolean;
		width?: 'auto' | 'full';
	};
	Blocks: {
		tooltip: [];
		default: [
			{
				isDisabled: boolean | undefined;
				isEnabled: boolean;
			},
		];
	};
}

/**
 * Adds advanced functionality for tooltips such as custom formatted tooltips.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/help.md)
 */
export default class Help extends Component<LunaHelpSignature> {
	@service('luna/announcer') declare announcerService: LunaAnnouncerService;
	@service('luna/async') declare async: AsyncService;
	@tracked showTooltipBlock = false;

	@service('luna/tooltip') declare tooltip: TooltipService;

	@action
	toggleTooltipBlock(value: boolean) {
		this.showTooltipBlock = value;
	}

	get placement(): TooltipPlacement {
		return this.args.placement ?? 'top';
	}

	get isEnabled() {
		return !this.args.disabled;
	}

	/**
	 * Resets the tooltip block visibility when the tooltip is destroyed.
	 * Handles the case where the tooltip is hidden by mousing into and out of
	 * the tooltip itself, which doesn't trigger onHide on the tooltip modifier.
	 */
	listenForDestroy = modifier(() => {
		if (!this.tooltip.currentTooltip) {
			this.toggleTooltipBlock(false);
		}
	});

	<template>
		<div
			class="Help"
			data-luna-feature="meld"
			data-width={{or @width "auto"}}
			data-qid-component-type="Luna::Help"
			...attributes
			{{tooltip
				@tooltip
				isPortalled=(has-block "tooltip")
				disabled=@disabled
				placement=this.placement
				onShow=(fn this.toggleTooltipBlock true)
				onHide=(fn this.toggleTooltipBlock false)
			}}
		>
			{{#if (has-block)}}
				{{yield (hash isDisabled=@disabled isEnabled=this.isEnabled)}}
			{{else}}
				<Trigger @tag="div" />
			{{/if}}

			{{#if (and (has-block "tooltip") this.showTooltipBlock)}}
				<Portal @to={{this.tooltip.tooltipId}}>
					<span
						{{this.listenForDestroy}}
						aria-live="polite"
						{{announceOnInit
							urgency="polite"
							announceOnInit=true
							announcerService=this.announcerService
							async=this.async
						}}
					>
						{{yield to="tooltip"}}
					</span>
				</Portal>
			{{/if}}
		</div>
	</template>
}

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Tooltip, type TooltipPlacement } from '../tooltip';

import styles from './help.module.css';

interface HelpBaseProps {
	/** Text content for the tooltip */
	tooltip?: string | null;
	/** Where to place the tooltip relative to the trigger */
	placement?: TooltipPlacement;
	/** Whether the tooltip is disabled */
	disabled?: boolean;
	/** Display width mode */
	width?: 'auto' | 'full';
	/** Custom tooltip content (React node) */
	tooltipSlot?: React.ReactNode;
	/** Child content - renders inside the help container. When no children provided, shows default question mark icon */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type HelpProps = ComponentPropsWithRef<HelpBaseProps, HTMLDivElement>;

/**
 * Default trigger component - question mark icon in a circle
 */
function HelpTrigger() {
	return (
		<div className={styles.Trigger} data-test-trigger>
			<div className={styles.indicator}>
				<svg className={styles.icon} xmlns="http://www.w3.org/2000/svg" viewBox="0 0 12 12">
					<g fill="none" fillRule="evenodd" stroke="currentColor" strokeLinecap="round" strokeLinejoin="round">
						<path
							strokeWidth="1.5"
							d="M3.75 4S4 1.75 6 1.75 8.25 3 8.25 4c0 .751-.423 1.503-1.27 1.83C6.465 6.029 6 6.448 6 7v.25"
						/>
						<path d="M6.5 10a.5.5 0 11-1 0 .5.5 0 011 0z" />
					</g>
				</svg>
			</div>
		</div>
	);
}

/**
 * Help component that provides tooltip functionality for content.
 *
 * Can be used with a simple string tooltip, custom tooltip content via tooltipSlot,
 * or with children that can access disabled state via render function.
 *
 * When no children are provided, renders a default question mark icon trigger.
 */
export function Help(props: HelpProps) {
	const { ref, tooltip, placement = 'top', disabled = false, width = 'auto', tooltipSlot, children, className } = props;

	// Use default trigger if no children provided
	const content = children ?? <HelpTrigger />;

	// Determine tooltip content - tooltipSlot takes precedence over tooltip string
	const tooltipContent = tooltipSlot ?? tooltip;

	// If disabled or no tooltip content, render without tooltip wrapper
	if (disabled || !tooltipContent) {
		return (
			<div ref={ref} className={cn(styles.Help, className)} data-width={width} data-qid-component-type="Luna::Help">
				{content}
			</div>
		);
	}

	// Wrap with Tooltip component
	return (
		<Tooltip content={tooltipContent} placement={placement}>
			<div
				ref={ref}
				className={cn(styles.Help, className)}
				data-width={width}
				data-qid-component-type="Luna::Help"
				tabIndex={0}
			>
				{content}
			</div>
		</Tooltip>
	);
}

export default Help;
