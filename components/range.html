/* eslint-disable ember/no-at-ember-render-modifiers */
import { assert } from '@ember/debug';
import { fn } from '@ember/helper';
import { on } from '@ember/modifier';
import { action } from '@ember/object';
import { guidFor } from '@ember/object/internals';
import type Owner from '@ember/owner';
import willDestroy from '@ember/render-modifiers/modifiers/will-destroy';
import { htmlSafe } from '@ember/template';
import Component from '@glimmer/component';

import t from 'ember-intl/helpers/t';
import or from 'ember-truth-helpers/helpers/or';

import toString0 from '../helpers/to-string.ts';
import { KEYBOARD_KEYS } from '../utils/key-codes.ts';

type DragType = 'min' | 'max';

interface Values {
	min: number;
	max: number;
}

interface LunaRangeSignature {
	Element: HTMLDivElement;
	Args: {
		disabled: boolean;
		min: number | string;
		max: number | string;
		minValue?: number;
		minValueText?: string;
		minLabel?: string;
		maxValue?: number;
		maxValueText?: string;
		maxLabel?: string;
		minRange?: number | string;
		steps?: number;
		onChange?: (newValues: unknown) => void;
	};
}

/**
 * Range slider
 *
 * ```handlebars
 * <Range
 *  @minValue={{this.min}}
 *  @maxValue={{this.max}}
 *  @min="0"
 *  @max="100"
 *  @onChange={{this.updateValues}} />
 * ```
 * @public
 */
export default class LunaRange extends Component<LunaRangeSignature> {
	rangeSliderId = `${guidFor(this)}`;

	constructor(owner: Owner, args: LunaRangeSignature['Args']) {
		super(owner, args);
		assert('You must provide a @min to Range', args.min !== undefined);
		assert('You must provide a @max to Range', args.max !== undefined);
	}

	dragType: DragType | null = null;

	get rangeSliderElement() {
		const element = document.getElementById(this.rangeSliderId);
		assert('expect rangeSlider element', element);
		return element;
	}

	get min(): number {
		// NOTE: It would probably be good to handle NaN here.
		return typeof this.args.min === 'string' ? parseInt(this.args.min) : this.args.min;
	}

	get max(): number {
		// NOTE: It would probably be good to handle NaN here.
		return typeof this.args.max === 'string' ? parseInt(this.args.max) : this.args.max;
	}

	get minValue(): number {
		return this.args.minValue || this.min;
	}

	get maxValue(): number {
		return this.args.maxValue || this.max;
	}

	/**
	 * Determines how much the values change when the user manipulates the
	 * values via the keyboard. It is also used as the default minRange
	 * value if not provided.
	 */
	get incrementBy() {
		const steps = this.args.steps || 30;
		return Math.round((this.max - this.min) / steps);
	}

	/**
	 * Styles applied ot the min knob for positioning. This is also used
	 * by the active rail element to position its left edge.
	 */
	get minKnobStyle() {
		/**
		 * If the minValue is less than the range's min we make sure it doesn't get
		 * positioned outside the range's bounding box.
		 */
		let minValue = Math.max(this.minValue, this.min);

		// Also do this in case the minValue is greater than the max
		minValue = Math.min(this.minValue, this.max);

		const percent = ((minValue - this.min) / (this.max - this.min)) * 100;

		return htmlSafe(`left: ${percent}%;`);
	}

	/**
	 * Styles applied to the max knob for positioning. This is also used
	 * by the active rail element to position its right edge.
	 */
	get maxKnobStyle() {
		/**
		 * If the maxValue is less than the range's min we make sure it doesn't get
		 * positioned outside the range's bounding box.
		 */
		let maxValue = Math.max(this.maxValue, this.min);

		// Also do this in case the maxValue is greater than the max
		maxValue = Math.min(this.maxValue, this.max);

		const percent = ((maxValue - this.min) / (this.max - this.min)) * 100;

		// Orient direction from right so we can reuse it for the active rail style
		return htmlSafe(`right: ${100 - percent}%;`);
	}

	/**
	 * Styles applied ot the active rail
	 */
	get activeRailStyle() {
		return htmlSafe(`${this.minKnobStyle.toString()} ${this.maxKnobStyle.toString()}`);
	}

	@action
	onRailPress(event: MouseEvent | TouchEvent) {
		if (this.args.disabled) {
			return;
		}

		const rect = this.rangeSliderElement.getBoundingClientRect();
		let offsetX: number;
		if ('targetTouches' in event) {
			const touchX = event.targetTouches[0]?.clientX;
			assert('expected at least one touch', touchX !== undefined);
			offsetX = touchX;
		} else {
			offsetX = event.clientX;
		}
		const percent = offsetX / rect.width;

		let newValues = {
			min: this.minValue,
			max: this.maxValue,
		};

		const clickValue = Math.round((this.max - this.min) * percent);
		const minKnobIsCloser = newValues.min - clickValue > clickValue - newValues.max;
		const key = minKnobIsCloser ? 'min' : 'max';

		newValues[key] = clickValue;
		newValues = this.constrainValues(newValues, key);

		/**
		 * Pretend like the user also clicked on the knob so if they want they
		 * can click on the rail and keep dragging to adjust the value
		 */
		this.startDrag(key);

		if (this.args.onChange) {
			this.args.onChange(newValues);
		}
	}

	/**
	 * Captures key events on the knobs and updates the values appropriately.
	 *
	 * @param key min or max depending on which knob the user is triggering the key input on
	 * @param event keydown DOM event
	 */
	@action
	onKeyDown(key: DragType, event: KeyboardEvent) {
		if (this.args.disabled) {
			return;
		}

		let newValues = {
			min: this.minValue,
			max: this.maxValue,
		};

		switch (event.key) {
			case KEYBOARD_KEYS.UP_ARROW:
				event.preventDefault();
				newValues[key] = newValues[key] + this.incrementBy;
				break;
			case KEYBOARD_KEYS.DOWN_ARROW:
				event.preventDefault();
				newValues[key] = newValues[key] - this.incrementBy;
				break;
			case KEYBOARD_KEYS.RIGHT_ARROW:
				event.preventDefault();
				newValues[key] = newValues[key] + this.incrementBy;
				break;
			case KEYBOARD_KEYS.LEFT_ARROW:
				event.preventDefault();
				newValues[key] = newValues[key] - this.incrementBy;
				break;
			case KEYBOARD_KEYS.HOME:
				event.preventDefault();
				newValues[key] = this.min;
				break;
			case KEYBOARD_KEYS.END:
				event.preventDefault();
				newValues[key] = this.max;
				break;
		}

		newValues = this.constrainValues(newValues, key);

		if (this.args.onChange) {
			this.args.onChange(newValues);
		}
	}

	/**
	 * Catches the user starting a drag
	 *
	 * @param type Which knob is triggering this drag event
	 * @param event mousedown DOM event
	 */
	@action
	startDrag(type: DragType) {
		if (this.args.disabled) {
			return;
		}

		this.dragType = type;

		/**
		 * We set these events on the body so you can move your cursor anywhere
		 * while dragging and still see the values update. This reduces the finicky
		 * nature of the slider.
		 */
		document.body.addEventListener('mouseup', this.endDrag);
		document.body.addEventListener('mousemove', this.onDrag);
		document.body.addEventListener('touchend', this.endDrag);
		document.body.addEventListener('touchmove', this.onDrag);
	}

	/**
	 * Stops the user drag event and cleans up event listeners
	 */
	@action
	endDrag() {
		this.dragType = null;
		this.removeEventListeners();
	}

	/**
	 * Updates the values as the user drags the knob around
	 *
	 * @param event mousemove DOM event
	 */
	@action
	onDrag(event: MouseEvent | TouchEvent) {
		if (!this.dragType) {
			return;
		}

		const rect = this.rangeSliderElement.getBoundingClientRect();
		let clientX: number;
		if ('targetTouches' in event) {
			const touchX = event.targetTouches[0]?.clientX;
			assert('expected at least one touch', touchX !== undefined);
			clientX = touchX;
		} else {
			clientX = event.clientX;
		}
		const offsetX = clientX - rect.left;
		const percent = offsetX / rect.width;

		let newValues = {
			min: this.minValue,
			max: this.maxValue,
		};

		newValues[this.dragType] = Math.round((this.max - this.min) * percent);
		newValues = this.constrainValues(newValues, this.dragType);

		if (this.args.onChange) {
			this.args.onChange(newValues);
		}
	}

	/**
	 * This ensures that the min and max knobs don't
	 * end up on the wrong side of each other or exceed the
	 * range slider's range. The slider that is prioritized will
	 * not get its value adjusted.
	 *
	 * @param values The values to constrain to the min/max
	 * @param priority Which value to prioritize when ensuring that the knobs don't pass each other.
	 */
	constrainValues(values: Values, priority = 'min'): Values {
		let newValues = Object.assign({}, values);

		/**
		 * Don't let the min/max values be less/greater than Range's min/max
		 * so we don't end up positioning them out of bounds
		 */
		newValues.min = Math.max(newValues.min, this.min);
		newValues.max = Math.min(newValues.max, this.max);

		if (priority === 'max') {
			// Don't let min/max values exceed their counterpart's
			// Min knob must always be on the left of the max knob.
			newValues.min = Math.min(newValues.min, newValues.max);

			// Max knob must always be on the right of the min knob.
			newValues.max = Math.max(newValues.max, newValues.min);
		} else {
			/**
			 * This is identical to the above adjustment, we just change the order
			 * so the max value gets adjusted first.
			 */
			newValues.max = Math.max(newValues.max, newValues.min);
			newValues.min = Math.min(newValues.min, newValues.max);
		}

		newValues = this.spaceValues(newValues, priority);

		return newValues;
	}

	/**
	 * Spaces apart the min/max values so the knobs don't end up
	 * on top of each other using the minRange argument. The prioritized
	 * knob won't get its value adjusted.
	 *
	 * @param values The values to space apart
	 * @param priority Which value to freeze when ensuring that the values aren't too close
	 */
	spaceValues(values: Values, priority: string) {
		const newValues = Object.assign({}, values);
		const minRange = (typeof this.args.minRange === 'string' ? parseInt(this.args.minRange) : null) || this.incrementBy;

		/**
		 * Values are already sufficiently far apart
		 */
		if (newValues.max - newValues.min >= minRange) {
			return newValues;
		}

		// Deals with dragging to far right
		if (newValues.min > this.max - minRange) {
			newValues.max = this.max;
			newValues.min = newValues.max - minRange;

			// Deals with dragging to the far left
		} else if (newValues.max < this.min + minRange) {
			newValues.min = this.min;
			newValues.max = newValues.min + minRange;

			// Deals with adjusting range in the middle
		} else if (priority === 'max') {
			newValues.min = newValues.max - minRange;
		} else {
			newValues.max = newValues.min + minRange;
		}

		return newValues;
	}

	/**
	 * Removes event listeners. Because they are manually attached to the body element
	 * they won't automatically get cleaned up when the Range component is removed
	 * from the dom.
	 */
	@action
	removeEventListeners() {
		document.body.removeEventListener('mouseup', this.endDrag);
		document.body.removeEventListener('mousemove', this.onDrag);
		document.body.removeEventListener('touchend', this.endDrag);
		document.body.removeEventListener('touchmove', this.onDrag);
	}

	@action
	onWillDestroy() {
		this.removeEventListeners();
	}

import { useCallback, useEffect, useId, useRef, useState } from 'react';

import { cn } from '../../utils/cn';

import styles from './range.module.css';

export interface RangeValues {
	min: number;
	max: number;
}

type DragType = 'min' | 'max' | null;

export interface RangeProps extends Omit<React.HTMLAttributes<HTMLDivElement>, 'onChange'> {
	/** Optional ref for the root element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** Minimum bound of the range */
	min: number | string;
	/** Maximum bound of the range */
	max: number | string;
	/** Current minimum value (left knob position) */
	minValue?: number;
	/** Current maximum value (right knob position) */
	maxValue?: number;
	/** Accessible text for the min value */
	minValueText?: string;
	/** Accessible text for the max value */
	maxValueText?: string;
	/** ARIA label for the min knob */
	minLabel?: string;
	/** ARIA label for the max knob */
	maxLabel?: string;
	/** Minimum distance between min and max values */
	minRange?: number | string;
	/** Number of steps to divide the range into (affects keyboard increment) */
	steps?: number;
	/** Whether the range is disabled */
	disabled?: boolean;
	/** Callback fired when values change */
	onChange?: (values: RangeValues) => void;
}

/**
 * A dual-handle range slider component.
 *
 * Allows users to select a range of values by dragging two knobs.
 * Supports keyboard navigation and touch/mouse interactions.
 *
 * @example
 * ```tsx
 * <Range
 *   min={0}
 *   max={100}
 *   minValue={minValue}
 *   maxValue={maxValue}
 *   onChange={({ min, max }) => {
 *     setMinValue(min);
 *     setMaxValue(max);
 *   }}
 * />
 * ```
 */
export function Range(props: RangeProps) {
	const {
		ref,
		min: minProp,
		max: maxProp,
		minValue: minValueProp,
		maxValue: maxValueProp,
		minValueText,
		maxValueText,
		minLabel = 'Minimum',
		maxLabel = 'Maximum',
		minRange: minRangeProp,
		steps = 30,
		disabled = false,
		onChange,
		className,
		...rest
	} = props;

	const generatedId = useId();
	const rangeSliderId = `range-slider-${generatedId}`;
	const rangeSliderRef = useRef<HTMLDivElement>(null);
	const [dragType, setDragType] = useState<DragType>(null);

	// Parse numeric props
	const min = typeof minProp === 'string' ? parseInt(minProp, 10) : minProp;
	const max = typeof maxProp === 'string' ? parseInt(maxProp, 10) : maxProp;
	const minValue = minValueProp ?? min;
	const maxValue = maxValueProp ?? max;

	// Calculate increment for keyboard navigation
	const incrementBy = Math.round((max - min) / steps);

	// Calculate minimum range between knobs
	const minRange = typeof minRangeProp === 'string' ? parseInt(minRangeProp, 10) : (minRangeProp ?? incrementBy);

	// Calculate knob positions as percentages
	const getMinKnobPercent = () => {
		let value = Math.max(minValue, min);
		value = Math.min(value, max);
		return ((value - min) / (max - min)) * 100;
	};

	const getMaxKnobPercent = () => {
		let value = Math.max(maxValue, min);
		value = Math.min(value, max);
		return ((value - min) / (max - min)) * 100;
	};

	const minKnobPercent = getMinKnobPercent();
	const maxKnobPercent = getMaxKnobPercent();

	// Ensure minimum distance between values (defined first so constrainValues can use it)
	const spaceValues = useCallback(
		(values: RangeValues, priority: string): RangeValues => {
			const newValues = { ...values };

			// Values are already sufficiently far apart
			if (newValues.max - newValues.min >= minRange) {
				return newValues;
			}

			// Handle dragging to far right
			if (newValues.min > max - minRange) {
				newValues.max = max;
				newValues.min = newValues.max - minRange;
			}
			// Handle dragging to far left
			else if (newValues.max < min + minRange) {
				newValues.min = min;
				newValues.max = newValues.min + minRange;
			}
			// Handle adjusting range in the middle
			else if (priority === 'max') {
				newValues.min = newValues.max - minRange;
			} else {
				newValues.max = newValues.min + minRange;
			}

			return newValues;
		},
		[min, max, minRange],
	);

	// Constrain values to valid range
	const constrainValues = useCallback(
		(values: RangeValues, priority: 'min' | 'max' = 'min'): RangeValues => {
			let newValues = { ...values };

			// Don't let values exceed bounds
			newValues.min = Math.max(newValues.min, min);
			newValues.max = Math.min(newValues.max, max);

			if (priority === 'max') {
				// Ensure min doesn't exceed max
				newValues.min = Math.min(newValues.min, newValues.max);
				// Ensure max doesn't go below min
				newValues.max = Math.max(newValues.max, newValues.min);
			} else {
				// Ensure max doesn't go below min
				newValues.max = Math.max(newValues.max, newValues.min);
				// Ensure min doesn't exceed max
				newValues.min = Math.min(newValues.min, newValues.max);
			}

			// Apply minimum range spacing
			newValues = spaceValues(newValues, priority);

			return newValues;
		},
		[min, max, spaceValues],
	);

	// Handle rail click
	const handleRailPress = useCallback(
		(event: React.MouseEvent | React.TouchEvent) => {
			if (disabled) return;

			const element = rangeSliderRef.current;
			if (!element) return;

			const rect = element.getBoundingClientRect();
			let clientX: number;

			if ('touches' in event) {
				const touch = event.touches[0];
				if (!touch) return;
				clientX = touch.clientX;
			} else {
				clientX = event.clientX;
			}

			const offsetX = clientX - rect.left;
			const percent = offsetX / rect.width;
			const clickValue = Math.round((max - min) * percent) + min;

			let newValues: RangeValues = { min: minValue, max: maxValue };

			// Determine which knob is closer
			const minKnobIsCloser = Math.abs(minValue - clickValue) < Math.abs(maxValue - clickValue);
			const key = minKnobIsCloser ? 'min' : 'max';

			newValues[key] = clickValue;
			newValues = constrainValues(newValues, key);

			// Start drag on the clicked knob
			setDragType(key);

			onChange?.(newValues);
		},
		[disabled, min, max, minValue, maxValue, constrainValues, onChange],
	);

	// Handle keyboard navigation
	const handleKeyDown = useCallback(
		(key: 'min' | 'max', event: React.KeyboardEvent) => {
			if (disabled) return;

			let newValues: RangeValues = { min: minValue, max: maxValue };

			switch (event.key) {
				case 'ArrowUp':
				case 'ArrowRight':
					event.preventDefault();
					newValues[key] = newValues[key] + incrementBy;
					break;
				case 'ArrowDown':
				case 'ArrowLeft':
					event.preventDefault();
					newValues[key] = newValues[key] - incrementBy;
					break;
				case 'Home':
					event.preventDefault();
					newValues[key] = min;
					break;
				case 'End':
					event.preventDefault();
					newValues[key] = max;
					break;
				default:
					return;
			}

			newValues = constrainValues(newValues, key);
			onChange?.(newValues);
		},
		[disabled, minValue, maxValue, min, max, incrementBy, constrainValues, onChange],
	);

	// Start drag
	const startDrag = useCallback(
		(type: 'min' | 'max') => {
			if (disabled) return;
			setDragType(type);
		},
		[disabled],
	);

	// End drag
	const endDrag = useCallback(() => {
		setDragType(null);
	}, []);

	// Handle drag
	const handleDrag = useCallback(
		(event: MouseEvent | TouchEvent) => {
			if (!dragType) return;

			const element = rangeSliderRef.current;
			if (!element) return;

			const rect = element.getBoundingClientRect();
			let clientX: number;

			if ('touches' in event) {
				// Prevent page scrolling while dragging the slider.
				event.preventDefault();
				const touch = event.touches[0];
				if (!touch) return;
				clientX = touch.clientX;
			} else {
				clientX = event.clientX;
			}

			const offsetX = clientX - rect.left;
			const percent = offsetX / rect.width;

			let newValues: RangeValues = { min: minValue, max: maxValue };
			newValues[dragType] = Math.round((max - min) * percent) + min;
			newValues = constrainValues(newValues, dragType);

			onChange?.(newValues);
		},
		[dragType, min, max, minValue, maxValue, constrainValues, onChange],
	);

	// Set up drag event listeners
	useEffect(() => {
		if (dragType) {
			document.body.addEventListener('mouseup', endDrag);
			document.body.addEventListener('mousemove', handleDrag);
			document.body.addEventListener('touchend', endDrag);
			document.body.addEventListener('touchmove', handleDrag, { passive: false });

			return () => {
				document.body.removeEventListener('mouseup', endDrag);
				document.body.removeEventListener('mousemove', handleDrag);
				document.body.removeEventListener('touchend', endDrag);
				document.body.removeEventListener('touchmove', handleDrag);
			};
		}
	}, [dragType, endDrag, handleDrag]);

	const wrapperClassName = cn(styles.Range, className);

	return (
		<div
			ref={(node) => {
				rangeSliderRef.current = node;
				if (typeof ref === 'function') {
					ref(node);
				} else if (ref) {
					ref.current = node;
				}
			}}
			className={wrapperClassName}
			data-disabled={disabled ? 'true' : undefined}
			data-qid-component-type="Luna::Range"
			id={rangeSliderId}
			{...rest}
		>
			<div className={styles.rail} />
			<div className={styles.railActive} style={{ left: `${minKnobPercent}%`, right: `${100 - maxKnobPercent}%` }} />
			<div className={styles.railClickTarget} onMouseDown={handleRailPress} onTouchStart={handleRailPress} />

			{/* Min knob */}
			<div
				className={styles.button}
				style={{ left: `${minKnobPercent}%` }}
				tabIndex={disabled ? -1 : 0}
				role="slider"
				aria-valuemax={max}
				aria-valuemin={min}
				aria-valuenow={minValue}
				aria-disabled={disabled ? 'true' : 'false'}
				aria-valuetext={minValueText ?? String(minValue)}
				aria-label={minLabel}
				data-type="min"
				data-test-min
				onKeyDown={(e) => handleKeyDown('min', e)}
				onMouseDown={() => startDrag('min')}
				onTouchStart={() => startDrag('min')}
			>
				<div className={styles.content}>
					<svg className={styles.bars} aria-hidden="true" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 56 64">
						<path d="M0 0h8v64H0z" />
						<path d="M24 0h8v64h-8z" />
						<path d="M48 0h8v64h-8z" />
					</svg>
				</div>
			</div>

			{/* Max knob */}
			<div
				className={styles.button}
				style={{ right: `${100 - maxKnobPercent}%` }}
				tabIndex={disabled ? -1 : 0}
				role="slider"
				aria-valuemax={max}
				aria-valuemin={min}
				aria-valuenow={maxValue}
				aria-disabled={disabled ? 'true' : 'false'}
				aria-valuetext={maxValueText ?? String(maxValue)}
				aria-label={maxLabel}
				data-type="max"
				data-test-max
				onKeyDown={(e) => handleKeyDown('max', e)}
				onMouseDown={() => startDrag('max')}
				onTouchStart={() => startDrag('max')}
			>
				<div className={styles.content}>
					<svg className={styles.bars} aria-hidden="true" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 56 64">
						<path d="M0 0h8v64H0z" />
						<path d="M24 0h8v64h-8z" />
						<path d="M48 0h8v64h-8z" />
					</svg>
				</div>
			</div>
		</div>
	);
}

export default Range;
