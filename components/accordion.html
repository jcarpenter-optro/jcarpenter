import { hash } from '@ember/helper';
import { on } from '@ember/modifier';
import { action } from '@ember/object';
import { guidFor } from '@ember/object/internals';
import didUpdate from '@ember/render-modifiers/modifiers/did-update';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';

import { AnimatedContainer, animatedIf } from 'ember-animated';
import fade from 'ember-animated/transitions/fade';
import element from 'ember-element-helper/helpers/element';
import t from 'ember-intl/helpers/t';
import sortableHandle from 'ember-sortable/modifiers/sortable-handle';

import toStringHelper from '../helpers/to-string.ts';

interface ExternallyManaged {
	expanded: boolean;
	onChange: (newValue: boolean) => void;
}

interface NotExternallyManaged {
	expanded?: boolean;
	onChange?: never;
}

export interface LunaAccordionSignature {
	Element: HTMLDivElement;
	Args: {
		variant: 'underline' | 'primary' | 'white' | 'text';
		sortable?: boolean;
		landmark?: boolean;
	} & (ExternallyManaged | NotExternallyManaged);
	Blocks: {
		title: [{ isExpanded: boolean }];
		panel: [{ isExpanded: boolean }];
		decorations?: [];
	};
}

/**
 * Control element that toggles a panel of content.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/accordion.md)
 */
export default class Accordion extends Component<LunaAccordionSignature> {
	titleId = `${guidFor(this)}--title`;
	panelId = `${guidFor(this)}--panel`;

	fadeTransition = fade;

	@tracked isExpanded = this.args.expanded === true;

	@action
	syncExpandedState() {
		if (stateIsManagedExternally(this.args)) {
			this.isExpanded = this.args.expanded;
		}
	}

	@action
	toggle() {
		if (stateIsManagedExternally(this.args)) {
			this.args.onChange(!this.isExpanded);
		} else {
			this.isExpanded = !this.isExpanded;
		}
	}

	<template>
		{{! template-lint-disable data-qid-interactive no-unsupported-role-attributes }}
		<div
			class="Accordion"
			data-variant={{@variant}}
			data-expanded={{toStringHelper this.isExpanded}}
			data-handle={{@sortable}}
			...attributes
			data-qid-component-type="Luna::Accordion"
			{{didUpdate this.syncExpandedState @expanded}}
		>
			<div class="titlebar">
				<button
					class="click-zone"
					type="button"
					aria-expanded={{toStringHelper this.isExpanded}}
					aria-controls={{this.panelId}}
					aria-labelledby={{this.titleId}}
					data-qid-accordion-toggle
					{{on "click" this.toggle}}
				/>

				{{#if @sortable}}
					<div class="handle" data-test-accordion-handle>
						<button
							class="handle-button"
							type="button"
							aria-label={{t "accordion.spacebar-instruction"}}
							{{sortableHandle}}
						>
							<svg
								class="handle-icon"
								xmlns="http://www.w3.org/2000/svg"
								fill="none"
								viewBox="0 0 24 24"
								stroke="currentColor"
							>
								<path
									stroke-linecap="round"
									stroke-linejoin="round"
									stroke-width="2"
									d="M4 6h16M4 12h16M4 18h16"
								></path>
							</svg>
						</button>
					</div>
				{{/if}}

				<div class="toggle" aria-hidden="true" data-open={{toStringHelper this.isExpanded}}>
					<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16">
						<path
							d="M3.88 2.73l1.49-1.48 5.26 5.27L12.12 8l-1.49 1.48-5.26 5.27-1.49-1.48L9.15 8z"
							fill="currentColor"
						></path>
					</svg>
				</div>

				<h2 id={{this.titleId}} class="title" role="presentation" data-qid-accordion-title>
					{{yield (hash isExpanded=this.isExpanded) to="title"}}
				</h2>

				<div class="decorations">
					{{yield to="decorations"}}
				</div>
			</div>

			{{#let (element (if @landmark "section" "div")) as |Tag|}}
				<Tag
					class="panel"
					id={{this.panelId}}
					role="tabpanel"
					aria-expanded={{toStringHelper this.isExpanded}}
					aria-labelledby={{this.titleId}}
				>
					<AnimatedContainer class="animation-container">
						{{#animatedIf this.isExpanded use=this.fadeTransition duration=150}}
							<div>
								{{yield (hash isExpanded=this.isExpanded) to="panel"}}
							</div>
						{{/animatedIf}}
					</AnimatedContainer>
				</Tag>
			{{/let}}
		</div>
	</template>
}

function stateIsManagedExternally(
	args: LunaAccordionSignature['Args'],
): args is LunaAccordionSignature['Args'] & ExternallyManaged {
	return args.onChange !== undefined;
}

import { Children, createContext, Fragment, type ReactElement, useCallback, useContext, useId, useState } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './accordion.module.css';

export type AccordionVariant = 'underline' | 'primary' | 'white' | 'text';

export interface AccordionContextValue {
	isExpanded: boolean;
}

export const AccordionContext = createContext<AccordionContextValue | null>(null);

/**
 * Hook to access accordion context.
 * Must be used within Accordion compound component children.
 */
export function useAccordionContext(): AccordionContextValue {
	const context = useContext(AccordionContext);
	if (!context) {
		throw new Error('useAccordionContext must be used within an Accordion component');
	}
	return context;
}

interface AccordionSlots {
	title?: React.ReactNode;
	panel?: React.ReactNode;
	decorations?: React.ReactNode;
}

/**
 * Extracts slot content from children by matching component displayName.
 * This allows children to be passed in any order while ensuring they render
 * in the correct positions.
 */
function extractAccordionSlots(children: React.ReactNode): AccordionSlots {
	const slots: AccordionSlots = {};

	const processChild = (child: React.ReactNode): void => {
		if (!child) return;

		if (Array.isArray(child)) {
			child.forEach(processChild);
			return;
		}

		if (typeof child === 'object' && 'type' in child) {
			const element = child as ReactElement<{ children?: React.ReactNode }>;
			const displayName = (element.type as { displayName?: string }).displayName;

			switch (displayName) {
				case 'Accordion.Title':
					slots.title = element.props.children;
					break;
				case 'Accordion.Panel':
					slots.panel = element.props.children;
					break;
				case 'Accordion.Decorations':
					slots.decorations = element.props.children;
					break;
				default:
					// Handle fragments by recursing into their children
					if (element.type === Fragment) {
						Children.forEach(element.props.children, processChild);
					}
			}
		}
	};

	Children.forEach(children, processChild);
	return slots;
}

export interface AccordionProps {
	/** Visual variant of the accordion */
	variant: AccordionVariant;
	/** Whether to render the panel as a <section> element (for landmarks) */
	landmark?: boolean;
	/** Controlled expanded state */
	expanded?: boolean;
	/** Default expanded state for uncontrolled mode */
	defaultExpanded?: boolean;
	/** Callback fired when expanded state changes */
	onChange?: (expanded: boolean) => void;
	/** Compound component children (Accordion.Title, Accordion.Panel, etc.) */
	children: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type AccordionComponentProps = ComponentPropsWithRef<AccordionProps, HTMLDivElement>;

interface AccordionSubcomponentProps {
	children: React.ReactNode;
}

/**
 * Control element that toggles a panel of content.
 *
 * Use compound components to compose the accordion:
 * - `Accordion.Title` - Required title content
 * - `Accordion.Panel` - Required panel content
 * - `Accordion.Decorations` - Optional decorations (badges, etc.)
 *
 * Children can be passed in any order - they will always render in the correct position.
 * Use `useAccordionContext()` hook within children to access `isExpanded` state.
 *
 * Supports both controlled and uncontrolled modes.
 *
 * @example
 * ```tsx
 * <Accordion variant="primary" defaultExpanded>
 *   <Accordion.Title>
 *     <Heading level={3}>Section Title</Heading>
 *   </Accordion.Title>
 *   <Accordion.Panel>
 *     <p>Panel content goes here</p>
 *   </Accordion.Panel>
 * </Accordion>
 * ```
 */
function AccordionRoot(props: AccordionComponentProps) {
	const {
		ref,
		variant,
		landmark = false,
		expanded: controlledExpanded,
		defaultExpanded = false,
		onChange,
		children,
		className,
		...rest
	} = props;

	const generatedId = useId();
	const titleId = `${generatedId}--title`;
	const panelId = `${generatedId}--panel`;

	// Determine if we're in controlled mode
	const isControlled = controlledExpanded !== undefined && onChange !== undefined;

	// Internal state for uncontrolled mode
	const [internalExpanded, setInternalExpanded] = useState(defaultExpanded);

	// Current expanded state
	const isExpanded = isControlled ? controlledExpanded : internalExpanded;

	const handleToggle = useCallback(() => {
		const newValue = !isExpanded;
		if (isControlled) {
			onChange?.(newValue);
		} else {
			setInternalExpanded(newValue);
		}
	}, [isControlled, isExpanded, onChange]);

	const handleKeyDown = useCallback(
		(event: React.KeyboardEvent<HTMLButtonElement>) => {
			if (event.key === 'Enter' || event.key === ' ') {
				event.preventDefault();
				handleToggle();
			}
		},
		[handleToggle],
	);

	const contextValue: AccordionContextValue = { isExpanded };

	// Extract slot content from children (order-independent)
	const slots = extractAccordionSlots(children);

	// Panel element tag based on landmark prop
	const PanelTag = landmark ? 'section' : 'div';

	return (
		<AccordionContext.Provider value={contextValue}>
			<div
				ref={ref}
				className={cn(styles.Accordion, className)}
				data-variant={variant}
				data-expanded={String(isExpanded)}
				data-qid-component-type="Luna::Accordion"
				{...rest}
			>
				<div className={styles.titlebar}>
					<button
						className={styles.clickZone}
						type="button"
						aria-expanded={isExpanded}
						aria-controls={panelId}
						aria-labelledby={titleId}
						data-qid-accordion-toggle
						onClick={handleToggle}
						onKeyDown={handleKeyDown}
					/>

					<div className={styles.toggle} aria-hidden="true" data-open={String(isExpanded)}>
						<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16">
							<path
								d="M3.88 2.73l1.49-1.48 5.26 5.27L12.12 8l-1.49 1.48-5.26 5.27-1.49-1.48L9.15 8z"
								fill="currentColor"
							/>
						</svg>
					</div>

					<h2 id={titleId} className={styles.title} role="presentation" data-qid-accordion-title>
						{slots.title}
					</h2>

					<div className={styles.decorations}>{slots.decorations}</div>
				</div>

				<PanelTag className={styles.panel} id={panelId} role="region" aria-labelledby={titleId} hidden={!isExpanded}>
					{isExpanded && <div className={styles.panelContent}>{slots.panel}</div>}
				</PanelTag>
			</div>
		</AccordionContext.Provider>
	);
}

AccordionRoot.displayName = 'Accordion';

// Subcomponents - these don't render themselves, they're extracted by the parent
const AccordionTitle = ({ children }: AccordionSubcomponentProps) => children;
AccordionTitle.displayName = 'Accordion.Title';

const AccordionPanel = ({ children }: AccordionSubcomponentProps) => children;
AccordionPanel.displayName = 'Accordion.Panel';

const AccordionDecorations = ({ children }: AccordionSubcomponentProps) => children;
AccordionDecorations.displayName = 'Accordion.Decorations';

/**
 * Accordion compound component with namespaced subcomponents.
 *
 * Subcomponents can be passed in any order - they will render in the correct position.
 * Use `useAccordionContext()` hook within children to access accordion state.
 */
export const Accordion = Object.assign(AccordionRoot, {
	Title: AccordionTitle,
	Panel: AccordionPanel,
	Decorations: AccordionDecorations,
});

export default Accordion;
