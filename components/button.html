/**
 * A button that allows you to use it as a traditional button or as a link styled to look like a button.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/button.md)
 */
export default class LunaButtonClass<Args extends LunaButtonArgs> extends Component<LunaButtonSignature<Args>> {
	get spinnerSize(): LunaSpinnerSignature['Args']['size'] {
		if (this.args.size === 'xs') {
			return 'sm';
		}
		return this.args.size ?? 'md';
	}

	get spinnerVariant(): LunaSpinnerSignature['Args']['variant'] {
		if (VARIANTS_REVERSED_LOADER.includes(this.args.variant as LightLoaderVariant)) {
			return 'reversed';
		}
		if (VARIANTS_DEFAULT_LOADER.includes(this.args.variant as DarkLoaderVariant)) {
			return 'default';
		}
		return 'default';
	}

	@consume(UI_CONTROL_SIZE_CONTEXT_KEY) declare uiControlSize: 'small' | 'medium' | 'large' | undefined;

	get size() {
		return this.uiControlSize ?? this.args.size ?? 'md';
	}

	<template>

import { type AnchorHTMLAttributes, type ButtonHTMLAttributes } from 'react';

import { cn } from '../../utils/cn';
import { Spinner, type SpinnerSize } from '../spinner';

import styles from './button.module.css';

export type ButtonVariant =
	| 'white'
	| 'transparent'
	| 'gray'
	| 'gray-tertiary'
	| 'blue'
	| 'red'
	| 'orange'
	| 'green'
	| 'purple'
	| 'unstyled';

// Variants that use the reversed (light) spinner for dark backgrounds
const VARIANTS_REVERSED_LOADER = ['blue', 'red', 'orange', 'green', 'purple'] as const;

export type ButtonSize = 'xs' | 'sm' | 'md' | 'lg';

export type ButtonWidth = 'auto' | 'full';

export type ButtonAlign = 'left' | 'center' | 'right';

interface ButtonBaseProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLButtonElement | HTMLAnchorElement>;
	/** Visual variant of the button */
	variant?: ButtonVariant;
	/** Size of the button */
	size?: ButtonSize;
	/** Width behavior */
	width?: ButtonWidth;
	/** Text alignment within the button */
	align?: ButtonAlign;
	/** Whether the button is in a loading state */
	loading?: boolean;
	/** Whether the button appears active */
	active?: boolean;
	/** Whether the button appears focused */
	focused?: boolean;
	/** Whether the button shows an error state */
	errored?: boolean;
	/** Whether the button is disabled */
	disabled?: boolean;
	/** Children to render inside the button */
	children?: React.ReactNode;
}

type ButtonAsButton = ButtonBaseProps &
	Omit<ButtonHTMLAttributes<HTMLButtonElement>, keyof ButtonBaseProps> & {
		/** When href is not provided, renders as a button */
		href?: undefined;
	};

type ButtonAsAnchor = ButtonBaseProps &
	Omit<AnchorHTMLAttributes<HTMLAnchorElement>, keyof ButtonBaseProps> & {
		/** When href is provided, renders as an anchor */
		href: string;
	};

export type ButtonProps = ButtonAsButton | ButtonAsAnchor;

/**
 * A button component that can render as a button or anchor element.
 *
 * Matches the Luna::Button Ember component API and styling.
 */
export function Button(props: ButtonProps) {
	const {
		ref,
		variant = 'white',
		size = 'md',
		width = 'auto',
		align,
		loading = false,
		active = false,
		focused = false,
		errored = false,
		disabled,
		children,
		className,
		href,
		...rest
	} = props;

	const isDisabled = disabled || loading;
	const isUnstyled = variant === 'unstyled';

	// Determine spinner size based on button size (xs uses sm spinner)
	const spinnerSize: SpinnerSize = size === 'xs' ? 'sm' : size;

	// Determine spinner variant based on button variant
	const spinnerVariant = (VARIANTS_REVERSED_LOADER as readonly string[]).includes(variant) ? 'reversed' : 'default';

	const buttonClassName = cn(styles.btn, !isUnstyled && styles.Button, className);

	const content = (
		<div className={styles.content} data-qid-luna-button-content>
			<div className={styles.label} data-align={align} data-qid-luna-button-label>
				{children}
			</div>
			{loading && (
				<div className={styles.loading}>
					<Spinner variant={spinnerVariant} size={spinnerSize} data-qid-loading />
				</div>
			)}
		</div>
	);

	const commonProps = {
		'className': buttonClassName,
		'data-variant': variant,
		'data-size': size,
		'data-width': width,
		'data-active': active ? 'true' : undefined,
		'data-focused': focused ? 'true' : undefined,
		'data-errored': errored ? 'true' : undefined,
		'data-luna-feature': 'meld',
		'data-qid-component-type': 'Luna::Button',
		'aria-disabled': isDisabled || undefined,
		'aria-busy': loading || undefined,
	};

	// Render as anchor if href is provided
	if (href && !isDisabled) {
		return (
			<a
				{...(rest as AnchorHTMLAttributes<HTMLAnchorElement>)}
				{...commonProps}
				href={href}
				ref={ref as React.Ref<HTMLAnchorElement>}
			>
				{content}
			</a>
		);
	}

	// Render as button
	return (
		<button
			{...(rest as ButtonHTMLAttributes<HTMLButtonElement>)}
			{...commonProps}
			type={(rest as ButtonHTMLAttributes<HTMLButtonElement>).type ?? 'button'}
			disabled={isDisabled}
			ref={ref as React.Ref<HTMLButtonElement>}
		>
			{content}
		</button>
	);
}

export default Button;
