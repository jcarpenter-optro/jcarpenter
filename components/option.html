import { useCallback } from 'react';

import { Menu } from '@base-ui-components/react/menu';

import { cn } from '../../utils/cn';

import styles from './option.module.css';

export interface OptionProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLButtonElement | HTMLAnchorElement>;
	/** Whether the option is disabled */
	disabled?: boolean;
	/** Whether the option is selected (shows checkmark) */
	selected?: boolean;
	/** Click handler */
	onClick?: () => void;
	/** If provided, renders as anchor instead of button */
	href?: string;
	/** Link target */
	target?: React.AnchorHTMLAttributes<HTMLAnchorElement>['target'];
	/** Link rel */
	rel?: React.AnchorHTMLAttributes<HTMLAnchorElement>['rel'];
	/** Whether to open in quick view */
	openInQuickView?: 'always' | 'never';
	/** Children content */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

/**
 * Renders an option for the user to select. Typically used within a Menu.
 *
 * Matches the Luna::Option Ember component API and styling.
 * Uses Base UI Menu.Item as the foundation for keyboard navigation.
 */
export function Option(props: OptionProps) {
	const {
		ref,
		disabled = false,
		selected,
		onClick,
		href,
		target,
		rel,
		openInQuickView,
		children,
		className,
		...rest
	} = props;

	const optionClassName = cn(styles.Option, className);

	// Handle mouse enter/leave to focus/blur for consistent styling
	const handleMouseEnter = useCallback((e: React.MouseEvent) => {
		if (e.currentTarget instanceof HTMLElement) {
			e.currentTarget.focus();
		}
	}, []);

	const handleMouseLeave = useCallback((e: React.MouseEvent) => {
		if (e.currentTarget instanceof HTMLElement) {
			e.currentTarget.blur();
		}
	}, []);

	const handleClick = useCallback(
		(e: React.MouseEvent) => {
			if (disabled) {
				e.preventDefault();
				return;
			}
			onClick?.();
		},
		[disabled, onClick],
	);

	// Use Base UI Menu.Item for keyboard navigation
	// Render as anchor or button based on href prop
	return (
		<Menu.Item
			disabled={disabled}
			onClick={handleClick}
			closeOnClick={false} // Menu handles closing
			render={(itemProps) => {
				const Tag = href ? 'a' : 'button';
				return (
					<Tag
						{...itemProps}
						ref={ref as React.Ref<HTMLButtonElement & HTMLAnchorElement>}
						className={optionClassName}
						role="menuitem"
						href={href}
						target={target}
						rel={rel}
						disabled={disabled && !href}
						aria-disabled={disabled}
						aria-current={selected === true ? 'true' : undefined}
						data-open-in-quick-view={openInQuickView}
						data-qid-component-type="Luna::Option"
						onMouseEnter={handleMouseEnter}
						onMouseLeave={handleMouseLeave}
						{...rest}
					>
						{selected !== undefined ? (
							<div className={styles['selection-layout']}>
								<div className={styles.checkmark} data-selected={selected === true ? 'true' : undefined}>
									<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16" className={styles.svg}>
										<path
											fill="currentColor"
											d="M14.17 2.6l-8.22 8.23L1.82 6.7.54 7.98l5.41 5.42 9.51-9.51-1.29-1.29z"
										></path>
									</svg>
								</div>
								<div className={styles['label-layout']}>{children}</div>
							</div>
						) : (
							<div className={styles['label-layout']}>{children}</div>
						)}
					</Tag>
				);
			}}
		/>
	);
}

export default Option;
