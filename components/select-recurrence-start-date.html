/**
 * Allows the user to add a start date to an iCalendar recurrence rule.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/select-recurrence-start-date.md)
 */
export default class SelectRecurrenceStartDate extends Component<SelectRecurrenceStartDateSignature> {
	@service declare intl: IntlService;

	@tracked rule: RRule;

	@consume('luna-input-id') inputId!: string;

	get id() {
		return this.args.id ?? this.inputId;
	}

	constructor(owner: Owner, args: SelectRecurrenceStartDateSignature['Args']) {
		super(owner, args);

		this.rule = new RRule(this.args.value, this.intl);
	}

	get formattedSelection() {
		return this.args.value ? this.rule.dtStartDate : null;
	}

	@action
	onRuleUpdate() {
		this.rule = new RRule(this.args.value, this.intl);
	}

	@action
	onChange(value: string | null | undefined) {
		if (!value) {
			this.rule.resetRecurrenceStartDate();
			this.args.onChange(this.rule.string);

			return;
		}

		this.rule.dtStartDate = value;

		if (this.rule.freq === FREQ.monthly && this.rule.willSkipMonths) {
			// make assumption that if user sets monthly to last day of month for the start date, then they really mean the last day of the month
			this.rule.byMonthDay = '-1';
		}

		this.args.onChange(this.rule.string);
	}
	<template>
		<SelectDate
			@disabled={{@disabled}}
			@size={{@size}}
			@form={{@form}}
			@id={{this.id}}
			@variant={{@variant}}
			@serverErrors={{@serverErrors}}
			@value={{this.formattedSelection}}
			@onChange={{this.onChange}}
			@requirements={{@requirements}}
			@disableDate={{@disableDate}}
			@ariaDescribedBy={{@ariaDescribedBy}}
			data-qid-recurrence-start-date
			data-qid-component-type="Luna::SelectRecurrenceStartDate"
			...attributes
			{{didUpdate this.onRuleUpdate @value}}
		/>
	</template>
}

import { useCallback, useEffect, useMemo, useState } from 'react';

import { FREQ, type IntlLike, RRule } from '@soxhub/luna-utils/rrule';

import { SelectDate, type SelectDateProps } from '../select-date/select-date';

export interface SelectRecurrenceStartDateProps extends Omit<SelectDateProps, 'value' | 'onChange' | 'selectionType'> {
	/** Optional ref for the root element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** The full recurrence rule string containing DTSTART */
	value: string | null;
	/** Called when the start date changes, returns the full updated rule string */
	onChange: (value: string) => void;
	/** Optional custom intl implementation */
	intl?: IntlLike;
}

/**
 * Allows the user to add a start date to an iCalendar recurrence rule.
 * Wraps SelectDate and handles RRule parsing/formatting.
 */
export function SelectRecurrenceStartDate(props: SelectRecurrenceStartDateProps) {
	const { ref, value, onChange, intl, ...selectDateProps } = props;
	// Parse the RRule when value changes
	const [rule, setRule] = useState(() => new RRule(value, intl));

	// Update rule when value prop changes
	useEffect(() => {
		setRule(new RRule(value, intl));
	}, [value, intl]);

	// Get the formatted date from the rule
	const formattedSelection = useMemo(() => {
		if (!value) {
			return null;
		}
		return rule.dtStartDate;
	}, [value, rule]);

	// Handle date selection
	const handleChange = useCallback(
		(dateValue: string | { start: string; end: string } | null) => {
			const newRule = new RRule(value, intl);

			// Only handle single date values
			if (typeof dateValue === 'object' && dateValue !== null) {
				return;
			}

			if (!dateValue) {
				newRule.resetRecurrenceStartDate();
				onChange(newRule.string);
				return;
			}

			newRule.setDtStartDate(dateValue);

			// If monthly frequency and the day is at end of month, assume user wants last day
			if (newRule.freq === FREQ.monthly && newRule.willSkipMonths) {
				newRule.byMonthDay = '-1';
			}

			onChange(newRule.string);
		},
		[value, onChange, intl],
	);

	return (
		<div ref={ref} data-qid-recurrence-start-date data-qid-component-type="Luna::SelectRecurrenceStartDate">
			<SelectDate value={formattedSelection} onChange={handleChange} selectionType="single" {...selectDateProps} />
		</div>
	);
}

export default SelectRecurrenceStartDate;
