import { cn } from '../../utils/cn';

import styles from './icon.module.css';

export type IconSize = 'match-font' | 'xs' | 'sm' | 'md' | 'lg' | 'xl' | '2xl' | '3xl' | '4xl';

export type LunaColor = string; // TODO: Import from types when available

interface IconProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<SVGSVGElement>;
	/** Icon name (without the "luna-" prefix) */
	icon?: string;
	/** Size of the icon */
	size?: IconSize;
	/** Additional CSS class */
	className?: string;
	/** ARIA role */
	role?: string;
	/** Title for accessibility */
	title?: string;
	/** Description for accessibility */
	desc?: string;
	/** Color of the icon */
	color?: LunaColor;
}

const getSizeValue = (size?: IconSize): string => {
	switch (size) {
		case 'match-font':
			return '0.9em';
		case 'xs':
			return '10px';
		case 'sm':
			return '12px';
		case 'md':
			return '16px';
		case 'lg':
			return '20px';
		case 'xl':
			return '24px';
		case '2xl':
			return '28px';
		case '3xl':
			return '32px';
		case '4xl':
			return '40px';
		default:
			return '16px';
	}
};

/**
 * Icons are graphical symbols that represent actions or information, offering visual context and enhancing usability.
 *
 * Matches the Luna::Icon Ember component API and styling.
 */
export function Icon(props: IconProps) {
	const { ref, icon, size, className, role, title, desc, color } = props;

	const width = getSizeValue(size);
	const height = getSizeValue(size);
	const iconId = icon ? `#luna-${icon}` : '';
	const ariaRole = role ?? (title ? undefined : 'presentation');

	const iconClassName = cn(styles.Icon, className);

	return (
		<svg
			ref={ref}
			className={iconClassName}
			width={width}
			height={height}
			role={ariaRole}
			data-qid-component-type="Luna::Icon"
			data-qid-icon-name={icon}
			data-color={color}
		>
			{title && <title>{title}</title>}
			{desc && <desc>{desc}</desc>}
			{iconId && <use href={iconId} xlinkHref={iconId} />}
		</svg>
	);
}

export default Icon;
