import { useCallback } from 'react';

import { Tooltip as BaseTooltip } from '@base-ui-components/react/tooltip';

import { cn } from '../../utils/cn';

import styles from './tooltip.module.css';

export type TooltipPlacement =
	| 'top'
	| 'top-start'
	| 'top-end'
	| 'bottom'
	| 'bottom-start'
	| 'bottom-end'
	| 'left'
	| 'left-start'
	| 'left-end'
	| 'right'
	| 'right-start'
	| 'right-end';

export interface TooltipProps {
	/** Optional ref for the trigger element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLSpanElement>;
	/** The content to display in the tooltip */
	content?: React.ReactNode;
	/** The trigger element(s) for the tooltip */
	children: React.ReactNode;
	/** Where to place the tooltip relative to the trigger */
	placement?: TooltipPlacement;
	/** Whether the tooltip is disabled */
	disabled?: boolean;
	/** Callback when the tooltip is shown */
	onShow?: () => void;
	/** Callback when the tooltip is hidden */
	onHide?: () => void;
	/** How long to wait before showing the tooltip (ms). Default is 0ms. */
	delay?: number;
	/** How long to wait before hiding the tooltip (ms) */
	closeDelay?: number;
	/** Additional class name for the popup */
	className?: string;
	/** Test ID for the tooltip popup */
	testId?: string;
}

/**
 * Tooltip component that shows informative text when an element is hovered or focused.
 *
 * Uses Base UI's Tooltip primitives under the hood for accessibility and positioning.
 */
export function Tooltip(props: TooltipProps) {
	const {
		ref,
		content,
		children,
		placement = 'top',
		disabled = false,
		onShow,
		onHide,
		delay = 0,
		closeDelay = 0,
		className,
		testId,
	} = props;
	const handleOpenChange = useCallback(
		(open: boolean) => {
			if (open) {
				onShow?.();
			} else {
				onHide?.();
			}
		},
		[onShow, onHide],
	);

	// If no content or disabled, render children without tooltip
	if (!content || disabled) {
		return <>{children}</>;
	}

	return (
		<BaseTooltip.Provider delay={delay} closeDelay={closeDelay}>
			<BaseTooltip.Root onOpenChange={handleOpenChange}>
				<BaseTooltip.Trigger ref={ref} className={styles.trigger} render={<span />}>
					{children}
				</BaseTooltip.Trigger>
				<BaseTooltip.Portal>
					<BaseTooltip.Positioner side={placement.split('-')[0] as 'top' | 'bottom' | 'left' | 'right'} sideOffset={8}>
						<BaseTooltip.Popup
							className={cn(styles.Popup, className)}
							data-test-tooltip
							data-testid={testId}
							data-qid-component-type="Luna::Tooltip"
						>
							<BaseTooltip.Arrow className={styles.Arrow}>
								<ArrowSvg />
							</BaseTooltip.Arrow>
							<div className={styles.content}>{content}</div>
						</BaseTooltip.Popup>
					</BaseTooltip.Positioner>
				</BaseTooltip.Portal>
			</BaseTooltip.Root>
		</BaseTooltip.Provider>
	);
}

/** Arrow SVG component matching Luna's dark popup arrow style */
function ArrowSvg() {
	return (
		<svg width="20" height="10" viewBox="0 0 20 10" className={styles.arrowSvg}>
			<path d="M0 10L10 0L20 10H0Z" className={styles.arrowFill} />
			<path d="M0 10L10 0L20 10" fill="none" className={styles.arrowStroke} />
		</svg>
	);
}

export default Tooltip;
