export interface MenuWrapperSignature {
	Element: HTMLDivElement;
	Args: {
		popupId?: PopupArgs['id'];
		relativeTo: PopupArgs['relativeTo'];
		isExpanded?: PopupArgs['show'];
		hideMenu?: PopupArgs['onClose'];
	};
	Blocks: {
		default: [
			{
				Option: LunaMenuSignature['Blocks']['default'][0]['Option'];
			},
		];
	};
}

class MenuWrapper extends Component<MenuWrapperSignature> {
	close = (popup: LunaPopupBlockValue) => {
		void popup.closeTask.perform();
	};

	<template>
		<Popup
			@id={{@popupId}}
			@show={{@isExpanded}}
			@trapFocus={{true}}
			@onClose={{@hideMenu}}
			@relativeTo={{@relativeTo}}
			@placement="bottom-start"
			@ySpacing={{8}}
			...attributes
			as |popup|
		>
			<Menu {{autofocus "[role=menuitem]"}} as |m|>
				{{yield (hash Option=(component m.Option onClick=(fn this.close popup)))}}
			</Menu>
		</Popup>
	</template>
}

interface TextWithMenuSignature {
	Blocks: {
		default: [
			{
				Text: WithBoundArgs<typeof TextWrapper, 'id' | 'popupId' | 'isExpanded' | 'showMenu'>;
				Popup: WithBoundArgs<typeof MenuWrapper, 'popupId' | 'isExpanded' | 'relativeTo' | 'hideMenu'>;
			},
		];
	};
}

let counter = 0;

/**
 * Text used to show a menu of options to the user. The menu closes once the user selects an option.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/text-with-menu.md)
 */
export default class LunaTextWithMenu extends Component<TextWithMenuSignature> {
	targetId = `luna${counter++}--target`;
	popupId = `luna${counter++}--popup`;

	@tracked renderMenu = false;

	showMenu = () => {
		this.renderMenu = true;
	};

	hideMenu = () => {
		this.renderMenu = false;
	};

	<template>
		{{yield
			(hash
				Text=(component
					TextWrapper id=this.targetId popupId=this.popupId isExpanded=this.renderMenu showMenu=this.showMenu
				)
				Popup=(component
					MenuWrapper
					popupId=this.popupId
					relativeTo=(concat "#" this.targetId)
					isExpanded=this.renderMenu
					hideMenu=this.hideMenu
				)
			)
		}}
	</template>
}

import { Icon } from '../icon';

import styles from './text-with-menu.module.css';

/** Symbol used to identify Text slot when extracting children (avoids string keys like $$). */
export const TWM_TEXT = Symbol('TextWithMenu.Text');
/** Symbol used to identify Menu slot when extracting children. */
export const TWM_MENU = Symbol('TextWithMenu.Menu');

/**
 * Context for sharing menu state between TextWithMenu and its subcomponents
 */
interface TextWithMenuContextValue {
	/** Whether the menu is currently open */
	isOpen: boolean;
	/** Open the menu */
	showMenu: (event?: React.MouseEvent) => void;
	/** Close the menu */
	hideMenu: () => void;
	/** ID of the trigger text */
	textId: string;
	/** ID of the popup menu */
	popupId: string;
	/** Ref to the trigger text element */
	textRef: RefObject<HTMLButtonElement | null>;
}

const TextWithMenuContext = createContext<TextWithMenuContextValue | null>(null);

function useTextWithMenuContext() {
	const context = useContext(TextWithMenuContext);
	if (!context) {
		throw new Error('TextWithMenu subcomponents must be used within TextWithMenu');
	}
	return context;
}

/**
 * Hook to access the TextWithMenu control functions and state from within menu content or text children
 */
export function useTextWithMenuHandle(): {
	showMenu: (event?: React.MouseEvent) => void;
	hideMenu: () => void;
	popupIsOpen: boolean;
} {
	const context = useTextWithMenuContext();
	return { showMenu: context.showMenu, hideMenu: context.hideMenu, popupIsOpen: context.isOpen };
}

interface TextTriggerBaseProps {
	/** ID for the element */
	'id'?: string;
	/** Children content */
	'children'?: React.ReactNode;

	'className'?: string;
	/** ARIA label for accessibility */
	'aria-label'?: string;

export type TextTriggerProps = ComponentPropsWithRef<TextTriggerBaseProps, HTMLButtonElement>;

/**
 * The trigger text for TextWithMenu.
 * Renders as a button styled to look like text with an underline and chevron icon.
 */
export function TextTrigger(props: TextTriggerProps) {
	const { ref, children, className, ...rest } = props;
	const { isOpen, showMenu, textId, popupId, textRef } = useTextWithMenuContext();

	const handleClick = useCallback(
		(event: React.MouseEvent<HTMLButtonElement>) => {
			event.stopPropagation();
			showMenu(event);
		},
		[showMenu],
	);

	// Merge refs
	const setRef = useCallback(
		(el: HTMLButtonElement | null) => {
			(textRef as MutableRefObject<HTMLButtonElement | null>).current = el;
			if (typeof ref === 'function') {
				ref(el);
			} else if (ref) {
				(ref as MutableRefObject<HTMLButtonElement | null>).current = el;
			}
		},
		[textRef, ref],
	);

	return (
		<button
			ref={setRef}
			id={rest.id ?? textId}
			type="button"
			className={cn(styles.trigger, className)}
			aria-expanded={isOpen}
			aria-controls={popupId}
			data-qid-component-type="Luna::TextWithMenu"
			onClick={handleClick}
			{...rest}
		>
			<div className={styles.triggerText}>{children}</div>
			<Icon icon="chevron-down" />
		</button>
	);
}

TextTrigger.displayName = 'TextWithMenu.Text';

/**
 * An option within the TextWithMenu popup.
 * Automatically closes the menu when clicked.
 */
export function TextMenuOption(props: TextMenuOptionProps) {
	const { ref, onClick, children, ...rest } = props;
	const popupContext = usePopupContext();

	const handleClick = useCallback(() => {
		onClick?.();
		// Close the popup after the option action
		popupContext?.close('manual');
	}, [onClick, popupContext]);

	return (
		<Option ref={ref} onClick={handleClick} {...rest}>
			{children}
		</Option>
	);
}

// Slot Extraction

/**
 * Extract text and menu slots from children
 */
function extractTextWithMenuSlots(children: ReactNode): {
	textSlot: ReactElement | null;
	menuSlot: ReactElement | null;
	otherChildren: ReactNode[];
} {
	let textSlot: ReactElement | null = null;
	let menuSlot: ReactElement | null = null;
	const otherChildren: ReactNode[] = [];

	Children.forEach(children, (child) => {
		if (!child || typeof child !== 'object' || !('type' in child)) {
			otherChildren.push(child);
			return;
		}

		const element = child as ReactElement & {
			type: Record<symbol, boolean>;
		};

		if (element.type?.[TWM_TEXT]) {
			textSlot = element;
		} else if (element.type?.[TWM_MENU]) {
			menuSlot = element;
		} else {
			otherChildren.push(child);
		}
	});

	return { textSlot, menuSlot, otherChildren };
}

export interface TextWithMenuProps {
	/** Children components - use TextWithMenu.Text and TextWithMenu.Menu */
	children: ReactNode;
}

/**
 * Text used to show a menu of options to the user.
 * The menu closes once the user selects an option.
 *
 * @example
 * ```tsx
 * <TextWithMenu>
 *   <TextWithMenu.Text>Options</TextWithMenu.Text>
 *   <TextWithMenu.Menu>
 *     <TextWithMenu.Option onClick={() => console.log('Edit')}>Edit</TextWithMenu.Option>
 *     <TextWithMenu.Option onClick={() => console.log('Delete')}>Delete</TextWithMenu.Option>
 *   </TextWithMenu.Menu>
 * </TextWithMenu>
 * ```
 */
function TextWithMenuComponent({ children }: TextWithMenuProps) {
	const id = useId();
	const textId = `${id}--target`;
	const popupId = `${id}--popup`;
	const textRef = useRef<HTMLButtonElement | null>(null);

	const [isOpen, setIsOpen] = useState(false);

	const showMenu = useCallback((event?: React.MouseEvent) => {
		event?.stopPropagation();
		setIsOpen(true);
	}, []);

	const hideMenu = useCallback(() => {
		setIsOpen(false);
	}, []);

	const contextValue: TextWithMenuContextValue = {
		isOpen,
		showMenu,
		hideMenu,
		textId,
		popupId,
		textRef,
	};

	const { textSlot, menuSlot } = extractTextWithMenuSlots(children);

	return (
		<TextWithMenuContext.Provider value={contextValue}>
			{textSlot}
			{menuSlot}
		</TextWithMenuContext.Provider>
	);
}

// Attach subcomponents
const TextWithMenu = Object.assign(TextWithMenuComponent, {
	Text: TextTrigger,
	Menu: TextMenuPopup,
	Option: TextMenuOption,
	Heading: OptionHeading,
});

export { TextWithMenu };

export default TextWithMenu;
