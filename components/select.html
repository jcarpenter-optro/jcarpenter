// Don't leak this to engines. It makes the types way too complicated. Instead, use SimplifiedValidSelection and just use array checks to differentiate between Single and Multi Selects
type ValidSelection<
	V extends UnknownSelectValue,
	Required extends boolean,
	SelectionType extends 'single' | 'multiple',
> = SelectionType extends 'single' ? (Required extends true ? V : V | null) : V[];

// Don't leak this to engines. It makes the types way too complicated.
export type ValidInitialSelection<
	T extends UnknownSelectValue,
	SelectionType extends 'single' | 'multiple',
> = SelectionType extends 'single'
	? T | string | null | undefined
	: SupportedArrayLike<T> | SupportedArrayLike<string> | null | undefined;

interface SelectRequirements<Required extends boolean> extends Requirements {
	required?: Required | [requirement: Required, message: string];
}

interface LunaSelectSignature<
	V extends UnknownSelectValue,
	LocalOption extends UnknownSelectValue,
	QueryOption extends UnknownSelectValue,
	Required extends boolean,
	SelectionType extends 'single' | 'multiple',
	I extends ValidInitialSelection<NoInfer<V> | NoInfer<LocalOption>, NoInfer<SelectionType>> = ValidInitialSelection<
		NoInfer<V> | NoInfer<LocalOption>,
		NoInfer<SelectionType>
	>,
	IsFake extends boolean = false,
> {
	Element: HTMLDivElement;
	Args: LunaSelectArgs<V, LocalOption, QueryOption, Required, SelectionType, I, IsFake>;
	Blocks: {
		icon?: [];
		// This value will never be falsy because we only render the trigger block when the placeholder is not shown
		trigger?: [value: NoInfer<ValidSelection<V | LocalOption, true, SelectionType>>];
		option: [option: NoInfer<V | QueryOption | LocalOption>];
		/** TODO: Should not have both option and default blocks. */
		// NOTE: This type will _not_ handle the case where the value is a string but the options are non-strings
		// This value will never be falsy because we only render the trigger block when the placeholder is not shown
		default: [displayValue: NoInfer<V> | NoInfer<LocalOption> | NoInfer<QueryOption>];
	};
}

/**
 * A component for selecting options, supporting both single and multi-select modes.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/select.md)
 */
export default class Select<
	V extends UnknownSelectValue,
	LocalOption extends UnknownSelectValue = V,
	QueryOption extends UnknownSelectValue = V,
	Required extends boolean = boolean,
	SelectionType extends 'single' | 'multiple' = 'single',
	I extends ValidInitialSelection<NoInfer<V> | NoInfer<LocalOption>, NoInfer<SelectionType>> = ValidInitialSelection<
		NoInfer<V> | NoInfer<LocalOption>,
		NoInfer<SelectionType>
	>,
	IsFake extends boolean = false,
> extends Component<LunaSelectSignature<V, LocalOption, QueryOption, Required, SelectionType, I, IsFake>> {
	@service declare intl: IntlService;
	@service('luna/async') declare private async: AsyncService;
	@service('store') declare private v1Store: V1Store;
	@service('v2-store') declare private v2Store: V2Store;

	@consume('store') declare private _contextStore: V1Store | V2Store | undefined;
	@consume('luna-input-id') providedInputId!: string;
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;

	@cached
	get contextStore() {
		const contextStore = this._contextStore;
		if (!contextStore) {
			warn('No store found in context for Select component. Falling back to v1 store.', {
				id: 'luna.select.no-context-store',
			});
		}
		return contextStore ?? this.v1Store;
	}

	manager: InputManager;

	constructor(owner: Owner, args: LunaSelectArgs<V, LocalOption, QueryOption, Required, SelectionType, I>) {
		super(owner, args);
		assert('You cannot provide both ariaLabel and ariaLabelledBy', !args.ariaLabel || !args.ariaLabelledBy);
		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;
		const managerOptions: InputManagerOptions = {
			mode: 'change',
			name: args.name,
			didAutosave: this.didAutosave.bind(this),

			focus: this.focusInput.bind(this),

			ariaDescribedBy: this.args.ariaDescribedBy ?? null,

			get value(): unknown {
				return args.value;
			},

			get isDisabled(): boolean {
				return args.disabled ?? false;
			},

			get requirements(): SelectRequirements<Required> | undefined {
				return args.requirements;
			},

			get serverErrors(): Array<{ message?: string } | string> | undefined {
				// eslint-disable-next-line @typescript-eslint/no-deprecated
				return args.serverErrors;
			},

			get formContext(): LunaFormBlockValue | undefined {
				return that.formContext;
			},

			get selectorIsVisible(): boolean {
				return that.showPopup;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	// SAFETY: This cast is necessary because of "'SelectionType' could be instantiated
	// with a different subtype" error (which is technically true, but to avoid it
	// we'd have to re-architect these components to not rely on SelectMultiple
	// inheriting from Select).
	selectionType: SelectionType = 'single' as SelectionType;

	_id = SelectID++;
	popupId = `select${this._id}--popup`;
	inputId = `select${this._id}--input`;

	@tracked showPopup = false;
	@tracked placement = this.args.placement ?? 'bottom';

	@tracked placementHasBeenUpdated = false;

	// we cache engine and engineArgs onto these properties
	// for stability
	declare _engine: SelectEngine<V, LocalOption, QueryOption> | undefined;
	declare _engineArgs:
		| {
				options?: Options<V>;
				queryFn?: SelectQueryFn<QueryOption> | null;
				useV1Store?: boolean;
				searchKeys?: string[];
				debounceDelay?: number;

				getUpstreamValue: () => SimplifiedValidSelection<V | LocalOption | string>;
				onRecommendationsLoaded?: (recommendations: ReadonlyArray<Recommendation<QueryOption>>) => void;

		| undefined;

	@cached
	get engineArgs() {
		const engineArgs = {
			selectionType: this.selectionType,
			options: (this.args.options ??
				// eslint-disable-next-line @typescript-eslint/no-deprecated -- TODO: API Contracts to remove once migration is complete
				this.args.dangerouslyUnpaginatedOptions) as Options<V> | undefined,
			hasDangerouslyUnpaginatedOptions: Boolean(
				// eslint-disable-next-line @typescript-eslint/no-deprecated -- TODO:  API Contracts to remove once migration is complete
				this.args.dangerouslyUnpaginatedOptions,
			),
			queryFn: this.args.queryFn,
			useV1Store: this.args.useV1Store,
			searchKeys: this.args.searchKeys,
			debounceDelay: this.args.debounceDelay,
			isAiSuggestion: this.args.isAiSuggestion,

			// NOTE: It's Currently important that we wrap `this.args.value` in a
			// function so that this getter doesn't consume it, which would reset
			// the Engine's caches every time the value is updated.
			getUpstreamValue: (): SimplifiedValidSelection<V | LocalOption | string> => {
				const { value } = this.args;

				if (this.isMultiSelect) {
					if (value === undefined || value === null) {
						return [];
					}
					assert(
						`BUG! Expected value in SelectMultiple to be a supported array-like type (or null-ish), but got ${typeof value}.`,
						isArray(value),
					);
					return makeNativeArray(value, false).slice() as Array<V | LocalOption | string>;
				}

				if (value === undefined || value === null) {
					return null;
				}

				return value as V | LocalOption | string;
			},
			onRecommendationsLoaded: this.args.onRecommendationsLoaded,
		};
		// eslint-disable-next-line ember/no-side-effects
		this._engineArgs = engineArgs;
		return engineArgs;
	}

	@cached
	get engine(): SelectEngine<V, LocalOption, QueryOption> {
		// we access _engineArgs before engineArgs to ensure
		// we have the prior computation state to compare against.
		const args = this._engineArgs;
		const engineArgs = this.engineArgs;

		// if we have an engine and args has not recomputed then
		// we can reuse the existing engine.
		if (this._engine && args === engineArgs) {
			return this._engine;
		}

		if (engineArgs.queryFn) {
			assert('Cannot set both `queryFn` and `options` as args', !engineArgs.options);

			if (this.contextStore === this.v2Store && engineArgs.useV1Store) {
				throw new Error('Cannot use v1 store when context store is v2');
			}

			// eslint-disable-next-line ember/no-side-effects
			return (this._engine = new QuerySelectEngine<V, LocalOption, QueryOption>(
				this.v1Store,
				this.v2Store,
				this.contextStore,
				{
					queryFn: engineArgs.queryFn,
					valueCompatFn: this.args.valueCompatFn ?? null,
					selectionType: engineArgs.selectionType,
					useV1Store: engineArgs.useV1Store ?? false,
					debounceDelay: engineArgs.debounceDelay,
					getUpstreamValue: engineArgs.getUpstreamValue,
					localOptions: this.args.localOptions ?? [],
					onRecommendationsLoaded: engineArgs.onRecommendationsLoaded,
				},
			));
		}

		// SAFETY: In the remaining engines, the types of LocalOption and
		// QueryOption are always the same as V so these casts are safe
		const engineOptions = {
			fuzzySearchKeys: engineArgs.searchKeys,
			getUpstreamValue: engineArgs.getUpstreamValue as () => SimplifiedValidSelection<V>,
			selectionType: engineArgs.selectionType,
			debounceDelay: engineArgs.debounceDelay,
			isAiSuggestion: engineArgs.isAiSuggestion as (option: V | { id: string }) => boolean,
			hasDangerouslyUnpaginatedOptions: engineArgs.hasDangerouslyUnpaginatedOptions,
		};
		// TODO: We need to ensure that regardless of the engine type (aside from query engine) search is handled
		if (typeof engineArgs.options === 'function') {
			// eslint-disable-next-line ember/no-side-effects
			return (this._engine = new DeferredSelectEngine<V>(this.async, engineArgs.options, engineOptions) as SelectEngine<
				V,
				LocalOption,
				QueryOption
			>);
		} else if (engineArgs.options && 'then' in engineArgs.options) {
			// eslint-disable-next-line ember/no-side-effects
			return (this._engine = new AsyncProxySelectEngine<V>(engineArgs.options, engineOptions) as SelectEngine<
				V,
				LocalOption,
				QueryOption
			>);
		}

		// eslint-disable-next-line ember/no-side-effects
		return (this._engine = new SyncSelectEngine<V>(engineArgs.options ?? [], engineOptions) as SelectEngine<
			V,
			LocalOption,
			QueryOption
		>);
	}

		if (value && (this.isSingleSelect || (this.isMultiSelect && isArray(value) && value.length > 0))) {
			return false;
		}

		return true;
	}

	/**

	 * to show the "Clear" button and what to say in the primary button.

	@tracked hasActiveFilters = this.engine.originalOptions !== this.engine.filteredOptions;

	get clearButtonIsShown() {
		return this.engine.selection instanceof Array && this.engine.selection.length > 0;
	}

	popupWidth(width: 'auto' | keyof typeof POPUP_WIDTHS, id: string) {
		if (width === 'auto') {
			const element = document.getElementById(id);
			assert(`Element not found for id: ${id}`, element);

			return htmlSafe(`width: ${clientWidth}px;`);
		}
		return htmlSafe(`width: ${POPUP_WIDTHS[width]}px`);
	}

	get isSingleSelect() {
		return this.selectionType === 'single';
	}

	get isMultiSelect(): boolean {
		return this.selectionType === 'multiple';

		if (this.shouldRenderAsDiv) {
			return;

type AnchorClickSubscriber = {
	inputId: string;
	lastClickWasAnchorRef: MutableRefObject<boolean>;
	resetTimerId: ReturnType<typeof setTimeout> | null;
};

const anchorClickSubscribers = new Map<string, AnchorClickSubscriber>();
let isAnchorClickListenerRegistered = false;

function setLastClickWasAnchor(subscriber: AnchorClickSubscriber, value: boolean): void {
	subscriber.lastClickWasAnchorRef.current = value;

	if (subscriber.resetTimerId) {
		clearTimeout(subscriber.resetTimerId);
		subscriber.resetTimerId = null;
	}

	if (value) {
		subscriber.resetTimerId = setTimeout(() => {
			subscriber.lastClickWasAnchorRef.current = false;
			subscriber.resetTimerId = null;
		}, 100);
	}
}

function handleDocumentCaptureClick(e: MouseEvent): void {
	if (anchorClickSubscribers.size === 0) return;
	if (!(e.target instanceof HTMLElement)) return;

	let triggerElement: HTMLElement | null = e.target;
	while (triggerElement && triggerElement !== document.body) {
		if (triggerElement.id && anchorClickSubscribers.has(triggerElement.id)) {
			break;
		}
		triggerElement = triggerElement.parentElement;
	}

	if (!triggerElement || triggerElement === document.body) return;
	const subscriber = anchorClickSubscribers.get(triggerElement.id);
	if (!subscriber) return;

	let element: HTMLElement | null = e.target;
	while (element && element !== triggerElement) {
		if (element.tagName === 'A') {
			setLastClickWasAnchor(subscriber, true);
			e.stopPropagation();
			e.stopImmediatePropagation();
			return;
		}
		element = element.parentElement;
	}

	// Reset flag if click was not on an anchor
	setLastClickWasAnchor(subscriber, false);
}

function registerAnchorClickListenerIfNeeded(): void {
	if (isAnchorClickListenerRegistered) return;
	if (typeof document === 'undefined') return;

	document.addEventListener('click', handleDocumentCaptureClick, true);
	isAnchorClickListenerRegistered = true;
}

function unregisterAnchorClickListenerIfNeeded(): void {
	if (!isAnchorClickListenerRegistered) return;
	if (typeof document === 'undefined') return;
	if (anchorClickSubscribers.size > 0) return;

	document.removeEventListener('click', handleDocumentCaptureClick, true);
	isAnchorClickListenerRegistered = false;
}

function subscribeToAnchorClickInterceptor(
	inputId: string,
	lastClickWasAnchorRef: MutableRefObject<boolean>,
): () => void {
	if (typeof document === 'undefined') return () => {};

	const subscriber: AnchorClickSubscriber = {
		inputId,
		lastClickWasAnchorRef,
		resetTimerId: null,
	};

	anchorClickSubscribers.set(inputId, subscriber);
	registerAnchorClickListenerIfNeeded();

	return () => {
		const current = anchorClickSubscribers.get(inputId);
		if (current?.resetTimerId) {
			clearTimeout(current.resetTimerId);
		}
		anchorClickSubscribers.delete(inputId);
		unregisterAnchorClickListenerIfNeeded();
	};
}

/** Type guard to check if a value is an object with an 'id' property */
function hasIdProperty(value: unknown): value is { id: unknown } {
	return typeof value === 'object' && value !== null && 'id' in value;
}

/** Check if two values match by comparing their 'id' properties if they have them, otherwise by reference */
function valuesMatchById(a: unknown, b: unknown): boolean {
	if (hasIdProperty(a) && hasIdProperty(b)) {
		return a.id === b.id;
	}
	return a === b;
}

function arraysEqual<T>(a: T[] | null | undefined, b: T[] | null | undefined): boolean {
	if (a === b) return true;
	if (!a || !b) return false;
	if (a.length !== b.length) return false;
	if (a.length > 0 && hasIdProperty(a[0])) {
		const aIds = new Set(a.map((item) => (item as { id: string }).id));
		const bIds = new Set(b.map((item) => (item as { id: string }).id));
		return aIds.size === bIds.size && [...aIds].every((id) => bIds.has(id));
	}
	const aSet = new Set(a);
	const bSet = new Set(b);
	return aSet.size === bSet.size && [...aSet].every((val) => bSet.has(val));
}

function isValueSelected<T>(selection: T[], value: T): boolean {
	return selection.some((val) => valuesMatchById(val, value));

function normalizeOptions<T>(
	options?:
		| readonly T[]
		| readonly SelectOption<T>[]
		| (() => readonly T[] | readonly SelectOption<T>[])
		| (() => Promise<readonly T[] | readonly SelectOption<T>[]>)
		| Promise<readonly T[] | readonly SelectOption<T>[]>,
): SelectOption<T>[] {
	if (!options) {
		return [];
	}
	// Handle Promise
	if (options instanceof Promise) {
		return [];
	}
	// Handle function
	if (typeof options === 'function') {
		try {
			const result = options();
			if (result instanceof Promise) {
				return [];
			}
			return normalizeOptions(result);
		} catch {
			return [];
		}
	}
	// Handle array
	return options.map((option) => {
		if (typeof option === 'object' && option !== null && 'value' in option) {
			return option as SelectOption<T>;
		}
		return { value: option, label: String(option) };
	});
}

function findOptionByValue<T>(options: SelectOption<T>[], value: T | null | undefined): SelectOption<T> | null {
	if (value === null || value === undefined) {
		return null;
	}
	return options.find((opt) => valuesMatchById(opt.value, value)) ?? null;
}

export function Select(props: SelectProps) {
	const {
		ref,
		selectionType = 'single',
		value,
		options: rawOptions = [],
		onChange,
		placeholder,
		size = 'md',
		variant = 'white',
		align = 'left',
		disabled = false,
		active: controlledActive,
		id: providedId,
		ariaLabel,
		ariaLabelledBy,
		ariaDescribedBy: providedAriaDescribedBy,
		chevronIsHidden = false,
		className,
		requirements,
		name,
		didAutosave,
		searchDisabled = false,
		emptyOptionText,
		hideEmptyOption,
		onOpen,
		onClose,
		didClose,
		onCancel,
		didCancel,
		children,
		renderTrigger,
		icon,
		triggerComponent: TriggerComponent,
		placement = 'bottom',
		popupWidth = 'auto',
		searchInputPlaceholder,
		debounceDelay = 300,
		searchKeys,
		onCreateNew,
		createNewLabel,
		isLoading: externalIsLoading = false,
		triggerClass,
		isAiSuggestion,
		onSearch: onSearchCallback,
		selectAllLabel = selectionType === 'multiple' ? t('select-multiple.selectAll') : undefined,
		selectedCountLabel,
		allowSelectAll = false,
		ariaLabelKeyboardNav = selectionType === 'multiple' ? t('select-multiple.ariaKeyboardNav') : undefined,
		ariaLabelSearch = selectionType === 'multiple' ? t('select-multiple.ariaSearch') : undefined,
		...restProps
	} = props;

	const isMultiple = selectionType === 'multiple';
	const valueArray = useMemo(
		() => (isMultiple && value != null && Array.isArray(value) ? (value as unknown[]) : []),
		[isMultiple, value],
	);
	const singleValue = !isMultiple ? (value as unknown) : undefined;

	const selectIdRef = useRef(`select${selectIdCounter++}`);
	const popupId = `${selectIdRef.current}--popup`;
	const fieldContext = useFieldContext();
	const inputId = providedId ?? fieldContext?.inputId ?? `${selectIdRef.current}--input`;

	const [open, setOpen] = useState(false);
	const [closeKey, setCloseKey] = useState(0);
	const [optionsLoading, setOptionsLoading] = useState(false);
	const [internalSelection, setInternalSelection] = useState<unknown[]>([]);
	const internalSelectionRef = useRef<unknown[]>([]);
	internalSelectionRef.current = internalSelection;
	// Initialize resolvedOptions synchronously for arrays to avoid timing issues
	const initialResolvedOptions = useMemo(() => {
		if (!rawOptions) {
			return [];
		}
		if (rawOptions instanceof Promise || typeof rawOptions === 'function') {
			return [];
		}
		return rawOptions;
	}, [rawOptions]);
	const [resolvedOptions, setResolvedOptions] = useState<readonly unknown[] | readonly SelectOption<unknown>[]>(
		initialResolvedOptions as readonly unknown[] | readonly SelectOption<unknown>[],
	);
	const valueIsUnchangedRef = useRef(true);
	const searchInputId = useId();
	const triggerRef = useRef<HTMLButtonElement | HTMLAnchorElement | null>(null);
	const rootRef = useRef<HTMLDivElement | null>(null);
	const lastClickWasAnchorRef = useRef(false);
	const comboboxActionsRef = useRef<{ open: () => void; close: () => void } | null>(null);
	const escKeyPressedRef = useRef(false);
	const searchDebounceTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
	const isSelectingRef = useRef(false); // Guard to prevent reopening during selection

	const formContext = useFormContext();

	// Handle async/function-based options
	useEffect(() => {
		let cancelled = false;

		async function resolveOptions() {
			if (!rawOptions) {
				setResolvedOptions([]);
				return;
			}

			// Handle Promise
			if (rawOptions instanceof Promise) {
				setOptionsLoading(true);
				try {
					const result = await rawOptions;
					if (!cancelled) {
						setResolvedOptions(result);
						setOptionsLoading(false);
					}
				} catch (error) {
					if (!cancelled) {
						console.error('Error loading options:', error);
						setResolvedOptions([]);
						setOptionsLoading(false);
					}
				}
				return;
			}

			// Handle function
			if (typeof rawOptions === 'function') {
				setOptionsLoading(true);
				try {
					const result = rawOptions();
					if (result instanceof Promise) {
						const resolved = await result;
						if (!cancelled) {
							setResolvedOptions(resolved);
							setOptionsLoading(false);
						}
					} else {
						if (!cancelled) {
							setResolvedOptions(result);
							setOptionsLoading(false);
						}
					}
				} catch (error) {
					if (!cancelled) {
						console.error('Error loading options:', error);
						setResolvedOptions([]);
						setOptionsLoading(false);
					}
				}
				return;
			}

			// Handle array (synchronous) - set immediately
			if (!cancelled) {
				setResolvedOptions(rawOptions);
			}
		}

		resolveOptions();

		return () => {
			cancelled = true;
		};
	}, [rawOptions]);

	const normalizedOptions = useMemo(() => normalizeOptions(resolvedOptions), [resolvedOptions]);
	const selectedOption = useMemo(
		() => (!isMultiple ? findOptionByValue(normalizedOptions, singleValue) : null),
		[normalizedOptions, singleValue, isMultiple],
	);
	const isLoading = externalIsLoading || optionsLoading;

	const required = useMemo(() => {
		if (!requirements) {
			return false;
		}
		const req = requirements.required;
		if (typeof req === 'boolean') {
			return req;
		}
		if (Array.isArray(req)) {
			return req[0];
		}
		return false;
	}, [requirements]);

	// Calculate effective hideEmptyOption (matches Ember: hideEmptyOption ?? required ?? false)
	const effectiveHideEmptyOption = useMemo(() => {
		if (hideEmptyOption !== undefined) {
			return hideEmptyOption;
		}
		return required;
	}, [hideEmptyOption, required]);

	const showEmptyOption = useMemo(() => {
		if (emptyOptionText === undefined) {
			return false;
		}
		// Don't show if effectively hidden (matches Ember: !this.args.required where required=hideEmptyOption)
		if (effectiveHideEmptyOption) {
			return false;
		}
		// Show if we have options or not loading (matches Ember behavior)
		// Note: Combobox handles filtering internally, so we show empty option if we have any options
		return normalizedOptions.length > 0 || !isLoading;
	}, [emptyOptionText, effectiveHideEmptyOption, normalizedOptions.length, isLoading]);

	// Add empty option to items array if it should be shown
	// Pass option objects to Combobox - filter function receives the entire object
	const itemsWithEmptyOption = useMemo(() => {
		if (showEmptyOption && emptyOptionText) {
			return [{ value: null, label: emptyOptionText }, ...normalizedOptions];
		}
		return normalizedOptions;
	}, [normalizedOptions, showEmptyOption, emptyOptionText]);

	// Base UI Select uses itemToStringLabel for label lookup, we don't need items prop

	const manager = useInputManager({
		mode: 'change',
		name,
		valueIsUnchanged: valueIsUnchangedRef.current,
		selectorIsVisible: open,
		requirements,
		value: isMultiple ? (valueArray.length > 0 ? valueArray : null) : value,
		isDisabled: disabled,
		formContext,
		ariaDescribedBy: providedAriaDescribedBy ?? null,
		focus: () => {
			const element = document.getElementById(inputId);
			element?.focus();
		},
		didAutosave: () => {
			valueIsUnchangedRef.current = true;
			manager.setValueIsUnchanged(true);
			didAutosave?.();
		},
	});

	// Track if last click was on an anchor tag (for custom triggers)
	// This is checked in handleOpenChange to prevent popup from opening
	const checkIfAnchorClick = useCallback(
		(e: MouseEvent | React.MouseEvent): boolean => {
			const target = e.target as HTMLElement;
			let element: HTMLElement | null = target;

			// Walk up the DOM tree to find if we're inside an anchor tag
			while (element && element !== document.body) {
				if (element.tagName === 'A') {
					return true;
				}
				// Stop if we've reached the trigger element (don't check beyond it)
				if (element.id === inputId || element.getAttribute('data-test-custom-trigger')) {
					break;
				}
				element = element.parentElement;
			}
			return false;
		},
		[inputId],
	);

	const handleOpenChange = useCallback(
		(newOpen: boolean) => {
			// Prevent reopening during selection (Combobox might try to reopen after we've closed)
			if (newOpen && isSelectingRef.current) {
				return;
			}
			// Prevent opening if the last click was on an anchor tag (for custom triggers)
			if (newOpen && TriggerComponent && lastClickWasAnchorRef.current) {
				lastClickWasAnchorRef.current = false;
				return;
			}
			if (newOpen) {
				escKeyPressedRef.current = false;
				if (isMultiple) {
					setInternalSelection([...valueArray]);
					valueIsUnchangedRef.current = true;
				}
			} else {
				setCloseKey((k) => k + 1);
			}
			// Use flushSync to force synchronous state update for closing
			if (!newOpen) {
				flushSync(() => {
					setOpen(newOpen);
				});
			} else {
				setOpen(newOpen);
			}
			if (!newOpen) {
				if (isMultiple) {
					const currentSelection = internalSelectionRef.current;
					const hasChanged = !arraysEqual(currentSelection, valueArray);
					if (hasChanged) {
						onChange?.(currentSelection as unknown[], undefined);
						manager.setValueIsUnchanged(false);
						manager.onChange();
					} else {
						if (escKeyPressedRef.current) {
							onCancel?.('escape');
							escKeyPressedRef.current = false;
						} else {
							onCancel?.('manual');
						}
						didCancel?.();
					}
				} else {
					if (!valueIsUnchangedRef.current) {
						manager.focusOutAndTrySave(new Event('blur'));
					}
					if (valueIsUnchangedRef.current) {
						if (escKeyPressedRef.current) {
							onCancel?.('escape');
							escKeyPressedRef.current = false;
						} else {
							onCancel?.('manual');
						}
						didCancel?.();
					}
				}
				onClose?.();
				didClose?.();
			}
		},
		[manager, onClose, didClose, onCancel, didCancel, TriggerComponent, isMultiple, valueArray, onChange],
	);

	const handleSelectAll = useCallback(() => {
		if (!allowSelectAll) return;
		const allSelected = normalizedOptions.every((opt) => isValueSelected(internalSelection, opt.value));
		setInternalSelection(allSelected ? [] : normalizedOptions.map((opt) => opt.value));
		valueIsUnchangedRef.current = false;
		manager.setValueIsUnchanged(false);
	}, [allowSelectAll, normalizedOptions, internalSelection, manager]);

	const getMultipleTriggerLabel = useCallback((): React.ReactNode => {
		if (!valueArray.length) {
			return placeholder ?? '';
		}
		if (renderTrigger) {
			return (renderTrigger as (value: unknown[]) => React.ReactNode)(valueArray);
		}
		if (valueArray.length > 1) {
			if (typeof selectedCountLabel === 'function') {
				return selectedCountLabel(valueArray.length);
			}
			if (typeof selectedCountLabel === 'string') {
				return selectedCountLabel.replace('{count}', String(valueArray.length));
			}
			return t('select-multiple.selectedCount', { count: valueArray.length });
		}
		if (valueArray.length === 1) {
			const single = valueArray[0];
			const option = findOptionByValue(normalizedOptions, single);
			return children ? (children as (v: unknown) => React.ReactNode)(single) : (option?.label ?? String(single));
		}
		return placeholder ?? '';
	}, [valueArray, placeholder, renderTrigger, selectedCountLabel, children, normalizedOptions]);

	const handleMultipleValueChange = useCallback(
		(newValue: unknown) => {
			// Combobox passes array of selected option objects for multiple
			const options = Array.isArray(newValue) ? (newValue as SelectOption<unknown>[]) : [];
			const newSelection = options.map((opt) => opt.value);
			setInternalSelection(newSelection);
			valueIsUnchangedRef.current = false;
			manager.setValueIsUnchanged(false);
			onChange?.(newSelection as typeof valueArray, undefined);
			manager.onChange();
		},
		[manager, onChange],
	);

	const handleValueChange = useCallback(
		(newValue: unknown) => {
			// Mark that we're selecting to prevent Combobox from reopening
			isSelectingRef.current = true;

			// newValue is a SelectOption object when we pass option objects to items
			// Extract the actual value from the option object
			const option = newValue as SelectOption<unknown>;
			const actualValue = option.value;
			// Check if value actually changed
			const valueChanged = !Object.is(actualValue, value);
			if (!valueChanged) {
				// Value didn't change - mark as unchanged and close popup
				valueIsUnchangedRef.current = true;
				manager.setValueIsUnchanged(true);
				// Close popup via handleOpenChange to ensure callbacks run
				handleOpenChange(false);
				// Reset selecting flag after a short delay
				setTimeout(() => {
					isSelectingRef.current = false;
				}, 50);
				return;
			}
			valueIsUnchangedRef.current = false;
			manager.setValueIsUnchanged(false);
			onChange?.(actualValue as typeof value, undefined);
			manager.onChange();
			// Close popup after selection
			handleOpenChange(false);
			// Reset selecting flag after a short delay
			setTimeout(() => {
				isSelectingRef.current = false;
			}, 50);
		},
		[onChange, manager, value, handleOpenChange],
	);

	// Handle ESC key to cancel selection
	const handleEscapeKey = useCallback(() => {
		if (open) {
			valueIsUnchangedRef.current = true;
			escKeyPressedRef.current = true;
			handleOpenChange(false);
		}
	}, [open, handleOpenChange]);

	// Handle keyboard navigation on trigger
	const handleTriggerKeyDown = useCallback(
		(e: React.KeyboardEvent<HTMLButtonElement>) => {
			if (disabled) {
				return;
			}

			const { key } = e;

			// Arrow keys - open popup and navigate
			if (key === 'ArrowDown' || key === 'ArrowUp') {
				e.preventDefault();
				e.stopPropagation();

				if (!open) {
					onOpen?.();
					handleOpenChange(true);
				}

				// Dispatch keyboard event to popup for navigation
				if (key === 'ArrowDown') {
					setTimeout(() => {
						document.body.dispatchEvent(
							new KeyboardEvent('keydown', {
								key: 'ArrowDown',
								bubbles: true,
							}),
						);
					}, 0);
				}
				return;
			}

			// Enter - open popup
			if (key === 'Enter') {
				e.preventDefault();
				e.stopPropagation();
				if (!open) {
					setOpen(true);
					onOpen?.();
				}
				return;
			}

			// Space - open popup without starting search
			if (key === 'Space') {
				e.preventDefault();
				e.stopPropagation();
				if (!open) {
					setOpen(true);
					onOpen?.();
				}
				// Don't start search for Space key
				return;
			}

			// ESC - close popup
			if (key === 'Escape') {
				e.preventDefault();
				e.stopPropagation();
				if (open) {
					escKeyPressedRef.current = true;
					handleEscapeKey();
				}
				return;
			}

			// Typing characters - open popup and set input value
			// Exclude Space key (handled separately above)
			if (key.length === 1 && key !== ' ' && !searchDisabled) {
				if (e.metaKey || e.ctrlKey || e.altKey) {
					return;
				}

				e.preventDefault();
				if (!open) {
					onOpen?.();
					handleOpenChange(true);
					// Set the input value to the typed character after popup opens
					setTimeout(() => {
						const searchInput = document.getElementById(searchInputId) as HTMLInputElement;
						if (searchInput) {
							searchInput.focus();
							searchInput.value = key;
							// Dispatch both input and change events to ensure Combobox and our handler both fire
							searchInput.dispatchEvent(new Event('input', { bubbles: true }));
							searchInput.dispatchEvent(new Event('change', { bubbles: true }));
							onSearchCallback?.(key);
						}
					}, 50);
				} else {
					// Popup already open - append character to existing input value
					const searchInput = document.getElementById(searchInputId) as HTMLInputElement;
					if (searchInput) {
						searchInput.focus();
						searchInput.value = searchInput.value + key;
						searchInput.dispatchEvent(new Event('input', { bubbles: true }));
						searchInput.dispatchEvent(new Event('change', { bubbles: true }));
						onSearchCallback?.(searchInput.value);
					}
				}
			}
		},
		[disabled, open, searchDisabled, onOpen, handleEscapeKey, handleOpenChange, searchInputId, onSearchCallback],
	);

	// Combobox handles search state internally, no need to reset

	// Handle anchor tag clicks in custom triggers - use capture phase to intercept before Base UI
	useEffect(() => {
		if (!TriggerComponent) {
			return;
		}

		return subscribeToAnchorClickInterceptor(inputId, lastClickWasAnchorRef);
	}, [TriggerComponent, inputId]);

	// Handle focus event bubbling (for accessibility)
	const handleFocus = useCallback((e: React.FocusEvent) => {
		// Create a new focus event that bubbles
		const focusEvent = new FocusEvent('focus', {
			bubbles: true,
			cancelable: true,
		});
		e.currentTarget.dispatchEvent(focusEvent);
	}, []);

	// Listen for ESC key when popup is open to track it for cancel callbacks
	useEffect(() => {
		if (!open) {
			return;
		}

		const handleKeyDown = (e: KeyboardEvent) => {
			if (e.key === 'Escape') {
				escKeyPressedRef.current = true;
			}
		};

		document.addEventListener('keydown', handleKeyDown, true);
		return () => {
			document.removeEventListener('keydown', handleKeyDown, true);
		};
	}, [open]);

	const isOpen = controlledActive !== undefined ? controlledActive : open;

	// Focus list and highlight first option when popup opens (single and multiple)
	useEffect(() => {
		if (!isOpen) return;
		const t = setTimeout(() => {
			const popup = document.getElementById(popupId);
			if (!popup) return;
			const listbox = popup.querySelector('[role="listbox"]') as HTMLElement | null;
			const searchInput = popup.querySelector('input');
			const focusTarget = listbox ?? searchInput;
			if (focusTarget) {
				focusTarget.focus();
				// Home always goes to first item (avoids highlighting selected+1 when a selection exists)
				listbox?.dispatchEvent(new KeyboardEvent('keydown', { key: 'Home', bubbles: true }));
			}
		}, 0);
		return () => clearTimeout(t);
	}, [isOpen, popupId]);

	// Close on outside mousedown (capture phase) so the click can reach other controls (e.g. header dropdown)
	useEffect(() => {
		if (!isOpen) return;
		const handleMouseDown = (e: MouseEvent) => {
			const target = e.target as Node;
			const rootEl = rootRef.current;
			const popup = document.getElementById(popupId);
			if (rootEl?.contains(target) || popup?.contains(target)) return;
			handleOpenChange(false);
		};
		document.addEventListener('mousedown', handleMouseDown, true);
		return () => document.removeEventListener('mousedown', handleMouseDown, true);
	}, [isOpen, popupId, handleOpenChange]);

	const labelId = fieldContext?.labelId ?? (formContext && name ? `field-${name}-label` : null);

	return (
		<div
			ref={(el) => {
				rootRef.current = el;
				if (typeof ref === 'function') ref(el);
				else if (ref) ref.current = el;
			}}
			className={cn(styles.Select, className)}
			data-size={size}
			data-disabled={disabled ? 'true' : undefined}
			data-qid-component-type={isMultiple ? 'Luna::SelectMultiple' : 'Luna::Select'}
			data-focused={manager.inputHasFocus ? 'true' : undefined}
			data-errored={manager.showErrorMessage ? 'true' : undefined}
			{...restProps}
		>
			<Combobox.Root
				key={`select-root-${closeKey}`}
				modal={false}
				items={isMultiple ? normalizedOptions : itemsWithEmptyOption}
				value={
					isMultiple
						? (normalizedOptions.filter((opt) =>
								isValueSelected(internalSelection, opt.value),
							) as SelectOption<unknown>[])
						: singleValue === null || singleValue === undefined
							? null
							: (itemsWithEmptyOption.find((opt) => {
									const option = opt as SelectOption<unknown>;
									return valuesMatchById(option.value, singleValue);
								}) ?? null)
				}
				onValueChange={isMultiple ? handleMultipleValueChange : handleValueChange}
				open={isOpen}
				onOpenChange={handleOpenChange}
				autoHighlight={true}
				{...(isMultiple ? {} : { closeOnSelect: true })}
				actionsRef={comboboxActionsRef as React.RefObject<{ open: () => void; close: () => void; unmount: () => void }>}
				onInputValueChange={(inputValue) => {
					// Debounce search callback
					if (searchDebounceTimerRef.current) {
						clearTimeout(searchDebounceTimerRef.current);
					}
					searchDebounceTimerRef.current = setTimeout(() => {
						onSearchCallback?.(inputValue);
					}, debounceDelay);
				}}
				itemToStringLabel={(item: unknown) => {
					// item is a SelectOption object when we pass option objects to items
					const option = item as SelectOption<unknown>;
					// Handle null
					if (option.value === null) {
						return emptyOptionText || '';
					}
					return option.label ?? String(option.value);
				}}
				isItemEqualToValue={(itemValue, currentValue) => {
					// Multiple: currentValue is array of selected option objects
					if (Array.isArray(currentValue)) {
						const option = itemValue as SelectOption<unknown>;
						return currentValue.some((selected) =>
							valuesMatchById(option.value, (selected as SelectOption<unknown>).value),
						);
					}
					// Handle null comparison
					if (itemValue === null && currentValue === null) {
						return true;
					}
					// Handle object comparison by ID if both have id property
					if (hasIdProperty(itemValue) && hasIdProperty(currentValue)) {
						return itemValue.id === currentValue.id;
					}
					// Default comparison
					return Object.is(itemValue, currentValue);
				}}
				filter={(itemValue: unknown, query: string, itemToString?: (itemValue: unknown) => string) => {
					// Custom filtering logic for searchKeys and HTML entity decoding
					if (!query || searchDisabled) {
						return true; // Show all items when no query
					}
					// itemValue is a SelectOption object when we pass option objects to items
					const option = itemValue as SelectOption<unknown>;
					// Empty option should not be filtered out (always show if in list)
					if (option.value === null) {
						return true;
					}
					const decodedQuery = decodeEntities(query.toLowerCase().trim());
					if (searchKeys && typeof option.value === 'object' && option.value !== null) {
						return searchKeys.some((key) => {
							const val = (option.value as Record<string, unknown>)[key];
							const decodedVal = decodeEntities(String(val).toLowerCase());
							return decodedVal.includes(decodedQuery);
						});
					}
					// Use itemToString if provided (it's our itemToStringLabel function), otherwise use option.label
					const label = itemToString ? itemToString(itemValue) : (option.label ?? String(option.value));
					const decodedLabel = decodeEntities(label.toLowerCase());
					return decodedLabel.includes(decodedQuery);
				}}
			>
				{/* Hidden Combobox.Trigger - we use custom Button instead */}
				<Combobox.Trigger
					render={(props: React.HTMLAttributes<HTMLButtonElement>) => {
						// Hidden trigger for Combobox to work, but we use our custom Button for UI
						return <button {...props} style={{ display: 'none' }} aria-hidden="true" />;
					}}
				/>
				{/* Custom trigger button */}
				{TriggerComponent ? (
					<TriggerComponent
						value={(isMultiple ? valueArray : value) ?? null}
						disabled={disabled}
						className={cn(styles.trigger, triggerClass)}
						id={inputId}
						role="combobox"
						aria-controls={popupId}
						aria-label={ariaLabel}
						aria-labelledby={ariaLabelledBy || (ariaLabel ? undefined : labelId || undefined)}
						aria-expanded={isOpen ? 'true' : 'false'}
						onClick={(e: React.MouseEvent) => {
							if (disabled) {
								return;
							}
							if (checkIfAnchorClick(e)) {
								lastClickWasAnchorRef.current = true;
								e.stopPropagation();
								e.preventDefault();
								setTimeout(() => {
									lastClickWasAnchorRef.current = false;
								}, 100);
								return;
							}
							lastClickWasAnchorRef.current = false;
							if (!open) {
								// Call onOpen immediately when opening
								onOpen?.();
								handleOpenChange(true);
							} else {
								handleOpenChange(false);
							}
						}}
						onKeyDown={handleTriggerKeyDown}
						onFocus={handleFocus}
					/>
				) : (
					<Button
						variant={variant}
						size={size}
						width="full"
						align={align}
						disabled={disabled}
						active={isOpen}
						errored={manager.showErrorMessage}
						id={inputId}
						className={cn(styles.trigger, triggerClass)}
						aria-controls={popupId}
						aria-label={ariaLabel}
						aria-labelledby={ariaLabelledBy || (ariaLabel ? undefined : labelId || undefined)}
						aria-required={required ? 'true' : 'false'}
						aria-expanded={isOpen ? 'true' : 'false'}
						aria-invalid={manager.showErrorMessage ? 'true' : 'false'}
						aria-describedby={manager.ariaDescribedBy}
						ref={(node) => {
							triggerRef.current = node;
						}}
						onClick={() => {
							if (!disabled) {
								if (!open) {
									// Call onOpen immediately when opening
									onOpen?.();
									handleOpenChange(true);
								} else {
									handleOpenChange(false);
								}
							}
						}}
						onFocus={(e) => {
							handleFocus(e);
							manager.focusIn();
						}}
						onBlur={(e) => {
							manager.focusOutAndTrySave(e.nativeEvent);
						}}
						onKeyDown={handleTriggerKeyDown}
					>
						{icon && <div className={styles.preIcon}>{icon}</div>}
						{isMultiple ? (
							<div className={styles.label} data-placeholder={!valueArray.length || undefined}>
								{getMultipleTriggerLabel()}
							</div>
						) : selectedOption === null ? (
							<div className={cn(styles.label, styles.placeholder)} data-placeholder>
								{placeholder}
							</div>
						) : (
							<div className={styles.label}>
								{renderTrigger
									? renderTrigger(singleValue as NonNullable<typeof value>)
									: children
										? children(singleValue as NonNullable<typeof value>)
										: (selectedOption.label ?? String(singleValue))}
							</div>
						)}
						{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
							<Spinner size="sm" transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined} />
						)}
						{!chevronIsHidden && <Icon icon="chevron-down" size="sm" className={styles.postIcon} />}
					</Button>
				)}
				{isOpen && (
					<Combobox.Portal keepMounted={false}>
						<Combobox.Positioner
							className={styles.positioner}
							side={placement}
							align="start"
							sideOffset={4}
							anchor={triggerRef}
						>
							<Combobox.Popup
								className={styles.popup}
								id={popupId}
								initialFocus={false}
								onKeyDown={(e: React.KeyboardEvent) => {
									if (e.key === 'Escape' && open) {
										escKeyPressedRef.current = true;
										handleEscapeKey();
									}
								}}
								style={{
									minWidth:
										popupWidth === 'auto'
											? isMultiple
												? triggerRef.current?.clientWidth || 200
												: undefined
											: popupWidth === 'sm'
												? 200
												: popupWidth === 'md'
													? 300
													: 400,
								}}
							>
								{isMultiple ? (
									<>
										{!searchDisabled && (
											<div className={styles.search}>
												<label htmlFor={searchInputId} className={styles.searchLabel}>
													{isLoading ? <Spinner size="sm" /> : <Icon icon="search" size="sm" />}
												</label>
												<Combobox.Input
													id={searchInputId}
													className={styles.searchInput}
													placeholder={searchInputPlaceholder || ariaLabelSearch}
													autoComplete="off"
													autoCorrect="off"
													autoCapitalize="off"
													spellCheck="false"
													aria-autocomplete="list"
													aria-controls={popupId}
													data-qid-select-option-list-search
												/>
											</div>
										)}
										{allowSelectAll && normalizedOptions.length > 0 && (
											<div className={styles.selectAllWrapper}>
												<button
													type="button"
													className={styles.itemMulti}
													data-qid-select-option="all"
													onClick={(e: React.MouseEvent) => {
														e.preventDefault();
														e.stopPropagation();
														handleSelectAll();
													}}
													role="option"
													aria-selected={
														normalizedOptions.every((opt) => isValueSelected(internalSelection, opt.value))
															? 'true'
															: 'false'
													}
												>
													<div className={styles.itemIndicator}>
														<div
															className={styles.checkbox}
															data-checked={
																normalizedOptions.every((opt) => isValueSelected(internalSelection, opt.value))
																	? 'true'
																	: 'false'
															}
														>
															<div className={styles.checkboxInner}>
																<svg
																	className={styles.checkboxIcon}
																	xmlns="http://www.w3.org/2000/svg"
																	viewBox="0 0 20 20"
																>
																	<path
																		fill="currentColor"
																		fillRule="evenodd"
																		d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z"
																		clipRule="evenodd"
																	/>
																</svg>
															</div>
														</div>
													</div>
													<div className={styles.itemText}>{selectAllLabel}</div>
												</button>
											</div>
										)}
										<Combobox.List
											className={styles.list}
											aria-label={searchDisabled ? ariaLabelKeyboardNav : ariaLabelSearch}
											data-test-option-list
											data-qid-select-option-list
										>
											{(item: unknown) => {
												const option = item as SelectOption<unknown>;
												const optionKey =
													typeof option.value === 'object' && option.value !== null && 'id' in option.value
														? `option-${(option.value as { id: string }).id}`
														: option.value === null
															? 'option-null'
															: `option-${String(option.value)}`;
												const isSelected = isValueSelected(internalSelection, option.value);
												return (
													<Combobox.Item
														key={optionKey}
														value={option}
														className={styles.itemMulti}
														data-qid-select-option
													>
														<div className={styles.itemIndicator}>
															<div className={styles.checkbox} data-checked={isSelected ? 'true' : 'false'}>
																<div className={styles.checkboxInner}>
																	<svg
																		className={styles.checkboxIcon}
																		xmlns="http://www.w3.org/2000/svg"
																		viewBox="0 0 20 20"
																	>
																		<path
																			fill="currentColor"
																			fillRule="evenodd"
																			d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z"
																			clipRule="evenodd"
																		/>
																	</svg>
																</div>
															</div>
														</div>
														<span className={styles.itemText}>
															{children
																? (children as (v: unknown) => React.ReactNode)(option.value)
																: (option.label ?? String(option.value))}
															{isAiSuggestion && isAiSuggestion(option.value) && (
																<Icon
																	icon="ai"
																	size="sm"
																	className={styles.aiIcon}
																	title="AI suggestion"
																	role="figure"
																/>
															)}
														</span>
													</Combobox.Item>
												);
											}}
										</Combobox.List>
										{onCreateNew && createNewLabel && (
											<div className={styles.footer}>
												<button
													type="button"
													className={styles.createNewButton}
													data-qid-option-list-create-new
													onClick={(e) => {
														e.preventDefault();
														handleOpenChange(false);
														onCreateNew();
													}}
												>
													<Icon icon="add" size="sm" />
													<div>{isMultiple ? createNewLabel || t('select-multiple.createNew') : createNewLabel}</div>
												</button>
											</div>
										)}
									</>
								) : (
									<>
										{!searchDisabled && (
											<div className={styles.search}>
												<label htmlFor={searchInputId} className={styles.searchLabel}>
													<span className={styles.srOnly}>
														{searchInputPlaceholder || 'Search'}
														{isLoading ? ' (Searching...)' : ''}
													</span>
													{isLoading ? <Spinner size="sm" /> : <Icon icon="search" size="sm" />}
												</label>
												<Combobox.Input
													id={searchInputId}
													className={styles.searchInput}
													placeholder={searchInputPlaceholder || 'Search'}
													autoComplete="off"
													autoCorrect="off"
													autoCapitalize="off"
													spellCheck="false"
													aria-autocomplete="list"
													aria-controls={popupId}
													data-qid-select-option-list-search
													onKeyDown={(e: React.KeyboardEvent<HTMLInputElement>) => {
														if (e.key === 'Enter') {
															e.preventDefault();
															e.stopPropagation();
															const popup = document.getElementById(popupId);
															const listbox = popup?.querySelector('[role="listbox"]');
															if (listbox) {
																const highlightedOption = listbox.querySelector(
																	'[role="option"][data-highlighted]:not([data-disabled="true"])',
																) as HTMLElement;
																const firstOption =
																	highlightedOption ||
																	(listbox.querySelector('[role="option"]:not([data-disabled="true"])') as HTMLElement);
																if (firstOption) firstOption.click();
															}
														}
													}}
												/>
											</div>
										)}
										<Combobox.Empty>
											<div className={styles.noResults}>No results</div>
										</Combobox.Empty>
										<Combobox.List className={styles.list}>
											{(item: unknown) => {
												const option = item as SelectOption<unknown>;
												const optionKey =
													option.value === null
														? 'empty'
														: hasIdProperty(option.value)
															? `option-${option.value.id}`
															: `option-${String(option.value)}`;
												return (
													<Combobox.Item key={optionKey} value={option} className={styles.item}>
														<Combobox.ItemIndicator className={styles.itemIndicator}>
															<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 16 16" className={styles.checkmark}>
																<path
																	fill="currentColor"
																	d="M14.17 2.6l-8.22 8.23L1.82 6.7.54 7.98l5.41 5.42 9.51-9.51-1.29-1.29z"
																/>
															</svg>
														</Combobox.ItemIndicator>
														<span className={styles.itemText}>
															{option.value === null
																? emptyOptionText
																: children
																	? children(option.value)
																	: (option.label ?? String(option.value))}
															{option.value !== null && isAiSuggestion && isAiSuggestion(option.value) && (
																<Icon
																	icon="ai"
																	size="sm"
																	className={styles.aiIcon}
																	title="AI suggestion"
																	role="figure"
																/>
															)}
														</span>
													</Combobox.Item>
												);
											}}
										</Combobox.List>
										{onCreateNew && createNewLabel && (
											<div className={styles.footer}>
												<button
													type="button"
													className={styles.createNewButton}
													onClick={(e) => {
														e.preventDefault();
														onCreateNew();
														setOpen(false);
													}}
												>
													<Icon icon="plus" size="sm" />
													{createNewLabel}
												</button>
											</div>
										)}
									</>
								)}
							</Combobox.Popup>
						</Combobox.Positioner>
					</Combobox.Portal>
				)}
			</Combobox.Root>
			<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
				{manager.errorMessageToDisplay}
			</InputError>
		</div>
	);
}
