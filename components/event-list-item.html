event-list-item.tsx
import { type ReactNode, useMemo } from 'react';

import { cn } from '../../utils/cn';
import { ButtonWithMenu, type MenuOptionProps } from '../button-with-menu';
import { Icon } from '../icon';
import { SkeletonText } from '../skeleton';
import { Text } from '../text';

import styles from './event-list-item.module.css';

export const LunaEventType = {
	Broadcast: 'broadcast',
	Success: 'success',
	Error: 'error',
	Info: 'info',
	Warning: 'warning',
} as const;
export type LunaEventType = (typeof LunaEventType)[keyof typeof LunaEventType];

export const LunaEventColor = {
	Broadcast: 'blue-500',
	Success: 'green-500',
	Error: 'red-500',
	Info: 'gray-500',
	Warning: 'yellow-500',
} as const;
export type LunaEventColor = (typeof LunaEventColor)[keyof typeof LunaEventColor];

export interface LunaEventListItemAction {
	label: string;
	icon?: string;
	href?: string;
	onClick?: () => void;
}

export interface EventListItemSlots {
	/** Title content for the event item */
	title?: ReactNode;
	/** Badge content for the event item */
	badge?: ReactNode;
	/** Meta content (e.g., date/time) for the event item */
	meta?: ReactNode;
	/** Custom actions render function */
	renderActions?: (helpers: { Option: React.ComponentType<MenuOptionProps> }) => ReactNode;
}

export interface EventListItemProps extends EventListItemSlots {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLLIElement>;
	/** URL to navigate to when the item is clicked */
	href?: string;
	/** Icon name to display */
	icon?: string;
	/** Type of event which determines icon color */
	eventType?: LunaEventType;
	/** Click handler when href is not provided */
	onClick?: () => void;
	/** Array of action items for the actions menu */
	actions?: LunaEventListItemAction[];
	/** Shows loading skeleton state */
	isLoading?: boolean;
	/** Additional CSS class */
	className?: string;
	/** Children content (alternative to title slot) */
	children?: ReactNode;
}

/**
 * Maps event type to icon color
 */
function getIconColor(eventType?: LunaEventType): LunaEventColor {
	switch (eventType) {
		case LunaEventType.Success:
			return LunaEventColor.Success;
		case LunaEventType.Error:
			return LunaEventColor.Error;
		case LunaEventType.Warning:
			return LunaEventColor.Warning;
		case LunaEventType.Info:
			return LunaEventColor.Info;
		case LunaEventType.Broadcast:
		default:
			return LunaEventColor.Broadcast;
	}
}

/**
 * An item within an EventList. Displays an event with optional icon, title, badge, meta, and actions.
 * Can be a link, button, or plain div depending on props.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/event-list-item.md)
 */
export function EventListItem(props: EventListItemProps) {
	const {
		ref,
		href,
		icon,
		eventType,
		onClick,
		actions,
		isLoading,
		className,
		title,
		badge,
		meta,
		renderActions,
		children,
	} = props;

	const iconColor = useMemo(() => getIconColor(eventType), [eventType]);

	// Determine the inner element tag
	const InnerTag = href ? 'a' : onClick ? 'button' : 'div';
	const innerProps = {
		...(href ? { href } : {}),
		...(InnerTag === 'button' ? { type: 'button' as const } : {}),
		onClick: onClick,
	};

	const showChevron = Boolean(href || onClick || isLoading);
	const hasSecondary = Boolean(badge || meta || actions || renderActions || isLoading);
	const titleContent = title ?? children;

	return (
		<li
			ref={ref}
			className={cn(styles.eventListItem, isLoading && styles.loading, className)}
			data-qid-luna-event-list-item
			data-qid-luna-event-list-item-loading={isLoading || undefined}
		>
			<InnerTag data-qid-event-list-item-inner className={styles.inner} {...innerProps}>
				<div className={styles.content}>
					<div className={styles.primary}>
						{icon && (
							<div data-qid-luna-event-list-item-icon className={styles.icon}>
								<Icon icon={icon} color={iconColor} />
							</div>
						)}
						<div data-qid-luna-event-list-item-title className={styles.title}>
							{isLoading ? <SkeletonText /> : <Text>{titleContent}</Text>}
						</div>
					</div>
					{hasSecondary && (
						<div className={styles.secondary}>
							{isLoading ? (
								<SkeletonText data-qid-luna-secondary-skeleton-text />
							) : (
								<>
									{badge && (
										<div data-qid-luna-event-list-item-badge className={styles.badge}>
											{badge}
										</div>
									)}
									{meta && (
										<div data-qid-luna-event-list-item-meta className={styles.meta}>
											<Text font="100" color="gray-500">
												{meta}
											</Text>
										</div>
									)}
									{(actions || renderActions) && (
										<div data-qid-luna-event-list-item-actions className={styles.actions}>
											<ButtonWithMenu>
												<ButtonWithMenu.Button variant="transparent" aria-label="Actions">
													<Icon icon="more-horizontal" />
												</ButtonWithMenu.Button>
												<ButtonWithMenu.Menu>
													{renderActions?.({ Option: ButtonWithMenu.Option })}
													{actions?.map((action, index) => (
														<ButtonWithMenu.Option
															key={index}
															onClick={action.onClick}
															href={action.href}
															data-qid-luna-event-list-item-action
														>
															{action.icon && (
																<div data-qid-luna-event-list-item-action-icon className={styles.icon}>
																	<Icon icon={action.icon} />
																</div>
															)}
															{action.label}
														</ButtonWithMenu.Option>
													))}
												</ButtonWithMenu.Menu>
											</ButtonWithMenu>
										</div>
									)}
								</>
							)}
						</div>
					)}
				</div>
				{showChevron && (
					<div data-qid-luna-event-list-item-chevron className={styles.chevron}>
						<Icon icon="chevron-right" />
					</div>
				)}
			</InnerTag>
		</li>
	);
}

export default EventListItem;
