let nextInputId = 0;

/**
 * A component for selecting a specific time.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/select-time.md)
 */
export default class SelectTime extends Component<SelectTimeSignature> {
	@service declare intl: IntlService;

	@consume('luna-input-id') declare inputId: string;
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;

	manager: InputManager;

	constructor(owner: Owner, args: SelectTimeSignature['Args']) {
		super(owner, args);
		assert('You cannot provide both ariaLabel and ariaLabelledBy', !args.ariaLabel || !args.ariaLabelledBy);

		this._configureLocale();
		this._initValues();

		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;
		const managerOptions: InputManagerOptions = {
			mode: 'change',
			name: args.name,

			focus: this.focusInput.bind(this),

			ariaDescribedBy: this.args.ariaDescribedBy ?? null,

			get value() {
				return args.value;
			},

			get isDisabled(): boolean {
				return args.disabled ?? false;
			},

			get requirements() {
				return args.requirements;
			},

			get serverErrors() {
				// eslint-disable-next-line @typescript-eslint/no-deprecated
				return args.serverErrors;
			},

			get formContext() {
				return that.formContext;
			},

			get selectorIsVisible() {
				return that.showPopup;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	popupId = `${guidFor(this)}--popup`;

	@tracked showPopup = false;

	/**
	 * Text value of focused input
	 */
	@tracked private _focusedValue: string | number | null = null;

	/**
	 * Input name that currently has focus
	 */
	@tracked private _focusedInput: string | null = null;

	@tracked useAmPm = false;

	/**
	 * The potentially new value of the input from the user interacting
	 * with the time picker. Used in all calculations and comparisons but
	 * not used in the input field until the user confirms the change.
	 * Reset when the popup opens.
	 */
	@tracked private newValue: Date | null = null;

	_configureLocale() {
		const date = new Date();
		date.setHours(20);
		date.setMinutes(0);

		// Compare the formatted time string to determine 12/24h format
		const useAmPm = this.intl.formatTime(date, { hour: 'numeric', minute: 'numeric' }).startsWith('8:00');
		this.useAmPm = useAmPm;
	}

	_initValues() {
		this._focusedInput = null;
		this._focusedValue = null;
	}

	get id() {
		return this.args.id ?? this.inputId ?? `luna-select-time-${nextInputId++}`;
	}

	private get initialValue(): Date | null {
		return this.args.value ? createDateFrom24HourTimeString(this.args.value) : null;
	}

	private get value(): Date {
		return this.newValue ?? createDateFrom24HourTimeString(this.args.value);
	}

	get displayValue() {
		if (this.showPopup) {
			assert('initialValue must exist', this.initialValue);
			return this.initialValue;
		}
		return this.value;
	}

	get triggerIsActive() {
		return this.args.active ?? this.showPopup;
	}

	/** Get hour value in 24h format */
	private get hour() {
		return this.value.getHours();
	}

	/** Set hour value in 24h format */
	private set hour(value: number | null) {
		if (value !== null) {
			this.newValue = new Date(this.value.setHours(Math.min(Math.max(Number(value), 0), 23)));
		}
	}

	/** Get hour value in 12/24h format based on locale */
	get hourInputValue() {
		if (this._focusedInput === 'hour') {
			return this._focusedValue;
		}
		if (this.useAmPm) {
			if (this.hour === 0) {
				return 12;
			}
			assert('Expected hour to be a number', typeof this.hour === 'number');
			if (this.hour > 12) {
				return this.hour - 12;
			}
			return this.hour;
		}
		return this.hour;
	}

	private get minute() {
		return this.value.getMinutes();
	}

	private set minute(value) {
		if (value !== null) {
			this.newValue = new Date(this.value.setMinutes(Number(value)));
		}
	}

	/**
	 * Whether the text entry is enabled. When true, alters the popup aria-label.
	 * Redeclared and bound in the instance initializer of the consuming app.
	 * See `client/app/initializers/luna.ts` for example implementation.
	 */
	// eslint-disable-next-line @typescript-eslint/class-literal-property-style
	get textEntryIsEnabled() {
		return false;
	}

	get popupAriaLabel() {
		if (!this.args.placeholder && !this.args.value) {
			if (this.textEntryIsEnabled) {
				return this.intl.t('input.select-date.placeholder', {
					textEntry: true,
					kind: 'single',
				});
			}
			return this.intl.t('input.select-date.placeholder', {
				textEntry: false,
				kind: 'single',
			});
		}
		return;
	}

	/** Get minute value with leading zero if single digit */
	get minuteInputValue() {
		if (this._focusedInput === 'minute') {
			return this._focusedValue;
		}
		return `${this.minute}`.padStart(2, '0');
	}

	get ampm() {
		return this.value.getHours() >= 12 ? 'PM' : 'AM';
	}

	set ampm(value) {
		if (value === 'AM' && this.value.getHours() >= 12) {
			this.hour = this.value.getHours() - 12;
		} else if (value === 'PM' && this.value.getHours() < 12) {
			this.hour = this.value.getHours() + 12;
		}
	}

	get labelId() {
		return this.inputId ? labelIdForField(this.inputId) : null;
	}

	@action
	private async handleChange(value: string | null) {
		if (value === null) {
			await this.args.onChange?.(null);
			this.manager.onChange();
		} else {
			this.newValue = createDateFrom24HourTimeString(value);
			assert('Expected hour to be a number', typeof this.hour === 'number');
			await this.args.onChange?.(`${this.hour}:${`${this.minute}`.padStart(2, '0')}`);
			this.manager.onChange();
		}
	}

	onFocusIn = () => {
		if (this.args.shouldRenderAsDiv) {
			return;
		}
		this.manager.focusIn();
	};

	@action
	maybeOpenTimePicker() {
		if (this.args.shouldRenderAsDiv) {
			return;
		}
		this._initValues();
		this.showPopup = true;
	}

	@action
	onCloseTimePicker(trigger: OnCloseSource) {
		if (trigger === 'esc') {
			this._initValues();
		} else if (trigger === 'manual') {
			// only happens if clear button is clicked
			void this.handleChange(null);
		} else {
			// 'enter' or 'click-outside'
			assert('Expected hour to be a number', typeof this.hour === 'number');
			void this.handleChange(`${this.hour}:${this.minute < 10 ? '0' : ''}${this.minute}`);
		}
		this.newValue = null;
		this.showPopup = false;
	}

	@action
	onChangeHour(value: string | number | null, { shouldMaybeAutofocus = true } = {}) {
		let hourString = value === null ? '' : `${value}`.replace(/[^0-9]/g, '');
		let hourNumeric = hourString.length ? Number(hourString) : null;

		if (hourString.length) {
			if (hourNumeric !== null && ((this.useAmPm && hourNumeric > 12) || (!this.useAmPm && hourNumeric > 23))) {
				const hourPart = `${hourNumeric}`.split('')[0];
				assert('Expected hourPart', hourPart);
				hourString = hourPart;
				hourNumeric = Number(hourString);
				if (this.useAmPm && this.ampm === 'PM') {
					this.hour = hourNumeric + 12;
				} else {
					this.hour = hourNumeric;
				}
			} else if (this.useAmPm && this.ampm === 'PM' && hourNumeric !== null && hourNumeric < 12) {
				this.hour = hourNumeric + 12;
			} else if (this.useAmPm && this.ampm === 'AM' && hourNumeric === 12) {
				this.hour = 0;
			} else {
				this.hour = hourNumeric;
			}
		}

		this._focusedValue = hourString;

		if (this._focusedValue.length === 2 && shouldMaybeAutofocus) {
			const minuteEl = document.querySelector('[data-input-minute]');
			assert('Expected minute input to exist', minuteEl instanceof HTMLElement);
			minuteEl.focus();
		}
	}

	@action
	onChangeMinute(value: string | number | null, { shouldMaybeAutofocus = true } = {}) {
		let minuteString = value === null ? '' : `${value}`.replace(/[^0-9]/g, '');

		if (minuteString.length) {
			let minute = Number(minuteString);
			if (minute > 59) {
				const minutePart = `${minute}`.split('')[0];
				assert('Expected minutePart', minutePart);
				minuteString = minutePart;
				minute = Number(minuteString);
			}
			this.minute = minute;
		}

		this._focusedValue = minuteString;

		if (this._focusedValue.length === 2 && this.useAmPm && shouldMaybeAutofocus) {
			if (this.ampm === 'AM') {
				const amEl = document.querySelector('[data-btn-am]');
				assert('Expected AM button to exist', amEl instanceof HTMLElement);
				amEl.focus();
			} else {
				const pmEl = document.querySelector('[data-btn-pm]');
				assert('Expected PM button to exist', pmEl instanceof HTMLElement);
				pmEl.focus();
			}
		}
	}

	@action
	handleKeypress(event: KeyboardEvent) {
		assert(
			'Expected event target to be an input or button',
			event.target instanceof HTMLInputElement || event.target instanceof HTMLButtonElement,
		);
		const { name } = event.target;
		const enter = event.key === KEYBOARD_KEYS.ENTER;
		const arrowUp = event.key === KEYBOARD_KEYS.UP_ARROW;
		const arrowDown = event.key === KEYBOARD_KEYS.DOWN_ARROW;
		const arrowLeft = event.key === KEYBOARD_KEYS.LEFT_ARROW;
		const arrowRight = event.key === KEYBOARD_KEYS.RIGHT_ARROW;

		const keyIsArrow = (arrowUp || arrowDown || arrowLeft || arrowRight) && !event.shiftKey;

		const valueNumeric = !isNaN(Number(this._focusedValue)) ? Number(this._focusedValue) : null;

		if (name === 'hour') {
			if (arrowUp) {
				if (valueNumeric === null || (this.useAmPm && valueNumeric >= 12) || (!this.useAmPm && valueNumeric >= 23)) {
					this._focusedValue = this.useAmPm ? 1 : 0;
				} else {
					this._focusedValue = valueNumeric + 1;
				}
			}
			if (arrowDown) {
				if (valueNumeric === null || (this.useAmPm && valueNumeric === 1) || (!this.useAmPm && valueNumeric === 0)) {
					this._focusedValue = this.useAmPm ? 12 : 23;
				} else {
					this._focusedValue = valueNumeric - 1;
				}
			}
			if (arrowUp || arrowDown) {
				this.onChangeHour(this._focusedValue, {
					shouldMaybeAutofocus: false,
				});
			}
		} else if (name === 'minute') {
			if (arrowUp) {
				if (valueNumeric === null || valueNumeric >= 59) {
					this._focusedValue = 0;
				} else {
					this._focusedValue = valueNumeric + 1;
				}
			}
			if (arrowDown) {
				if (valueNumeric === null || valueNumeric === 0) {
					this._focusedValue = 59;
				} else {
					this._focusedValue = valueNumeric - 1;
				}
			}
			if (arrowUp || arrowDown) {
				this.onChangeMinute(this._focusedValue, {
					shouldMaybeAutofocus: false,
				});
			}
		} else if ((name === 'ampm-am' || name === 'ampm-pm') && keyIsArrow) {
			this.ampm = this.ampm === 'PM' ? 'AM' : 'PM';
			const ampmEl = document.querySelector(`[name="ampm-${this.ampm.toLowerCase()}"]`);
			assert('Expected AM/PM element to exist', ampmEl instanceof HTMLElement);
			ampmEl.focus();
		}

		if (enter) {
			if (name === 'hour') {
				this.onChangeHour(this._focusedValue);
			} else if (name === 'minute') {
				this.onChangeMinute(this._focusedValue);
			}
			this.onCloseTimePicker('enter');
		}

		if (keyIsArrow || enter) {
			event.preventDefault();
		}
	}

	@action
	handleFocus(event: FocusEvent) {
		assert('Expected event target to be an input', event.target instanceof HTMLInputElement);
		this._focusedInput = event.target.name;
		this._focusedValue = event.target.value;
		event.target.select();
	}

	@action
	handleBlur() {
		this._focusedInput = null;
		this._focusedValue = null;
	}

	@action
	onClose(closeTask: Task<void, []>, value: string | null) {
		void this.handleChange(value);
		void closeTask.perform();
	}

	focusInput = () => {
		focusById(this.id);
	};

	/** Custom ID for the trigger */
	id?: string;
	/** Input requirements for validation */
	requirements?: {
		required?: boolean;
	};
	/** Name for form submission */
	name?: string;
	/** Popup placement relative to trigger */
	placement?: PopupPlacement;
	/** Render as a non-interactive div */
	shouldRenderAsDiv?: boolean;
	/** @deprecated Use form validators instead */
	serverErrors?: Array<{ message?: string } | string>;
	/** ARIA label for the trigger */
	ariaLabel?: string;
	/** ARIA labelledby reference */
	ariaLabelledBy?: string;
	/** ARIA describedby reference */
	ariaDescribedBy?: string;
	/** Called after autosave completes */
	didAutosave?: () => void;
	/** Additional CSS class for the wrapper */
	className?: string;
	/** Force 24-hour format (by default, uses locale detection) */
	use24Hour?: boolean;
	/** Custom trigger render function */
	children?: (value: string | null) => ReactNode;
}

let nextInputId = 0;

/**
 * Creates a Date object from a 24-hour time string (HH:MM)
 */
function createDateFrom24HourTimeString(timeString: string | null | undefined): Date {
	const date = new Date();
	if (!timeString) {
		// Default to 12:00 AM (midnight, which is hour 0 in 24-hour format)
		date.setHours(0, 0, 0, 0);
		return date;
	}

	const [hourStr, minuteStr] = timeString.split(':');
	const hour = parseInt(hourStr ?? '0', 10);
	const minute = parseInt(minuteStr ?? '0', 10);

	date.setHours(hour, minute, 0, 0);
	return date;
}

/**
 * Detects whether to use 12-hour or 24-hour format based on locale
 */
function detectUse12HourFormat(): boolean {
	try {
		const date = new Date();
		date.setHours(20, 0, 0, 0);
		// If formatted time starts with "8:00", it's 12-hour format
		const formatted = new Intl.DateTimeFormat(navigator.language, {
			hour: 'numeric',
			minute: 'numeric',
		}).format(date);
		return formatted.startsWith('8:00') || formatted.startsWith('8：00');
	} catch {
		// Default to 12-hour format if detection fails
		return true;
	}
}

/**
 * Formats a Date object for display
 */
function formatTimeForDisplay(date: Date, useAmPm: boolean): string {
	return new Intl.DateTimeFormat(navigator.language, {
		hour: 'numeric',
		minute: '2-digit',
		hour12: useAmPm,
	}).format(date);
}

/**
 * A component for selecting a specific time.
 * Uses a popup with hour and minute inputs, and AM/PM toggle for 12-hour format.
 */
export function SelectTime(props: SelectTimeProps) {
	const {
		ref,
		value,
		onChange,
		placeholder,
		size = 'md',
		variant = 'white',
		active,
		disabled = false,
		id: providedId,
		requirements,
		name,
		placement = 'bottom-start',
		shouldRenderAsDiv = false,
		serverErrors,
		ariaLabel,
		ariaLabelledBy,
		ariaDescribedBy,
		didAutosave,
		className,
		use24Hour,
		children,
	} = props;
	const generatedId = useId();
	const fieldContext = useFieldContext();
	const id = providedId ?? fieldContext?.inputId ?? `luna-select-time-${nextInputId++}`;
	const popupId = `${id}--popup`;

	const formContext = useFormContext();
	const triggerRef = useRef<HTMLButtonElement>(null);

	// Determine if we use 12-hour or 24-hour format
	const useAmPm = useMemo(() => {
		if (use24Hour !== undefined) {
			return !use24Hour;
		}
		return detectUse12HourFormat();
	}, [use24Hour]);

	// Track popup visibility
	const [showPopup, setShowPopup] = useState(false);

	// Internal state for editing
	const [newValue, setNewValue] = useState<Date | null>(null);
	const [focusedInput, setFocusedInput] = useState<'hour' | 'minute' | null>(null);
	const [focusedValue, setFocusedValue] = useState<string | number | null>(null);

	const valueIsUnchangedRef = useRef(true);

	// Refs for inputs
	const hourInputRef = useRef<HTMLInputElement>(null);
	const minuteInputRef = useRef<HTMLInputElement>(null);

	// Initialize/reset the editing state
	const initValues = useCallback(() => {
		setFocusedInput(null);
		setFocusedValue(null);
	}, []);

	// Get initial value as Date
	const initialValue = useMemo(() => {
		return value ? createDateFrom24HourTimeString(value) : null;
	}, [value]);

	// Get current editing value as Date
	const currentValue = useMemo(() => {
		return newValue ?? createDateFrom24HourTimeString(value);
	}, [newValue, value]);

	// Get display value (shows initial value when popup is open)
	const displayValue = useMemo(() => {
		if (showPopup) {
			return initialValue;
		}
		return currentValue;
	}, [showPopup, initialValue, currentValue]);

	// Get hour in 24h format
	const hour = currentValue.getHours();

	// Get minute
	const minute = currentValue.getMinutes();

	// Get AM/PM
	const ampm = hour >= 12 ? 'PM' : 'AM';

	// Get hour input value (respects 12/24 hour format)
	const hourInputValue = useMemo(() => {
		if (focusedInput === 'hour') {
			return focusedValue;
		}
		if (useAmPm) {
			if (hour === 0) return 12;
			if (hour > 12) return hour - 12;
			return hour;
		}
		return hour;
	}, [focusedInput, focusedValue, useAmPm, hour]);

	// Get minute input value (with leading zero)
	const minuteInputValue = useMemo(() => {
		if (focusedInput === 'minute') {
			return focusedValue;
		}
		return `${minute}`.padStart(2, '0');
	}, [focusedInput, focusedValue, minute]);

	// Input manager for form integration
	const manager = useInputManager({
		mode: 'change',
		name,
		valueIsUnchanged: valueIsUnchangedRef.current,
		didAutosave: useCallback(() => {
			valueIsUnchangedRef.current = true;
			didAutosave?.();
		}, [didAutosave]),
		ariaDescribedBy: ariaDescribedBy ?? null,
		value,
		isDisabled: disabled,
		requirements,
		serverErrors,
		formContext,
		selectorIsVisible: showPopup,
	} as InputManagerOptions);

	// Computed
	const triggerIsActive = active ?? showPopup;

	// Update hour value
	const setHour = useCallback(
		(newHour: number | null) => {
			if (newHour !== null) {
				const date = new Date(currentValue);
				date.setHours(Math.min(Math.max(Number(newHour), 0), 23));
				setNewValue(date);
			}
		},
		[currentValue],
	);

	// Update minute value
	const setMinute = useCallback(
		(newMinute: number | null) => {
			if (newMinute !== null) {
				const date = new Date(currentValue);
				date.setMinutes(Math.min(Math.max(Number(newMinute), 0), 59));
				setNewValue(date);
			}
		},
		[currentValue],
	);

	// Toggle AM/PM
	const setAmPm = useCallback(
		(newAmPm: 'AM' | 'PM') => {
			const date = new Date(currentValue);
			const currentHour = date.getHours();
			if (newAmPm === 'AM' && currentHour >= 12) {
				date.setHours(currentHour - 12);
			} else if (newAmPm === 'PM' && currentHour < 12) {
				date.setHours(currentHour + 12);
			}
			setNewValue(date);
		},
		[currentValue],
	);

	// Handle change
	const handleChange = useCallback(
		async (timeValue: string | null) => {
			if (timeValue === null) {
				await onChange?.(null);
				manager.onChange();
			} else {
				const date = createDateFrom24HourTimeString(timeValue);
				const hours = date.getHours();
				const minutes = date.getMinutes();
				await onChange?.(`${hours}:${`${minutes}`.padStart(2, '0')}`);
				manager.onChange();
			}
		},
		[onChange, manager],
	);

	// Handle focus in
	const handleFocusIn = useCallback(() => {
		if (!shouldRenderAsDiv) {
			manager.focusIn();
		}
	}, [shouldRenderAsDiv, manager]);

	// Open popup
	const handleOpen = useCallback(() => {
		if (shouldRenderAsDiv || disabled) return;

		initValues();
		setNewValue(null);
		setShowPopup(true);
	}, [shouldRenderAsDiv, disabled, initValues]);

	// Close popup
	const handleClose = useCallback(
		(source: OnCloseSource) => {
			if (source === 'esc') {
				// Reset state on escape
				initValues();
			} else if (source === 'manual') {
				// Clear button clicked
				handleChange(null);
			} else {
				// 'enter' or 'click-outside' - apply value
				const hours = currentValue.getHours();
				const minutes = currentValue.getMinutes();
				handleChange(`${hours}:${minutes < 10 ? '0' : ''}${minutes}`);
			}
			setNewValue(null);
			setShowPopup(false);
		},
		[initValues, handleChange, currentValue],
	);

	// Handle hour change
	const handleHourChange = useCallback(
		(inputValue: string, { shouldMaybeAutofocus = true } = {}) => {
			let hourString = inputValue.replace(/[^0-9]/g, '');
			let hourNumeric = hourString.length ? Number(hourString) : null;

			if (hourString.length && hourNumeric !== null) {
				if ((useAmPm && hourNumeric > 12) || (!useAmPm && hourNumeric > 23)) {
					const hourPart = `${hourNumeric}`.split('')[0];
					hourString = hourPart ?? '';
					hourNumeric = Number(hourString);
				}

				if (useAmPm && ampm === 'PM' && hourNumeric < 12) {
					setHour(hourNumeric + 12);
				} else if (useAmPm && ampm === 'AM' && hourNumeric === 12) {
					setHour(0);
				} else {
					setHour(hourNumeric);
				}
			}

			setFocusedValue(hourString);

			// Auto-focus minute input when 2 digits entered
			if (hourString.length === 2 && shouldMaybeAutofocus) {
				minuteInputRef.current?.focus();
			}
		},
		[useAmPm, ampm, setHour],
	);

	// Handle minute change
	const handleMinuteChange = useCallback(
		(inputValue: string, { shouldMaybeAutofocus = true } = {}) => {
			let minuteString = inputValue.replace(/[^0-9]/g, '');

			if (minuteString.length) {
				let minuteNum = Number(minuteString);
				if (minuteNum > 59) {
					const minutePart = `${minuteNum}`.split('')[0];
					minuteString = minutePart ?? '';
					minuteNum = Number(minuteString);
				}
				setMinute(minuteNum);
			}

			setFocusedValue(minuteString);

			// Auto-focus AM/PM button when 2 digits entered
			if (minuteString.length === 2 && useAmPm && shouldMaybeAutofocus) {
				const btnSelector = ampm === 'AM' ? '[data-btn-am]' : '[data-btn-pm]';
				const btn = document.querySelector(btnSelector) as HTMLElement | null;
				btn?.focus();
			}
		},
		[useAmPm, ampm, setMinute],
	);

	// Handle keydown
	const handleKeyDown = useCallback(
		(event: React.KeyboardEvent<HTMLInputElement | HTMLButtonElement>) => {
			const target = event.target as HTMLInputElement | HTMLButtonElement;
			const inputName = target.name;

			const isEnter = event.key === 'Enter';
			const isArrowUp = event.key === 'ArrowUp';
			const isArrowDown = event.key === 'ArrowDown';
			const isArrowLeft = event.key === 'ArrowLeft';
			const isArrowRight = event.key === 'ArrowRight';
			const isArrow = (isArrowUp || isArrowDown || isArrowLeft || isArrowRight) && !event.shiftKey;

			const valueNumeric = !isNaN(Number(focusedValue)) ? Number(focusedValue) : null;

			if (inputName === 'hour') {
				if (isArrowUp) {
					let newVal: number;
					if (valueNumeric === null || (useAmPm && valueNumeric >= 12) || (!useAmPm && valueNumeric >= 23)) {
						newVal = useAmPm ? 1 : 0;
					} else {
						newVal = valueNumeric + 1;
					}
					setFocusedValue(newVal);
					handleHourChange(String(newVal), { shouldMaybeAutofocus: false });
				}
				if (isArrowDown) {
					let newVal: number;
					if (valueNumeric === null || (useAmPm && valueNumeric === 1) || (!useAmPm && valueNumeric === 0)) {
						newVal = useAmPm ? 12 : 23;
					} else {
						newVal = valueNumeric - 1;
					}
					setFocusedValue(newVal);
					handleHourChange(String(newVal), { shouldMaybeAutofocus: false });
				}
			} else if (inputName === 'minute') {
				if (isArrowUp) {
					let newVal: number;
					if (valueNumeric === null || valueNumeric >= 59) {
						newVal = 0;
					} else {
						newVal = valueNumeric + 1;
					}
					setFocusedValue(newVal);
					handleMinuteChange(String(newVal), { shouldMaybeAutofocus: false });
				}
				if (isArrowDown) {
					let newVal: number;
					if (valueNumeric === null || valueNumeric === 0) {
						newVal = 59;
					} else {
						newVal = valueNumeric - 1;
					}
					setFocusedValue(newVal);
					handleMinuteChange(String(newVal), { shouldMaybeAutofocus: false });
				}
			} else if ((inputName === 'ampm-am' || inputName === 'ampm-pm') && isArrow) {
				const newAmPm = ampm === 'PM' ? 'AM' : 'PM';
				setAmPm(newAmPm);
				const btnSelector = `[name="ampm-${newAmPm.toLowerCase()}"]`;
				const btn = document.querySelector(btnSelector) as HTMLElement | null;
				btn?.focus();
			}

			if (isEnter) {
				handleClose('enter');
			}

			if (isArrow || isEnter) {
				event.preventDefault();
			}
		},
		[focusedValue, useAmPm, ampm, handleHourChange, handleMinuteChange, setAmPm, handleClose],
	);

	// Handle input focus
	const handleInputFocus = useCallback((event: React.FocusEvent<HTMLInputElement>) => {
		const target = event.target;
		setFocusedInput(target.name as 'hour' | 'minute');
		setFocusedValue(target.value);
		target.select();
	}, []);

	// Handle input blur
	const handleInputBlur = useCallback(() => {
		setFocusedInput(null);
		setFocusedValue(null);
	}, []);

	// Handle clear
	const handleClear = useCallback(() => {
		handleClose('manual');
	}, [handleClose]);

	// Get aria-labelledby
	const computedAriaLabelledBy = useMemo(() => {
		if (ariaLabelledBy) return ariaLabelledBy;
		if (!ariaLabel) {
			// Use generated label ID if available
			return generatedId ? `${generatedId}-label` : undefined;
		}
		return undefined;
	}, [ariaLabelledBy, ariaLabel, generatedId]);

	// Determine if we should show the clear button
	const showClearButton = value && !requirements?.required;

	// Format the display value
	const formattedDisplayValue = useMemo(() => {
		if (!displayValue) return null;
		return formatTimeForDisplay(displayValue, useAmPm);
	}, [displayValue, useAmPm]);

	return (
		<div
			ref={ref}
			className={cn(styles.SelectTime, className)}
			data-size={size}
			data-disabled={disabled ? 'true' : undefined}
			data-focused={manager.inputHasFocus ? 'true' : undefined}
			data-errored={manager.showErrorMessage ? 'true' : undefined}
			data-qid-component-type="Luna::SelectTime"
		>
			<Button
				ref={triggerRef}
				id={id}
				variant={variant}
				size={size}
				width="full"
				errored={manager.showErrorMessage}
				disabled={disabled}
				active={triggerIsActive}
				role={!shouldRenderAsDiv ? 'combobox' : undefined}
				aria-haspopup={!shouldRenderAsDiv ? 'dialog' : undefined}
				aria-controls={!shouldRenderAsDiv ? popupId : undefined}
				aria-expanded={!shouldRenderAsDiv ? showPopup : undefined}
				aria-label={ariaLabel}
				aria-labelledby={computedAriaLabelledBy}
				aria-describedby={manager.ariaDescribedBy}
				aria-required={requirements?.required ? 'true' : 'false'}
				aria-invalid={manager.showErrorMessage ? 'true' : undefined}
				onFocus={handleFocusIn}
				onClick={handleOpen}
			>
				{variant !== 'transparent' && <Icon className={styles.preIcon} icon="module-timesheets" size="sm" />}

				{value ? (
					children ? (
						<div className={styles.label} data-test-select-time-label>
							{children(value)}
						</div>
					) : displayValue ? (
						<div className={styles.label} data-test-select-time-label>
							{formattedDisplayValue}
						</div>
					) : null
				) : (
					<div className={styles.label} data-placeholder data-test-select-time-label>
						{placeholder}
					</div>
				)}

				{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
					<Spinner size="sm" transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined} />
				)}

				<Icon className={styles.postIcon} icon="chevron-down" size="sm" />
			</Button>

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
				{manager.errorMessageToDisplay}
			</InputError>

			{showPopup && (
				<Popup
					className={styles.popup}
					id={popupId}
					show={showPopup}
					relativeTo={triggerRef}
					placement={placement}
					trapFocus
					onClose={handleClose}
				>
					<div className={styles.inputs}>
						<XStack className={styles.timeInputs}>
							<Input
								ref={hourInputRef}
								size="sm"
								name="hour"
								value={String(hourInputValue ?? '')}
								onChange={handleHourChange}
								onFocus={handleInputFocus}
								onBlur={handleInputBlur}
								onKeyDown={handleKeyDown}
								placeholder="00"
								autoComplete="off"
								inputMode="numeric"
								maxLength={2}
								data-test-select-time-hour
								data-input-hour
								autoFocus
							/>
							<Text font="500" weight="strong">
								:
							</Text>
							<Input
								ref={minuteInputRef}
								size="sm"
								name="minute"
								value={String(minuteInputValue ?? '')}
								onChange={handleMinuteChange}
								onFocus={handleInputFocus}
								onBlur={handleInputBlur}
								onKeyDown={handleKeyDown}
								placeholder="00"
								autoComplete="off"
								inputMode="numeric"
								maxLength={2}
								data-test-select-time-minute
								data-input-minute
							/>
						</XStack>

						{useAmPm && (
							<XStack className={styles.ampmButtons} data-luna-feature="meld">
								<Button
									variant={ampm === 'AM' ? 'blue' : 'white'}
									size="sm"
									name="ampm-am"
									data-test-select-time-ampm="am"
									data-btn-am
									onClick={() => setAmPm('AM')}
									onKeyDown={handleKeyDown}
								>
									AM
								</Button>
								<Button
									variant={ampm === 'PM' ? 'blue' : 'white'}
									size="sm"
									name="ampm-pm"
									data-test-select-time-ampm="pm"
									data-btn-pm
									onClick={() => setAmPm('PM')}
									onKeyDown={handleKeyDown}
								>
									PM
								</Button>
							</XStack>
						)}
					</div>

					{showClearButton && (
						<>
							<Divider className={styles.divider} />
							<Button
								className={styles.clear}
								data-test-clear-date
								width="full"
								size="sm"
								variant="transparent"
								onClick={handleClear}
							>
								Clear Time
							</Button>
						</>
					)}
				</Popup>
			)}
		</div>
	);
}

export default SelectTime;
