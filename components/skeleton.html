● React (TSX)
skeleton.tsx
import { cn } from '../../utils/cn';
import { type LunaFont, Text } from '../text';

import styles from './skeleton.module.css';

interface SkeletonTextBaseProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** Font style to match the text being loaded */
	font?: LunaFont;
	/** Character length of the skeleton. Number of characters or 'full' for 100% width */
	length?: string | number;
	/** Additional CSS class */
	className?: string;
}

export type SkeletonTextProps = SkeletonTextBaseProps &
	Omit<React.HTMLAttributes<HTMLDivElement>, keyof SkeletonTextBaseProps>;

/**
 * A loading placeholder that mimics text content with a shimmer animation.
 * Use to indicate content is loading while maintaining layout stability.
 *
 * Matches the Luna::Skeleton.Text Ember component API and styling.
 */
export function SkeletonText(props: SkeletonTextProps) {
	const { ref, font, length, className, ...rest } = props;

	const widthStyle = length ? `${length}ch` : '100%';

	return (
		<Text
			ref={ref}
			tag="div"
			font={font}
			className={cn(styles.Text, className)}
			style={{ width: widthStyle }}
			data-qid-luna-skeleton-text
			{...rest}
		>
			<div className={styles.animation}>&nbsp;</div>
		</Text>
	);
}

/**
 * Skeleton namespace for loading placeholder components.
 */
export const Skeleton = {
	Text: SkeletonText,
};

export default Skeleton;
