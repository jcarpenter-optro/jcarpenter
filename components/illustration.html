export interface IllustrationProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<SVGSVGElement>;
	/** Name of the illustration to display */
	illustration: IllustrationName;
	/** Size of the illustration (uses Luna size tokens) */
	size?: LunaSize;
	/** Additional CSS class */
	className?: string;
	/** ARIA role (defaults to 'presentation' if no title provided) */
	role?: string;
	/** Accessible title for the illustration */
	title?: string;
	/** Accessible description for the illustration */
	desc?: string;
}

/**
 * An illustration is a visual representation of a concept or idea often used to supplement success, error, or empty states.
 *
 * Matches the Luna::Illustration Ember component API and styling.
 */
export function Illustration(props: IllustrationProps) {
	const { ref, illustration = 'check-circle', size, className, role, title, desc, ...rest } = props;

	// Convert size to token value
	const key = size !== undefined ? String(size) : '48';
	const sizeValue = sizeTokens[key] || '12rem';

	// Default role to 'presentation' if no title provided
	const ariaRole = role ?? (title ? undefined : 'presentation');

	// Reference illustration SVG symbol by ID
	const illustrationId = `#${illustration}`;

	return (
		<svg
			ref={ref}
			className={cn(styles.Illustration, className)}
			width={sizeValue}
			height={sizeValue}
			role={ariaRole}
			data-qid-component-type="Luna::Illustration"
			data-qid-illustration-name={illustration}
			{...rest}
		>
			{title && <title>{title}</title>}
			{desc && <desc>{desc}</desc>}
			<use href={illustrationId} xlinkHref={illustrationId} />
		</svg>
	);
}

export default Illustration;
