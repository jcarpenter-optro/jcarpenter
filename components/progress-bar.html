import type { TOC } from '@ember/component/template-only';
import { assert } from '@ember/debug';
import { concat, hash } from '@ember/helper';
import { htmlSafe } from '@ember/template';

import formatNumber from 'ember-intl/helpers/format-number';
import t from 'ember-intl/helpers/t';
import { eq, not, or } from 'ember-truth-helpers';

import tooltip from '../modifiers/tooltip.ts';

export interface ProgressBarSignature {
	Element: HTMLElement;
	Args: {
		value: number;
		max?: number;
		label?: string | undefined;
		showDetails?: 'percent' | 'count';
		size?: 'sm' | 'md' | 'lg';
		color?: LunaColor;
		labelLayout?: 'hidden' | 'top-left' | 'top-right' | 'inline';
		precision?: number;
	};
	Blocks: {
		default: [];
	};
}

// remove when we upgrade to ember-source@5.2, and replace with built-in uniqueId
let barCount = 0;
function uniqueId(): string {
	return `bar-${barCount++}`;
}

const DEFAULT_LAYOUT: ProgressBarSignature['Args']['labelLayout'] = 'top-left';
const DEFAULT_SIZE: ProgressBarSignature['Args']['size'] = 'sm';
const DEFAULT_MAX = 100;

function argWithDefault<Value>(value: Value, defaultValue: Exclude<Value, undefined | null>) {
	return value ?? defaultValue;
}

function percentage(value: number, max: number = DEFAULT_MAX): number {
	if (max === 0) {
		return 0;
	}
	return Math.min(1, value / max);
}

function variableForColor(color: LunaColor) {
	return setCssVar('luna-progress-bar-color', lunaColorVar(color));
}

/**
 * How far along are you? Only the progress bar knows.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/progress-bar.md)
 */
export const ProgressBar: TOC<ProgressBarSignature> = <template>
	{{#let
		(hash
			uniqueId=(uniqueId)
			hiddenLabel=(or (eq @labelLayout "inline") (eq @labelLayout "hidden"))
			percentage=(formatNumber (percentage @value @max) style="percent" minimumFractionDigits=@precision)
		)
		as |state|
	}}
		<div
			class="ProgressBar"
			data-qid-component-type="Luna::ProgressBar"
			data-label-layout={{argWithDefault @labelLayout DEFAULT_LAYOUT}}
			data-size={{argWithDefault @size DEFAULT_SIZE}}
			style={{if @color (variableForColor @color)}}
			...attributes
			{{tooltip @label disabled=(not (eq @labelLayout "hidden"))}}
		>
			{{assert "You must pass a @label arg or use the <:label> block" @label}}

			{{#unless state.hiddenLabel}}
				<Text
					data-qid-progress-bar-label
					id={{state.uniqueId}}
					class="label-wrapper"
					@tag="label"
					@weight="weak"
					@font="200"
					@clamp={{1}}
				>
					{{@label}}
				</Text>
			{{/unless}}

			{{#if @showDetails}}
				<div data-qid-progress-bar-detail class="detail-wrapper">
					<Text @clamp={{1}} @font="200" @tag="div" @weight="weak">
						{{#if (eq @showDetails "percent")}}
							{{state.percentage}}
						{{else if (eq @showDetails "count")}}
							{{t "progress-bar.details" value=@value max=@max}}
						{{/if}}
					</Text>
				</div>
			{{/if}}

			<div
				data-qid-progress-bar-bar
				role="progressbar"
				aria-valuemin="0"
				aria-valuemax={{@max}}
				aria-valuenow={{@value}}
				aria-valuetext={{state.percentage}}
				aria-labelledby={{unless state.hiddenLabel state.uniqueId false}}
				aria-label={{if state.hiddenLabel (t "progress-bar.label" label=@label percent=state.percentage)}}
				class="progressbar-bg"

import { useId } from 'react';

import { cn } from '../../utils/cn';
import { Text } from '../text';

import styles from './progress-bar.module.css';

export type ProgressBarSize = 'sm' | 'md' | 'lg';
export type ProgressBarLabelLayout = 'hidden' | 'top-left' | 'top-right' | 'inline';
export type ProgressBarShowDetails = 'percent' | 'count';
export type LunaColor = string; // TODO: Import from types when available

interface ProgressBarProps extends React.HTMLAttributes<HTMLDivElement> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** Current progress value */
	value: number;
	/** Maximum value (default: 100) */
	max?: number;
	/** Label text (required) */
	label: string;
	/** Show details: 'percent' or 'count' */
	showDetails?: ProgressBarShowDetails;
	/** Size of the progress bar */
	size?: ProgressBarSize;
	/** Color of the progress bar */
	color?: LunaColor;
	/** Label layout position */
	labelLayout?: ProgressBarLabelLayout;
	/** Number of decimal places for percentage (default: 0) */
	precision?: number;
}

const DEFAULT_LAYOUT: ProgressBarLabelLayout = 'top-left';
const DEFAULT_SIZE: ProgressBarSize = 'sm';
const DEFAULT_MAX = 100;

function percentage(value: number, max: number = DEFAULT_MAX): number {
	if (max === 0) {
		return 0;
	}
	return Math.min(1, value / max);
}

function formatPercentage(value: number, max: number, precision: number = 0): string {
	const percent = percentage(value, max);
	return new Intl.NumberFormat('en-US', {
		style: 'percent',
		minimumFractionDigits: precision,
		maximumFractionDigits: precision,
	}).format(percent);
}

function formatCount(value: number, max: number): string {
	return `${value}/${max}`;
}

function getColorStyle(color?: LunaColor): React.CSSProperties | undefined {
	if (!color) {
		return undefined;
	}
	// Convert Luna color format (e.g., "green-300") to CSS variable
	// Handle both formats: "green-300" and "gray-a-100"
	const cssVarName = `--luna-${color}`;
	return {
		[`--luna-progress-bar-color`]: `var(${cssVarName})`,
	} as React.CSSProperties;
}

/**
 * How far along are you? Only the progress bar knows.
 *
 * Matches the Luna::ProgressBar Ember component API and styling.
 */
export function ProgressBar(props: ProgressBarProps) {
	const {
		ref,
		value,
		max = DEFAULT_MAX,
		label,
		showDetails,
		size = DEFAULT_SIZE,
		color,
		labelLayout = DEFAULT_LAYOUT,
		precision = 0,
		className,
		...rest
	} = props;

	const uniqueId = useId();
	const hiddenLabel = labelLayout === 'inline' || labelLayout === 'hidden';
	const percentText = formatPercentage(value, max, precision);
	const detailText =
		showDetails === 'percent' ? percentText : showDetails === 'count' ? formatCount(value, max) : undefined;

	const progressBarClassName = cn(styles.ProgressBar, className);
	const colorStyle = getColorStyle(color);

	return (
		<div
			ref={ref}
			className={progressBarClassName}
			data-qid-component-type="Luna::ProgressBar"
			data-label-layout={labelLayout}
			data-size={size}
			style={colorStyle}
			{...rest}
		>
			{!hiddenLabel && (
				<Text
					data-qid-progress-bar-label
					id={uniqueId}
					className={styles['label-wrapper']}
					tag="label"
					weight="weak"
					font="200"
					clamp={1}
				>
					{label}
				</Text>
			)}

			{showDetails && (
				<div data-qid-progress-bar-detail className={styles['detail-wrapper']}>
					<Text clamp={1} font="200" tag="div" weight="weak">
						{detailText}
					</Text>
				</div>
			)}

			<div
				data-qid-progress-bar-bar
				role="progressbar"
				aria-valuemin={0}
				aria-valuemax={max}
				aria-valuenow={value}
				aria-valuetext={percentText}
				aria-labelledby={hiddenLabel ? undefined : uniqueId}
				aria-label={hiddenLabel ? `${label}: ${percentText}` : undefined}
				className={styles['progressbar-bg']}
				data-value-percentage={percentText}
			>
				<div className={styles['progressbar-value']} style={{ width: percentText }}></div>
			</div>
		</div>
	);
}

export default ProgressBar;
