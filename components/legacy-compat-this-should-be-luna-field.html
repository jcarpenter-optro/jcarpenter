import type { ModifierLike } from '@glint/template';

import { assert } from '@ember/debug';
import { hash } from '@ember/helper';
import { guidFor } from '@ember/object/internals';
import Component from '@glimmer/component';

import { modifier } from 'ember-modifier';
import { provide } from 'ember-provide-consume-context';

import { labelIdForField } from './field.gts';

interface LegacyCompatThisShouldBeLunaFieldSignature {
	Args: {
		id?: string;
	};
	Blocks: {
		default: [
			{
				inputId: string;
				labelId: string;
				labelContext: ModifierLike<{ Element: HTMLElement }>;
				inputContext: ModifierLike<{ Element: HTMLElement }>;
			},
		];
	};
}

/**
 * Internal component used by `Field` to ensure labels and fields are properly linked.
 * This can be used as a stopgap to fix accessibility issues with fields, but is not a permanent
 * solution, hence the ugly name.
 *
 * Longer term, these should be replaced with Luna Field which has specific expectations around
 * layout and styling.
 *
 * @internal
 */
export default class LegacyCompatThisShouldBeLunaField extends Component<LegacyCompatThisShouldBeLunaFieldSignature> {
	baseId = this.args.id ?? `${guidFor(this)}--target`;

	@provide('luna-input-id') inputId = this.baseId;

	get labelId() {
		return labelIdForField(this.inputId);
	}

	labelContext: ModifierLike<{ Element: HTMLElement }> = modifier((element: HTMLElement) => {
		if (element.tagName === 'LABEL') {
			element.setAttribute('for', this.inputId);
		}
		element.setAttribute('id', this.labelId);
	});

	inputContext: ModifierLike<{ Element: HTMLElement }> = modifier((element: HTMLElement) => {
		const combobox = element.querySelector('[role="combobox"]');
		const target = combobox ?? element;
		const existingId = target.getAttribute('id');
		assert(
			'\nDo not set id on input elements using LegacyCompatThisShouldBeLunaField. This is handled by the provided inputContext modifier.',
			!existingId || existingId === this.inputId,
		);
		const isLuna = element.dataset.qidComponentType?.startsWith('Luna::');
		if (target.getAttribute('role') === 'combobox') {
			assert(
				'\nLegacy combobox implementations are not supported in LegacyCompatThisShouldBeLunaField due to other accessibility concerns.\nPlease migrate to Luna Select.',
				isLuna,
			);
		} else if (
			(target.tagName === 'INPUT' || target.tagName === 'SELECT' || target.tagName === 'TEXTAREA') &&
			!isLuna
		) {
			// eslint-disable-next-line no-console
			console.warn(
				'\nUsing LegacyCompatThisShouldBeLunaField with non-Luna input elements may have remaining accessibility issues with border contrast, placeholder contrast, and focus styles.\nPlease make sure to verify these aspects manually.',
			);
		}

		target.setAttribute('id', this.inputId);
		target.setAttribute('aria-labelledby', `${this.inputId}-label`);
	});

	<template>
		{{yield
			(hash inputId=this.inputId labelId=this.labelId labelContext=this.labelContext inputContext=this.inputContext)
		}}
	</template>
}
