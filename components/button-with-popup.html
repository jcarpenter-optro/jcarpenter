	targetId = `${guidFor(this)}--target`;
	popupId = `${guidFor(this)}--popup`;

	@tracked showPopup = false;

	@action

		this.showPopup = true;

		await timeout(isTesting() ? 0 : 100);

import {
	Children,
	createContext,
	type MutableRefObject,
	type ReactElement,
	type ReactNode,
	type RefObject,
	useCallback,
	useContext,
	useEffect,
	useId,
	useRef,
	useState,
} from 'react';

import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Button, type ButtonProps } from '../button';
import { type OnCloseSource, Popup, type PopupPlacement } from '../popup';

/** Symbol used to identify Button slot when extracting children (avoids string keys like $$). */
export const BWP_BUTTON = Symbol('ButtonWithPopup.Button');
/** Symbol used to identify Popup slot when extracting children. */
export const BWP_POPUP = Symbol('ButtonWithPopup.Popup');

/**
 * Handle provided to popup content for controlling the popup
 */
export interface ButtonWithPopupHandle {
	/** Close the popup */
	close: (source?: OnCloseSource, event?: Event) => Promise<void>;
	/** Open the popup */
	showPopup: () => void;
}

/**
 * Context for sharing popup state between ButtonWithPopup and its subcomponents
 */
interface ButtonWithPopupContextValue {
	/** Whether the popup is currently open */
	isOpen: boolean;
	/** Open the popup */
	showPopup: (event?: React.MouseEvent) => void;
	/** Close the popup */
	hidePopup: () => void;
	/** ID of the trigger button */
	buttonId: string;
	/** ID of the popup */
	popupId: string;
	/** Ref to the trigger button element */
	buttonRef: RefObject<HTMLButtonElement | null>;
	/** Handle for controlling the popup */
	handle: ButtonWithPopupHandle;
	/** Ref to store the onClose callback from PopupContent */
	onCloseCallbackRef: MutableRefObject<((source: OnCloseSource, event: unknown) => void) | null>;
}

const ButtonWithPopupContext = createContext<ButtonWithPopupContextValue | null>(null);

function useButtonWithPopupContext() {
	const context = useContext(ButtonWithPopupContext);
	if (!context) {
		throw new Error('ButtonWithPopup subcomponents must be used within ButtonWithPopup');
	}
	return context;
}

/**
 * Hook to access the ButtonWithPopup handle from within popup content
 */
export function useButtonWithPopupHandle(): ButtonWithPopupHandle {
	const context = useButtonWithPopupContext();
	return context.handle;
}

interface PopupButtonBaseProps extends Omit<
	ButtonProps & { href?: undefined },
	'active' | 'aria-haspopup' | 'aria-expanded' | 'aria-controls'
> {
	/** Children content */
	children?: ReactNode;
}

export type PopupButtonProps = ComponentPropsWithRef<PopupButtonBaseProps, HTMLButtonElement>;

/**
 * The trigger button for ButtonWithPopup.
 * Automatically handles ARIA attributes and active state.
 */
export function PopupButton({ ref, children, onClick, ...props }: PopupButtonProps) {
	const { isOpen, showPopup, buttonId, popupId, buttonRef } = useButtonWithPopupContext();

	const handleClick = useCallback(
		(event: React.MouseEvent<HTMLButtonElement>) => {
			showPopup(event);
			onClick?.(event);
		},
		[showPopup, onClick],
	);

	// Merge refs
	const setRef = useCallback(
		(el: HTMLButtonElement | null) => {
			(buttonRef as React.MutableRefObject<HTMLButtonElement | null>).current = el;
			if (typeof ref === 'function') {
				ref(el);
			} else if (ref && typeof ref === 'object') {
				(ref as React.MutableRefObject<HTMLButtonElement | null>).current = el;
			}
		},
		[buttonRef, ref],
	);

	return (
		<Button
			ref={setRef}
			id={buttonId}
			active={isOpen}
			aria-haspopup="dialog"
			aria-expanded={isOpen}
			aria-controls={popupId}
			data-qid-luna-button-with-popup-button
			onClick={handleClick}
			{...props}
		>
			{children}
		</Button>
	);
}

PopupButton.displayName = 'ButtonWithPopup.Button';
(PopupButton as unknown as Record<symbol, boolean>)[BWP_BUTTON] = true;

export interface PopupContentProps {
	/** Popup placement relative to the button */
	'placement'?: PopupPlacement;
	/** Whether to show an arrow */
	'enableArrow'?: boolean;
	/** Children content - use useButtonWithPopupHandle() hook to access close/showPopup functions */
	'children'?: ReactNode;
	/** Additional class name */
	'className'?: string;
	/** Called when popup is closed */
	'onClose'?: (source: OnCloseSource, event: unknown) => void;
	/** ARIA label for the popup */
	'aria-label'?: string;
	/** ARIA labelledby for the popup */
	'aria-labelledby'?: string;
}

/**
 * The popup content for ButtonWithPopup.
 * Automatically positions relative to the trigger button.
 * Unlike ButtonWithMenu, interacting with content does NOT close the popup.
 * The popup traps focus and closes on ESC or clicking outside.
 */
export function PopupContent({
	placement = 'bottom-start',
	enableArrow,
	children,
	className,
	onClose,
	'aria-label': ariaLabel,
	'aria-labelledby': ariaLabelledBy,
}: PopupContentProps) {
	const { isOpen, hidePopup, popupId, buttonRef, onCloseCallbackRef } = useButtonWithPopupContext();

	// Register the onClose callback in the ref so handle.close() can call it
	useEffect(() => {
		onCloseCallbackRef.current = onClose ?? null;
	}, [onClose, onCloseCallbackRef]);

	const handleClose = useCallback(
		(source: OnCloseSource, event: unknown) => {
			hidePopup();
			onClose?.(source, event);
		},
		[hidePopup, onClose],
	);

	return (
		<Popup
			id={popupId}
			show={isOpen}
			trapFocus={true}
			onClose={handleClose}
			relativeTo={buttonRef}
			enableArrow={enableArrow}
			placement={placement}
			className={className}
			role="dialog"
			aria-label={ariaLabel}
			aria-labelledby={ariaLabelledBy}
			data-test-popup
			data-qid-luna-button-with-popup-popup
		>
			{children}
		</Popup>
	);
}

export interface ButtonWithPopupProps {
	/** Children should be ButtonWithPopup.Button and ButtonWithPopup.Popup components */
	children: ReactNode;
}

/**
 * Extract Button and Popup subcomponents from children
 */
function extractButtonWithPopupSlots(children: ReactNode): {
	buttonSlot: ReactElement | null;
	popupSlot: ReactElement | null;
	otherChildren: ReactNode[];
} {
	let buttonSlot: ReactElement | null = null;
	let popupSlot: ReactElement | null = null;
	const otherChildren: ReactNode[] = [];

	Children.forEach(children, (child) => {
		if (!child || typeof child !== 'object' || !('type' in child)) {
			otherChildren.push(child);
			return;
		}

		const element = child as ReactElement & {
			type: Record<symbol, boolean>;
		};

		if (element.type?.[BWP_BUTTON]) {
			buttonSlot = element;
		} else if (element.type?.[BWP_POPUP]) {
			popupSlot = element;
		} else {
			otherChildren.push(child);
		}
	});

	return { buttonSlot, popupSlot, otherChildren };
}

/**
 * A button used to show a popup with multiple types of content.
 * Interacting with the popup will not close it.
 * The popup traps focus and will close if the user presses ESC or clicks outside of it.
 *
 * @example
 * ```tsx
 * <ButtonWithPopup>
 *   <ButtonWithPopup.Button>Actions</ButtonWithPopup.Button>
 *   <ButtonWithPopup.Popup aria-label="Actions popup">
 *     <p>Content here</p>
 *   </ButtonWithPopup.Popup>
 * </ButtonWithPopup>
 * ```
 */
function ButtonWithPopupComponent({ children }: ButtonWithPopupProps) {
	const id = useId();
	const buttonId = `${id}--target`;
	const popupId = `${id}--popup`;
	const buttonRef = useRef<HTMLButtonElement | null>(null);
	const onCloseCallbackRef = useRef<((source: OnCloseSource, event: unknown) => void) | null>(null);

	const [isOpen, setIsOpen] = useState(false);

	const showPopup = useCallback((event?: React.MouseEvent) => {
		// Toggle behavior - clicking the button when popup is open should close it
		setIsOpen((prev) => !prev);
		event?.stopPropagation();
	}, []);

	const hidePopup = useCallback(() => {
		setIsOpen(false);
	}, []);

	const handle: ButtonWithPopupHandle = {
		close: async (source?: OnCloseSource, event?: Event) => {
			const effectiveSource = source ?? 'manual';
			hidePopup();
			// Also call the onClose callback if registered
			const onClose = onCloseCallbackRef.current;
			if (onClose) onClose(effectiveSource, event);
		},
		showPopup: () => showPopup(),
	};

	const contextValue: ButtonWithPopupContextValue = {
		isOpen,
		showPopup,
		hidePopup,
		buttonId,
		popupId,
		buttonRef,
		handle,
		onCloseCallbackRef,
	};

	const { buttonSlot, popupSlot, otherChildren } = extractButtonWithPopupSlots(children);

	return (
		<ButtonWithPopupContext.Provider value={contextValue}>
			{buttonSlot}
			{popupSlot}
			{otherChildren}
		</ButtonWithPopupContext.Provider>
	);
}

ButtonWithPopupComponent.displayName = 'ButtonWithPopup';

/**
 * ButtonWithPopup with subcomponents attached
 */
export const ButtonWithPopup = Object.assign(ButtonWithPopupComponent, {
	Button: PopupButton,
	Popup: PopupContent,
});

export default ButtonWithPopup;
