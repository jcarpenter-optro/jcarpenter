● React (TSX)
frame.tsx
import { cn } from '../../utils/cn';

import styles from './frame.module.css';

export type HighlightPosition = 'left' | 'top';
export type HighlightColor = 'red' | 'green' | 'orange' | 'blue' | 'gray' | 'teal' | 'purple' | 'yellow' | 'grayA';
export type FrameHighlight = `${HighlightPosition}-${HighlightColor}` | 'none';
export type FramePadding = 'xs' | 'sm' | 'md' | 'lg' | 'xl' | '2xl' | 'none';
export type FrameRadius = 'sm' | 'md' | 'lg';

export interface FrameProps extends React.HTMLAttributes<HTMLDivElement> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** Highlight position and color, e.g., 'left-blue', 'top-red'. Defaults to 'none' */
	highlight?: FrameHighlight;
	/** Padding size. Defaults to 'none' */
	padding?: FramePadding;
	/** Border radius size. Defaults to 'sm' */
	radius?: FrameRadius;
	/** Whether to show border. Defaults to true */
	border?: boolean;
	/** Children content */
	children?: React.ReactNode;
}

/**
 * Provides a frame around content to group it together, or add visual flair to boost its visual hierarchy.
 *
 * Matches the Luna::Frame Ember component API and styling.
 */
export function Frame(props: FrameProps) {
	const {
		ref,
		highlight = 'none',
		padding = 'none',
		radius = 'sm',
		border = true,
		children,
		className,
		...rest
	} = props;

	const frameClassName = cn(styles.Frame, className);

	return (
		<div
			ref={ref}
			className={frameClassName}
			data-highlight={highlight}
			data-padding={padding}
			data-radius={radius}
			data-border={String(border)}
			data-qid-component-type="Luna::Frame"
			{...rest}
		>
			<div className={styles.wrapper}>{children}</div>
		</div>
	);
}

export default Frame;
