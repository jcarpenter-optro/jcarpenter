import type { WithBoundArgs } from '@glint/template';

import { concat, hash } from '@ember/helper';
import { on } from '@ember/modifier';
import { guidFor } from '@ember/object/internals';
import Component from '@glimmer/component';
import { tracked } from '@glimmer/tracking';

import t from 'ember-intl/helpers/t';
import { scopedClass } from 'ember-scoped-css';
import setHelper from 'ember-set-helper/helpers/set';

import tooltip from '../modifiers/tooltip.ts';
import type { LunaAvatarArgs } from './avatar.gts';
import type LunaAvatar from './avatar.gts';
import Avatar from './avatar.gts';
import Popup from './popup.gts';
import Text from './text.gts';
import XStack from './x-stack.gts';
import YStack from './y-stack.gts';

type UserLike = Pick<LunaAvatarArgs, 'image' | 'name' | 'initials' | 'status' | 'bgColor' | 'fullName' | 'id'>;
const LunaAvatarLi = Avatar as WithBoundArgs<typeof Avatar<'li'>, 'class' | 'tag'>;

interface AvatarListSignature {
	Element: HTMLUListElement;
	Args: {
		maxCount: number;
		users: UserLike[];
	};
	Blocks: {
		visible: [
			{
				Avatar: typeof LunaAvatarLi;
				user: UserLike;
			},
		];
		remaining: [
			{
				Avatar: typeof LunaAvatar;
				user: UserLike;
			},
		];
	};
}

/**
 * Renders a list of Avatar components with support for overflow.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/avatar-list.md)
 */
export default class AvatarList extends Component<AvatarListSignature> {
	targetId = `${guidFor(this)}--target`;
	@tracked show = false;

	get visibleUsers() {
		return Boolean(this.args.maxCount) && Boolean(this.args.users.length)
			? this.args.users.slice(0, this.args.maxCount)
			: this.args.users;
	}

	get remainingUsers() {
		return Boolean(this.args.maxCount) && Boolean(this.args.users.length)
			? this.args.users.slice(this.args.maxCount, this.args.users.length)
			: [];
	}

	<template>
		<ul class="AvatarList" data-qid-component-type="Luna::AvatarList" ...attributes>
			{{#each this.visibleUsers as |user|}}
				{{#if (has-block "visible")}}
					{{yield (hash Avatar=(component LunaAvatarLi class=(scopedClass "avatar") tag="li") user=user) to="visible"}}
				{{else}}
					<Avatar
						@image={{user.image}}
						@name={{user.name}}
						@initials={{user.initials}}
						@status={{user.status}}
						@bgColor={{user.bgColor}}
						@tag="li"
						data-avatar-list="true"
						class={{scopedClass "avatar"}}
						{{! The Avatar component has screen reader content that makes the announcement of this tooltip redundant }}
						{{tooltip user.name shouldAnnounce=false}}
					/>
				{{/if}}
			{{/each}}

			{{#if this.remainingUsers.length}}
				<li class="button-wrapper">
					<Avatar
						@tag="button"
						{{! gray-200 }}
						@bgColor="#E2E8F0"
						id={{this.targetId}}
						aria-label={{concat "Show a list of " this.remainingUsers.length " remaining users."}}
						class={{scopedClass "popup-button avatar"}}
						data-test-overflow-users
						{{on "click" (setHelper this "show" true)}}
					>
						+{{this.remainingUsers.length}}
					</Avatar>
				</li>

				<Popup
					@show={{this.show}}
					@trapFocus={{true}}
					@relativeTo={{concat "#" this.targetId}}
					@placement="bottom"
					@onClose={{setHelper this "show" false}}
					class="popup"
					aria-label={{t "avatar-list.additional-users"}}
				>
					<YStack @spacing="md" @tag="ul">
						{{#each this.remainingUsers as |user|}}
							{{#if (has-block "remaining")}}
								{{yield (hash Avatar=Avatar user=user) to="remaining"}}
							{{else}}
								<XStack @tag="li" tabindex="0">
									<Avatar
										@image={{user.image}}
										@name={{user.name}}
										@initials={{user.initials}}
										@status={{user.status}}
										@bgColor={{user.bgColor}}
										aria-hidden="true"
										data-test-overflow-user
									/>
									<Text @font="200">
										{{user.name}}
									</Text>
								</XStack>
							{{/if}}
						{{/each}}
					</YStack>
				</Popup>
			{{/if}}
		</ul>
	</template>
}

import { useCallback, useMemo, useRef, useState } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Avatar, type AvatarProps } from '../avatar';
import { Popup } from '../popup';
import { Text } from '../text';
import { Tooltip } from '../tooltip';
import { XStack } from '../x-stack';
import { YStack } from '../y-stack';

import styles from './avatar-list.module.css';

export type AvatarListUser = {
	id?: AvatarProps['id'];
	image?: AvatarProps['image'];
	name: AvatarProps['name'];
	initials?: AvatarProps['initials'];
	status?: AvatarProps['status'];
	bgColor?: AvatarProps['bgColor'];
	fullName?: AvatarProps['fullName'];
};

interface AvatarListBaseProps {
	/** Maximum number of avatars to show inline */
	maxCount: number;
	/** Array of user objects to display */
	users: AvatarListUser[];
	/** Custom render function for visible users */
	renderVisibleUser?: (user: AvatarListUser, index: number) => React.ReactNode;
	/** Custom render function for remaining users in popup */
	renderRemainingUser?: (user: AvatarListUser, index: number) => React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type AvatarListProps = ComponentPropsWithRef<AvatarListBaseProps, HTMLUListElement>;

/**
 * Displays a list of avatars with overflow handling.
 * Shows the first maxCount users inline with overlapping avatars,
 * and remaining users in a popup triggered by a "+N" button.
 */
export function AvatarList(props: AvatarListProps) {
	const { ref, maxCount, users, renderVisibleUser, renderRemainingUser, className, ...rest } = props;
	const [showPopup, setShowPopup] = useState(false);
	const overflowButtonRef = useRef<HTMLButtonElement>(null);

	// Split users into visible and remaining
	const visibleUsers = useMemo(() => users.slice(0, maxCount), [users, maxCount]);
	const remainingUsers = useMemo(() => users.slice(maxCount), [users, maxCount]);
	const remainingCount = remainingUsers.length;

	const handleOverflowClick = useCallback(() => {
		setShowPopup(true);
	}, []);

	const handlePopupClose = useCallback(() => {
		setShowPopup(false);
	}, []);

	// Default render for visible users
	const defaultRenderVisibleUser = useCallback(
		(user: AvatarListUser) => (
			<Tooltip content={user.name || user.fullName}>
				<Avatar
					image={user.image}
					name={user.name}
					initials={user.initials}
					status={user.status}
					bgColor={user.bgColor}
					id={user.id}
					size="sm"
					className={styles.avatar}
				/>
			</Tooltip>
		),
		[],
	);

	// Default render for remaining users in popup
	const defaultRenderRemainingUser = useCallback(
		(user: AvatarListUser) => (
			<XStack spacing="sm" align="center">
				<Avatar
					image={user.image}
					name={user.name}
					initials={user.initials}
					status={user.status}
					bgColor={user.bgColor}
					id={user.id}
					size="sm"
				/>
				<Text font="200">{user.name || user.fullName}</Text>
			</XStack>
		),
		[],
	);

	return (
		<>
			<ul ref={ref} className={cn(styles.AvatarList, className)} data-qid-component-type="Luna::AvatarList" {...rest}>
				{visibleUsers.map((user, index) => (
					<li key={user.id ?? index} className={styles.avatar}>
						{renderVisibleUser ? renderVisibleUser(user, index) : defaultRenderVisibleUser(user)}
					</li>
				))}
				{remainingCount > 0 && (
					<li className={styles.buttonWrapper}>
						<Avatar
							ref={overflowButtonRef}
							tag="button"
							size="sm"
							bgColor="#E2E8F0"
							className={cn(styles.avatar, styles.popupButton)}
							onClick={handleOverflowClick}
							aria-label={`Show ${remainingCount} more users`}
							data-testid="avatar-list-overflow-button"
						>
							+{remainingCount}
						</Avatar>
					</li>
				)}
			</ul>
			{remainingCount > 0 && (
				<Popup
					show={showPopup}
					relativeTo={overflowButtonRef}
					placement="bottom-start"
					onClose={handlePopupClose}
					className={styles.popup}
					data-testid="avatar-list-popup"
				>
					<YStack spacing="md">
						{remainingUsers.map((user, index) => (
							<div key={user.id ?? index}>
								{renderRemainingUser ? renderRemainingUser(user, index) : defaultRenderRemainingUser(user)}
							</div>
						))}
					</YStack>
				</Popup>
			)}
		</>
	);
}

export default AvatarList;
