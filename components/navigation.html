import type { ElementType, HTMLAttributes, KeyboardEvent, ReactNode } from 'react';
import { createContext, useCallback, useContext, useMemo, useRef } from 'react';

import { cn } from '../../utils/cn';
import { Heading } from '../heading';
import { YStack } from '../y-stack';

import styles from './navigation.module.css';

// ============================================================================
// Navigation Context
// ============================================================================

interface NavigationContextValue {
	/** Current active href/path for highlighting */
	currentPath?: string;
	/** Callback when a link is activated */
	onNavigate?: (href: string) => void;
}

const NavigationContext = createContext<NavigationContextValue>({});

// ============================================================================
// NavigationLink
// ============================================================================

export interface NavigationLinkProps extends Omit<HTMLAttributes<HTMLElement>, 'children'> {
	ref?: React.Ref<HTMLElement>;
	/** The destination URL */
	href?: string;
	/** Explicitly set active state (overrides currentPath check) */
	active?: boolean;
	/** Route names to check for active state */
	currentWhen?: string[];
	/** URLs to check for active state */
	currentWhenUrls?: string[];
	/** Link content */
	children?: ReactNode;
}

/**
 * Navigation link component that renders as an anchor or button.
 */
export function NavigationLink({
	href,
	active,
	currentWhen,
	currentWhenUrls,
	children,
	className,
	onClick,
	ref,
	...rest
}: NavigationLinkProps): React.ReactElement {
	const { currentPath, onNavigate } = useContext(NavigationContext);

	// Determine if this link is active
	const isActive = useMemo(() => {
		// Explicit active prop takes precedence
		if (typeof active === 'boolean') {
			return active;
		}

		// Check currentWhenUrls
		if (currentWhenUrls && currentPath) {
			return currentWhenUrls.includes(currentPath);
		}

		// Check currentWhen (route names - simplified implementation)
		if (currentWhen && currentPath) {
			return currentWhen.some((route) => currentPath.startsWith(route));
		}

		// Check if href matches current path
		if (href && currentPath) {
			return href === currentPath;
		}

		return false;
	}, [active, currentWhenUrls, currentWhen, href, currentPath]);

	const handleClick = useCallback(
		(e: React.MouseEvent<HTMLElement>) => {
			onClick?.(e);
			if (href && onNavigate && !e.defaultPrevented) {
				onNavigate(href);
			}
		},
		[href, onClick, onNavigate],
	);

	const commonProps = {
		'className': cn(styles.Link, className),
		'data-active': String(isActive),
		'onClick': handleClick,
		...rest,
	};

	if (href) {
		return (
			<li>
				<a ref={ref as React.Ref<HTMLAnchorElement>} href={href} {...commonProps}>
					{children}
				</a>
			</li>
		);
	}

	return (
		<li>
			<button ref={ref as React.Ref<HTMLButtonElement>} type="button" {...commonProps}>
				{children}
			</button>
		</li>
	);
}

// ============================================================================
// NavigationSection
// ============================================================================

export interface NavigationSectionProps extends HTMLAttributes<HTMLLIElement> {
	ref?: React.Ref<HTMLLIElement>;
	/** Optional section title */
	title?: string;
	/** Section content (NavigationLinks) */
	children?: ReactNode;
}

/**
 * Groups navigation links with an optional heading.
 */
export function NavigationSection({
	title,
	children,
	className,
	ref,
	...rest
}: NavigationSectionProps): React.ReactElement {
	return (
		<li ref={ref} className={cn(styles.Section, className)} {...rest}>
			{title && (
				<Heading level={3} font="50" className={styles.sectionHeading}>
					{title}
				</Heading>
			)}
			<ul className={styles.sectionList}>{children}</ul>
		</li>
	);
}

// ============================================================================
// Navigation
// ============================================================================

type NavigationTag = 'nav' | 'div' | 'aside';

export interface NavigationProps extends Omit<HTMLAttributes<HTMLElement>, 'children'> {
	ref?: React.Ref<HTMLElement>;
	/** HTML element to render as */
	tag?: NavigationTag;
	/** Current active path for link highlighting */
	currentPath?: string;
	/** Callback when navigation link is clicked */
	onNavigate?: (href: string) => void;
	/** Navigation content */
	children?: ReactNode;
}

/**
 * Container for navigation links with keyboard navigation support.
 *
 * Provides context for child NavigationLink and NavigationSection components.
 *
 * @example
 * ```tsx
 * <Navigation currentPath="/dashboard">
 *   <Navigation.Section title="Main">
 *     <Navigation.Link href="/dashboard">Dashboard</Navigation.Link>
 *     <Navigation.Link href="/settings">Settings</Navigation.Link>
 *   </Navigation.Section>
 * </Navigation>
 * ```
 */
function NavigationRoot({
	tag: Tag = 'nav',
	currentPath,
	onNavigate,
	children,
	className,
	ref,
	...rest
}: NavigationProps): React.ReactElement {
	const containerRef = useRef<HTMLElement>(null);

	// Keyboard navigation for arrow keys
	const handleKeyDown = useCallback((e: KeyboardEvent<HTMLElement>) => {
		const container = containerRef.current;
		if (!container) return;

		const links = Array.from(container.querySelectorAll<HTMLElement>(`.${styles.Link}`));
		const currentIndex = links.findIndex((link) => link === document.activeElement);

		let nextIndex = -1;

		switch (e.key) {
			case 'ArrowDown':
				e.preventDefault();
				nextIndex = currentIndex < links.length - 1 ? currentIndex + 1 : 0;
				break;
			case 'ArrowUp':
				e.preventDefault();
				nextIndex = currentIndex > 0 ? currentIndex - 1 : links.length - 1;
				break;
			case 'Home':
				e.preventDefault();
				nextIndex = 0;
				break;
			case 'End':
				e.preventDefault();
				nextIndex = links.length - 1;
				break;
		}

		if (nextIndex >= 0 && links[nextIndex]) {
			links[nextIndex]?.focus();
		}
	}, []);

	const contextValue = useMemo<NavigationContextValue>(
		() => ({
			currentPath,
			onNavigate,
		}),
		[currentPath, onNavigate],
	);

	// Merge refs
	const mergedRef = useCallback(
		(node: HTMLElement | null) => {
			(containerRef as React.MutableRefObject<HTMLElement | null>).current = node;
			if (typeof ref === 'function') {
				ref(node);
			} else if (ref) {
				ref.current = node;
			}
		},
		[ref],
	);

	const Component = Tag as ElementType;

	return (
		<NavigationContext.Provider value={contextValue}>
			<Component
				ref={mergedRef}
				className={cn(styles.focusContainer, className)}
				onKeyDown={handleKeyDown}
				data-qid-component-type="Luna::Layout::Navigation"
				{...rest}
			>
				<YStack tag="ul" spacing="sm" className={styles.stack}>
					{children}
				</YStack>
			</Component>
		</NavigationContext.Provider>
	);
}

export const Navigation = Object.assign(NavigationRoot, {
	Link: NavigationLink,
	Section: NavigationSection,
});

export default Navigation;
