export interface LunaClickToEditSignature<T> {
	Element: HTMLDivElement;
	Args: {
		id?: string;
		value: T;
		disabled?: boolean;
		onChange: (value: T) => void;
		form?: LunaFormBlockValue;
		placeholder?: string;
		size?: Size;
	};
	Blocks: {
		input: [
			{
				didAutosave: () => void;
				disabled?: boolean;
				form?: LunaFormBlockValue;
				id: string;
				onChange: (value: T) => void;
				placeholder?: string;
				size: Size;
				value: T;
				variant: 'transparent';
			},
		];
		text: [];
	};
}

/**
 * A wrapper component for displaying text that the user can click on to edit.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/click-to-edit.md)
 */
export default class LunaClickToEdit<T> extends Component<LunaClickToEditSignature<T>> {
	@tracked inputId = this.args.id || `${guidFor(this)}--target`;

	@dedupeTracked isEditing = false;

	/**
	 * This property keeps track if the user has initiated the editing
	 * version of the field. If so, then we autofocus back to the fake
	 * input after we exit the editing state.
	 *
	 * This only applied when starting the edit process via the keyboard.
	 */
	@tracked enableAutofocusToText = false;

	/**
	 * Stores original value before user changes so we can reset it
	 * if they decide to cancel via ESC
	 */
	originalValue?: T;

	get showPlaceholder() {
		const value = unwrapProxy(this.args.value);
		return value === null || value === undefined || value === '' || (isArray(value) && value.length === 0);
	}

	@action
	enableEditing(event: MouseEvent) {
		if (this.args.disabled) {
			return;
		}

		if (
			event.target instanceof HTMLAnchorElement ||
			(event.target instanceof HTMLElement && event.target.closest('a'))
		) {
			// We're clicking on a link, so don't enable editing
			return;
		}

		this.originalValue = this.args.value;
		this.isEditing = true;
	}

	@action
	disableEditing() {
		// Don't disable editing if user refocused input during save animation
		if (document.activeElement === document.getElementById(this.inputId)) {
			return;
		}

		this.isEditing = false;

		scheduleOnce('afterRender', this, this.afterDisableEditing);
	}

	afterDisableEditing() {
		this.enableAutofocusToText = false;
	}

	@action
	onChange(value: T) {
		this.args.onChange(value);
	}

	@action
	onInputKeyDown(event: KeyboardEvent) {
		if (event.key === KEYBOARD_KEYS.ESC) {
			// This may trigger a focusout event on the inner input, causing an autosave.
			// We set the data attribute so we can check for that in the focusout and
			// prevent the autosave behavior.
			const inputEl = document.getElementById(this.inputId);
			if (inputEl) {
				inputEl.setAttribute('data-is-esc', 'true');
			}

			// SAFETY: We know it will be `T` since we set it in `enableEditing`
			this.args.onChange(this.originalValue as T);
			this.isEditing = false;
		}
	}

	@action
	onTextKeyDown(event: KeyboardEvent) {
		if (event.key === KEYBOARD_KEYS.ENTER || event.key === KEYBOARD_KEYS.SPACE) {
			event.preventDefault();

			this.isEditing = true;
			this.enableAutofocusToText = true;
		}
	}
	<template>
		{{! template-lint-disable no-invalid-interactive }}
		{{#if this.isEditing}}
			<div ...attributes {{on "keydown" this.onInputKeyDown}} data-editing={{String this.isEditing}}>
				{{yield
					(hash
						didAutosave=this.disableEditing
						disabled=@disabled
						form=@form
						id=this.inputId
						onChange=this.onChange
						placeholder=@placeholder
						size=(or @size "md")
						value=@value
						variant="transparent"
					)
					to="input"
				}}
			</div>
		{{else}}
			<div
				id={{@id}}
				class="text"
				role={{unless @disabled "button"}}
				tabindex={{if @disabled "-1" "0"}}
				data-size={{or @size "md"}}
				data-disabled={{@disabled}}
				data-qid-component-type="Luna::ClickToEdit"
				...attributes
				{{on "click" this.enableEditing}}
				{{on "keydown" this.onTextKeyDown}}
				{{autofocus disabled=(not this.enableAutofocusToText)}}
			>
				<div class="internal">
					{{#if this.showPlaceholder}}
						<span class="placeholder">
							{{@placeholder}}
						</span>
					{{else}}
						{{yield to="text"}}
					{{/if}}
				</div>
			</div>
		{{/if}}
	</template>
}

/**
 * Context for ClickToEdit input rendering.
 */
const ClickToEditContext = createContext<ClickToEditInputContext<unknown> | null>(null);

/**
 * Hook to access ClickToEdit context within Input subcomponent.
 */
export function useClickToEditContext<T = unknown>(): ClickToEditInputContext<T> {
	const context = useContext(ClickToEditContext);
	if (!context) {
		throw new Error('useClickToEditContext must be used within a ClickToEdit component');
	}
	return context as ClickToEditInputContext<T>;
}

/**
 * Optional context accessor for components that can render inside or outside ClickToEdit.
 */
export function useOptionalClickToEditContext<T = unknown>(): ClickToEditInputContext<T> | null {
	return useContext(ClickToEditContext) as ClickToEditInputContext<T> | null;
}

interface ExtractedSlots<T> {
	renderInput?: (context: ClickToEditInputContext<T>) => ReactNode;
	renderText?: () => ReactNode;
	inputSlotProps?: Omit<ClickToEditInputSlotProps<T>, 'children'>;
}

/**
 * Extract Input and Text slots from children (same displayName-based pattern as other compound components).
 */
function extractClickToEditSlots<T>(children: React.ReactNode): ExtractedSlots<T> {
	let renderInput: ((context: ClickToEditInputContext<T>) => ReactNode) | undefined;
	let renderText: (() => ReactNode) | undefined;
	let inputSlotProps: Omit<ClickToEditInputSlotProps<T>, 'children'> | undefined;

	Children.forEach(children, (child) => {
		if (typeof child === 'object' && child !== null && 'type' in child) {
			const displayName = (child.type as { displayName?: string }).displayName;
			if (displayName === 'ClickToEdit.Input') {
				const inputProps = child.props as ClickToEditInputSlotProps<T>;
				const { children: inputChildren, ...restInputProps } = inputProps;
				inputSlotProps = restInputProps;
				if (typeof inputChildren === 'function') {
					renderInput = inputChildren as (context: ClickToEditInputContext<T>) => ReactNode;
				} else if (inputChildren !== undefined) {
					renderInput = () => inputChildren;
				}
			}
			if (displayName === 'ClickToEdit.Text') {
				const textProps = child.props as ClickToEditTextSlotProps;
				const textChildren = textProps.children;
				if (typeof textChildren === 'function') {
					renderText = textChildren as () => ReactNode;
				} else if (textChildren !== undefined) {
					renderText = () => textChildren;
				}
			}
		}
	});

	return { renderInput, renderText, inputSlotProps };
}

export type ClickToEditSize = 'xs' | 'sm' | 'md' | 'lg';

export interface ClickToEditInputContext<T> {
	/** Callback to call when autosave completes - exits edit mode */
	didAutosave: () => void;
	/** Whether the input is disabled */
	disabled?: boolean;
	/** Form context, if provided */
	form?: FormContextValue;
	/** Input ID for accessibility */
	id: string;
	/** Callback when value changes */
	onChange: (value: T) => void;
	/** Placeholder text */
	placeholder?: string;
	/** Size of the input */
	size: ClickToEditSize;
	/** Current value */
	value: T;
	/** Visual variant for the input */
	variant: 'transparent';
}

export interface ClickToEditInputSlotProps<T = unknown> extends Omit<React.HTMLAttributes<HTMLDivElement>, 'children'> {
	children?: ReactNode | ((context: ClickToEditInputContext<T>) => ReactNode);
}

export interface ClickToEditTextSlotProps {
	children?: ReactNode | (() => ReactNode);
}

export interface ClickToEditProps<T> {
	ref?: React.Ref<HTMLDivElement>;
	/** Optional ID for the component */
	id?: string;
	/** Current value */
	value: T;
	/** Whether the component is disabled */
	disabled?: boolean;
	/** Callback when value changes */
	onChange: (value: T) => void;
	/** Form context for autosave integration */
	form?: FormContextValue;
	/** Placeholder text when value is empty */
	placeholder?: string;
	/** Size of the component */
	size?: ClickToEditSize;
	/** Render function for the input when editing (legacy prop, prefer ClickToEdit.Input) */
	renderInput?: (context: ClickToEditInputContext<T>) => ReactNode;
	/** Render function for the text when not editing (legacy prop, prefer ClickToEdit.Text) */
	renderText?: () => ReactNode;
	/** Child elements, typically ClickToEdit.Input and ClickToEdit.Text */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

/**
 * Checks if a value should be considered "empty" for placeholder purposes.
 */
function isEmptyValue<T>(value: T): boolean {
	if (value === null || value === undefined || value === '') {
		return true;
	}
	if (Array.isArray(value) && value.length === 0) {
		return true;
	}
	return false;
}

const KEYBOARD_KEYS = {
	ENTER: 'Enter',
	SPACE: ' ',
	ESC: 'Escape',
} as const;

/**
 * A wrapper component for displaying text that the user can click on to edit.
 *
 * Matches the Luna::ClickToEdit Ember component API.
 */
function ClickToEditInner<T>({
	id: providedId,
	value,
	disabled = false,
	onChange,
	form,
	placeholder,
	size = 'md',
	renderInput: renderInputProp,
	renderText: renderTextProp,
	children,
	className,
	ref,
	...rest
}: ClickToEditProps<T> & Omit<React.HTMLAttributes<HTMLDivElement>, 'onChange'>) {
	// Extract slots from children
	const {
		renderInput: renderInputSlot,
		renderText: renderTextSlot,
		inputSlotProps,
	} = extractClickToEditSlots<T>(children);

	// Use slots if provided, otherwise fallback to props
	const renderInput = renderInputSlot ?? renderInputProp;
	const renderText = renderTextSlot ?? renderTextProp;

	if (!renderText) {
		throw new Error('ClickToEdit requires renderText prop or ClickToEdit.Text children');
	}

	const generatedId = useId();
	const inputId = `${providedId || generatedId}--target`;
	const [isEditing, setIsEditing] = useState(false);
	const [enableAutofocusToText, setEnableAutofocusToText] = useState(false);
	const originalValueRef = useRef<T>(value);
	const textRef = useRef<HTMLDivElement>(null);

	const showPlaceholder = isEmptyValue(value);

	// Autofocus the input element when entering edit mode.
	// Note: We experienced focus being lost when switching from an iframe to this input
	// (e.g. compare view with Ember in iframe). Not considered a real issue for now.
	useEffect(() => {
		if (isEditing) {
			const inputElement = document.getElementById(inputId);
			if (inputElement) {
				inputElement.focus();
			}
		}
	}, [isEditing, inputId]);

	// Autofocus back to text element after exiting edit mode via keyboard
	useEffect(() => {
		if (!isEditing && enableAutofocusToText && textRef.current) {
			textRef.current.focus();
			// Reset the flag after focusing
			const timeoutId = setTimeout(() => {
				setEnableAutofocusToText(false);
			}, 0);
			return () => clearTimeout(timeoutId);
		}
	}, [isEditing, enableAutofocusToText]);

	const handleEnableEditing = useCallback(
		(event: React.MouseEvent<HTMLDivElement>) => {
			if (disabled) {
				return;
			}

			// Don't enable editing if clicking on a link
			if (
				event.target instanceof HTMLAnchorElement ||
				(event.target instanceof HTMLElement && event.target.closest('a'))
			) {
				return;
			}

			originalValueRef.current = value;
			setIsEditing(true);
		},
		[disabled, value],
	);

	const handleDisableEditing = useCallback(() => {
		// Don't disable editing if user refocused input during save animation
		const inputElement = document.getElementById(inputId);
		if (document.activeElement === inputElement) {
			return;
		}

		setIsEditing(false);
	}, [inputId]);

	const handleChange = useCallback(
		(newValue: T) => {
			onChange(newValue);
		},
		[onChange],
	);

	const handleInputKeyDown = useCallback(
		(event: React.KeyboardEvent<HTMLDivElement>) => {
			if (event.key === KEYBOARD_KEYS.ESC) {
				// Set data attribute so we can check for it in the focusout handler
				// and prevent autosave behavior
				const inputEl = document.getElementById(inputId);
				if (inputEl) {
					inputEl.setAttribute('data-is-esc', 'true');
				}

				// Reset to original value
				onChange(originalValueRef.current);
				setIsEditing(false);
			}
		},
		[inputId, onChange],
	);

	const handleTextKeyDown = useCallback(
		(event: React.KeyboardEvent<HTMLDivElement>) => {
			if (event.key === KEYBOARD_KEYS.ENTER || event.key === KEYBOARD_KEYS.SPACE) {
				event.preventDefault();

				originalValueRef.current = value;
				setIsEditing(true);
				setEnableAutofocusToText(true);
			}
		},
		[value],
	);

	const inputContext: ClickToEditInputContext<T> = {
		didAutosave: handleDisableEditing,
		disabled,
		form,
		id: inputId,
		onChange: handleChange,
		placeholder,
		size,
		value,
		variant: 'transparent',
	};

	if (isEditing) {
		const defaultInputRenderer = (context: ClickToEditInputContext<T>) => {
			const { children: _ignoredChildren, ...slotProps } = (inputSlotProps ?? {}) as Record<string, unknown>;
			const dataTestId =
				typeof slotProps['data-testid'] === 'string' ? (slotProps['data-testid'] as string) : undefined;
			return (
				<div {...(slotProps as React.HTMLAttributes<HTMLDivElement>)}>
					<div
						className={inputStyles.TextInput}
						data-variant={context.variant}
						data-size={context.size}
						data-textarea="false"
						data-disabled="false"
						data-luna-feature="meld"
						data-qid-component-type="Luna::Input"
						data-focused="true"
						data-errored="false"
					>
						<input
							id={context.id}
							className={inputStyles.input}
							value={String(context.value ?? '')}
							placeholder={context.placeholder}
							onChange={(event) => context.onChange(event.target.value as unknown as T)}
							onBlur={() => context.didAutosave()}
							data-testid={dataTestId ? `${dataTestId}-input` : undefined}
						/>
					</div>
				</div>
			);
		};
		return (
			<ClickToEditContext.Provider value={inputContext as ClickToEditInputContext<unknown>}>
				<div
					ref={ref}
					className={className}
					onKeyDown={handleInputKeyDown}
					data-qid-component-type="Luna::ClickToEdit"
					data-editing="true"
					{...rest}
				>
					{(renderInput ?? defaultInputRenderer)(inputContext)}
				</div>
			</ClickToEditContext.Provider>
		);
	}

	return (
		<div
			ref={(node) => {
				// Handle both the forwarded ref and our internal ref
				textRef.current = node;
				if (typeof ref === 'function') {
					ref(node);
				} else if (ref) {
					ref.current = node;
				}
			}}
			id={providedId}
			className={cn(styles.text, className)}
			role={disabled ? undefined : 'button'}
			tabIndex={disabled ? -1 : 0}
			data-size={size}
			data-disabled={disabled ? 'true' : undefined}
			data-qid-component-type="Luna::ClickToEdit"
			onClick={handleEnableEditing}
			onKeyDown={handleTextKeyDown}
			{...rest}
		>
			<div className={styles.internal}>
				{showPlaceholder ? <span className={styles.placeholder}>{placeholder}</span> : renderText()}
			</div>
		</div>
	);
}

/**
 * A wrapper component for displaying text that the user can click on to edit.
 *
 * Matches the Luna::ClickToEdit Ember component API.
 *
 * @example
 * ```tsx
 * // Using compound components (recommended)
 * <ClickToEdit value={text} onChange={setText}>
 *   <ClickToEdit.Input>
 *     {(ctx) => <Input id={ctx.id} value={ctx.value} onChange={ctx.onChange} />}
 *   </ClickToEdit.Input>
 *   <ClickToEdit.Text>
 *     {() => <span>{text}</span>}
 *   </ClickToEdit.Text>
 * </ClickToEdit>
 *
 * // Using legacy render props (deprecated)
 * <ClickToEdit
 *   value={text}
 *   onChange={setText}
 *   renderInput={(ctx) => <Input id={ctx.id} value={ctx.value} onChange={ctx.onChange} />}
 *   renderText={() => <span>{text}</span>}
 * />
 * ```
 */
export const ClickToEdit = Object.assign(ClickToEditInner, {
	Input,
	Text,
});

export default ClickToEdit;
