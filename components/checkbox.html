export interface LunaCheckboxModifiers {
	class?: string;
	size?: 'sm' | 'md' | 'lg';
	ariaLabel?: string;
	id?: string;
	requirements?: Requirements;
	form?: LunaFormBlockValue;
	name?: string;

interface LunaCheckboxSignature<Value extends AttrValue> {
	Element: HTMLDivElement;
	Args: LunaCheckboxModifiers & {
		onChange?: (value: boolean, event: Event) => unknown;
		value: Value;
	};
	Blocks: {
		default?: [];
	};
}

let nextInputId = 0;

/**
 * A checkbox. Similar to a toggle, but supports a mixed state.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/checkbox.md)
 */
export default class Checkbox<Value extends AttrValue> extends Component<LunaCheckboxSignature<Value>> {
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;
	@consume('luna-input-id') inputId!: string;

	labelId = `${guidFor(this)}--label`;

	manager: InputManager;

	constructor(owner: Owner, args: LunaCheckboxSignature<Value>['Args']) {
		super(owner, args);
		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;
		const managerOptions: InputManagerOptions = {
			mode: 'change',
			name: args.name,

			focus: this.focusInput.bind(this),

			ariaDescribedBy: args.ariaDescribedBy ?? null,

			get value(): Value {
				return args.value;
			},

			get isDisabled(): boolean {
				return args.disabled ?? false;
			},

			get requirements() {
				return args.requirements;
			},

			get formContext() {
				return that.formContext;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	get id() {
		return this.args.id ?? this.inputId ?? `luna-checkbox-${nextInputId++}`;
	}

	/**
	 * Converts the provided value to an appropriate string to assign to the
	 * aria-checked attribute.
	 */
	get value() {
		const value = this.args.value;

		if (typeof value === 'boolean') {
			return value.toString();
		}

		if (value === undefined || value === null) {
			return 'false';
		}

		return 'mixed';
	}

	@action
	async toggle(event: Event) {
		if (this.args.disabled) {
			return;
		}

		// This ensures a indeterminate value goes to true on click
		const value = typeof this.args.value === 'boolean' ? !this.args.value : true;
		await this.args.onChange?.(value, event);
		this.manager.onChange();
	}

	@action
	onKeyDown(event: KeyboardEvent) {
		if (event.key === KEYBOARD_KEYS.SPACE) {
			event.preventDefault();
			void this.toggle(event);
		}
	}

	focusInput = () => {
		focusById(this.id);
	};
	<template>
		{{! template-lint-disable require-presentational-children }}
		<div
			class="CheckboxWrapper {{@class}}"
			data-qid-component-type="Luna::Checkbox"
			data-focused={{toString0 this.manager.inputHasFocus}}
			data-errored={{toString0 this.manager.showErrorMessage}}
		>
			<div
				id={{this.id}}
				class="Checkbox"
				tabindex="0"
				role="checkbox"
				aria-checked={{this.value}}
				aria-readonly={{toString0 @disabled}}
				aria-label={{@ariaLabel}}
				aria-labelledby={{if (has-block) this.labelId}}
				aria-describedby={{this.manager.ariaDescribedBy}}
				aria-invalid={{toString0 this.manager.showErrorMessage}}
				data-size={{or @size "md"}}
				data-checkbox-element-container
				...attributes
				{{on "click" this.toggle}}
				{{on "keydown" this.onKeyDown}}
			>

				<Element
					@value={{this.value}}
					@size={{or @size "md"}}
					@disabled={{@disabled}}
					@invalid={{this.manager.showErrorMessage}}
					data-qid-component-type="Luna::CheckboxElement"
					class="element"
				/>

				{{#if (has-block)}}
					<div id={{this.labelId}} data-test-checkbox-label>
						{{yield}}

						{{#if (or this.manager.isAutosaving this.manager.didAutosaveSuccessfully)}}
							<Spinner @size="sm" @transitionTo={{if this.manager.didAutosaveSuccessfully "success"}} class="spinner" />
						{{/if}}
					</div>
				{{/if}}
			</div>

			<InputError id={{this.manager.errorId}} @showWhen={{this.manager.showErrorMessage}} @align="left">
				{{this.manager.errorMessageToDisplay}}
			</InputError>
		</div>
	</template>
}

export type CheckboxElementSize = 'sm' | 'md' | 'lg';

export interface CheckboxElementProps extends React.HTMLAttributes<HTMLDivElement> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** Checkbox value state */
	value: boolean | 'mixed' | undefined;
	/** Whether checkbox is disabled */
	disabled?: boolean;
	/** Whether checkbox is in invalid state */
	invalid?: boolean;
	/** Size of the checkbox */
	size?: CheckboxElementSize;
}

/**
 * A visual-only checkbox element.
 * Used by CheckboxTree for rendering checkbox appearance without form integration.
 */
export function CheckboxElement(props: CheckboxElementProps) {
	const { ref, value, disabled = false, invalid = false, size = 'md', className, ...rest } = props;
	// Convert value to string for data attribute
	const dataChecked = value === 'mixed' ? 'mixed' : value === true ? 'true' : 'false';

	return (
		<div
			ref={ref}
			className={cn(styles.checkboxElement, className)}
			data-checked={dataChecked}
			data-readonly={disabled ? 'true' : undefined}
			data-invalid={invalid ? 'true' : undefined}
			data-size={size}
			data-test-checkbox-element
			{...rest}
		>
			<div className={styles.inner}>
				{/* Check icon */}
				<svg className={styles.check} xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
					<path
						fill="currentColor"
						fillRule="evenodd"
						d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z"
						clipRule="evenodd"
					/>
				</svg>

checkbox.tsx
import { useCallback, useEffect, useId, useRef, useState } from 'react';

import { Checkbox as BaseCheckbox } from '@base-ui-components/react/checkbox';
import type { Requirements } from '@soxhub/luna-utils/validators';

import { cn } from '../../utils/cn';
import { useInputManager } from '../../utils/use-input-manager';
import { useFieldContext } from '../field/field-context';
import { useFormContext } from '../form/form';
import { InputError } from '../input-error';
import { Spinner } from '../spinner';

import { CheckboxElement } from './checkbox-element';

import styles from './checkbox.module.css';

export type CheckboxSize = 'sm' | 'md' | 'lg';

interface CheckboxBaseProps {
	/** Forwarded ref to the checkbox root element */
	'ref'?: React.Ref<HTMLDivElement>;
	/** Checkbox value (boolean or undefined for mixed/indeterminate state) */
	'value': boolean | undefined;
	/** Callback fired when value changes */
	'onChange'?: (value: boolean, event: Event) => void;
	/** Checkbox size */
	'size'?: CheckboxSize;
	/** Whether checkbox is disabled */
	'disabled'?: boolean;
	/** Checkbox ID (auto-generated if not provided) */
	'id'?: string;
	/** Checkbox name */
	'name'?: string;
	/** ARIA label (for non-label scenarios) */
	'ariaLabel'?: string;
	/** ARIA described by */
	'aria-describedby'?: string;
	/** Additional CSS class */
	'className'?: string;
	/** Children (label content) */
	'children'?: React.ReactNode;
	/** Validation requirements */
	'requirements'?: Requirements;
	/** @deprecated Server errors (use externalErrorsFor on Form instead) */
	'serverErrors'?: Array<{ message?: string } | string>;
	/** Callback fired when autosave completes successfully */
	'didAutosave'?: () => void;
}

type CheckboxProps = CheckboxBaseProps &
	Omit<React.HTMLAttributes<HTMLDivElement>, keyof CheckboxBaseProps | 'onChange'>;

/**
 * A checkbox. Similar to a toggle, but supports a mixed state.
 *
 * Matches the Luna::Checkbox Ember component API and styling.
 * Uses Base UI Checkbox as the foundation.
 */
export function Checkbox(props: CheckboxProps) {
	const {
		ref,
		value,
		onChange,
		size = 'md',
		disabled = false,
		'id': providedId,
		name,
		ariaLabel,
		'aria-describedby': providedAriaDescribedBy,
		className,
		children,
		requirements,
		serverErrors,
		didAutosave,
		...rest
	} = props;

	const generatedId = useId();
	const fieldContext = useFieldContext();
	const id = providedId ?? fieldContext?.inputId ?? generatedId;
	const labelId = fieldContext?.labelId ?? `${id}--label`;
	const formContext = useFormContext();
	const checkboxRef = useRef<HTMLDivElement>(null);
	const valueIsUnchangedRef = useRef(true);

	const focusInput = useCallback(() => {
		if (checkboxRef.current) {
			checkboxRef.current.focus();
		} else if (ref && typeof ref === 'object' && ref.current) {
			ref.current.focus();
		}
	}, [ref]);

	const manager = useInputManager({
		mode: 'change',
		valueIsUnchanged: valueIsUnchangedRef.current,
		serverErrors,
		requirements,
		name,
		value,
		isDisabled: disabled,
		formContext,
		ariaDescribedBy: providedAriaDescribedBy ?? null,
		focus: focusInput,
		didAutosave: () => {
			valueIsUnchangedRef.current = true;
			didAutosave?.();
		},
	});

	// Register/unregister with form
	const managerRef = useRef(manager);
	managerRef.current = manager;

	useEffect(() => {
		if (formContext) {
			formContext.actions.registerInputManager(managerRef.current);
			return () => {
				formContext.actions.unregisterInputManager(managerRef.current);
			};
		}
	}, [formContext]);

	const [focused, setFocused] = useState(false);
	const focusedState = focused || manager.inputHasFocus;

	// Convert value to checked state and indeterminate
	const checked = value === true;
	const indeterminate = value === undefined;

	const handleCheckedChange = useCallback(
		(checked: boolean) => {
			valueIsUnchangedRef.current = false;
			manager.setValueIsUnchanged(false);
			// When indeterminate, clicking should go to true
			const newValue = indeterminate ? true : checked;
			onChange?.(newValue, new Event('change'));
			manager.onChange();
		},
		[onChange, indeterminate, manager],
	);

	const handleKeyDown = useCallback(
		(event: React.KeyboardEvent) => {
			if (event.key === ' ') {
				event.preventDefault();
				handleCheckedChange(!checked);
			}
		},
		[handleCheckedChange, checked],
	);

	const handleFocus = useCallback(() => {
		setFocused(true);
		manager.focusIn();
	}, [manager]);

	const handleBlur = useCallback(() => {
		setFocused(false);
		manager.focusOutAndTrySave(new Event('blur'));
	}, [manager]);

	const wrapperClassName = cn(styles.CheckboxWrapper, className);
	const checkboxClassName = cn(styles.Checkbox, disabled && styles.disabled);

	// Convert value to aria-checked string
	const ariaChecked = value === true ? 'true' : value === false ? 'false' : 'mixed';

	return (
		<div className={wrapperClassName} data-qid-component-type="Luna::Checkbox" ref={ref} {...rest}>
			<BaseCheckbox.Root
				checked={checked}
				indeterminate={indeterminate}
				disabled={disabled}
				onCheckedChange={handleCheckedChange}
				id={id}
				render={(checkboxProps) => (
					<div
						{...checkboxProps}
						className={checkboxClassName}
						tabIndex={disabled ? -1 : 0}
						role="checkbox"
						aria-checked={ariaChecked}
						aria-readonly={disabled ? 'true' : undefined}
						aria-label={ariaLabel}
						aria-labelledby={children ? labelId : undefined}
						aria-describedby={manager.ariaDescribedBy ?? providedAriaDescribedBy}
						aria-invalid={manager.showErrorMessage ? 'true' : undefined}
						data-size={size}
						data-checkbox-element-container
						data-focused={focusedState ? 'true' : 'false'}
						data-errored={manager.showErrorMessage ? 'true' : 'false'}
						onKeyDown={handleKeyDown}
						onFocus={handleFocus}
						onBlur={handleBlur}
					>
						<div className={styles.checkboxElementContainer}>
							<CheckboxElement
								value={value === undefined ? 'mixed' : value}
								size={size}
								disabled={disabled}
								invalid={manager.showErrorMessage}
							/>
						</div>
						{children && (
							<div id={labelId} data-test-checkbox-label>
								{children}
								{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
									<Spinner
										size="sm"
										transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined}
										className={styles.spinner}
									/>
								)}
							</div>
						)}
					</div>
				)}
			/>

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage} align="left">
				{manager.errorMessageToDisplay}
			</InputError>
		</div>
	);
}

export default Checkbox;
