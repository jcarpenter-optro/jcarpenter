import Component from '@glimmer/component';

import { consume } from 'ember-provide-consume-context';

import type { StructuredErrorDocument } from '@auditboard/warp-drive/v1/core-types/request';
import type { CollectionResourceDataDocument } from '@auditboard/warp-drive/v1/core-types/spec/document';
import type { StoreRequestInput } from '@auditboard/warp-drive/v1/store';
import type { Store } from '@auditboard/warp-drive/v2/core';
import { Request } from '@auditboard/warp-drive/v2/ember';

import EventListItem from './event-list-item.gts';

interface EventListSignature<V> {
	Element: HTMLElement;
	Args: {
		query: StoreRequestInput<CollectionResourceDataDocument<V>>;
	};
	Blocks: {
		event: [V];
		error?: [StructuredErrorDocument<unknown>];
		empty: [];
	};
}

/**
 * An ordered list of critical events that require user attention.
 * The list items highlight tasks based on severity, urgency, or date with a link to the relevant area to resolve.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/event-list.md)
 */
export class EventList<V> extends Component<EventListSignature<V>> {
	@consume('store') declare private store: Store;

	<template>
		<div class="luna-event-list" data-qid-luna-event-list ...attributes>
			<Request @query={{@query}} @store={{this.store}}>
				<:loading>
					<ul class="list" data-qid-event-list-loading>
						<EventListItem @isLoading={{true}} />
						<EventListItem @isLoading={{true}} />
						<EventListItem @isLoading={{true}} />
						<EventListItem @isLoading={{true}} />
					</ul>
				</:loading>
				<:error as |error|>
					<div class="empty-state">
						{{yield error to="error"}}
					</div>
				</:error>
				<:content as |result|>
					{{#if result.data}}
						<ol data-qid-luna-event-list-content class="list">
							{{#each result.data as |event|}}
								{{yield event to="event"}}
							{{/each}}
						</ol>
					{{else}}
						<div class="empty-state" data-qid-luna-event-list-empty>{{yield to="empty"}}</div>
					{{/if}}
				</:content>
			</Request>
		</div>
	</template>
}

export default EventList;

import { createContext, forwardRef, type ReactNode, useContext } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { EventListItem } from '../event-list-item';

import styles from './event-list.module.css';

export interface EventListContextValue<T = unknown> {
	items?: T[];
	loading: boolean;
	error: Error | string | null;
}

const EventListContext = createContext<EventListContextValue<unknown> | null>(null);

/**
 * Hook to access EventList context (items, loading, error state)
 */
export function useEventListContext<T = unknown>(): EventListContextValue<T> {
	const context = useContext(EventListContext);
	if (!context) {
		throw new Error('useEventListContext must be used within an EventList component');
	}
	return context as EventListContextValue<T>;
}

export interface EventListProps<T = unknown> {
	/** Array of items to render */
	items?: T[];
	/** Render function for each item */
	renderItem?: (item: T, index: number) => ReactNode;
	/** Whether the list is in a loading state */
	loading?: boolean;
	/** Error to display (if any) */
	error?: Error | string | null;
	/** Content to render for the error state (receives the error value) */
	renderError?: (error: Error | string) => ReactNode;
	/** Content to render when items is empty */
	emptyContent?: ReactNode;
	/** Content to render for the loading state (defaults to skeleton items) */
	loadingContent?: ReactNode;
	/** Number of skeleton items to show while loading (default: 4) */
	skeletonCount?: number;
	/** Additional CSS class */
	className?: string;
}

export type EventListComponentProps<T = unknown> = ComponentPropsWithRef<EventListProps<T>, HTMLDivElement>;

/**
 * An ordered list of critical events that require user attention.
 * The list items highlight tasks based on severity, urgency, or date with a link to the relevant area to resolve.
 *
 * Automatically handles loading, error, and empty states.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/event-list.md)
 *
 * @example
 * ```tsx
 * <EventList
 *   items={events}
 *   renderItem={(event) => <EventListItem key={event.id} title={event.title} icon="alert" />}
 *   loading={isLoading}
 *   error={error}
 *   renderError={(err) => <div>Failed: {typeof err === 'string' ? err : err.message}</div>}
 *   emptyContent={<div>No events to display</div>}
 * />
 * ```
 */
function EventListRoot<T>(props: EventListComponentProps<T>, ref: React.ForwardedRef<HTMLDivElement>) {
	const {
		items,
		renderItem,
		loading = false,
		error = null,
		renderError,
		emptyContent,
		loadingContent,
		skeletonCount = 4,
		className,
		...rest
	} = props;

	const contextValue: EventListContextValue<T> = { items, loading, error };

	if (loading) {
		return (
			<EventListContext.Provider value={contextValue}>
				<div ref={ref} className={cn(styles.lunaEventList, className)} data-qid-luna-event-list {...rest}>
					<ul className={styles.list} data-qid-event-list-loading>
						{loadingContent ??
							Array.from({ length: skeletonCount }).map((_, index) => <EventListItem key={index} isLoading />)}
					</ul>
				</div>
			</EventListContext.Provider>
		);
	}

	if (error) {
		return (
			<EventListContext.Provider value={contextValue}>
				<div ref={ref} className={cn(styles.lunaEventList, className)} data-qid-luna-event-list {...rest}>
					<div className={styles.emptyState} data-qid-luna-event-list-error>
						{renderError ? renderError(error) : <span>{typeof error === 'string' ? error : error.message}</span>}
					</div>
				</div>
			</EventListContext.Provider>
		);
	}

	if (!items || items.length === 0) {
		return (
			<EventListContext.Provider value={contextValue}>
				<div ref={ref} className={cn(styles.lunaEventList, className)} data-qid-luna-event-list {...rest}>
					<div className={styles.emptyState} data-qid-luna-event-list-empty>
						{emptyContent ?? null}
					</div>
				</div>
			</EventListContext.Provider>
		);
	}

	return (
		<EventListContext.Provider value={contextValue}>
			<div ref={ref} className={cn(styles.lunaEventList, className)} data-qid-luna-event-list {...rest}>
				<ol data-qid-luna-event-list-content className={styles.list}>
					{renderItem ? items.map((item, index) => renderItem(item, index)) : null}
				</ol>
			</div>
		</EventListContext.Provider>
	);
}

export const EventList = forwardRef(EventListRoot) as <T>(
	props: EventListComponentProps<T> & { ref?: React.ForwardedRef<HTMLDivElement> },
) => React.ReactElement;

export default EventList;
