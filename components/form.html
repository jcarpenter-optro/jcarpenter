import { on } from '@ember/modifier';
import { action } from '@ember/object';
import { next } from '@ember/runloop';
import Component from '@glimmer/component';

import fade from 'ember-animated/transitions/fade';
import { task } from 'ember-concurrency';
import perform from 'ember-concurrency/helpers/perform';
import { provide } from 'ember-provide-consume-context';

import { isFormError } from '../utils/error.ts';
import type InputManager from '../utils/input-manager.ts';
import { isVisibleInViewport } from '../utils/scroll.ts';
import Animation from './animation.gts';
import Notice from './notice.gts';

export interface LunaFormBlockValue {
	actions: {
		onSubmit: () => Promise<{ errorDuringSave: boolean }>;
		validate: () => { errorDuringSave: boolean } | undefined;
		registerInputManager: (manager: InputManager) => void;
		unregisterInputManager: (manager: InputManager) => void;

export interface LunaFormSignature {
	Element: HTMLFormElement;
	Args: {
		autosave?: boolean;
		onSubmit: (event?: SubmitEvent, ...additionalArgs: unknown[]) => Promise<void> | void;
		externalErrorsFor?: (name: string) => Array<{ message?: string } | string> | null | undefined;
	};
	Blocks: {
		default: [LunaFormBlockValue];
	};
}

/**
 * A wrapper around the HTML form. It will ensure its child inputs have valid data before allowing the user to submit it.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/form.md)
 */
export default class LunaForm extends Component<LunaFormSignature> {
	@provide('luna-form')
	get formApi() {
		return {
			actions: {
				validate: this.validate.bind(this),
				onSubmit: this.onSubmit.perform.bind(this),
				registerInputManager: this.registerInputManager.bind(this),
				unregisterInputManager: this.unregisterInputManager.bind(this),
			},
			autosave: this.args.autosave,
			isLoading: this.onSubmit.isRunning,
			externalErrorsFor: this.externalErrorsFor.bind(this),
		};
	}

	fadeTransition = fade;

	/**
	 * Hold the error returned from the onSubmit action
	 */
	@tracked onSubmitError: string | null = null;

	/**
	 * A registry of all child input components.
	 */
	inputManagers = new Set<InputManager>();

	@action
	registerInputManager(manager: InputManager) {
		this.inputManagers.add(manager);
	}

	@action
	unregisterInputManager(manager: InputManager) {
		this.inputManagers.delete(manager);
	}

	get hasSubmitError() {
		return Boolean(this.onSubmitError);
	}

	// Should we just pass this directly?
	@action
	externalErrorsFor(name: string) {
		return this.args.externalErrorsFor?.(name);
	}

	private validate = () => {
		let invalidInput: InputManager | undefined;
		this.inputManagers.forEach((manager) => {
			manager.showErrorIfPresent = true;
			if (!invalidInput && manager.isInvalid) {
				invalidInput = manager;
			}
		});

		if (invalidInput) {
			if (!this.args.autosave) {
				invalidInput.focus();
			}
			return { errorDuringSave: true };
		}
	};

	/**
	 * Action that is called when the form is submitted. If none of the
	 * child components have any errors then it calls the onSubmit action.
	 *
	 * @param event The event that triggered the form submission
	 */
	onSubmit = task(async (event?: SubmitEvent, ...additionalArgs: unknown[]) => {
		event?.preventDefault?.();

		const validation = this.validate();

		if (validation !== undefined) {
			return validation;
		}

		this.onSubmitError = null;

		try {
			await this.args.onSubmit(event, ...additionalArgs);

			/**
			 * This resets the form so it can be reused without error
			 * messages immediately showing up when all the fields are
			 * reset to empty.
			 */
			this.inputManagers.forEach((manager) => {
				manager.showErrorIfPresent = false;
			});

			return { errorDuringSave: false };
		} catch (error: unknown) {
			if (isFormError(error)) {
				if (error.hasGlobalError) {
					this.onSubmitError = error.message;

					next(this, function () {
						const errorElement = document.querySelector('[data-form-error]');

						if (errorElement && !isVisibleInViewport(errorElement)) {
							errorElement.scrollIntoView({
								behavior: 'smooth',
							});
						}
					});
				}
			} else {
				// We got an unexpected error, so we re-throw it
				throw error;
			}

			return { errorDuringSave: true };
		}
	});
	<template>
		<form
			class="Form"
			{{on "submit" (perform this.onSubmit)}}
			novalidate
			data-qid-component-type="Luna::Form"
			...attributes
		>

			<Animation @animation="slide-and-fade" @show={{this.hasSubmitError}}>
				<Notice class="error" @variant="red" data-form-error data-qid-form-error>
					{{!
          We set this to unbound because we don't want to text to disappear
          during the out animation.
        }}
					{{! template-lint-disable no-unbound }}
					{{unbound this.onSubmitError}}
				</Notice>
			</Animation>

			{{yield this.formApi}}
		</form>
	</template>
}

import { createContext, type ReactNode, useCallback, useContext, useRef, useState } from 'react';

import type { InputManager } from '../../utils/use-input-manager';

export interface FormContextValue {
	actions: {
		onSubmit: () => Promise<{ errorDuringSave: boolean }>;
		validate: () => { errorDuringSave: boolean } | undefined;
		registerInputManager: (manager: InputManager) => void;
		unregisterInputManager: (manager: InputManager) => void;
	};
	autosave: boolean | undefined;
	isLoading: boolean;
	externalErrorsFor: (name: string) => Array<{ message?: string } | string> | null | undefined;
}

const FormContext = createContext<FormContextValue | undefined>(undefined);

export function useFormContext(): FormContextValue | undefined {
	return useContext(FormContext);
}

interface FormProps {
	/** Whether to enable autosave */
	autosave?: boolean;
	/** Callback fired when form is submitted */
	onSubmit: (event?: SubmitEvent, ...additionalArgs: unknown[]) => Promise<void> | void;
	/** Function to get external errors for a field name */
	externalErrorsFor?: (name: string) => Array<{ message?: string } | string> | null | undefined;
	/** Children (use useFormContext() hook to access form context) */
	children: ReactNode | ((formContext: FormContextValue) => ReactNode);
	/** Additional CSS class */
	className?: string;
}

/**
 * A wrapper around the HTML form. It will ensure its child inputs have valid data before allowing the user to submit it.
 *
 * Matches the Luna::Form Ember component API and styling.
 *
 * @example
 * ```tsx
 * // Using useFormContext hook (recommended)
 * function MyForm() {
 *   return (
 *     <Form onSubmit={handleSubmit}>
 *       <MyInput />
 *       <MySubmitButton />
 *     </Form>
 *   );
 * }
 *
 * function MySubmitButton() {
 *   const form = useFormContext();
 *   return <Button loading={form?.isLoading} type="submit">Save</Button>;
 * }
 *
 * // Using legacy render callback (deprecated)
 * <Form onSubmit={handleSubmit}>
 *   {(form) => (
 *     <>
 *       <Input />
 *       <Button loading={form.isLoading} type="submit">Save</Button>
 *     </>
 *   )}
 * </Form>
 * ```
 */
export function Form({ autosave, onSubmit, externalErrorsFor, children, className }: FormProps) {
	const [isLoading, setIsLoading] = useState(false);
	const inputManagersRef = useRef<Map<string, InputManager>>(new Map());

	const registerInputManager = useCallback((manager: InputManager) => {
		inputManagersRef.current.set(manager.id, manager);
	}, []);

	const unregisterInputManager = useCallback((manager: InputManager) => {
		inputManagersRef.current.delete(manager.id);
	}, []);

	const validate = useCallback((): { errorDuringSave: boolean } | undefined => {
		let invalidInput: InputManager | undefined;

		inputManagersRef.current.forEach((manager) => {
			manager.setShowErrorIfPresent(true);
			if (!invalidInput && manager.isInvalid) {
				invalidInput = manager;
			}
		});

		if (invalidInput) {
			if (!autosave) {
				invalidInput.focus();
			}
			return { errorDuringSave: true };
		}
	}, [autosave]);

	const handleSubmit = useCallback(
		async (event?: SubmitEvent, ...additionalArgs: unknown[]): Promise<{ errorDuringSave: boolean }> => {
			event?.preventDefault?.();

			const validation = validate();

			if (validation !== undefined) {
				return validation;
			}

			setIsLoading(true);

			try {
				await onSubmit(event, ...additionalArgs);

				/**
				 * This resets the form so it can be reused without error
				 * messages immediately showing up when all the fields are
				 * reset to empty.
				 */
				inputManagersRef.current.forEach((manager) => {
					manager.setShowErrorIfPresent(false);
				});

				return { errorDuringSave: false };
			} catch (_error: unknown) {
				// TODO: Handle FormError type
				return { errorDuringSave: true };
			} finally {
				setIsLoading(false);
			}
		},
		[onSubmit, validate],
	);

	const formContext: FormContextValue = {
		actions: {
			onSubmit: handleSubmit,
			validate,
			registerInputManager,
			unregisterInputManager,
		},
		autosave,
		isLoading,
		externalErrorsFor: externalErrorsFor ?? (() => undefined),
	};

	return (
		<form
			className={className}
			onSubmit={(e) => {
				e.preventDefault();
				void handleSubmit(e.nativeEvent as SubmitEvent);
			}}
			noValidate
			data-qid-component-type="Luna::Form"
		>
			<FormContext.Provider value={formContext}>
				{typeof children === 'function' ? children(formContext) : children}
			</FormContext.Provider>
		</form>
	);
}

export default Form;
