import Icon from './icon.gts';

interface SelectedListSignature<T> {
	Element: HTMLUListElement;
	Args: {
		value: T[];
		onChange: (value: T[]) => void;
		placeholder?: string;
		disabled?: boolean;
	};
	Blocks: {
		default: [option: T];
	};
}

/**
 * For rendering selected options. Often as a companion to Luna::Select.
 *
 * @typeParam T The type of the value.
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/selected-list.md)
 */
export default class LunaSelectedList<T> extends Component<SelectedListSignature<T>> {
	removeOption = (option: T) => {
		const selections = this.args.value.filter((selectedOption: T) => selectedOption !== option);
		this.args.onChange(selections);
	};

	<template>
		<ul class="list" data-qid-component-type="Luna::SelectedList" ...attributes>
			{{#each @value as |option i|}}
				<li class="list-item" data-test-list-item={{i}}>
					<div>
						{{yield option}}
					</div>
					{{#unless @disabled}}
						<Button
							@size="xs"
							@variant="transparent"
							aria-label={{t "input.selected-list.remove"}}
							data-test-remove-list-option={{i}}
							{{on "click" (fn this.removeOption option)}}
						>
							<Icon @icon="close" />
						</Button>
					{{/unless}}
				</li>
			{{else}}
				{{#if @placeholder}}
					<li class="list-item list-item--placeholder">
						{{@placeholder}}
					</li>
				{{/if}}
			{{/each}}
		</ul>
	</template>
}

import { Children, useCallback } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { t } from '../../utils/i18n';
import { Button } from '../button/button';
import { Icon } from '../icon/icon';

import styles from './selected-list.module.css';

/**
 * Item subcomponent for SelectedList.
 * Accepts a render function as children that receives (item: T, index: number).
 */
function Item<T>({ children: _children }: { children: (item: T, index: number) => React.ReactNode }): null {
	return null;
}

Item.displayName = 'SelectedList.Item';
// Use a symbol to mark this as a SelectedList.Item slot for runtime detection
(Item as unknown as { $$lunaSelectedListItem: true }).$$lunaSelectedListItem = true;

interface ExtractedSlots<T> {
	renderItem?: (item: T, index: number) => React.ReactNode;
}

/**
 * Extract the Item slot from children.
 */
function extractSelectedListSlots<T>(children: React.ReactNode): ExtractedSlots<T> {
	let renderItem: ((item: T, index: number) => React.ReactNode) | undefined;

	Children.forEach(children, (child) => {
		if (
			typeof child === 'object' &&
			child !== null &&
			'type' in child &&
			typeof child.type === 'function' &&
			(child.type as unknown as { $$lunaSelectedListItem?: boolean }).$$lunaSelectedListItem === true
		) {
			// Extract the render function from Item's children prop
			const itemProps = child.props as { children?: (item: T, index: number) => React.ReactNode };
			if (itemProps.children) {
				renderItem = itemProps.children;
			}
		}
	});

	return { renderItem };
}

interface SelectedListBaseProps<T> {
	/** The array of selected values */
	value: T[];
	/** Callback fired when the value changes (item removed) */
	onChange: (value: T[]) => void;
	/** Placeholder text shown when the list is empty */
	placeholder?: string;
	/** Whether the list is disabled (hides remove buttons) */
	disabled?: boolean;
	/** Render function for each item (legacy prop, prefer using SelectedList.Item) */
	renderItem?: (item: T, index: number) => React.ReactNode;
	/** Child elements, typically SelectedList.Item */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type SelectedListProps<T> = ComponentPropsWithRef<SelectedListBaseProps<T>, HTMLUListElement>;

/**
 * For rendering selected options. Often as a companion to Luna::Select.
 *
 * @typeParam T The type of the value.
 */
function SelectedListComponent<T>({
	value,
	onChange,
	placeholder,
	disabled = false,
	renderItem: renderItemProp,
	children,
	className,
	ref,
	...props
}: SelectedListProps<T>): React.ReactElement {
	// Extract slots from children
	const { renderItem: renderItemSlot } = extractSelectedListSlots<T>(children);

	// Use slot if provided, otherwise fallback to prop
	const renderItem = renderItemSlot ?? renderItemProp;

	if (!renderItem) {
		throw new Error('SelectedList requires either a renderItem prop or a SelectedList.Item child');
	}

	const removeOption = useCallback(
		(index: number) => {
			const newValue = value.filter((_, i) => i !== index);
			onChange(newValue);
		},
		[value, onChange],
	);

	return (
		<ul ref={ref} className={cn(styles.list, className)} data-qid-component-type="Luna::SelectedList" {...props}>
			{value.length > 0 ? (
				value.map((item, index) => (
					<li key={index} className={styles.listItem} data-test-list-item={index}>
						<div>{renderItem(item, index)}</div>
						{!disabled && (
							<Button
								size="xs"
								variant="transparent"
								aria-label={t('selected-list.ariaRemove')}
								data-test-remove-list-option={index}
								onClick={() => removeOption(index)}
							>
								<Icon icon="close" />
							</Button>
						)}
					</li>
				))
			) : placeholder ? (
				<li className={cn(styles.listItem, styles.listItemPlaceholder)}>{placeholder}</li>
			) : null}
		</ul>
	);
}

SelectedListComponent.displayName = 'SelectedList';

/**
 * For rendering selected options. Often as a companion to Luna::Select.
 *
 * @typeParam T The type of the value.
 *
 * @example
 * ```tsx
 * // Using compound component (recommended)
 * <SelectedList value={items} onChange={setItems}>
 *   <SelectedList.Item>{(item, index) => <span>{item.name}</span>}</SelectedList.Item>
 * </SelectedList>
 *
 * // Using legacy renderItem prop (deprecated)
 * <SelectedList
 *   value={items}
 *   onChange={setItems}
 *   renderItem={(item, index) => <span>{item.name}</span>}
 * />
 * ```
 */
export const SelectedList = Object.assign(SelectedListComponent, {
	Item,
});

export default SelectedList;
