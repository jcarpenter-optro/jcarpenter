interface CheckboxTreeNodeSignature {
	Element: HTMLDivElement;
	Args: {
		node: CheckboxNode;
		onChange: (value: boolean, changedNodes: CheckboxNode[], event: Event) => unknown;
	};
	Blocks: { default?: [] };
}

export interface CheckboxTreeItem {
	id: string;
	label: string;
	children?: CheckboxTreeItem[];
}

// Note does a shallow validation and doesn't validate children is CheckboxTreeItem
export function isCheckboxTreeItem(value: unknown): value is CheckboxTreeItem {
	if (typeof value !== 'object' || value === null) {
		return false;
	}

	const item = value as Record<string, unknown>;

	if (typeof item.id !== 'string' || typeof item.label !== 'string') {
		return false;
	}

	if (item.children !== undefined && !Array.isArray(item.children)) {
		return false;
	}

	return true;
}

/**
 * A Node within a checkbox tree component. Nodes render the CheckboxNode, recursively rendering for their children.
 */
class CheckboxTreeNode extends Component<CheckboxTreeNodeSignature> {
	nodeElement: HTMLDivElement | null = null;
	childrenId = `${guidFor(this)}--children`;

	get isExpanded() {
		return !!this.args.node.expanded;
	}

	get hasChildren() {
		return this.args.node.children.length;
	}

	get value() {
		return this.args.node.value;
	}

	get id() {
		return this.args.node.fullPath();
	}

	@action
	toggleExpanded() {
		this.args.node.expanded = !this.args.node.expanded;
	}

	@action
	toggleValue(event: Event) {
		const nextValue = this.value === 'mixed' ? true : !this.value;
		const leafNodes = this.args.node.updateValue(nextValue);

		// Note that checkbox tree state is decoupled from the caller, if this.args.onChange fails checkbox tree will not revert state.
		this.args.onChange?.(nextValue, leafNodes, event);
	}

	@action
	handleKeydown(event: KeyboardEvent) {
		const { key } = event;
		if (key === KEYBOARD_KEYS.SPACE || key === KEYBOARD_KEYS.RIGHT_ARROW || key === KEYBOARD_KEYS.LEFT_ARROW) {
			event.preventDefault();
		}
		switch (key) {
			case KEYBOARD_KEYS.RIGHT_ARROW:
				if (this.hasChildren) {
					// First expand then enter children
					if (!this.args.node.expanded) {
						this.args.node.expanded = true;
					} else {
						this.moveFocusToFirstTreeItem();
					}
				}
				break;
			case KEYBOARD_KEYS.LEFT_ARROW:
				// Collapse children first (if any) or move to parent
				if (this.args.node.expanded && this.hasChildren) {
					this.args.node.expanded = false;
				} else {
					this.moveFocusToParentTreeItem();
				}
				break;
			case KEYBOARD_KEYS.SPACE:
				this.toggleValue(event);
				break;
			case KEYBOARD_KEYS.ENTER:
				if (this.nodeElement && this.hasChildren) {
					this.toggleExpanded();
				}
				break;
		}
	}

	moveFocusToFirstTreeItem = () => {
		if (this.nodeElement) {
			const firstItem = this.nodeElement.querySelector(`#${this.childrenId} [role="treeitem"]:first-of-type`);
			if (firstItem instanceof HTMLElement) firstItem.focus();
		}
	};

	moveFocusToParentTreeItem = () => {
		if (this.nodeElement) {
			const groupElement = this.nodeElement.closest('[role="group"]');
			if (groupElement) {
				const parentTreeItem = groupElement.previousElementSibling;
				if (parentTreeItem instanceof HTMLElement) parentTreeItem.focus();
			}
		}
	};

	registerNodeElement: ModifierLike<{ Element: HTMLElement }> = modifier((element: HTMLElement) => {
		if (element instanceof HTMLDivElement) this.nodeElement = element;
		assert('Expected this.nodeElement to be an HTMLDivElement', this.nodeElement instanceof HTMLDivElement);
	});

	<template>
		{{#if @node.isVisible}}
			<YStack {{this.registerNodeElement}}>
				{{#unless @node.isRootNode}}
					{{! We're using aria-checked here instead of aria-selected since it supports mixed state }}
					{{! template-lint-disable require-mandatory-role-attributes }}
					<XStack
						class="tree-item left-padding"
						{{! Tree role provided in parent component }}
						{{! template-lint-disable require-context-role }}
						role="treeitem"
						tabindex="0"
						aria-expanded={{if this.hasChildren (String this.isExpanded)}}
						aria-checked={{String this.value}}
						aria-owns={{if (and this.isExpanded this.hasChildren) this.childrenId}}
						data-qid-tree-item={{this.id}}
						{{on "keydown" this.handleKeydown}}
					>
						{{#if this.hasChildren}}
							{{! This is acceptable since the keyboard interaction, focus state, etc. are on the node, not the toggle itself }}
							{{! template-lint-disable no-invalid-interactive }}
							<div
								class="toggle-tree"
								data-open={{String this.isExpanded}}
								{{on "click" this.toggleExpanded}}
								data-qid-tree-toggle
							>
								<Icon @icon="chevron-right" @size="sm" />
							</div>
						{{else}}
							<span class="toggle-tree" data-qid-tree-toggle-placeholder />
						{{/if}}
						<XStack @spacing="sm" class="checkbox-and-label" @width="auto">
							<LunaCheckboxElement
								@value={{String this.value}}
								aria-hidden="true"
								data-qid-checkbox
								{{on "click" this.toggleValue}}
							/>
							{{@node.label}}
						</XStack>
					</XStack>
				{{/unless}}
				{{#if (and this.isExpanded this.hasChildren)}}
					<div role="group" id={{this.childrenId}}>
						{{#each @node.children as |childNode|}}
							<CheckboxTreeNode @node={{childNode}} @onChange={{@onChange}} />
						{{/each}}
					</div>
				{{/if}}
			</YStack>
		{{/if}}
	</template>
}

type CheckboxTreeItemsLoader = () => PromiseLike<CheckboxTreeItem[]>;

/**
 * External controller for CheckboxTree that allows consumers to trigger select all and search actions, as well as track loading state and select all value.
 */
export class CheckboxTreeController {
	@tracked selectAllValue: boolean | 'mixed' = false;

	// These methods will be set by the CheckboxTree component when a controller instance is passed in as an argument
	selectAll: (value: boolean) => void = () => {
		assert('Expected selectAll method to be set by CheckboxTree component', false);
	};

	searchItems: (searchStr: string) => void = () => {
		assert('Expected searchItems method to be set by CheckboxTree component', false);
	};
}

interface CheckboxTreeSignature {
	Element: HTMLDivElement;
	Args: {
		items: CheckboxTreeItem[] | CheckboxTreeItemsLoader;
		onChange: (items: string[], event: Event) => unknown;
		initialSelection?: string[];
		autoExpandSelection?: boolean;
		controller?: CheckboxTreeController;
	};
	Blocks: { default?: [] };
}

/**
 * A checkbox tree. Checkbox component that open to show more checkboxes in a tree structure.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/checkbox-tree.md)
 */
export default class CheckboxTree extends Component<CheckboxTreeSignature> {
	@tracked selectedItems = new Set<string>(this.args.initialSelection ?? []);
	@tracked rootNode = buildTree([]);
	@tracked isLoading = true;
	treeElement: HTMLDivElement | null = null;

	constructor(owner: Owner, args: CheckboxTreeSignature['Args']) {
		super(owner, args);

		// Attach the internal method to the external controller object
		if (this.args.controller) {
			this.args.controller.selectAll = this.selectAll;
			this.args.controller.searchItems = this.searchItems;
		}

		void this.loadRootNode.perform();
	}

	get controller() {
		return this.args.controller ?? new CheckboxTreeController();
	}

	updateSelectAllValue() {
		this.controller.selectAllValue = this.rootNode.value;
	}

	selectAll = (value: boolean) => {
		const selectedNodes = this.rootNode.updateValue(value);

		this.onChange(value, selectedNodes, new Event('remote-select-all'));
		this.updateSelectAllValue();
	};

	searchItems = (searchStr: string) => {
		this.rootNode.filterVisibleNodes(searchStr);
		this.updateSelectAllValue();
	};

	loadRootNode = task({ keepLatest: true }, async () => {
		this.isLoading = true;
		try {
			let items: CheckboxTreeItem[];

			if (typeof this.args.items === 'function') {
				items = await this.args.items();
			} else {
				items = this.args.items;
			}

			this.rootNode = buildTree(items, this.args.initialSelection ?? [], this.args.autoExpandSelection ?? false);
			this.updateSelectAllValue();
		} finally {
			this.isLoading = false;
		}
	});

	@action
	onChange(value: boolean, changedNodes: CheckboxNode[], event: Event) {
		const newSelection = new Set(this.selectedItems);

		if (value) {
			for (const node of changedNodes) {
				if (node.id) {
					newSelection.add(node.id);
				}
			}
		} else {
			for (const node of changedNodes) {
				if (node.id) {
					newSelection.delete(node.id);
				}
			}
		}

		this.selectedItems = newSelection;
		this.args.onChange(Array.from(this.selectedItems), event);
		this.updateSelectAllValue();
	}

	<template>
		<div role="tree" {{a11yMenu loop=true memorize=true}} data-qid-component-type="Luna::CheckboxTree" ...attributes>
			<YStack>
				{{#if this.isLoading}}
					<Spinner />
				{{else}}
					<CheckboxTreeNode @node={{this.rootNode}} @onChange={{this.onChange}} />
				{{/if}}
			</YStack>
		</div>
	</template>
}

import { useCallback, useId, useRef, useState } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { CheckboxElement } from '../checkbox/checkbox-element';
import { Icon } from '../icon/icon';

import { buildTree, type CheckboxNode, type CheckboxTreeItem, type CheckboxValue } from './checkbox-node';

import styles from './checkbox-tree.module.css';

// Keyboard codes
const KEYS = {
	SPACE: ' ',
	ENTER: 'Enter',
	ARROW_UP: 'ArrowUp',
	ARROW_DOWN: 'ArrowDown',
	ARROW_LEFT: 'ArrowLeft',
	ARROW_RIGHT: 'ArrowRight',
	HOME: 'Home',
	END: 'End',
} as const;

interface CheckboxTreeNodeProps {
	node: CheckboxNode;
	onChange: (value: boolean, changedNodes: CheckboxNode[], event: React.MouseEvent | React.KeyboardEvent) => void;
	onTreeUpdate: () => void;
}

/**
 * A Node within a checkbox tree component.
 * Nodes render the CheckboxNode, recursively rendering for their children.
 */
function CheckboxTreeNode({ node, onChange, onTreeUpdate }: CheckboxTreeNodeProps) {
	const nodeRef = useRef<HTMLDivElement>(null);
	const childrenId = useId();

	const isExpanded = node.expanded;
	const hasChildren = node.children.length > 0;
	const value = node.value;
	const id = node.fullPath();

	const toggleExpanded = useCallback(() => {
		node.expanded = !node.expanded;
		onTreeUpdate();
	}, [node, onTreeUpdate]);

	const toggleValue = useCallback(
		(event: React.MouseEvent | React.KeyboardEvent) => {
			const nextValue = value === 'mixed' ? true : !value;
			const leafNodes = node.updateValue(nextValue);
			onChange(nextValue, leafNodes, event);
			onTreeUpdate();
		},
		[node, value, onChange, onTreeUpdate],
	);

	const moveFocusToFirstTreeItem = useCallback(() => {
		if (nodeRef.current) {
			const firstItem = nodeRef.current.querySelector(`#${CSS.escape(childrenId)} [role="treeitem"]:first-of-type`);
			if (firstItem instanceof HTMLElement) {
				firstItem.focus();
			}
		}
	}, [childrenId]);

	const moveFocusToParentTreeItem = useCallback(() => {
		if (nodeRef.current) {
			const groupElement = nodeRef.current.closest('[role="group"]');
			if (groupElement) {
				const parentTreeItem = groupElement.previousElementSibling;
				if (parentTreeItem instanceof HTMLElement) {
					parentTreeItem.focus();
				}
			}
		}
	}, []);

	const handleKeyDown = useCallback(
		(event: React.KeyboardEvent) => {
			const { key } = event;

			if (key === KEYS.SPACE || key === KEYS.ARROW_RIGHT || key === KEYS.ARROW_LEFT) {
				event.preventDefault();
			}

			switch (key) {
				case KEYS.ARROW_RIGHT:
					if (hasChildren) {
						if (!node.expanded) {
							node.expanded = true;
							onTreeUpdate();
						} else {
							moveFocusToFirstTreeItem();
						}
					}
					break;

				case KEYS.ARROW_LEFT:
					if (node.expanded && hasChildren) {
						node.expanded = false;
						onTreeUpdate();
					} else {
						moveFocusToParentTreeItem();
					}
					break;

				case KEYS.SPACE:
					toggleValue(event);
					break;

				case KEYS.ENTER:
					if (hasChildren) {
						toggleExpanded();
					}
					break;
			}
		},
		[hasChildren, node, onTreeUpdate, moveFocusToFirstTreeItem, moveFocusToParentTreeItem, toggleValue, toggleExpanded],
	);

	const handleCheckboxClick = useCallback(
		(event: React.MouseEvent) => {
			event.stopPropagation();
			toggleValue(event);
		},
		[toggleValue],
	);

	const handleToggleClick = useCallback(
		(event: React.MouseEvent) => {
			event.stopPropagation();
			toggleExpanded();
		},
		[toggleExpanded],
	);

	// Don't render the root node itself, only its children
	if (node.isRootNode) {
		return (
			<>
				{node.children.map((childNode) => (
					<CheckboxTreeNode
						key={childNode.fullPath()}
						node={childNode}
						onChange={onChange}
						onTreeUpdate={onTreeUpdate}
					/>
				))}
			</>
		);
	}

	return (
		<div ref={nodeRef}>
			<div
				className={styles.treeItem}
				role="treeitem"
				tabIndex={0}
				aria-expanded={hasChildren ? isExpanded : undefined}
				aria-checked={String(value) as 'true' | 'false' | 'mixed'}
				aria-owns={isExpanded && hasChildren ? childrenId : undefined}
				data-qid-tree-item={id}
				onKeyDown={handleKeyDown}
			>
				{hasChildren ? (
					<button
						type="button"
						className={styles.toggleTree}
						data-open={String(isExpanded)}
						data-has-children="true"
						data-qid-tree-toggle
						onClick={handleToggleClick}
						tabIndex={-1}
						aria-hidden="true"
					>
						<Icon icon="chevron-right" size="sm" />
					</button>
				) : (
					<span className={styles.toggleTree} data-qid-tree-toggle-placeholder />
				)}
				<div className={styles.nodeContent}>
					<CheckboxElement
						value={value}
						aria-hidden="true"
						data-qid-checkbox
						onClick={handleCheckboxClick}
						tabIndex={-1}
					/>
					{node.label}
				</div>
			</div>

			{isExpanded && hasChildren && (
				<div role="group" id={childrenId} className={styles.childrenGroup}>
					{node.children.map((childNode) => (
						<CheckboxTreeNode
							key={childNode.fullPath()}
							node={childNode}
							onChange={onChange}
							onTreeUpdate={onTreeUpdate}
						/>
					))}
				</div>
			)}
		</div>
	);
}

interface CheckboxTreeBaseProps {
	/** Items to display in the tree */
	items: CheckboxTreeItem[];
	/** Initial selection (array of item IDs) */
	initialSelection?: string[];
	/** Whether to auto-expand nodes that have selected children */
	autoExpandSelection?: boolean;
	/** Called when selection changes */
	onChange?: (selectedIds: string[], event: React.MouseEvent | React.KeyboardEvent) => void;
	/** Additional CSS class */
	className?: string;
	/** Test ID */
	testId?: string;
}

export type CheckboxTreeProps = ComponentPropsWithRef<CheckboxTreeBaseProps, HTMLDivElement>;

/**
 * A checkbox tree component.
 * Displays a tree of checkboxes that can be expanded/collapsed and supports mixed states.
 */
export function CheckboxTree(props: CheckboxTreeProps) {
	const {
		ref,
		items,
		initialSelection = [],
		autoExpandSelection = false,
		onChange,
		className,
		testId,
		...rest
	} = props;
	// Build the tree and track selected items
	const [rootNode, _setRootNode] = useState<CheckboxNode>(() =>
		buildTree(items, initialSelection, autoExpandSelection),
	);

	const [_selectedItems, setSelectedItems] = useState<Set<string>>(() => new Set(initialSelection));

	// Force re-render when tree state changes (expand/collapse, value changes)
	const [, forceUpdate] = useState({});
	const onTreeUpdate = useCallback(() => {
		forceUpdate({});
	}, []);

	const handleChange = useCallback(
		(value: boolean, changedNodes: CheckboxNode[], event: React.MouseEvent | React.KeyboardEvent) => {
			setSelectedItems((prev) => {
				const next = new Set(prev);

				if (value) {
					for (const node of changedNodes) {
						if (node.id) {
							next.add(node.id);
						}
					}
				} else {
					for (const node of changedNodes) {
						if (node.id) {
							next.delete(node.id);
						}
					}
				}

				// Call onChange with the new selection
				onChange?.(Array.from(next), event);

				return next;
			});
		},
		[onChange],
	);

	return (
		<div
			ref={ref}
			role="tree"
			className={cn(styles.CheckboxTree, className)}
			data-qid-component-type="Luna::CheckboxTree"
			data-testid={testId}
			{...rest}
		>
			<CheckboxTreeNode node={rootNode} onChange={handleChange} onTreeUpdate={onTreeUpdate} />
		</div>
	);
}

export default CheckboxTree;
