/**
 * Base chart component using AG Charts. Not intended to be used directly

export default class Chart extends Component<LunaChartSignature> {
	chartElement: HTMLElement | null = null;
	@tracked _chart: AgChartInstance | null = null;

	get chartId(): string {
		return guidFor(this);
	}

	get padding(): AgChartOptions['padding'] {
		switch (typeof this.args.padding) {
			case 'number':
				return {
					top: this.args.padding,
					right: this.args.padding,
					left: this.args.padding,
					bottom: this.args.padding,
				};
			case 'object':
				return this.args.padding;
			default:
				return {
					top: PADDING_TOP,
					right: PADDING_RIGHT,
					left: PADDING_LEFT,
					bottom: PADDING_BOTTOM,
				};
		}
	}

	get minHeight(): number {
		return this.args.minHeight ?? MIN_CHART_HEIGHT;
	}

	get minWidth(): number {
		return this.args.minWidth ?? MIN_CHART_WIDTH;
	}

	get theme(): AgChartTheme | undefined {
		return this.args.theme ?? defaultTheme;
	}

	get type(): 'bar' | 'line' | 'donut' | 'pie' {
		return this.args.type;
	}

	/**
	 * Builds AgChart options from arguments
	 */
	get chartOptions(): AgChartOptions {
		if (!this.chartElement) {
			throw new Error('Chart element not available');
		}

		return {
			container: this.chartElement,
			minWidth: this.minWidth,
			minHeight: this.minHeight,
			seriesArea: {
				clip: false,
				padding: this.padding ?? PADDING_SIZE,
			},
			padding: this.padding,
			theme: this.theme,
			footnote: { enabled: false },
			title: { enabled: false },
			subtitle: { enabled: false },
			...(this.args.axes && { axes: this.args.axes }),
			...(this.args.legend && { legend: this.args.legend }),
			...(this.args.data && { data: this.args.data }),
			...(this.args.series && { series: this.args.series }),
			...(this.args.width && { width: this.args.width }),
			...(this.args.height && { height: this.args.height }),
		} as AgChartOptions;
	}

	/**
	 *  This run-once modifier is strictly for setting the container element
	 * 	and setting up teardown handler
	 */
	setChartElement = modifier((element: HTMLElement) => {
		if (this.chartElement) return;
		this.chartElement = element;

		this._chart = AgCharts.create(this.chartOptions);
		// register destructor
		registerDestructor(this, () => {
			this._chart?.destroy();
		});
	});

	/**
	 * This modifier entangles the arguments and updates the chart if the arguments change
	 */
	onUpdate = modifier((_element: HTMLElement) => {
		// Use void to handle async operations properly in modifiers
		if (this._chart) {
			void this._chart?.update(this.chartOptions);
		}
	});

	<template>
		<div
			class="container"
			data-qid-chart={{this.chartId}}
			data-qid-component-type={{concat "Luna::Chart-" this.type}}
			{{this.setChartElement}}
			{{this.onUpdate}}
			...attributes
		/>
		{{yield this._chart}}
	</template>
}

import type {

	AxisOptionsUnion,
	BarChartConfig,
	CategoryAxisOptions,
	LogAxisOptions,
	NumberAxisOptions,
} from '@soxhub/luna-charts';
import {
	buildBarSeriesConfig,
	buildCategoryAxisConfig,
	buildHiddenAxisConfig,
	buildLogAxisConfig,
	buildNumberAxisConfig,
} from '@soxhub/luna-charts';
import type { AgChartOptions } from 'ag-charts-community';

import { Chart } from './chart';

const DIMENSIONS: Record<'grid' | 'default', { minWidth: number; minHeight: number }> = {
	default: { minWidth: 300, minHeight: 200 },
	grid: { minWidth: 400, minHeight: 300 },
};

export interface BarChartProps {
	/** Chart data */
	data?: unknown[];
	/** Bar chart configuration */
	config: BarChartConfig;
	/** Additional CSS class */
	className?: string;
}

/**
 * Dispatches to the correct axis builder based on the `type` field.
 * Defaults to a number axis when no type is specified.
 */
function buildValueAxisConfig(options: AxisOptionsUnion) {
	if (options.type === 'log') {
		return buildLogAxisConfig(options as LogAxisOptions);
	}
	return buildNumberAxisConfig(options as NumberAxisOptions);
}

/** Whether the value axis uses a logarithmic scale. */
function hasLogAxis(config: BarChartConfig): boolean {
	const axes = config.axes;
	if (!axes) return false;
	const orientation = config.orientation ?? 'vertical';
	const valueAxis = orientation === 'horizontal' ? axes.x : axes.y;
	return valueAxis?.type === 'log';
}

/**
 * Sanitizes data for log axes by replacing zero values with undefined.
 * AG Charts cannot render data with zeros on a log scale (log(0) is undefined).
 */
function sanitizeDataForLogAxis(data: unknown[], config: BarChartConfig): unknown[] {
	const valueKeys = (config.items ?? []).map((item) => item.key).filter((key): key is string => Boolean(key));
	if (valueKeys.length === 0) return data;

	return data.map((datum) => {
		const record = datum as Record<string, unknown>;
		const hasZero = valueKeys.some((key) => record[key] === 0);
		if (!hasZero) return datum;

		const copy = { ...record };
		for (const key of valueKeys) {
			if (copy[key] === 0) {
				copy[key] = undefined;
			}
		}
		return copy;
	});
}

function getBarChartAxes(config: BarChartConfig): AgAxisConfig | undefined {
	const showAxis = typeof config.showAxis === 'boolean' ? config.showAxis : true;
	if (!showAxis) {
		return buildHiddenAxisConfig();
	}

	const orientation = config.orientation ?? 'vertical';
	const layout = config.layout ?? 'default';
	const defaults = { direction: orientation as 'horizontal' | 'vertical', layout };
	const axesConfig = config.axes ?? {};
	const xConfig = { ...defaults, ...(axesConfig.x ?? {}) } as AxisOptionsUnion;
	const yConfig = { ...defaults, ...(axesConfig.y ?? {}) } as AxisOptionsUnion;

	return orientation === 'horizontal'
		? {
				x: buildValueAxisConfig(xConfig),
				y: buildCategoryAxisConfig(yConfig as CategoryAxisOptions),
			}
		: {
				x: buildCategoryAxisConfig(xConfig as CategoryAxisOptions),
				y: buildValueAxisConfig(yConfig),
			};
}

export function BarChart({ data = [], config, className }: BarChartProps): React.ReactElement {
	const chartData = hasLogAxis(config) ? sanitizeDataForLogAxis(data, config) : data;
	const series = buildBarSeriesConfig(config);
	const axes = getBarChartAxes(config);
	const layout = config.layout ?? 'default';
	const legend = typeof config.showLegend === 'boolean' && !config.showLegend ? { enabled: false } : config.legend;

	return (
		<Chart
			type="bar"
			data={chartData}
			series={series as AgChartOptions['series']}
			axes={axes as unknown}
			legend={legend}
			minWidth={config.minWidth ?? DIMENSIONS[layout].minWidth}
			minHeight={config.minHeight ?? DIMENSIONS[layout].minHeight}
			padding={config.padding}
			width={config.width}
			height={config.height}
			className={className}
		/>
	);
}

// AG Charts v13 requires explicit module registration

(ModuleRegistry as any).registerModules([AllCommunityModule]);

import { cn } from '../../utils/cn';

import styles from './chart.module.css';

// Re-export chart utility types
export type {

const defaultTheme: AgChartTheme = {

		fills: DefaultChartLegendColors,
	},

const PADDING_TOP = 5;
const PADDING_BOTTOM = 5;
const PADDING_LEFT = 15;
const PADDING_RIGHT = 15;
const MIN_CHART_WIDTH = 200;
const MIN_CHART_HEIGHT = 100;

// TODO: Consider using ag-charts-react wrapper for declarative React integration.

/**
 * Base chart component using AG Charts.
 *
 * Provides a flexible charting solution supporting bar, line, donut, and pie charts.
 *
 * @example
 * ```tsx
 * <Chart
 *   type="bar"
 *   data={[
 *     { category: 'A', value: 10 },
 *     { category: 'B', value: 20 },
 *   ]}
 *   series={[{ type: 'bar', xKey: 'category', yKey: 'value' }]}
 * />
 * ```
 */
export function Chart(props: ChartProps) {
	const {
		ref,
		axes,
		data,
		padding: paddingProp,
		legend,
		series,
		chartTitle: _chartTitle,
		theme = defaultTheme,
		type,
		minWidth = MIN_CHART_WIDTH,
		minHeight = MIN_CHART_HEIGHT,
		width,
		height,
		className,
		children,
		...rest
	} = props;

	const chartId = useId();
	const containerRef = useRef<HTMLDivElement>(null);
	const chartRef = useRef<AgChartInstance | null>(null);

	// Calculate padding
	const padding = useCallback((): AgChartOptions['padding'] => {
		if (typeof paddingProp === 'number') {
			return {
				top: paddingProp,
				right: paddingProp,
				left: paddingProp,
				bottom: paddingProp,
			};
		}
		if (typeof paddingProp === 'object') {
			return paddingProp;
		}
		return {
			top: PADDING_TOP,
			right: PADDING_RIGHT,
			left: PADDING_LEFT,
			bottom: PADDING_BOTTOM,
		};
	}, [paddingProp]);

	// Build chart options
	const getChartOptions = useCallback((): AgChartOptions | null => {
		if (!containerRef.current) {
			return null;
		}

		const options: AgChartOptions = {
			container: containerRef.current,
			minWidth,
			minHeight,
			seriesArea: {
				clip: false,
				padding: padding(),
			},
			padding: padding(),
			theme,
			footnote: { enabled: false },
			title: { enabled: false },
			subtitle: { enabled: false },
		};

		if (axes) {
			// @ts-expect-error - AgChartOptions.axes type varies by chart type
			options.axes = axes;
		}
		if (legend) {
			options.legend = legend;
		}
		if (data) {
			options.data = data;
		}
		if (series) {
			options.series = series;
		}
		if (width) {
			options.width = width;
		}
		if (height) {
			options.height = height;
		}

		return options;
	}, [axes, data, legend, series, theme, minWidth, minHeight, width, height, padding]);

	// Create chart on mount
	useEffect(() => {
		const options = getChartOptions();
		if (options && !chartRef.current) {
			chartRef.current = AgCharts.create(options);
		}

		return () => {
			if (chartRef.current) {
				chartRef.current.destroy();
				chartRef.current = null;
			}
		};
		// eslint-disable-next-line react-hooks/exhaustive-deps
	}, []); // Only run on mount/unmount

	// Update chart when props change
	useEffect(() => {
		const options = getChartOptions();
		if (chartRef.current && options) {
			void chartRef.current.update(options);
		}
	}, [getChartOptions]);

	const containerClassName = cn(styles.container, className);

	return (
		<div
			ref={(node) => {
				containerRef.current = node;
				if (typeof ref === 'function') {
					ref(node);
				} else if (ref && typeof ref === 'object') {
					(ref as React.MutableRefObject<HTMLDivElement | null>).current = node;
				}
			}}
			className={containerClassName}
			data-qid-chart={chartId}
			data-qid-component-type={`Luna::Chart-${type}`}
			{...rest}
		>
			{children?.(chartRef.current)}
		</div>
	);
}

export default Chart;

import { Chart } from './chart';

const DIMENSIONS: Record<'grid' | 'default', { minWidth: number; minHeight: number }> = {
	default: { minWidth: 300, minHeight: 200 },
	grid: { minWidth: 400, minHeight: 300 },
};

export interface LineChartProps {
	/** Chart data */
	data?: unknown[];

	config: LineChartConfig;

	className?: string;

function getLineChartAxes(config: LineChartConfig): AgAxisConfig | undefined {
	const axesConfig = config.axes ?? {};
	const layout = config.layout ?? 'default';

	const buildAxis = (options: unknown): AgAxisUnion | undefined => {
		if (!options) return undefined;
		const typed = options as Record<string, unknown> & { type?: string };
		const type = typed.type?.toLowerCase();
		const configWithLayout = { ...typed, layout };

		if (type === 'number') {
			return buildNumberAxisConfig({
				direction: 'horizontal',
				position: 'left',
				...configWithLayout,
			} as NumberAxisOptions) as AgAxisUnion;
		}
		if (type === 'time') {
			return buildTimeAxisConfig({
				direction: 'horizontal',
				position: 'bottom',
				...configWithLayout,
			} as TimeAxisOptions) as AgAxisUnion;
		}
		return undefined;
	};

	const result: AgAxisConfig = {};
	if (axesConfig.y) result.y = buildAxis(axesConfig.y);
	if (axesConfig.x) result.x = buildAxis(axesConfig.x);
	return Object.keys(result).length > 0 ? result : undefined;
}

export function LineChart({ data = [], config, className }: LineChartProps): React.ReactElement {
	const series = buildLineSeriesConfig(config);
	const axes = getLineChartAxes(config);
	const layout = config.layout ?? 'default';
	const legend = typeof config.showLegend === 'boolean' && !config.showLegend ? { enabled: false } : config.legend;

	return (
		<Chart
			type="line"
			data={data}
			series={series as AgChartOptions['series']}
			axes={axes as unknown}
			legend={legend}
			minWidth={config.minWidth ?? DIMENSIONS[layout].minWidth}
			minHeight={config.minHeight ?? DIMENSIONS[layout].minHeight}
			padding={config.padding}
			width={config.width}
