empty-state-announcer.gts
import { assert } from '@ember/debug';
import type Owner from '@ember/owner';
import Component from '@glimmer/component';

import { consume } from 'ember-provide-consume-context';

import { SCHEMA_TABLE_HAD_ROWS_CONTEXT } from '@auditboard/consts/schema-table';

import type { A11yAnnouncerSignature } from './a11y-announcer.gts';
import A11yAnnouncer from './a11y-announcer.gts';

interface EmptyStateAnnouncerSignature {
	Element: HTMLSpanElement;
	Args: Omit<A11yAnnouncerSignature['Args'], 'announceOnInit'>;
}

/**
 * A specialized announcer component for empty states in schema tables that only announces
 * when the table transitions from having data to being empty (not on initial empty load).
 *
 * This component wraps A11yAnnouncer and conditionally announces based on whether the schema
 * table previously had rows. It should be used in Schema Table `:empty` blocks instead of
 * using A11yAnnouncer directly.
 *
 * @example
 * ```hbs
 * <SchemaTable::Table ...>
 *   <:empty>
 *     <EmptyStateAnnouncer @text={{formatMessage "No results found."}} />
 *   </:empty>
 * </SchemaTable::Table>
 * ```
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/empty-state-announcer.md)
 */
export default class EmptyStateAnnouncer extends Component<EmptyStateAnnouncerSignature> {
	@consume(SCHEMA_TABLE_HAD_ROWS_CONTEXT) hadRows: boolean | undefined;

	constructor(owner: Owner, args: EmptyStateAnnouncerSignature['Args']) {
		super(owner, args);
		// Ensure the component is being used within a Schema Table context
		assert(
			'EmptyStateAnnouncer must be used within a Schema Table. If you need to make announcements outside of a Schema Table, use A11yAnnouncer instead.',
			this.hadRows !== undefined,
		);
	}

	/**
	 * Only announce if the table previously had rows.
	 * This prevents announcements when tables load empty initially.
	 */
	get shouldAnnounce(): boolean {
		return this.hadRows === true;
	}

	<template>
		<A11yAnnouncer
			@text={{@text}}
			@screenReaderText={{@screenReaderText}}
			@urgency={{@urgency}}
			@announceOnInit={{this.shouldAnnounce}}
			@destroyMessage={{@destroyMessage}}
			...attributes
		/>
	</template>
}
