import type { LunaColor } from '../types.ts';

	Element: HTMLDivElement;
	Args: {
		padding?: string;

		slot?: Slot;
	};

		default: [];

		{{yield}}
	</div>
</template>;

export default LunaBlock;

interface BlockBaseProps {
	/** Padding size */
	padding?: BlockPadding;
	/** Background color from Luna design tokens */
	background?: BlockBackground;
	/** Enable overflow scrolling */
	scroll?: boolean;
	/** Make the block sticky positioned */
	sticky?: BlockSticky;
	/** Grid area slot assignment */
	slot?: BlockSlot;
	/** Additional CSS class */
	className?: string;
	/** Child content */
	children?: ReactNode;
}

export type BlockProps = ComponentPropsWithRef<BlockBaseProps, HTMLDivElement>;

/**
 * Renders a content block, similar to a div. The benefit is access to all the Luna design tokens via arguments.
 *
 * Matches the Luna::Block Ember component API and styling.
 */
export function Block(props: BlockProps) {
	const { ref, padding, background, scroll, sticky, slot, className, children, ...rest } = props;

	return (
		<div
			ref={ref}
			className={cn(styles.Block, className)}
			data-padding={padding}
			data-background={background}
			data-scroll={scroll || undefined}
			data-sticky={sticky}
			data-slot={slot}
			data-qid-component-type="Luna::Block"
			{...rest}
		>
			{children}
		</div>
	);
}

export default Block;
