● React (TSX)
spinner.tsx
import { cn } from '../../utils/cn';

import styles from './spinner.module.css';

export type SpinnerVariant = 'default' | 'reversed';
export type SpinnerSize = 'sm' | 'md' | 'lg' | 'xl' | '2xl' | '3xl' | '4xl';

interface SpinnerProps extends React.HTMLAttributes<HTMLDivElement> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** With current (light) theme:
	 * default - grey/dark spinner (for light bg)
	 * reversed - white spinner (for dark bg) */
	variant?: SpinnerVariant;
	/** Defaults to md */
	size?: SpinnerSize;
	/** Transition state (e.g., 'success') */
	transitionTo?: string;
}

/**
 * An indicator used to make it clear to the user that an action is processing.
 *
 * Matches the Luna::Spinner Ember component API and styling.
 */
export function Spinner(props: SpinnerProps) {
	const { ref, variant = 'default', size = 'md', transitionTo, className, ...rest } = props;

	const spinnerClassName = cn(styles.Spinner, className);

	return (
		<div
			ref={ref}
			className={spinnerClassName}
			data-variant={variant}
			data-size={size}
			data-transition-to={transitionTo}
			data-qid-component-type="Luna::Spinner"
			{...rest}
		>
			<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20">
				<g className={styles.animation}>
					<circle className={styles.path} cx="10" cy="10" r="9"></circle>
					<circle className={styles.progress} cx="10" cy="10" r="9"></circle>
				</g>
				<path className={styles.checkmark} strokeLinecap="round" strokeLinejoin="round" d="M5.5 10.5l3 3 6-6"></path>
			</svg>
		</div>
	);
}

export default Spinner;
