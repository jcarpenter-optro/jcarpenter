● React (TSX)
link.tsx
import { useCallback } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './link.module.css';

export type LinkVariant = 'blue' | 'red' | 'orange' | 'gray' | 'gray-with-underline' | 'teal';
export type LinkOpenInQuickView = 'always' | 'never';

interface LinkBaseProps {
	/** Link variant */
	variant?: LinkVariant;
	/** Link href (if provided, renders as anchor) */
	href?: string;
	/** Link target */
	target?: React.AnchorHTMLAttributes<HTMLAnchorElement>['target'];
	/** Link rel */
	rel?: React.AnchorHTMLAttributes<HTMLAnchorElement>['rel'];
	/** Link download */
	download?: React.AnchorHTMLAttributes<HTMLAnchorElement>['download'];
	/** Whether to open in quick view ('always' | 'never') */
	openInQuickView?: LinkOpenInQuickView | boolean | null | undefined;
	/** Additional CSS class */
	className?: string;
	/** Children (link content) */
	children?: React.ReactNode;
	/** onClick handler (for button-style links) */
	onClick?: (event: React.MouseEvent) => void;
}

export type LinkProps = ComponentPropsWithRef<LinkBaseProps, HTMLElement>;

/**
 * Renders text that looks like a link.
 *
 * Matches the Luna::Link Ember component API and styling.
 */
export function Link(props: LinkProps) {
	const {
		ref,
		variant = 'blue',
		href,
		target,
		rel,
		download,
		openInQuickView,
		className,
		children,
		onClick,
		...rest
	} = props;

	const linkClassName = cn(styles.Link, className);

	// Handle openInQuickView prop - convert boolean/null/undefined to string or undefined
	const openInQuickViewAttr =
		openInQuickView === 'always' || openInQuickView === 'never'
			? openInQuickView
			: openInQuickView === true
				? 'always'
				: openInQuickView === false || openInQuickView === null || openInQuickView === undefined
					? undefined
					: undefined;

	const handleKeyDown = useCallback(
		(event: React.KeyboardEvent) => {
			if (event.key === 'Enter' || event.key === ' ') {
				event.preventDefault();
				if (onClick) {
					onClick(event as unknown as React.MouseEvent<HTMLElement>);
				}
			}
		},
		[onClick],
	);

	if (href) {
		return (
			<a
				className={linkClassName}
				data-variant={variant}
				href={href}
				target={target}
				rel={rel}
				download={download}
				data-open-in-quick-view={openInQuickViewAttr}
				data-qid-component-type="Luna::Link"
				ref={ref as React.Ref<HTMLAnchorElement>}
				{...rest}
			>
				{children}
			</a>
		);
	}

	return (
		<div
			className={linkClassName}
			data-variant={variant}
			role="button"
			tabIndex={0}
			onClick={onClick}
			onKeyDown={handleKeyDown}
			data-open-in-quick-view={openInQuickViewAttr}
			data-qid-component-type="Luna::Link"
			ref={ref as React.Ref<HTMLDivElement>}
			{...rest}
		>
			{children}
		</div>
	);
}

export default Link;
