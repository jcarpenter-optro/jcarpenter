export interface LunaMenuSignature {
	Element: HTMLDivElement;
	Args: Pick<LunaOptionArgs, 'onClick'>;
	Blocks: {
		default: [
			{
				Option: WithBoundArgs<typeof LunaOption, 'onClick'>;
				Heading?: typeof OptionGroupHeader;
			},
		];
	};

/**
 * The Menu allows a user to navigate the menu options with the arrow keys, and HOME and END will jump to the first and last option items, respectively. The Menu component uses the [option](https://docs.auditboardteam.com/Luna/components/components/option) component for each menu item.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/menu.md)
 */
export default class MenuComponent extends Component<LunaMenuSignature> {
	menuId = `${guidFor(this)}--menu`;

	get optionsElement() {
		const element = document.getElementById(this.menuId);
		assert('Expected menu element', element);
		return element;
	}

	get a11ySetup(): object {
		return {
			...group({ tabbing: 'trap' }),
			...menu({ loop: true }),
		};
	}

	<template>
		<div
			class="Menu"
			role="menu"
			id={{this.menuId}}
			{{a11y this.a11ySetup}}
			data-qid-component-type="Luna::Menu"
			...attributes
		>

import { type MutableRefObject, useEffect, useId, useRef } from 'react';

import { Menu as BaseMenu } from '@base-ui-components/react/menu';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Option } from '../option';
import { OptionHeading } from '../option-heading';

import styles from './menu.module.css';

interface MenuBaseProps {
	/** Click handler for options (passed to all Option children) */
	onClick?: () => void;
	/** Child elements, typically Menu.Option and Menu.Heading */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
	/**
	 * CSS selector to autofocus when the menu mounts.
	 * The selector is relative to the menu container.
	 * E.g., '[role=menuitem]' to focus the first menu item.
	 */
	autoFocusSelector?: string;
}

export type MenuProps = ComponentPropsWithRef<MenuBaseProps, HTMLDivElement>;

/**
 * The Menu allows a user to navigate the menu options with the arrow keys, and HOME and END will jump to the first and last option items, respectively.
 *
 * Matches the Luna::Menu Ember component API and styling.
 * Uses Base UI Menu.Root as the foundation for keyboard navigation.
 */
export function Menu(props: MenuProps) {
	const { ref, children, className, autoFocusSelector, ...rest } = props;

	const menuId = useId();
	const menuClassName = cn(styles.Menu, className);
	const internalRef = useRef<HTMLDivElement>(null);

	// Merge refs
	const setRef = (el: HTMLDivElement | null) => {
		internalRef.current = el;
		if (typeof ref === 'function') {
			ref(el);
		} else if (ref) {
			(ref as MutableRefObject<HTMLDivElement | null>).current = el;
		}
	};

	// Autofocus the first matching element when specified
	useEffect(() => {
		if (autoFocusSelector && internalRef.current) {
			// Use setTimeout to ensure focus happens after the popup's focus trap is established
			const timeoutId = setTimeout(() => {
				if (internalRef.current) {
					const element = internalRef.current.querySelector<HTMLElement>(
						`${autoFocusSelector}:not(:disabled):not([tabindex="-1"])`,
					);
					if (element) {
						element.focus();
					}
				}
			}, 0);
			return () => clearTimeout(timeoutId);
		}
	}, [autoFocusSelector]);

	// Base UI Menu.Popup requires Menu.Positioner, but Luna Menu is standalone
	// We use Menu.Root for keyboard navigation context; children (Menu.Option, Menu.Heading) render inside
	return (
		<BaseMenu.Root defaultOpen={true} loopFocus={true} modal={false}>
			<div
				ref={setRef}
				className={menuClassName}
				role="menu"
				id={menuId}
				data-qid-component-type="Luna::Menu"
				{...rest}
			>
				{children}
			</div>
		</BaseMenu.Root>
	);
}

// Compound component API: consumers use <Menu><Menu.Option>...</Menu.Option><Menu.Heading>...</Menu.Heading></Menu>
Menu.Option = Option;
Menu.Heading = OptionHeading;

export default Menu;
