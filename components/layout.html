modal.gts
import type { TOC } from '@ember/component/template-only';
import { on } from '@ember/modifier';

import t from 'ember-intl/helpers/t';
import or from 'ember-truth-helpers/helpers/or';

import detectScrollable from '../../modifiers/detect-scrollable.ts';
import Button from '../button.gts';
import Heading from '../heading.gts';
import Icon from '../icon.gts';
import type { ModalHandle } from '../modal.gts';
import Text from '../text.gts';
import XStack from '../x-stack.gts';

export interface LunaLayoutModalSignature {
	Element: HTMLDivElement;
	Args: {
		size?: string;
		padding?: string;
		modal: ModalHandle;
		disableCloseButton?: boolean;
		onCancel?: () => void;
	};
	Blocks: {
		body: [];
		title: [];
		intro?: [];
		sidebar?: [];
		footer?: [];
	};
}

const LunaLayoutModal: TOC<LunaLayoutModalSignature> = <template>
	<div
		class="container"
		data-size={{or @size "md"}}
		data-padding={{or @padding "auto"}}
		data-named-blocks="
    {{if (has-block 'intro') 'intro'}}
    {{if (has-block 'sidebar') 'sidebar'}}
    {{if (has-block 'footer') 'footer'}}
  "
		...attributes
		{{detectScrollable target="[data-scrollable-target]"}}
	>
		<div class="layout">
			<div class="header">
				<Heading @font="500" @weight="strong" id={{@modal.titleId}}>
					{{yield to="title"}}
				</Heading>

				{{#if (has-block "intro")}}
					<Text @font="400" @weight="weak" @color="gray-500" @tag="p" class="intro">
						{{yield to="intro"}}
					</Text>
				{{/if}}
			</div>

			{{#if (has-block "sidebar")}}
				<div class="split-layout">
					<div class="sidebar">
						{{yield to="sidebar"}}
					</div>

					<div class="body" data-scrollable-target>
						{{yield to="body"}}
					</div>
				</div>

			{{else}}
				<div class="body" data-scrollable-target>
					{{yield to="body"}}
				</div>
			{{/if}}

			{{#if (has-block "footer")}}
				<XStack @spacing="md" class="footer">
					{{yield to="footer"}}
				</XStack>
			{{/if}}
		</div>

		<Button
			data-test-close-modal
			@size="sm"
			@variant="transparent"
			@disabled={{@disableCloseButton}}
			class="x"
			aria-label={{t "layout.modal.cancel"}}
			{{on "click" (or @onCancel @modal.cancel)}}
		>
			<Icon @icon="close" />
		</Button>
	</div>
</template>;

export default LunaLayoutModal;
navigation.gts
import { hash } from '@ember/helper';
import Component from '@glimmer/component';

import element_ from 'ember-element-helper/helpers/element';

import a11yMenu from '../../modifiers/a11y-menu.ts';
import type { ElementFromTagName, OptionalTagAttr, OptionalTagName } from '../../types.ts';
import YStack from '../y-stack.gts';
import NavigationLink from './navigation/link.gts';
import NavigationSection from './navigation/section.gts';

export interface LunaLayoutNavigationSignature<TagName extends OptionalTagName> {
	Element: ElementFromTagName<TagName, HTMLElement>;
	Args: OptionalTagAttr<TagName>;
	Blocks: {
		default: [
			{
				Section: typeof NavigationSection;
				Link: typeof NavigationLink;
			},
		];
	};
}

export default class LunaLayoutNavigation<TagName extends OptionalTagName> extends Component<
	LunaLayoutNavigationSignature<TagName>
> {
	NavigationLink = NavigationLink;
	NavigationSection = NavigationSection;

	get tagName(): keyof HTMLElementTagNameMap {
		return this.args.tag ?? 'nav';
	}
	<template>
		{{#let (element_ this.tagName) as |Tag|}}
			<Tag class="focus-container" ...attributes>
				<YStack @tag="ul" @spacing="sm" class="stack" {{a11yMenu memorize=true}}>
					{{yield (hash Section=this.NavigationSection Link=this.NavigationLink)}}
				</YStack>
			</Tag>
		{{/let}}
	</template>
}
