interface LunaRadioGroupSignature {
	Element: HTMLDivElement;
	Args: {
		/**
		 * The ID of the element that describes the radio group.
		 * Should be only ever be used in addition to the aria label.
		 */
		ariaDescribedBy?: string;

		/**
		 * The label of the radio group. Must be used if the
		 * `ariaLabelledBy` argument is not provided.
		 */
		ariaLabel?: string;

		/**
		 * The ID of the radiogroup's label element. Must be used
		 * if an `ariaLabel` argument is not provided.
		 */
		ariaLabelledBy?: string;

		/**
		 * The ID of the radio group. If none is provided, a unique ID will be
		 * generated or automatically provided when using Luna Field.
		 */
		id?: string;

		/**
		 * Whether the radio group is disabled. When true, sets the
		 * `aria-required` and `aria-disabled` attributes accordingly.
		 * Additionally, `tabindex="0"` will not be applied to the group and
		 * child radio buttons will be `disabled`.
		 */
		isDisabled?: boolean;

		/**
		 * Whether the radio buttons are to be laid out horizontally.
		 * When true, sets the container display to `flex`.
		 */
		isHorizontal?: boolean;

		/**
		 * The action to run when the selected value changes.
		 */
		onChange?: (selectedValue: string) => unknown;

		/**
		 * An array of  options to loop through and create radio buttons from.
		 */
		options?: RadioButtonOption[];

		/**
		 * The initially selected value. Required for block-constructed radio groups.
		 * If a value is not supplied, the first non-disabled option will be selected.
		 */
		selectedValue?: string | null;

		/**
		 * The size of the radio buttons and their spacing.
		 */
		size?: RadioButtonSize;
	};
	Blocks: {
		default?: [{ option: WithBoundArgs<typeof RadioButton, 'selectedValue' | 'size' | 'onSelect'> }];
	};
}

let nextGroupId = 0;

/**
 * A group of accessible radio buttons.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/radio-group.md)
 */
export default class RadioGroup extends Component<LunaRadioGroupSignature> {
	@consume('luna-input-id') inputId!: string;

	groupId = nextGroupId++;

	/**
	 * The name of the radio group. Used by `getOptionId` to generate unique
	 * IDs for radio-button focus targeting.
	 */
	groupName = `luna-radio-group-${this.groupId}`;

	/**
	 * The internally tracked selected value. Starts as the `selectedValue` arg;
	 * updated when a radio button is selected. Used by the `selectedValue` getter.
	 */
	@tracked private _selectedValue: string | null = this.args.selectedValue ?? null;

	/**
	 * Check that the group is properly labelled.
	 */
	constructor(owner: Owner, args: LunaRadioGroupSignature['Args']) {
		super(owner, args);

		assert(
			'either ariaLabel or ariaLabelledBy must be passed in when used outside of Luna Field',
			this.inputId || args.ariaLabel || args.ariaLabelledBy,
		);

		assert('ariaLabel and ariaLabelledBy cannot both be passed in', !(args.ariaLabel && args.ariaLabelledBy));
	}

	get id() {
		return this.args.id ?? this.inputId ?? `luna-select-date-${this.groupId}`;
	}

	get labelId() {
		return this.inputId ? labelIdForField(this.inputId) : null;
	}

	/**
	 * The currently selected value. If no value is selected, defaults to the
	 * first non-disabled option's value if options are provided.
	 */
	get selectedValue(): string | null {
		if (this._selectedValue === null || this._selectedValue === undefined) {
			if (!this.args.options || this.args.options.length === 0) {
				throw new Error('block-constructed radio groups must pass in an initially selected value');
			}
			const firstOption = this.args.options?.find((option) => !option.isDisabled);
			assert(
				'if no selectedValue is provided, there must be at least one non-disabled option',
				firstOption !== undefined,
			);
			return firstOption.value;
		}

		return this._selectedValue;
	}

	/*
	 * The size of the radio buttons. Defaults to 'md' unless specified in the arguments.
	 */
	get size() {
		return this.args.size ?? 'md';
	}

	/**
	 * Utility to generate a unique ID for focus targeting.
	 * Used by `RadioButton` components when an ID is not passed in.
	 */
	getOptionId = (value: string) => {
		return `${this.groupName}-${value}`;
	};

	/**
	 * Keydown handler for the radiogroup container. Handles arrow key navigation.
	 * Per guidelines, moving with the arrow keys changes the selected.
	 */
	handleKeydown = (event: KeyboardEvent) => {
		const { key } = event;

		if (
			key === KEYBOARD_KEYS.RIGHT_ARROW ||
			key === KEYBOARD_KEYS.DOWN_ARROW ||
			key === KEYBOARD_KEYS.LEFT_ARROW ||
			key === KEYBOARD_KEYS.UP_ARROW
		) {
			event.preventDefault();

			const radioGroupElement = event.currentTarget as HTMLElement;
			const radioButtons = Array.from(radioGroupElement.querySelectorAll('[role="radio"]:not([disabled])'));

			if (radioButtons.length === 0) return;

			const currentIndex = radioButtons.findIndex((button) => button.getAttribute('aria-checked') === 'true');

			assert('currentIndex should be valid', currentIndex !== -1);

			let nextIndex: number;

			if (key === KEYBOARD_KEYS.RIGHT_ARROW || key === KEYBOARD_KEYS.DOWN_ARROW) {
				nextIndex = (currentIndex + 1) % radioButtons.length;
			} else {
				nextIndex = (currentIndex - 1 + radioButtons.length) % radioButtons.length;
			}

			const nextButton = radioButtons[nextIndex] as HTMLElement;

			if (nextButton) {
				nextButton.focus();
				nextButton.click();
			}
		}
	};

	/*
	 * Function to select an option. Called on click and when navigating with keyboard.
	 * Updates the locally tracked selection and runs the `onChange` callback.
	 */
	select = (value: string) => {
		this._selectedValue = value;
		this.args.onChange?.(value);
	};

	<template>
		<div
			aria-describedby={{@ariaDescribedBy}}
			aria-disabled={{if @isDisabled "true" "false"}}
			aria-label={{@ariaLabel}}
			aria-labelledby={{if @ariaLabelledBy @ariaLabelledBy (unless @ariaLabel this.labelId)}}
			aria-required={{if @isDisabled "false" "true"}}
			class="RadioGroupWrapper {{if @isHorizontal 'horizontal'}}"
			data-is-horizontal={{if @isHorizontal "true"}}
			data-qid-component-type="Luna::RadioGroup"
			data-size={{this.size}}
			id={{this.id}}
			role="radiogroup"
			tabindex={{unless @isDisabled "0"}}
			{{on "keydown" this.handleKeydown}}
			...attributes
		>
			{{#if @options}}
				{{#each @options as |option|}}
					<RadioButton
						@id={{this.getOptionId option.value}}
						@value={{option.value}}
						@label={{option.label}}
						@isDisabled={{or @isDisabled option.isDisabled}}
						@selectedValue={{this.selectedValue}}
						@size={{this.size}}
						@onSelect={{this.select}}
					/>
				{{/each}}
			{{else}}
				{{yield
					(hash
						option=(component
							RadioButton
							getOptionId=this.getOptionId
							isDisabled=@isDisabled
							onSelect=this.select
							selectedValue=this.selectedValue
							size=this.size
						)
					)
				}}
			{{/if}}
		</div>
	</template>
}

import { createContext, useCallback, useContext, useId, useMemo, useState } from 'react';

import { cn } from '../../utils/cn';
import { useFieldContext } from '../field/field-context';
import { RadioButton, type RadioButtonSize } from '../radio-button';

import styles from './radio-group.module.css';

export interface RadioGroupOption {
	/** The value of the option */
	value: string;
	/** The label to display */
	label: string;
	/** Whether this specific option is disabled */
	isDisabled?: boolean;
}

export interface RadioGroupContextValue {
	/** The currently selected value */
	selectedValue: string | null;
	/** The size of the radio buttons */
	size: RadioButtonSize;
	/** Whether the entire group is disabled */
	isDisabled: boolean;
	/** Callback when an option is selected */
	onSelect: (value: string) => void;
	/** Function to generate unique IDs for options */
	getOptionId: (value: string) => string;
}

const RadioGroupContext = createContext<RadioGroupContextValue | null>(null);

export function useRadioGroupContext() {
	return useContext(RadioGroupContext);
}

export interface RadioGroupProps extends Omit<React.HTMLAttributes<HTMLDivElement>, 'onChange'> {
	/** Forwarded ref to the group root element */
	ref?: React.Ref<HTMLDivElement>;
	/** ARIA label for the group (required if ariaLabelledBy not provided) */
	ariaLabel?: string;
	/** ID of element that labels the group (required if ariaLabel not provided) */
	ariaLabelledBy?: string;
	/** ID of element that describes the group */
	ariaDescribedBy?: string;
	/** Custom ID for the group */
	id?: string;
	/** Whether the entire group is disabled */
	isDisabled?: boolean;
	/** Whether to display options horizontally */
	isHorizontal?: boolean;
	/** Callback fired when selection changes */
	onChange?: (value: string) => void;
	/** Array of options to render */
	options?: RadioGroupOption[];
	/** The currently selected value */
	selectedValue?: string | null;
	/** Size of the radio buttons */
	size?: RadioButtonSize;
	/** Children (for block content usage) */
	children?: React.ReactNode;
}

/**
 * A group of accessible radio buttons.
 *
 * Can be used with either an `options` array or children via `RadioGroup.Option`.
 *
 * @example
 * ```tsx
 * // With options array
 * <RadioGroup
 *   ariaLabel="Select color"
 *   options={[
 *     { value: 'red', label: 'Red' },
 *     { value: 'blue', label: 'Blue' },
 *   ]}
 *   selectedValue={color}
 *   onChange={setColor}
 * />
 *
 * // With children
 * <RadioGroup ariaLabel="Select size" selectedValue={size} onChange={setSize}>
 *   <RadioGroup.Option value="sm">Small</RadioGroup.Option>
 *   <RadioGroup.Option value="md">Medium</RadioGroup.Option>
 *   <RadioGroup.Option value="lg">Large</RadioGroup.Option>
 * </RadioGroup>
 * ```
 */
function RadioGroupRoot(props: RadioGroupProps) {
	const {
		ref,
		ariaLabel,
		ariaLabelledBy,
		ariaDescribedBy,
		id: providedId,
		isDisabled = false,
		isHorizontal = false,
		onChange,
		options,
		selectedValue: selectedValueProp,
		size = 'md',
		className,
		children,
		...rest
	} = props;

	const generatedId = useId();
	const fieldContext = useFieldContext();
	const groupId = providedId ?? fieldContext?.inputId ?? `luna-radio-group-${generatedId}`;
	const groupName = `luna-radio-group-${generatedId}`;
	const ariaLabelledByResolved = ariaLabelledBy ?? fieldContext?.labelId ?? undefined;

	// Internal selected value tracking
	const [internalSelectedValue, setInternalSelectedValue] = useState<string | null>(selectedValueProp ?? null);

	// Determine the effective selected value
	const selectedValue = useMemo(() => {
		const value = selectedValueProp !== undefined ? selectedValueProp : internalSelectedValue;

		// If no value is selected and we have options, default to first non-disabled option
		if (value === null && options && options.length > 0) {
			const firstEnabled = options.find((opt) => !opt.isDisabled);
			return firstEnabled?.value ?? null;
		}

		return value;
	}, [selectedValueProp, internalSelectedValue, options]);

	// Generate unique IDs for options
	const getOptionId = useCallback((value: string) => `${groupName}-${value}`, [groupName]);

	// Handle selection
	const handleSelect = useCallback(
		(value: string) => {
			setInternalSelectedValue(value);
			onChange?.(value);
		},
		[onChange],
	);

	// Handle keyboard navigation
	const handleKeyDown = useCallback((event: React.KeyboardEvent) => {
		const { key } = event;

		if (!['ArrowRight', 'ArrowDown', 'ArrowLeft', 'ArrowUp'].includes(key)) {
			return;
		}

		event.preventDefault();

		const radioGroupElement = event.currentTarget as HTMLElement;
		const radioButtons = Array.from(radioGroupElement.querySelectorAll<HTMLElement>('[role="radio"]:not([disabled])'));

		if (radioButtons.length === 0) return;

		const currentIndex = radioButtons.findIndex((button) => button.getAttribute('aria-checked') === 'true');

		if (currentIndex === -1) return;

		let nextIndex: number;

		if (key === 'ArrowRight' || key === 'ArrowDown') {
			nextIndex = (currentIndex + 1) % radioButtons.length;
		} else {
			nextIndex = (currentIndex - 1 + radioButtons.length) % radioButtons.length;
		}

		const nextButton = radioButtons[nextIndex];

		if (nextButton) {
			nextButton.focus();
			nextButton.click();
		}
	}, []);

	const wrapperClassName = cn(styles.RadioGroupWrapper, isHorizontal && styles.horizontal, className);

	// Context value for child RadioButtons
	const contextValue = useMemo<RadioGroupContextValue>(
		() => ({
			selectedValue,
			size,
			isDisabled,
			onSelect: handleSelect,
			getOptionId,
		}),
		[selectedValue, size, isDisabled, handleSelect, getOptionId],
	);

	return (
		<div
			ref={ref}
			aria-describedby={ariaDescribedBy}
			aria-disabled={isDisabled ? 'true' : 'false'}
			aria-label={ariaLabel}
			aria-labelledby={ariaLabelledByResolved}
			aria-required={isDisabled ? 'false' : 'true'}
			className={wrapperClassName}
			data-is-horizontal={isHorizontal ? 'true' : undefined}
			data-qid-component-type="Luna::RadioGroup"
			data-size={size}
			id={groupId}
			role="radiogroup"
			tabIndex={isDisabled ? undefined : 0}
			onKeyDown={handleKeyDown}
			{...rest}
		>
			<RadioGroupContext.Provider value={contextValue}>
				{options
					? options.map((option) => (
							<RadioButton
								key={option.value}
								id={getOptionId(option.value)}
								value={option.value}
								label={option.label}
								isDisabled={isDisabled || option.isDisabled}
								selectedValue={selectedValue}
								size={size}
								onSelect={handleSelect}
							/>
						))
					: children}
			</RadioGroupContext.Provider>
		</div>
	);
}

// RadioGroup.Option component for block content usage
interface RadioGroupOptionProps {
	/** The value of the option */
	value: string;
	/** The label (or use children) */
	label?: string;
	/** Whether this specific option is disabled */
	isDisabled?: boolean;
	/** Children (label content) */
	children?: React.ReactNode;
}

function RadioGroupOption(props: RadioGroupOptionProps) {
	const { value, label, isDisabled: isDisabledProp = false, children } = props;
	const context = useRadioGroupContext();

	if (!context) {
		throw new Error('RadioGroup.Option must be used within a RadioGroup');
	}

	const { selectedValue, size, isDisabled: groupDisabled, onSelect, getOptionId } = context;

	return (
		<RadioButton
			id={getOptionId(value)}
			value={value}
			label={label}
			isDisabled={groupDisabled || isDisabledProp}
			selectedValue={selectedValue}
			size={size}
			onSelect={onSelect}
		>
			{children}
		</RadioButton>
	);
}

export const RadioGroup = Object.assign(RadioGroupRoot, {
	Option: RadioGroupOption,
});

export default RadioGroup;
