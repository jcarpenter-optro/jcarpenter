● React (TSX)
definition-list.tsx
import { cn } from '../../utils/cn';

import styles from './definition-list.module.css';

export type DefinitionListSize = 'sm' | 'md';

export interface DefinitionListProps extends React.HTMLAttributes<HTMLDListElement> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDListElement>;
	/** Size variant of the definition list */
	size?: DefinitionListSize;
	/** Additional CSS class */
	className?: string;
	/** Definition list content (dt and dd elements) */
	children?: React.ReactNode;
}

/**
 * A styled definition list for displaying term-description pairs.
 *
 * Matches the Luna::DefinitionList Ember component API and styling.
 */
export function DefinitionList(props: DefinitionListProps) {
	const { ref, size = 'md', className, children, ...rest } = props;

	const definitionListClassName = cn(styles.definitionList, className);

	return (
		<dl
			ref={ref}
			className={definitionListClassName}
			data-size={size}
			data-qid-component-type="Luna::DefinitionList"
			{...rest}
		>
			{children}
		</dl>
	);
}

export default DefinitionList;
