● Ember (Glimmer)
search-highlight.gts
import Component from '@glimmer/component';

import { modifier } from 'ember-modifier';
import { consume } from 'ember-provide-consume-context';

import { findAndCreateRanges, highlightRegistry } from '../utils/highlight-registry.ts';

export const SEARCH_QUERY_CONTEXT = 'luna-search-query';

interface SearchHighlightSignature {
	Element: HTMLSpanElement;
	Blocks: {
		default: [];
	};
}

export class SearchHighlight extends Component<SearchHighlightSignature> {
	@consume(SEARCH_QUERY_CONTEXT) contextQuery?: string | null;

	get searchTerm(): string | null {
		return this.contextQuery ?? null;
	}

	applyHighlight = modifier((element: HTMLSpanElement, [term]: [string | null]) => {
		let currentRanges: Range[] = [];

		const walkAndHighlight = () => {
			// Remove previous ranges
			if (currentRanges.length > 0) {
				highlightRegistry.removeRanges(currentRanges);
				currentRanges = [];
			}

			// Create new ranges
			currentRanges = findAndCreateRanges(element, term);
			highlightRegistry.addRanges(currentRanges);
		};

		walkAndHighlight();

		// Watch for yielded content re-renders
		let pendingMicrotask = false;
		const observer = new MutationObserver(() => {
			if (pendingMicrotask) return;
			pendingMicrotask = true;
			queueMicrotask(() => {
				pendingMicrotask = false;
				walkAndHighlight();
			});
		});

		observer.observe(element, {
			childList: true,
			subtree: true,
			characterData: true,
		});

		// Cleanup: runs on destroy and before re-invocation (term change)
		return () => {
			observer.disconnect();
			highlightRegistry.removeRanges(currentRanges);
			currentRanges = [];
		};
	});

	<template>
		<span {{this.applyHighlight this.searchTerm}} data-qid-search-highlight ...attributes>
			{{yield}}
		</span>
	</template>
}
