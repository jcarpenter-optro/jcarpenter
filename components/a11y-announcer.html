import { assert } from '@ember/debug';
import type Owner from '@ember/owner';
import { service } from '@ember/service';
import { htmlSafe } from '@ember/template';
import Component from '@glimmer/component';

import { or } from 'ember-truth-helpers';

import type LunaAnnouncerService from '../services/announcer.ts';
import type { AnnouncerUrgencyLevels } from '../services/announcer.ts';
import type AsyncService from '../services/async.ts';
import announceOnInit from './a11y-announcer/announce-on-init.ts';

export interface A11yAnnouncerSignature {
	Element: HTMLSpanElement;
	Args: {
		/**
		 * The text content to be rendered visibly for sighted users, but hidden from screen readers.
		 * If screenReaderText is not defined, the same text is used for display and screen reader.
		 */
		text?: string;

		/**
		 * Optional text content to be visually hidden but announced to screen readers.
		 * If text is not defined nothing is rendered (good for loading announcements).
		 */
		screenReaderText?: string;

		/**
		 * The urgency level for the announcement.
		 * @default 'polite'
		 */
		urgency?: AnnouncerUrgencyLevels;

		/**
		 * When set to true the text will be announced when the component is initialized.
		 * Example: "No results found" when filtering a list
		 * @default false
		 */
		announceOnInit?: boolean;

		/**
		 * Optional message to announce when the component is destroyed.
		 * @default null
		 */
		destroyMessage?: string;
	};
}

/**
 * A component that can render text content for sighted users while providing
 * accessible announcements for screen reader users via aria-live regions.
 *
 * @example
 * ```hbs
 * <A11yAnnouncer
 *   @text="Item added to cart"
 *   @screenReaderText="Item successfully added to your shopping cart"
 *   @urgency="polite"
 *   @announceOnInit={{true}}
 *   @destroyMessage="Item removed from cart"
 * />
 * ```
 */
export default class A11yAnnouncer extends Component<A11yAnnouncerSignature> {
	@service('luna/announcer') declare announcerService: LunaAnnouncerService;
	@service('luna/async') declare async: AsyncService;

	constructor(owner: Owner, args: A11yAnnouncerSignature['Args']) {
		super(owner, args);
		// Require at least one of these
		assert('Expected @text or @screenReaderText to be defined', args.text || args.screenReaderText);
	}

	get screenReaderText(): string {
		return `<span>${this.args.screenReaderText ?? this.args.text}</span>`;
	}

	get urgency(): AnnouncerUrgencyLevels {
		return this.args.urgency ?? 'polite';
	}

	get hasBothText() {
		return this.args.screenReaderText && this.args.text;
	}

	get renderSeparate() {
		return this.hasBothText && this.args.screenReaderText !== this.args.text;
	}

	get screenReaderOnly() {
		return this.args.screenReaderText && !this.args.text;
	}

	<template>
		<span data-luna-announcer="true" ...attributes>
			{{! Only render if different values for screen and screen reader }}
			{{#if this.renderSeparate}}
				<span data-qid-a11y-announcer-text-content aria-hidden="true">
					{{@text}}
				</span>
			{{/if}}

			<span
				{{! Hide this if screen reader value is unique }}
				data-qid-a11y-announcer-live-region
				class={{if (or this.renderSeparate this.screenReaderOnly) "sr-only"}}
				aria-live={{this.urgency}}
				{{announceOnInit
					urgency=this.urgency
					announceOnInit=@announceOnInit
					announcerService=this.announcerService
					async=this.async
					destroyMessage=@destroyMessage
				}}
			>
				{{htmlSafe this.screenReaderText}}
			</span>
		</span>
	</template>
}

/**
 * A11yAnnouncer: renders aria-live regions for screen reader announcements.
 * When used inside AnnouncerProvider, announcements are coordinated (e.g. only the
 * most recently mounted instance speaks). There is no separate "setup" component;
 * each A11yAnnouncer instance is a live region. Place AnnouncerProvider at app root.
 */
import { useEffect, useId, useRef } from 'react';

import { assert } from '@soxhub/luna-utils';

import { cn } from '../../utils/cn';

import type { AnnouncerUrgencyLevels } from './announcer-context';
import { useAnnouncerOptional } from './announcer-context';

import srOnlyStyles from '../sr-only/sr-only.module.css';

export interface A11yAnnouncerProps extends React.HTMLAttributes<HTMLSpanElement> {
	/**
	 * The text content to be rendered visibly for sighted users, but hidden from screen readers.
	 * If screenReaderText is not defined, the same text is used for display and screen reader.
	 */
	text?: string;

	/**
	 * Optional text content to be visually hidden but announced to screen readers.
	 * If text is not defined nothing is rendered (good for loading announcements).
	 */
	screenReaderText?: string;

	/**
	 * The urgency level for the announcement.
	 * @default 'polite'
	 */
	urgency?: AnnouncerUrgencyLevels;

	/**
	 * When set to true the text will be announced when the component is initialized.
	 * Example: "No results found" when filtering a list
	 * @default false
	 */
	announceOnInit?: boolean;

	/**
	 * Optional message to announce when the component is destroyed.
	 * @default null
	 */
	destroyMessage?: string;
}

/**
 * A component that can render text content for sighted users while providing
 * accessible announcements for screen reader users via aria-live regions.
 *
 * @example
 * ```tsx
 * <A11yAnnouncer
 *   text="Item added to cart"
 *   screenReaderText="Item successfully added to your shopping cart"
 *   urgency="polite"
 *   announceOnInit
 *   destroyMessage="Item removed from cart"
 * />
 * ```
 */
export function A11yAnnouncer(props: A11yAnnouncerProps & { ref?: React.Ref<HTMLSpanElement> }) {
	const { ref, text, screenReaderText, urgency = 'polite', announceOnInit = false, destroyMessage, ...rest } = props;
	const announcer = useAnnouncerOptional();
	const instanceId = useId();
	const hasAnnouncedRef = useRef(false);
	const liveRegionRef = useRef<HTMLSpanElement>(null);

	// Validate that at least one text prop is provided
	assert('Expected text or screenReaderText to be defined', text || screenReaderText);

	// Register/unregister with announcer service
	useEffect(() => {
		if (!announcer) return;

		announcer.registerId(instanceId);

		return () => {
			announcer.unregisterId(instanceId);

			// Announce destroy message if provided
			if (destroyMessage) {
				announcer.announce(destroyMessage, { urgency });
			}
		};
	}, [announcer, instanceId, destroyMessage, urgency]);

	// Announce on init if requested
	useEffect(() => {
		if (!announcer || !announceOnInit || hasAnnouncedRef.current) return;

		// Announce the text content
		const textToAnnounce = screenReaderText ?? text;
		if (textToAnnounce) {
			announcer.announce(textToAnnounce, { urgency });
			hasAnnouncedRef.current = true;
		}
	}, [announcer, announceOnInit, screenReaderText, text, urgency]);

	// Determine what to render
	const resolvedScreenReaderText = screenReaderText ?? text;
	const hasBothText = screenReaderText && text;
	const renderSeparate = hasBothText && screenReaderText !== text;
	const screenReaderOnly = screenReaderText && !text;

	// Determine if we should visually hide the live region
	const shouldHideLiveRegion = renderSeparate || screenReaderOnly;

	return (
		<span data-luna-announcer="true" ref={ref} {...rest}>
			{/* Only render if different values for screen and screen reader */}
			{renderSeparate && (
				<span data-qid-a11y-announcer-text-content aria-hidden="true">
					{text}
				</span>
			)}

			<span
				ref={liveRegionRef}
				data-qid-a11y-announcer-live-region
				className={cn(shouldHideLiveRegion && srOnlyStyles.SrOnly)}
				aria-live={urgency}
			>
				{resolvedScreenReaderText}
			</span>
		</span>
	);
}

export default A11yAnnouncer;
announcer-context.tsx
import { createContext, useCallback, useContext, useMemo, useRef, useState } from 'react';

import { createPriorityQueue } from '@soxhub/luna-utils';

export type AnnouncerUrgencyLevels = 'polite' | 'assertive';

interface AriaAnnounceOptions {
	batchId?: string;
	urgency?: AnnouncerUrgencyLevels;
	priority?: number;
}

export interface AriaLiveMessage {
	message: string;
	createdAt: number;
	priority: number;
	batchId?: string;
	urgency: AnnouncerUrgencyLevels;
}

interface BatchMessage {
	batchId: string;
	message: AriaLiveMessage;
}

// Clean HTML from strings for screen readers
const cleanseStringOfAnyHtml = (str: string): string => {
	const doc = new DOMParser().parseFromString(str, 'text/html');
	return doc.body.textContent ?? '';
};

export interface AnnouncerContextValue {
	announce: (message: string, options?: AriaAnnounceOptions) => void;
	getPoliteMessages: () => AriaLiveMessage[];
	getAssertiveMessages: () => AriaLiveMessage[];
	checkHasMessages: () => boolean;
	registerId: (id: string) => void;
	unregisterId: (id: string) => void;
	shouldInstanceAnnounce: (id: string) => boolean;
}

const AnnouncerContext = createContext<AnnouncerContextValue | null>(null);

export interface AnnouncerProviderProps {
	children: React.ReactNode;
}

/**
 * Provider that manages screen reader announcements via aria-live regions.
 * This provider does not render DOM; the actual aria-live regions are rendered
 * by A11yAnnouncer components that consume this context.
 *
 * Must wrap any components using A11yAnnouncer or the useAnnouncer hook.
 */
export function AnnouncerProvider({ children }: AnnouncerProviderProps) {
	// Use refs for mutable state that shouldn't trigger re-renders
	const messageOrderRef = useRef(0);
	const batchMessagesRef = useRef<BatchMessage[]>([]);
	const instanceIdsRef = useRef<string[]>([]);

	// Use state for the message queue to allow components to subscribe to updates
	const [messageQueue] = useState(() =>
		createPriorityQueue<AriaLiveMessage>((a, b) => {
			if (a.priority !== b.priority) {
				return b.priority - a.priority;
			}
			return a.createdAt - b.createdAt;
		}),
	);

	// Force update state to trigger re-renders when messages are added
	const [, setUpdateTrigger] = useState(0);

	const registerId = useCallback((id: string) => {
		instanceIdsRef.current = [...instanceIdsRef.current, id];
	}, []);

	const unregisterId = useCallback((id: string) => {
		instanceIdsRef.current = instanceIdsRef.current.filter((i) => i !== id);
	}, []);

	const shouldInstanceAnnounce = useCallback((id: string) => {
		return id === instanceIdsRef.current[instanceIdsRef.current.length - 1];
	}, []);

	const announce = useCallback(
		(rawMessage: string, options: AriaAnnounceOptions = {}) => {
			const message = cleanseStringOfAnyHtml(rawMessage);

			// UI state changes immediately after announcing can cause focus changes that
			// cancel reading so wait for the UI first
			setTimeout(() => {
				const { priority = 0, batchId, urgency = 'assertive' } = options;

				const liveMessage: AriaLiveMessage = {
					message,
					createdAt: messageOrderRef.current++,
					priority,
					batchId,
					urgency,
				};

				// Check if batchId exists and update our batch cache if needed
				if (batchId) {
					const batchMessage = batchMessagesRef.current.find((msg) => msg.batchId === batchId);

					if (batchMessage) {
						// Replace existing message in queue
						messageQueue.remove(batchMessage.message);
						// Update list of existing batchIds with most recent message
						batchMessage.message = liveMessage;
					} else {
						// Add new batch entry
						batchMessagesRef.current.push({ batchId, message: liveMessage });
					}
				}

				// Add new message to queue
				messageQueue.enqueue(liveMessage);

				// Trigger re-render for any listening components
				setUpdateTrigger((prev) => prev + 1);
			}, 250);
		},
		[messageQueue],
	);

	const getAndClearMessages = useCallback(
		(targetUrgency: AnnouncerUrgencyLevels): AriaLiveMessage[] => {
			const allCurrentMessages = messageQueue.all();
			const messagesToReturn: AriaLiveMessage[] = [];
			const messagesToKeepInQueue: AriaLiveMessage[] = [];

			for (const msg of allCurrentMessages) {
				if (msg.message.trim().length > 0 && msg.urgency === targetUrgency) {
					messagesToReturn.push(msg);
					// If this message was part of a batch, remove that batch entry
					if (msg.batchId) {
						batchMessagesRef.current = batchMessagesRef.current.filter((batchMsg) => batchMsg.message !== msg);
					}
				} else {
					messagesToKeepInQueue.push(msg);
				}
			}

			// Rebuild queue with messages that weren't taken
			messageQueue.clear();
			messagesToKeepInQueue.forEach((msg) => messageQueue.enqueue(msg));

			return messagesToReturn;
		},
		[messageQueue],
	);

	const getPoliteMessages = useCallback(() => {
		return getAndClearMessages('polite');
	}, [getAndClearMessages]);

	const getAssertiveMessages = useCallback(() => {
		return getAndClearMessages('assertive');
	}, [getAndClearMessages]);

	const checkHasMessages = useCallback(() => {
		return messageQueue.peek() !== undefined;
	}, [messageQueue]);

	const value = useMemo<AnnouncerContextValue>(
		() => ({
			announce,
			getPoliteMessages,
			getAssertiveMessages,
			checkHasMessages,
			registerId,
			unregisterId,
			shouldInstanceAnnounce,
		}),
		[
			announce,
			getPoliteMessages,
			getAssertiveMessages,
			checkHasMessages,
			registerId,
			unregisterId,
			shouldInstanceAnnounce,
		],
	);

	return <AnnouncerContext.Provider value={value}>{children}</AnnouncerContext.Provider>;
}

/**
 * Hook to access the announcer context for making screen reader announcements.
 *
 * @throws Error if used outside of AnnouncerProvider
 */
export function useAnnouncer(): AnnouncerContextValue {
	const context = useContext(AnnouncerContext);

	if (!context) {
		throw new Error('useAnnouncer must be used within an AnnouncerProvider');
	}

	return context;
}

/**
 * Optional hook that returns the announcer context or null when used outside
 * AnnouncerProvider (no provider in the tree). Useful for components that
 * should work with or without announcer support.
 */
export function useAnnouncerOptional(): AnnouncerContextValue | null {
	return useContext(AnnouncerContext);
}
