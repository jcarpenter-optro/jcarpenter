● React (TSX)
heading.tsx
import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './heading.module.css';

export type HeadingLevel = 1 | 2 | 3 | 4 | 5 | 6;
export type LunaFont = '25' | '50' | '100' | '150' | '200' | '300' | '400' | '450' | '500' | '600';
export type HeadingWeight = 'weak' | 'normal' | 'medium' | 'strong';
export type HeadingAlign = 'left' | 'center' | 'right';

interface HeadingBaseProps {
	/** Heading level (h1-h6), defaults to 2 */
	level?: HeadingLevel;
	/** Font size from typography scale, defaults to '300' */
	font?: LunaFont;
	/** Font weight, defaults to 'strong' */
	weight?: HeadingWeight;
	/** Text alignment, defaults to 'left' */
	align?: HeadingAlign;
	/** Whether to truncate text with ellipsis */
	truncate?: boolean;
	/** Children content */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type HeadingProps = ComponentPropsWithRef<HeadingBaseProps, HTMLHeadingElement>;

/**
 * Shows a heading styled using the typography scale.
 *
 * Matches the Luna::Heading Ember component API and styling.
 */
export function Heading(props: HeadingProps) {
	const {
		ref,
		level = 2,
		font = '300',
		weight = 'strong',
		align = 'left',
		truncate = false,
		children,
		className,
		...rest
	} = props;

	const Tag = `h${level}` as 'h1' | 'h2' | 'h3' | 'h4' | 'h5' | 'h6';

	const headingClassName = cn(styles.Heading, className);

	return (
		<Tag
			ref={ref}
			className={headingClassName}
			data-font={font}
			data-weight={weight}
			data-align={align}
			data-truncate={truncate ? 'true' : undefined}
			data-qid-component-type="Luna::Heading"
			{...rest}
		>
			{children}
		</Tag>
	);
}

export default Heading;
