import { useCallback, useEffect, useMemo, useState } from 'react';

import { createDefaultIntl, type IntlLike, RRule } from '@soxhub/luna-utils/rrule';

import { Button } from '../button/button';

export interface TerminationOption {
	id: string;
	value: string;
	labelKey: string;
}

export interface SelectRecurrenceTerminationBuilderProps {
	value: string | null;
	options: TerminationOption[];
	requirements?: {
		required?: boolean;
	};
	onChange: (value: string | null) => void;
	onCancel: () => void;
	intl?: IntlLike;

/**
 * Builder component for configuring recurrence termination options.
 */
export function SelectRecurrenceTerminationBuilder({
	value,
	options,
	requirements,
	onChange,
	onCancel,
	intl: intlProp,
}: SelectRecurrenceTerminationBuilderProps) {
	const intl = useMemo(() => intlProp ?? createDefaultIntl(), [intlProp]);

	// Initialize rule from value
	const [rule, setRule] = useState(() => new RRule(value, intl));

	// Determine selected option from value
	const [selectedOption, setSelectedOption] = useState<TerminationOption | null>(() => {
		if (!value) return null;
		if (value.includes('COUNT')) {
			return options.find((opt) => opt.value === 'count') ?? null;
		}
		if (value.includes('UNTIL')) {
			return options.find((opt) => opt.value === 'until') ?? null;
		}
		return null;
	});

	// Update rule when value changes
	useEffect(() => {
		setRule(new RRule(value, intl));
	}, [value, intl]);

	const handleSaveRule = useCallback(async () => {
		const newValue = rule.string || null;
		onChange(newValue);
	}, [rule, onChange]);

	const handleSelectOption = useCallback(
		(option: TerminationOption | null) => {
			setSelectedOption(option);
			// Reset the rule's termination values
			const newRule = new RRule(value, intl);
			newRule.until = null;
			newRule.count = null;
			setRule(newRule);

			if (!option) {
				// User selected "none" - clear termination and save
				onChange(newRule.string || null);
			}
		},
		[value, intl, onChange],
	);

	const handleUntilDateChange = useCallback(
		(dateValue: string | { start: string; end: string } | null) => {
			if (typeof dateValue === 'string' && dateValue) {
				const newRule = new RRule(value, intl);
				newRule.setUntilDate(dateValue);
				setRule(newRule);
			}
		},
		[value, intl],
	);

	const handleUntilTimeChange = useCallback(
		(timeValue: string) => {
			if (timeValue) {
				const newRule = new RRule(value, intl);
				if (newRule.until) {
					newRule.setUntilTime(timeValue);
					setRule(newRule);
				}
			}
		},
		[value, intl],
	);

	const handleCountChange = useCallback(
		(countValue: string) => {
			const newRule = new RRule(value, intl);
			newRule.count = countValue || null;
			setRule(newRule);
		},
		[value, intl],
	);

	// If an option is selected, show the builder form
	if (selectedOption) {
		return (
			<div className={styles.Builder} data-qid-select-recurrence-builder>
				<YStack spacing="md">
					<XStack spacing="md">
						<IconButton
							size="sm"
							variant="gray"
							tooltip={intl.t('input.selectRecurrence.custom.back')}
							icon="arrow-left"
							data-qid-recurrence-termination-back
							onClick={() => setSelectedOption(null)}
						/>
						<Text font="200" weight="strong" color="gray-900">
							{intl.t(selectedOption.labelKey)}
						</Text>
					</XStack>

					<Form onSubmit={handleSaveRule}>
						{() => (
							<YStack spacing="md">
								{selectedOption.value === 'until' && (
									<XStack spacing="md" width="full">
										<Field
											className={styles.date}
											label={intl.t('input.select-recurrence-termination.field.end-date')}
											input={({ id }) => (
												<SelectDate
													id={id}
													value={rule.untilDate}
													onChange={handleUntilDateChange}
													requirements={{ required: true }}
													data-qid-recurrence-termination-date
												/>
											)}
										/>
										<Field
											className={styles.time}
											label={intl.t('input.select-recurrence-termination.field.end-time')}
											input={({ id }) => (
												<Input
													id={id}
													type="time"
													value={rule.untilTime ?? ''}
													onChange={handleUntilTimeChange}
													data-qid-recurrence-termination-time
												/>
											)}
										/>
									</XStack>
								)}

								{selectedOption.value === 'count' && (
									<Field
										label={intl.t('input.select-recurrence-termination.field.number')}
										input={({ id }) => (
											<Input
												id={id}
												type="number"
												value={rule.count ?? ''}
												onChange={handleCountChange}
												requirements={{ required: true }}
												data-qid-recurrence-termination-count
											/>
										)}
									/>
								)}

								<XStack spacing="md" width="full" className={styles.actions}>
									<div style={{ flex: 1 }} />
									<Button variant="transparent" data-qid-recurrence-termination-cancel onClick={onCancel}>
										{intl.t('input.select-recurrence-termination.cancel')}
									</Button>
									<Button type="submit" variant="blue" data-qid-recurrence-termination-save>
										{intl.t('input.select-recurrence-termination.apply')}
									</Button>
								</XStack>
							</YStack>
						)}
					</Form>
				</YStack>
			</div>
		);
	}

	// Show the option list
	return (
		<Menu data-qid-select-recurrence-presets>
			{requirements?.required !== true && (
				<Menu.Option
					selected={!value}
					data-qid-select-recurrence-termination-option="none"
					onClick={() => handleSelectOption(null)}
				>
					{intl.t('input.select-recurrence-termination.placeholder')}
				</Menu.Option>
			)}

			{options.map((option) => (
				<Menu.Option
					key={option.id}
					selected={value === option.value}
					data-qid-select-recurrence-termination-option={option.id}
					onClick={() => handleSelectOption(option)}
				>
					{intl.t(option.labelKey)}
				</Menu.Option>
			))}
		</Menu>
	);
}

export default SelectRecurrenceTerminationBuilder;

				size={size}
				width="full"
				errored={manager.showErrorMessage}
				disabled={disabled}
				active={triggerIsActive}
				role="combobox"
				aria-controls={popupId}
				aria-expanded={showPopup}
				aria-owns={popupId}
				aria-haspopup="dialog"
				aria-required={requirements?.required ? 'true' : 'false'}
				aria-label={ariaLabel}
				aria-labelledby={computedAriaLabelledBy}
				aria-describedby={manager.ariaDescribedBy}
				aria-invalid={manager.showErrorMessage ? 'true' : undefined}
				onFocus={handleFocusIn}
				onClick={handleOpen}
			>
				{iconSlot ? <div className={styles.preIcon}>{iconSlot}</div> : null}

				<div className={styles.label}>
					{children
						? children(value)
						: (formattedSelection ?? placeholder ?? intl.t('input.select-recurrence-termination.placeholder'))}
				</div>

				{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
					<Spinner size="sm" transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined} />
				)}

				<Icon className={styles.icon} icon="chevron-down" size="sm" />
			</Button>

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
				{manager.errorMessageToDisplay}
			</InputError>

			{showPopup && (
				<Popup
					className={styles.popup}
					id={popupId}
					show={showPopup}
					relativeTo={triggerRef}
					placement={placement}
					trapFocus
					role="dialog"
					aria-label={!placeholder && !value ? intl.t('input.select-recurrence-termination.placeholder') : undefined}
					willClose={handleWillClose}
					onClose={handleClose}
				>
					<SelectRecurrenceTerminationBuilder
						value={value}
						options={options}
						requirements={requirements}
						onChange={handleBuilderChange}
						onCancel={handleBuilderCancel}
						intl={intl}
					/>
				</Popup>
			)}
		</div>
	);
}

export default SelectRecurrenceTermination;
