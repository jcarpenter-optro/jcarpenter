import type { TOC } from '@ember/component/template-only';
import { on } from '@ember/modifier';

import t from 'ember-intl/helpers/t';
import or from 'ember-truth-helpers/helpers/or';

import detectScrollable from '../../modifiers/detect-scrollable.ts';
import Button from '../button.gts';
import Heading from '../heading.gts';
import Icon from '../icon.gts';
import type { ModalHandle } from '../modal.gts';
import Text from '../text.gts';
import XStack from '../x-stack.gts';

export interface LunaLayoutModalSignature {
	Element: HTMLDivElement;
	Args: {
		size?: string;
		padding?: string;
		modal: ModalHandle;
		disableCloseButton?: boolean;
		onCancel?: () => void;
	};
	Blocks: {
		body: [];
		title: [];
		intro?: [];
		sidebar?: [];
		footer?: [];
	};
}

const LunaLayoutModal: TOC<LunaLayoutModalSignature> = <template>
	<div
		class="container"
		data-size={{or @size "md"}}
		data-padding={{or @padding "auto"}}
		data-named-blocks="
    {{if (has-block 'intro') 'intro'}}
    {{if (has-block 'sidebar') 'sidebar'}}
    {{if (has-block 'footer') 'footer'}}
  "
		...attributes
		{{detectScrollable target="[data-scrollable-target]"}}
	>
		<div class="layout">
			<div class="header">
				<Heading @font="500" @weight="strong" id={{@modal.titleId}}>
					{{yield to="title"}}
				</Heading>

				{{#if (has-block "intro")}}
					<Text @font="400" @weight="weak" @color="gray-500" @tag="p" class="intro">
						{{yield to="intro"}}
					</Text>
				{{/if}}
			</div>

			{{#if (has-block "sidebar")}}
				<div class="split-layout">
					<div class="sidebar">
						{{yield to="sidebar"}}
					</div>

					<div class="body" data-scrollable-target>
						{{yield to="body"}}
					</div>
				</div>

			{{else}}
				<div class="body" data-scrollable-target>
					{{yield to="body"}}
				</div>
			{{/if}}

			{{#if (has-block "footer")}}
				<XStack @spacing="md" class="footer">
					{{yield to="footer"}}
				</XStack>
			{{/if}}
		</div>

		<Button
			data-test-close-modal
			@size="sm"
			@variant="transparent"
			@disabled={{@disableCloseButton}}
			class="x"
			aria-label={{t "layout.modal.cancel"}}
			{{on "click" (or @onCancel @modal.cancel)}}
		>
			<Icon @icon="close" />
		</Button>
	</div>
</template>;

export default LunaLayoutModal;
React (TSX)
modal-layout.tsx
import type { HTMLAttributes, ReactNode } from 'react';
import { useRef } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { extractCompoundSlots } from '../../utils/extract-compound-slots';
import { Button } from '../button';
import { Heading } from '../heading';
import { Icon } from '../icon';
import { Text } from '../text';
import { XStack } from '../x-stack';

import styles from './modal-layout.module.css';

const MODAL_LAYOUT_SLOT_MAP = {
	'ModalLayout.Title': 'title',
	'ModalLayout.Intro': 'intro',
	'ModalLayout.Body': 'body',
	'ModalLayout.Sidebar': 'sidebar',
	'ModalLayout.Footer': 'footer',
} as const;

export type ModalLayoutSize = 'sm' | 'md' | 'lg' | 'xl' | 'full';
export type ModalLayoutPadding = 'auto' | 'none';

// ============================================================================
// Title Subcomponent
// ============================================================================

export interface TitleProps {
	children: ReactNode;
}

/**
 * The title for the modal layout.
 */
export function Title({ children }: TitleProps) {
	return <>{children}</>;
}

Title.displayName = 'ModalLayout.Title';

// ============================================================================
// Intro Subcomponent
// ============================================================================

export interface IntroProps {
	children: ReactNode;
}

/**
 * Optional intro/description text for the modal.
 */
export function Intro({ children }: IntroProps) {
	return <>{children}</>;
}

Intro.displayName = 'ModalLayout.Intro';

// ============================================================================
// Body Subcomponent
// ============================================================================

export interface BodyProps {
	children: ReactNode;
}

/**
 * The main body content for the modal.
 */
export function Body({ children }: BodyProps) {
	return <>{children}</>;
}

Body.displayName = 'ModalLayout.Body';

// ============================================================================
// Sidebar Subcomponent
// ============================================================================

export interface SidebarProps {
	children: ReactNode;
}

/**
 * Optional sidebar content for the modal.
 */
export function Sidebar({ children }: SidebarProps) {
	return <>{children}</>;
}

Sidebar.displayName = 'ModalLayout.Sidebar';

// ============================================================================
// Footer Subcomponent
// ============================================================================

export interface FooterProps {
	children: ReactNode;
}

/**
 * Optional footer content (typically buttons) for the modal.
 */
export function Footer({ children }: FooterProps) {
	return <>{children}</>;
}

Footer.displayName = 'ModalLayout.Footer';

// ============================================================================
// Main Component
// ============================================================================

interface ModalLayoutBaseProps extends Omit<HTMLAttributes<HTMLDivElement>, 'title'> {
	/** Size variant */
	size?: ModalLayoutSize;
	/** Padding mode */
	padding?: ModalLayoutPadding;
	/** ID for the title element (for aria-labelledby) */
	titleId?: string;
	/** Children components - use ModalLayout.Title, ModalLayout.Body, etc. */
	children: ReactNode;
	/** Disable the close button */
	disableCloseButton?: boolean;
	/** Custom cancel handler */
	onCancel?: () => void;
	/** Close button aria-label */
	closeButtonLabel?: string;
}

export type ModalLayoutProps = ComponentPropsWithRef<ModalLayoutBaseProps, HTMLDivElement>;

/**
 * Standard layout for modal content with title, body, optional sidebar, and footer.
 *
 * @example
 * ```tsx
 * <Modal open onClose={handleClose}>
 *   <ModalLayout onCancel={handleClose}>
 *     <ModalLayout.Title>Edit Item</ModalLayout.Title>
 *     <ModalLayout.Intro>Make changes to your item below.</ModalLayout.Intro>
 *     <ModalLayout.Body>
 *       <form>...</form>
 *     </ModalLayout.Body>
 *     <ModalLayout.Footer>
 *       <Button variant="secondary" onClick={handleClose}>Cancel</Button>
 *       <Button onClick={handleSave}>Save</Button>
 *     </ModalLayout.Footer>
 *   </ModalLayout>
 * </Modal>
 * ```
 */
function ModalLayoutComponent({
	ref,
	size = 'md',
	padding = 'auto',
	titleId,
	children,
	disableCloseButton = false,
	onCancel,
	closeButtonLabel = 'Close',
	className,
	...rest
}: ModalLayoutProps) {
	const bodyRef = useRef<HTMLDivElement>(null);
	const slots = extractCompoundSlots(children, MODAL_LAYOUT_SLOT_MAP);

	const titleContent = slots.title;
	const introContent = slots.intro;
	const bodyContent = slots.body;
	const sidebarContent = slots.sidebar;
	const footerContent = slots.footer;
	const hasSidebar = Boolean(sidebarContent);
	const hasFooter = Boolean(footerContent);

	return (
		<div
			ref={ref}
			className={cn(styles.container, className)}
			data-size={size}
			data-padding={padding}
			data-has-sidebar={hasSidebar}
			data-has-footer={hasFooter}
			data-qid-component-type="Luna::Layout::Modal"
			{...rest}
		>
			<div className={styles.layout}>
				<div className={styles.header}>
					<Heading font="500" weight="strong" id={titleId}>
						{titleContent}
					</Heading>

					{introContent && (
						<Text font="400" weight="weak" color="gray-500" tag="p" className={styles.intro}>
							{introContent}
						</Text>
					)}
				</div>

				{hasSidebar ? (
					<div className={styles.splitLayout}>
						<div className={styles.sidebar}>{sidebarContent}</div>
						<div ref={bodyRef} className={styles.body} data-scrollable-target tabIndex={-1}>
							{bodyContent}
						</div>
					</div>
				) : (
					<div ref={bodyRef} className={styles.body} data-scrollable-target tabIndex={-1}>
						{bodyContent}
					</div>
				)}

				{hasFooter && (
					<XStack spacing="md" className={styles.footer}>
						{footerContent}
					</XStack>
				)}
			</div>

			<Button
				data-test-close-modal
				size="sm"
				variant="transparent"
				disabled={disableCloseButton}
				className={styles.closeButton}
				aria-label={closeButtonLabel}
				onClick={onCancel}
			>
				<Icon icon="close" />
			</Button>
		</div>
	);
}

// Attach subcomponents
const ModalLayout = Object.assign(ModalLayoutComponent, {
	Title,
	Intro,
	Body,
	Sidebar,
	Footer,
});

export { ModalLayout };

export default ModalLayout;
