import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { t } from '../../utils/i18n';

import styles from './input-error.module.css';

export type InputErrorAlign = 'left' | 'right';

interface InputErrorBaseProps {
	/** Whether to show the error */
	showWhen: boolean;
	/** Alignment of the error message */
	align?: InputErrorAlign;
	/** Additional CSS class */
	className?: string;
	/** Children (error message content) */
	children?: React.ReactNode;
}

export type InputErrorProps = ComponentPropsWithRef<InputErrorBaseProps, HTMLDivElement>;

/**
 * An error message component for form inputs.
 *
 * Matches the Luna::InputError Ember component API and styling.
 */
export function InputError(props: InputErrorProps) {
	const { ref, showWhen, align = 'right', className, children, ...rest } = props;

	if (!showWhen) {
		return null;
	}

	const inputErrorClassName = cn(styles.InputError, className);

	return (
		<div
			className={inputErrorClassName}
			aria-live="assertive"
			data-align={align}
			data-qid-input-error
			data-qid-component-type="Luna::InputError"
			ref={ref}
			{...rest}
		>
			<div className={styles.layout}>
				<svg
					className={styles.icon}
					xmlns="http://www.w3.org/2000/svg"
					viewBox="0 0 16 16"
					aria-label={t('input-error.ariaError')}
				>
					<path
						d="M8 0a8 8 0 11-8 8 8 8 0 018-8zm0 14.48A6.48 6.48 0 101.52 8 6.48 6.48 0 008 14.48zM8 12a.93.93 0 00.93-.93.93.93 0 10-1.86 0A.92.92 0 008 12zm-.86-2.76h1.72V4H7.14z"
						fill="currentColor"
					></path>
				</svg>
				<div className={styles.text}>{children}</div>
			</div>
		</div>
	);
}

export default InputError;
