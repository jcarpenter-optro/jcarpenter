import type { PropertyMatchWithMessage } from '@auditboard/utils/types';

import tooltip from '../modifiers/tooltip.ts';
import type { LunaColor } from '../types.ts';
import type { LunaButtonArgs } from './button.gts';
import Button from './button.gts';
import Icon from './icon.gts';

// IMPORTANT: Any arguments added here should also be passed through to the Luna::Button in the template block for this component
type AcceptedLunaButtonArgs =
	| 'align'
	| 'variant'
	| 'size'
	| 'width'
	| 'type'
	| 'disabled'
	| 'disabledTooltip'
	| 'loading'
	| 'active'
	| 'focused'
	| 'errored'
	| 'transitionTo'
	| 'openInQuickView'
	| 'shouldRenderAsDiv'
	| 'href'
	| 'target'
	| 'rel'
	| 'download';
type ArgsToButton = Pick<LunaButtonArgs, AcceptedLunaButtonArgs>;

// Throw our error if AcceptedLunaButtonArgs doesn't match LunaButtonArgs properties
// Why are we doing this?
// Luna::IconButton should be able to receive all valid Luna::Button arguments and pass them along to Luna::Button
// There are suggestions around providing a way to spread/splat arguments but currently this is not possible in Ember
// See: https://github.com/emberjs/rfcs/blob/1c90e48eeec9a990627a6ce1e9f70391ba95bbb7/text/0000-spreadable-arguments.md
const _: true = true satisfies PropertyMatchWithMessage<
	Required<ArgsToButton>,
	Required<LunaButtonArgs>,
	// eslint-disable-next-line @soxhub/i18n/no-bare-strings
	'Types for Luna::Button and Luna::IconButton are out of alignment. Update AcceptedLunaButtonArgs in Luna::IconButton and adjust the hbs file to match.'
>;

interface LunaIconButtonArgs extends LunaButtonArgs {
	// IconButton specific args
	icon: string;
	tooltip: string;
	iconColor?: LunaColor;
}

export interface LunaIconButtonSignature<Args extends LunaIconButtonArgs> {
	Element: Args extends { href: string } ? HTMLAnchorElement : HTMLButtonElement;
	Args: Args;
}

/**
 * A simple interactive element for repeated, common, or persistent actions when screen real estate is limited. Built with accessibility features and all functionality from [Button](https://docs.auditboardteam.com/Luna/components/components/button.md) using [icons from the library](https://docs.auditboardteam.com/Luna/components/components/icon.md).
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/icon-button.md)
 */
export default class IconButton<Args extends LunaIconButtonArgs> extends Component<LunaIconButtonSignature<Args>> {
	<template>
		<Button
			@variant={{@variant}}
			@size={{@size}}
			@width={{@width}}
			@type={{@type}}
			@disabled={{@disabled}}
			@disabledTooltip={{@disabledTooltip}}
			@loading={{@loading}}
			@active={{@active}}
			@focused={{@focused}}
			@errored={{@errored}}
			@transitionTo={{@transitionTo}}
			@openInQuickView={{@openInQuickView}}
			@shouldRenderAsDiv={{@shouldRenderAsDiv}}
			@href={{@href}}
			@target={{@target}}
			@rel={{@rel}}
			@download={{@download}}
			aria-label={{@tooltip}}
			{{tooltip @tooltip shouldAnnounce=false disabled=@disabled}}
			data-qid-button-type="IconButton"
			...attributes
		>
			<Icon @icon={{@icon}} @color={{@iconColor}} />
		</Button>
	</template>
}

export type { LunaIconButtonArgs };

icon-button.tsx
import { Button, type ButtonProps } from '../button';
import { Icon } from '../icon';

export type LunaColor = string; // TODO: Import from proper types when available

export interface IconButtonProps extends Omit<ButtonProps, 'children'> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLButtonElement | HTMLAnchorElement>;
	/** Icon name (required) */
	icon: string;
	/** Tooltip text (also used as aria-label) */
	tooltip: string;
	/** Color of the icon */
	iconColor?: LunaColor;
}

/**
 * A simple interactive element for repeated, common, or persistent actions when screen real estate is limited.
 * Built with accessibility features and all functionality from Button using icons from the library.
 *
 * Matches the Luna::IconButton Ember component API and styling.
 */
export function IconButton(props: IconButtonProps) {
	const { ref, icon, tooltip, iconColor, ...buttonProps } = props;

	return (
		// @ts-expect-error - Button union type creates incompatible intersection
		<Button {...buttonProps} ref={ref} aria-label={tooltip} data-qid-button-type="IconButton">
			<Icon icon={icon} color={iconColor} />
		</Button>
	);
}

export default IconButton;
