import { service } from '@ember/service';
import { htmlSafe as trusted } from '@ember/template';
import Component from '@glimmer/component';

import { element } from 'ember-element-helper';
import type IntlService from 'ember-intl/services/intl';
import { eq, or } from 'ember-truth-helpers';

import { legacyColorMap } from '../const/avatar-color.ts';
import type { ElementFromTagName, OptionalTagAttr, OptionalTagName } from '../types.ts';
import { getHexColorContrast } from '../utils/a11y.ts';
import { normalizeColorValueToHex } from '../utils/color.ts';
import Icon from './icon.gts';
import SrOnly from './sr-only.gts';

const WHITE = '#FFFFFF';
const BLACK = '#000000';
const MINIMUM_CONTRAST = 4.5;
const DEFAULT_TEXT_COLOR = BLACK;

export type Slot = 'header' | 'sidebar' | 'body' | 'toolkit';

export interface LunaAvatarArgs {
	class?: string;
	image?: string;
	size?: 'sm' | 'md' | 'lg';
	// May actually be required
	status?: 'success' | 'error';
	bgColor?: string | null;
	name?: string;
	initials?: string;
	fullName?: string;
	id?: string;
}

export interface LunaAvatarSignature<TagName extends OptionalTagName> {
	Element: ElementFromTagName<TagName>;
	Args: LunaAvatarArgs & OptionalTagAttr<TagName>;
	Blocks: {
		default?: [];
	};
}

/**
 * An avatar is a visual representation of a user or entity.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/avatar.md)
 */
export default class LunaAvatar<TagName extends keyof HTMLElementTagNameMap | undefined> extends Component<
	LunaAvatarSignature<TagName>
> {
	@service declare intl: IntlService;

	get tagName(): keyof HTMLElementTagNameMap {
		return this.args.tag ?? 'div';
	}

	get bgColor(): string | undefined {
		return this.args.bgColor || undefined;
	}

	/**
	 * Determines by background color what text color to use for an appropriate contrast ratio to meet WCAG guidelines
	 */
	get textColor(): string | undefined {
		const value = this.bgColor;
		if (typeof value !== 'undefined') {
			const colorValue = normalizeColorValueToHex(value);
			// if we have a predefined mapping, use it
			if (colorValue) {
				if (legacyColorMap.has(colorValue)) {
					return legacyColorMap.get(colorValue) ?? DEFAULT_TEXT_COLOR;
				}
				// In the wild, there are already several arbitrary values that were stored before the predefined values were set,
				// so this check will cover any color by checking which has higher contrast
				const blackTextContrast = getHexColorContrast(BLACK, colorValue);
				const whiteTextContrast = getHexColorContrast(WHITE, colorValue);

				// if we don't meet minimum contrast on either, we default to the opposite
				if (blackTextContrast < MINIMUM_CONTRAST) return WHITE;
				if (whiteTextContrast < MINIMUM_CONTRAST) return BLACK;

				return whiteTextContrast < blackTextContrast ? BLACK : WHITE;
			}
		}

		return DEFAULT_TEXT_COLOR;
	}

	get style() {
		const { status } = this.args;

		if (!status && this.bgColor) {
			return trusted(`background-color: ${this.bgColor}; color: ${this.textColor};`);
		}

		return;
	}

	get srName() {
		/** Use default 'User' if no name is provided */
		const defaultName = this.intl.t('avatar.user');

		return this.intl.t('avatar.srAvatarName', {
			name: this.args.name || defaultName,
		});
	}

	<template>
		{{#let (element this.tagName) as |Tag|}}
			<Tag
				class="Avatar {{@class}}"
				data-image={{if @image "true" "false"}}
				data-size={{or @size "sm"}}
				data-status={{@status}}
				data-qid-component-type="Luna::Avatar"
				data-qid-avatar-name={{@name}}
				type={{if (eq @tag "button") "button"}}
				style={{this.style}}
				...attributes
			>
				<div class="content">
					{{#if (has-block)}}
						{{yield}}
					{{else if @image}}
						<img alt={{this.srName}} class="image" src={{@image}} />
					{{else if @initials}}
						<span aria-hidden="true" class="initials">{{@initials}}</span>
						<SrOnly @text={{this.srName}} />
					{{else}}
						<Icon @icon="person" @title={{this.srName}} />
					{{/if}}
				</div>
			</Tag>
		{{/let}}
	</template>
}

import { useMemo } from 'react';

import { legacyColorMap } from '@soxhub/luna-utils';
import { getHexColorContrast } from '@soxhub/luna-utils';
import { normalizeColorValueToHex } from '@soxhub/luna-utils';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Icon } from '../icon';

import styles from './avatar.module.css';

const WHITE = '#FFFFFF';
const BLACK = '#000000';
const MINIMUM_CONTRAST = 4.5;
const DEFAULT_TEXT_COLOR = BLACK;

export type AvatarSize = 'sm' | 'md' | 'lg';
export type AvatarStatus = 'success' | 'error';

interface AvatarBaseProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLElement>;
	/** Image URL to display */
	image?: string;
	/** Size of the avatar */
	size?: AvatarSize;
	/** Status indicator */
	status?: AvatarStatus;
	/** Background color (hex or rgb) */
	bgColor?: string | null;
	/** Name for accessibility */
	name?: string;
	/** Initials to display */
	initials?: string;
	/** Full name for accessibility */
	fullName?: string;
	/** ID attribute */
	id?: string;
	/** HTML tag to render as */
	tag?: keyof React.JSX.IntrinsicElements;
	/** Additional CSS class */
	className?: string;
	/** Children content (if provided, overrides image/initials/icon) */
	children?: React.ReactNode;
}

export type AvatarProps = ComponentPropsWithRef<AvatarBaseProps, HTMLElement>;

/**
 * Determines by background color what text color to use for an appropriate contrast ratio to meet WCAG guidelines
 */
function getTextColor(bgColor?: string | null): string {
	if (typeof bgColor === 'undefined' || bgColor === null) {
		return DEFAULT_TEXT_COLOR;
	}

	const colorValue = normalizeColorValueToHex(bgColor);
	if (colorValue) {
		// if we have a predefined mapping, use it
		if (legacyColorMap.has(colorValue)) {
			return legacyColorMap.get(colorValue) ?? DEFAULT_TEXT_COLOR;
		}
		// In the wild, there are already several arbitrary values that were stored before the predefined values were set,
		// so this check will cover any color by checking which has higher contrast
		const blackTextContrast = getHexColorContrast(BLACK, colorValue);
		const whiteTextContrast = getHexColorContrast(WHITE, colorValue);

		// if we don't meet minimum contrast on either, we default to the opposite
		if (blackTextContrast < MINIMUM_CONTRAST) return WHITE;
		if (whiteTextContrast < MINIMUM_CONTRAST) return BLACK;

		return whiteTextContrast < blackTextContrast ? BLACK : WHITE;
	}

	return DEFAULT_TEXT_COLOR;
}

/**
 * An avatar is a visual representation of a user or entity.
 *
 * Matches the Luna::Avatar Ember component API and styling.
 */
export function Avatar(props: AvatarProps) {
	const {
		ref,
		image,
		size = 'sm',
		status,
		bgColor,
		name,
		initials,
		id,
		tag = 'div',
		className,
		children,
		...rest
	} = props;

	const Tag = tag as React.ElementType;

	// Calculate text color based on background color
	const textColor = useMemo(() => getTextColor(bgColor), [bgColor]);

	// Generate inline style for background and text color (only if no status)
	const style = useMemo(() => {
		if (!status && bgColor) {
			return {
				backgroundColor: bgColor,
				color: textColor,
			};
		}
		return undefined;
	}, [status, bgColor, textColor]);

	// Screen reader name
	const srName = useMemo(() => {
		const defaultName = 'User';
		return name ? `${name} avatar` : `${defaultName} avatar`;
	}, [name]);

	const avatarClassName = cn(styles.Avatar, className);

	// Determine content to render
	const content = children ? (
		children
	) : image ? (
		<img alt={srName} className={styles.image} src={image} />
	) : initials ? (
		<>
			<span aria-hidden="true" className={styles.initials}>
				{initials}
			</span>
			<span className={styles.SrOnly}>{srName}</span>
		</>
	) : (
		<Icon icon="person" title={srName} />
	);

	return (
		<Tag
			ref={ref}
			className={avatarClassName}
			data-image={image ? 'true' : 'false'}
			data-size={size}
			data-status={status}
			data-qid-component-type="Luna::Avatar"
			data-qid-avatar-name={name}
			type={tag === 'button' ? 'button' : undefined}
			style={style}
			id={id}
			{...rest}
		>
			<div className={styles.content}>{content}</div>
		</Tag>
	);
}

export default Avatar;
