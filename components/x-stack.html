● React (TSX)
x-stack.tsx
import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './x-stack.module.css';

export type XStackSpacing = 'none' | 'xs' | 'sm' | 'md' | 'lg' | 'xl';
export type XStackAlign = 'start' | 'center' | 'stretch';
export type XStackFlexWrap = 'nowrap' | 'wrap';
export type XStackTag = keyof React.JSX.IntrinsicElements;

interface XStackBaseProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLElement>;
	/** Controls how much distance to place between each child element. Default is 'md'. */
	spacing?: XStackSpacing;
	/** Controls flex alignment of each child element. Default is 'center'. */
	align?: XStackAlign;
	/** How wide the XStack should be. Default is 'full'. */
	width?: string;
	/** Controls if flexbox should wrap or expand items. Default is 'nowrap'. */
	flexWrap?: XStackFlexWrap;
	/** HTML tag to render as, defaults to 'div' */
	tag?: XStackTag;
	/** Children content */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type XStackProps = ComponentPropsWithRef<XStackBaseProps, HTMLElement>;

/**
 * Enables the layout of elements along the X axis using flexbox. Items will not wrap.
 *
 * Matches the Luna::XStack Ember component API and styling.
 */
export function XStack(props: XStackProps) {
	const {
		ref,
		spacing = 'md',
		align = 'center',
		width = 'full',
		flexWrap = 'nowrap',
		tag = 'div',
		children,
		className,
		...rest
	} = props;

	const Tag = tag as React.ElementType;

	const xstackClassName = cn(styles.XStack, className);

	return (
		<Tag
			ref={ref as React.Ref<HTMLElement>}
			className={xstackClassName}
			data-xstack-align={align}
			data-spacing={spacing}
			data-width={width}
			data-flexwrap={flexWrap}
			data-qid-component-type="Luna::XStack"
			{...rest}
		>
			{children}
		</Tag>
	);
}

export default XStack;
