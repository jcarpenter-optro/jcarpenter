/**
 * A component for selecting multiple options.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/select-multiple.md)
 */
export default class SelectMultiple<
	V extends UnknownSelectValue,
	LocalOption extends UnknownSelectValue = V,
	QueryOption extends UnknownSelectValue = V,
	Required extends boolean = boolean,
	I extends ValidInitialSelection<NoInfer<V> | NoInfer<LocalOption>, 'multiple'> = ValidInitialSelection<
		NoInfer<V> | NoInfer<LocalOption>,
		'multiple'
	>,
> extends Select<V, LocalOption, QueryOption, Required, 'multiple', I> {

● React (TSX)

/**
 * SelectMultiple – thin wrapper around Select with selectionType="multiple".
 *
 * Matches the Ember pattern where SelectMultiple extends Select and only sets selectionType.
 */
import type { SelectProps } from '../select/select';
import { Select } from '../select/select';

export type { SelectAlign, SelectSize, SelectVariant } from '../select/select';
export type { SelectOption } from '../select/select';

export interface SelectMultipleProps<T = unknown> extends Omit<
	SelectProps<T>,
	'selectionType' | 'value' | 'onChange' | 'renderTrigger'
> {
	/** Current selected values (array) */
	value?: T[] | null;
	/** Callback when selection changes */
	onChange?: (value: T[], event?: Event) => void;
	/** Custom trigger render function (receives selected array) */
	renderTrigger?: (value: T[]) => React.ReactNode;
}

export function SelectMultiple<T = unknown>(props: SelectMultipleProps<T>): React.ReactElement {
	// Select expects SelectProps<unknown>; our props are compatible at runtime (multi value/onChange)
	return <Select {...(props as unknown as SelectProps)} selectionType="multiple" />;
}

export default SelectMultiple;
