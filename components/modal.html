/**
 * Shows content inside an isolated modal. The modal traps the user's focus so it cannot leave the modal via tabbing. Frequently instantiated by the [`ModalService`](https://docs.auditboardteam.com/Luna/utilities/services/modal).
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/modal.md)
 */
export default class LunaModal extends Component<LunaModalSignature> {
	<template>
		{{#if (notEq @show false)}}
			<Portal @to="luna-modals">
				<div class="Modal" data-variant={{or @variant "light"}} {{didInsert this.onDidInsert}}>
					{{! template-lint-disable no-invalid-interactive }}
					<div
						class="overlay
							{{if (eq this.renderModal (eq this.closingModal false)) 'fade-in'}}
							{{if this.closingModal 'fade-out'}}
							"
						data-enabled={{toStringOrEmpty (or @quickClose false)}}
						data-animation-duration="{{this.animationDuration}}ms"
						aria-hidden="true"
						data-test-overlay
						data-qid-component-type="Luna::Modal"
						{{on "click" (if @quickClose this.onQuickClose noop)}}
					></div>

					<Frame
						@highlight={{this.highlight}}
						@padding={{this.padding}}
						@radius={{this.radius}}
						data-test-modal
						id={{this.windowId}}
						data-animation-duration="{{this.animationDuration}}ms"
						class="window
							{{if (eq this.renderModal (eq this.closingModal false)) 'scale-and-fade-transition-in'}}
							{{if this.closingModal 'scale-and-fade-transition-out'}}
							"
						role="dialog"
						aria-modal="true"
						aria-labelledby={{this.labelledbyId}}
						{{a11yTrapFocus modal=true}}
						...attributes
						{{on "keyup" this.onKeyUp}}
					>
						<Announcer />
						{{yield
							(hash
								close=(perform this.hide (hash cancelled=false))
								cancel=(perform this.hide (hash cancelled=true))
								titleId=this.labelledbyId
							)
						}}

					</Frame>
				</div>
			</Portal>
		{{/if}}
	</template>

	@tracked renderModal = false;
	@tracked closingModal = false;

	windowId = `${guidFor(this)}--window`;
	labelledbyId = `${guidFor(this)}--labelledby`;

	get windowElement() {
		return document.getElementById(this.windowId);
	}

	get animationDuration(): number {
		const owner = getOwner(this);
		assert('Expected an owner', owner instanceof EngineInstance);
		const { environment } = owner.resolveRegistration('config:environment') as { environment: string };

		/**
		 * If the window is not yet in the DOM and we aren't trying to hide it then
		 * we can assume it was not rendered in fastboot.
		 */
		const wasRenderedInFastboot = Boolean(document.querySelector('[role=dialog]')) && !this.hide.isRunning;
		const animationIsDisabled = this.args.animation === 'off';

		return animationIsDisabled || environment === 'test' || wasRenderedInFastboot ? 0 : 190;
	}

	get highlight() {
		return this.args.highlight ?? 'top-teal';
	}

	get padding() {
		return this.args.padding ?? 'none';
	}

	get radius() {
		return this.args.radius ?? 'md';
	}

	@action
	onDidInsert() {
		this.show();
		this.args.onDidInsert?.(this);
	}

	@action
	onQuickClose() {
		if (this.args.quickClose === true) {
			void this.hide.perform({ cancelled: true });
		}
	}

	@action
	onKeyUp(event: KeyboardEvent) {
		if (event.key === KEYBOARD_KEYS.ESC) {
			event.stopPropagation();
			this.onQuickClose();
		}
	}

	/**
	 * Shows the modal and manages the focus state.
	 */
	show() {
		this.renderModal = true;
	}

	/**
	 * Hides the modal and ensures that focus is restored
	 * to where it was before the modal was shown.
	 */
	hide = task(async (options: { cancelled: boolean }) => {
		this.closingModal = true;

		/**
		 * We delay until after the animation had time to finish before the
		 * onClose is called and the Modal is removed from the dom.
		 */
		await timeout(this.animationDuration);
		this.closingModal = false;
		this.renderModal = false;

		this.args.onClose(options);
	});

	willDestroy() {
		super.willDestroy();

		warn(
			'You must close <Luna::Modal> using the yielded `close` method. You most likely used an {{if}} statement to remove the modal.',
			this.hide.performCount > 0,
			{
				id: 'luna-modal-incorrectly-closed',
			},
		);
	}
}

import { createContext, type ReactNode, useCallback, useContext, useId } from 'react';

import { Dialog } from '@base-ui-components/react/dialog';

import { cn } from '../../utils/cn';

import styles from './modal.module.css';

export type ModalVariant = 'light' | 'dark';
export type ModalHighlight = 'top-teal' | 'none';
export type ModalPadding = 'none' | 'sm' | 'md' | 'lg';
export type ModalRadius = 'none' | 'sm' | 'md' | 'lg';

/**
 * Modal handle for controlling the modal from within
 */
export interface ModalHandle {
	/** ID for aria-labelledby */
	titleId: string;
	/** Close the modal (not cancelled) */
	close: () => void;
	/** Cancel the modal (cancelled = true) */
	cancel: () => void;
}

const ModalContext = createContext<ModalHandle | null>(null);

/**
 * Hook to access the modal handle from within the modal
 */
export function useModalContext(): ModalHandle | null {
	return useContext(ModalContext);
}

export interface ModalProps {
	/** Forwarded ref to the modal dialog element */
	'ref'?: React.Ref<HTMLDivElement>;
	/** Visual variant of the modal overlay */
	'variant'?: ModalVariant;
	/** Whether clicking the overlay closes the modal */
	'quickClose'?: boolean;
	/** Disable open/close animations */
	'animation'?: 'off';
	/** Top highlight style */
	'highlight'?: ModalHighlight;
	/** Padding inside the modal */
	'padding'?: ModalPadding;
	/** Border radius */
	'radius'?: ModalRadius;
	/** Whether the modal is visible */
	'show'?: boolean;
	/** Called when the modal is closed */
	'onClose'?: (options: { cancelled: boolean }) => void;
	/** Modal content - use useModalContext() hook to access modal handle */
	'children'?: ReactNode;
	/** Additional class name */
	'className'?: string;
	/** Data attributes */
	'data-testid'?: string;
}

/**
 * Shows content inside an isolated modal. The modal traps the user's focus so it cannot leave the modal via tabbing.
 */
export function Modal({
	ref,
	variant = 'light',
	quickClose = false,
	animation,
	highlight = 'top-teal',
	padding = 'none',
	radius = 'md',
	show = true,
	onClose,
	children,
	className,
	...props
}: ModalProps) {
	const generatedId = useId();
	const titleId = `modal-title-${generatedId}`;
	const animationDisabled = animation === 'off';

	// Handle close
	const handleClose = useCallback(
		(cancelled: boolean) => {
			onClose?.({ cancelled });
		},
		[onClose],
	);

	// Handle open state change from Base UI
	const handleOpenChange = useCallback(
		(open: boolean) => {
			if (!open) {
				handleClose(true); // Treat backdrop click / Esc as cancel
			}
		},
		[handleClose],
	);

	const modalHandle: ModalHandle = {
		titleId,
		close: () => handleClose(false),
		cancel: () => handleClose(true),
	};

	return (
		<Dialog.Root open={show} onOpenChange={handleOpenChange} modal={true}>
			<Dialog.Portal>
				<Dialog.Backdrop
					className={cn(styles.overlay, animationDisabled && styles.noAnimation)}
					data-variant={variant}
					data-enabled={quickClose ? 'true' : 'false'}
					data-testid="modal-overlay"
				/>
				<Dialog.Popup
					ref={ref}
					className={cn(styles.window, animationDisabled && styles.noAnimation, className)}
					data-highlight={highlight}
					data-padding={padding}
					data-radius={radius}
					data-qid-component-type="Luna::Modal"
					data-testid="modal"
					aria-labelledby={titleId}
					{...props}
				>
					<ModalContext.Provider value={modalHandle}>{children}</ModalContext.Provider>
				</Dialog.Popup>
			</Dialog.Portal>
		</Dialog.Root>
	);
}

export default Modal;
