import { on } from '@ember/modifier';
import Component from '@glimmer/component';

import { or } from 'ember-truth-helpers';

enum LabelVariant {
	Default = 'default',
	Light = 'light',
	Caps = 'caps',
}

interface LabelSignature {
	Element: HTMLLabelElement;
	Args: {
		for: string; // Required, but frequently missing
		variant?: `${LabelVariant}`;
		level?: number;
	};
	Blocks: {
		default: [];
	};
}

/**
 * Adds a label for a form input.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/label.md)
 */
export default class Label extends Component<LabelSignature> {
	/**
	 * Click handler for the label. Conditionally clicks or focuses
	 * the associated input depending on its type and whether it's part
	 * of a "click to edit" interface.
	 */
	handleClick = (e: MouseEvent) => {
		e.preventDefault();

		if (!this.args.for) {
			return;
		}

		const input = document.getElementById(this.args.for);

		/**
		 * If the input is a checkbox, we send a click event to toggle its value.
		 */
		if ((input as HTMLInputElement).type === 'checkbox' || input?.getAttribute('role') === 'checkbox') {
			input?.click();
			return;
		}

		/**
		 * `EditableValue` labels initially reference input-styled buttons;
		 * the actual inputs are hidden until the user enables editing.
		 * When a label references one of these buttons, we conditionally click it in
		 * accordance with how the native label/input would have behaved.
		 * All other input references are focused.
		 */

		const editableValueFieldType = input?.getAttribute('data-qid-editable-value-field-type');

		switch (editableValueFieldType) {
			case 'checkbox':
			case 'text':
			case 'textarea':
			case 'number':
				input?.click();
				break;
			default:
				input?.focus();
				break;
		}
	};

	<template>
		<label
			class="Label"
			for={{@for}}
			data-variant={{or @variant "default"}}
			data-level={{or @level 1}}
			data-qid-component-type="Luna::Label"
			...attributes
			{{on "click" this.handleClick}}
		>
			{{yield}}
		</label>
	</template>
}
● React (TSX)
label.tsx
import { useCallback } from 'react';

import { Field } from '@base-ui-components/react/field';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './label.module.css';

export type LabelVariant = 'default' | 'light' | 'caps';
export type LabelLevel = 1 | 2;

interface LabelBaseProps {
	/** ID of the input this label is for (maps to htmlFor) */
	for?: string;
	/** Visual variant */
	variant?: LabelVariant;
	/** Level (1 or 2) - affects color for default variant */
	level?: LabelLevel;
	/** Children content */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type LabelProps = ComponentPropsWithRef<
	LabelBaseProps & Omit<React.LabelHTMLAttributes<HTMLLabelElement>, keyof LabelBaseProps | 'htmlFor'>,
	HTMLLabelElement
>;

/**
 * Click handler for the label. Conditionally clicks or focuses
 * the associated input depending on its type and whether it's part
 * of a "click to edit" interface.
 */
function handleLabelClick(e: React.MouseEvent<HTMLLabelElement>, forId?: string) {
	e.preventDefault();

	if (!forId) {
		return;
	}

	const input = document.getElementById(forId);

	/**
	 * If the input is a checkbox, we send a click event to toggle its value.
	 */
	if ((input as HTMLInputElement)?.type === 'checkbox' || input?.getAttribute('role') === 'checkbox') {
		input?.click();
		return;
	}

	/**
	 * `EditableValue` labels initially reference input-styled buttons;
	 * the actual inputs are hidden until the user enables editing.
	 * When a label references one of these buttons, we conditionally click it in
	 * accordance with how the native label/input would have behaved.
	 * All other input references are focused.
	 */

	const editableValueFieldType = input?.getAttribute('data-qid-editable-value-field-type');

	switch (editableValueFieldType) {
		case 'checkbox':
		case 'text':
		case 'textarea':
		case 'number':
			input?.click();
			break;
		default:
			input?.focus();
			break;
	}
}

/**
 * Adds a label for a form input.
 *
 * Matches the Luna::Label Ember component API and styling.
 * Uses Base UI Field.Label as the foundation.
 */
export function Label(props: LabelProps) {
	const { ref, for: forId, variant = 'default', level = 1, children, className, ...rest } = props;

	const labelClassName = cn(styles.Label, className);

	const handleClick = useCallback(
		(e: React.MouseEvent<HTMLLabelElement>) => {
			handleLabelClick(e, forId);
		},
		[forId],
	);

	// Use Field.Label with render prop to add the `for` attribute and custom click handler
	// Note: We use Field.Root minimally just to provide context, but Field.Label renders the actual label element
	return (
		<Field.Root style={{ display: 'contents' }}>
			<Field.Label
				ref={ref}
				className={labelClassName}
				data-variant={variant}
				data-level={level}
				data-qid-component-type="Luna::Label"
				render={(labelProps) => (
					<label {...labelProps} htmlFor={forId} onClick={handleClick} {...rest}>
						{children}
					</label>
				)}
			/>
		</Field.Root>
	);
}

export default Label;
