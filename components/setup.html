setup.gts
import 'ember-power-select/styles';

import { isDestroyed } from '@ember/destroyable';
import type Owner from '@ember/owner';
import { service } from '@ember/service';
import Component from '@glimmer/component';

import BasicDropdownWormhole from 'ember-basic-dropdown/components/basic-dropdown-wormhole';
import { keyResponder, onKey } from 'ember-keyboard';

import type A11yService from '../services/a11y.ts';
import type LunaCommandService from '../services/command.ts';
import { isMac } from '../utils/browsers.ts';
import Announcer from './setup/announcer.gts';
import Modals from './setup/modals.gts';
import Tooltip from './setup/tooltip.gts';

interface LunaSetupArgs {
	deferA11ySetup?: boolean;
}

export interface LunaSetupSignature<Args extends LunaSetupArgs> {
	Args: Args;
}

@keyResponder
class LunaSetup extends Component<LunaSetupSignature<LunaSetupArgs>> {
	@service('luna/command') declare commandService: LunaCommandService;
	@service('luna/a11y') declare a11y: A11yService;

	constructor(owner: Owner, props: Record<never, unknown>) {
		super(owner, props);
		this.setBodyData();
		// Allow deferring a11y setup in cases where layout may not be ready yet
		if (!this.args.deferA11ySetup) {
			this.a11y.setup();
		}
	}

	setBodyData() {
		if (isMac) {
			document.body.setAttribute('data-device-type', 'mac');
		}
	}

	@onKey('ctrl+k')
	@onKey('meta+k')
	// unfortunate workaround for a bug in @bryntum/schedulerpro, that prevents shortcuts with the meta key to be properly recognized. See https://github.com/bryntum/support/issues/5480
	@onKey('ctrl+meta+k')
	async onKeyDown(event: Event) {
		// Prevent widows from bubbling the default action
		event.preventDefault();

		// For test environment
		if (isDestroyed(this)) {
			return;
		}

		await this.commandService.toggleCommandBar();
	}
	<template>
		<div id="luna-modals"></div>
		<div id="luna-popups"></div>

		<BasicDropdownWormhole />
		<Announcer />
		<Modals />
		<Tooltip />
	</template>
}

export default LunaSetup;
