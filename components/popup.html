export interface LunaPopupBlockValue {
	/**
	 * Task that closes the popup.
	 *
	 * `source` is a string that describes the source of the close request, defaults to 'manual'.
	 * `event` is the DOM event that triggered the close request, if any.
	 */
	closeTask: Task<void, [source?: string, event?: Event]>;

	/**
	 * ID of the content element. Useful for referencing the popup from a nested component,
	 * such as when the OptionList checks whether the popup is focused when determining
	 * how the Enter key should function.
	 */
	id?: string;
}

export interface LunaPopupSignature {
	Element: HTMLDivElement;
	Args: {
		id?: string;
		variant?: 'light' | 'dark' | 'editable-value--schema-table-cell' | 'editable-value';
		relativeTo?: Element | string;
		placement?: Placement | 'overlay';
		strategy?: Strategy;
		trapFocus?: boolean;
		enableArrow?: boolean;
		spacing?: number;
		xSpacing?: number;
		ySpacing?: number;
		show?: boolean;
		disableAnimation?: boolean;
		excludeAnnouncer?: boolean;
		willClose?: (source: OnCloseSource, event: unknown) => Promise<unknown> | Promise<boolean> | void | boolean;
		onClose?: (source: OnCloseSource, event: unknown) => void;
		inheritFieldContext?: boolean;
	};
	Blocks: {
		default: [LunaPopupBlockValue];
	};
}

/**
 * Shows a popup positioned relative to another element. The popup closes if the user presses ESC or clicks outside of it.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/popup.md)
 */
export default class Popup extends Component<LunaPopupSignature> {
	@tracked isClosing = false;

	/**
	 * The arrow element, if it exists. Registered on render and
	 * passed to the `tether-to` modifier for positioning.
	 */
	@tracked arrowElement: HTMLElement | null = null;

	@consume('luna-input-id') upstreamLunaId!: string;

	@provide('luna-input-id')
	get lunaInputId(): string | undefined {
		if (this.args.inheritFieldContext) {
			return this.upstreamLunaId;
		}
	}

	@provide('luna-form')
	get() {
		return null;
	}

	get middleware() {
		const middleware: Middleware[] = [];

		middleware.push(
			// https://floating-ui.com/docs/offset#creating-custom-placements
			offset(({ rects }) => {
				if (this.args.placement === 'overlay') {
					return -rects.reference.height / 2 - rects.floating.height / 2;
				}

				return {
					mainAxis: this.ySpacing,
					crossAxis: this.xSpacing,
				};
			}),
		);

		const sizeMiddleware = size({
			apply: ({ rects, elements, availableHeight }) => {
				if (this.args.placement === 'overlay') {
					Object.assign(elements.floating.style, {
						minWidth: `${rects.reference.width}px`,
						height: `${rects.reference.height}px`,
					});
				}

				/**
				 * If there's still not enough space after the shift, space, set a max-height
				 * on the popup to prevent it from being cropped by the viewport.
				 */
				if (availableHeight < rects.floating.height) {
					Object.assign(elements.floating.style, {
						maxHeight: `${availableHeight}px`,
					});

					/**
					 * Also set a CSS variable so that components inside the popup
					 * can optionally reference to adjust their height accordingly.
					 * Minus 5px to give some visual breathing room for shadows, borders, etc.
					 *
					 * We do not set this in tests to avoid flakiness due to varying viewport sizes in test environments.
					 *
					 * You must use `var(--luna-popup-max-height, <defaultValue>)` in your child's CSS to take advantage of this.
					 */
					if (!macroCondition(isTesting())) {
						elements.floating.style.setProperty('--luna-popup-max-height', `${availableHeight - 5}px`);
					}
				}
			},
		});

		middleware.push(sizeMiddleware);

		return middleware;
	}

	get xSpacing() {
		if (typeof this.args.xSpacing === 'number') {
			return this.args.xSpacing;
		}

		return 0;
	}

	get ySpacing() {
		if (typeof this.args.ySpacing === 'number') {
			return this.args.ySpacing;
		}

		return this.args.enableArrow ? 8 : 2;
	}

	get popupContainer() {
		const element = document.getElementById('luna-popups');
		assert('Expected to have #luna-popups', element);
		return element;
	}

	get placement(): Placement {
		if (this.args.placement === 'overlay') {
			return 'bottom';
		}
		return this.args.placement ?? 'bottom';
	}

	get strategy(): Strategy {
		return this.args.strategy ?? 'absolute';
	}

	get animation() {
		if (this.args.disableAnimation) {
			return null;
		}
		return this.isClosing ? 'out' : 'in';
	}

	closeTask = task(
		{ drop: true },
		async (
			/** What caused the closeTask to be called */
			// We actually need to declare the types here for TS to figure it out.
			source: OnCloseSource = 'manual',
			/** Event that triggered it */
			event?: Event,
		) => {
			this.isClosing = (await this.args.willClose?.(source, event)) !== false;
			if (this.isClosing) {
				await timeout(isTesting() ? 0 : 100);
				this.args.onClose?.(source, event);
				this.isClosing = false;
			}
		},
	);

	registerArrowElement = modifier((element: HTMLElement) => {
		this.arrowElement = element;
	});

	@action
	async onKeyUp(event: KeyboardEvent) {
		if (event.key === KEYBOARD_KEYS.ESC) {
			event.stopPropagation();
			await this.closeTask.perform('esc', event);
		}
	}
	<template>
		{{#if @show}}
			{{#in-element this.popupContainer insertBefore=null}}
				<div
					class="Popup"
					data-variant={{or @variant "light"}}
					data-qid-component-type="Luna::Popup"
					data-animation-disabled={{toString0 (or @disableAnimation false)}}
					data-test-popup
					{{tetherTo
						@relativeTo
						placement=this.placement
						strategy=this.strategy
						middleware=this.middleware
						arrowElement=this.arrowElement
					}}
					{{onClickOutside (perform this.closeTask "click-outside") disabled=(not @onClose)}}
				>
					{{#unless @excludeAnnouncer}}
						<Announcer />
					{{/unless}}
					{{! template-lint-disable no-invalid-interactive }}
					<div
						class="content"
						id={{@id}}
						data-animate={{this.animation}}
						{{on "keyup" this.onKeyUp}}
						{{a11yTrapFocus disabled=(not @trapFocus)}}
					>
						{{#if @enableArrow}}
							<div {{this.registerArrowElement}} class="arrow-wrapper">
								<div class="arrow"></div>
							</div>

						{{/if}}

						<div class="overflow" ...attributes>
							{{yield (hash id=@id closeTask=this.closeTask)}}
						</div>
					</div>
				</div>
			{{/in-element}}
		{{/if}}
	</template>
}

import {
	createContext,
	type MutableRefObject,
	type ReactNode,
	type Ref,
	type RefObject,
	useCallback,
	useContext,
	useEffect,
	useId,
	useRef,
} from 'react';

import { Popover } from '@base-ui-components/react/popover';

import { cn } from '../../utils/cn';

import styles from './popup.module.css';

export type PopupVariant = 'light' | 'dark' | 'editable-value' | 'editable-value--schema-table-cell';
export type PopupPlacement =
	| 'top'
	| 'top-start'
	| 'top-end'
	| 'bottom'
	| 'bottom-start'
	| 'bottom-end'
	| 'left'
	| 'left-start'
	| 'left-end'
	| 'right'
	| 'right-start'
	| 'right-end'
	| 'overlay';
export type PopupStrategy = 'absolute' | 'fixed';
export type OnCloseSource = 'esc' | 'enter' | 'click-outside' | 'manual';

/**
 * Context value provided by Popup to its children
 */
export interface PopupContextValue {
	/** Close the popup */
	close: (source?: OnCloseSource, event?: Event) => Promise<void>;
	/** ID of the popup content element */
	id?: string;
}

const PopupContext = createContext<PopupContextValue | null>(null);

/**
 * Hook to access the popup context from within the popup
 */
export function usePopupContext(): PopupContextValue | null {
	return useContext(PopupContext);
}

export interface PopupProps {
	/** Forwarded ref to the popup element */
	'ref'?: Ref<HTMLDivElement>;
	/** Unique identifier for the popup content */
	'id'?: string;
	/** Visual variant of the popup */
	'variant'?: PopupVariant;
	/** Element or selector to position the popup relative to */
	'relativeTo'?: Element | string | RefObject<Element | null>;
	/** Where to place the popup relative to the anchor. "overlay" positions popup directly over the anchor */
	'placement'?: PopupPlacement;
	/** CSS positioning strategy */
	'strategy'?: PopupStrategy;
	/** Whether to trap focus within the popup */
	'trapFocus'?: boolean;
	/** Whether to show an arrow pointing to the anchor */
	'enableArrow'?: boolean;
	/** Space between anchor and popup (deprecated, use ySpacing) */
	'spacing'?: number;
	/** Horizontal offset from anchor */
	'xSpacing'?: number;
	/** Vertical offset from anchor */
	'ySpacing'?: number;
	/** Whether the popup is visible */
	'show'?: boolean;
	/** Disable open/close animations */
	'disableAnimation'?: boolean;
	/** Called before popup closes. Return false or resolve to false to prevent closing */
	'willClose'?: (source: OnCloseSource, event: unknown) => Promise<unknown> | Promise<boolean> | void | boolean;
	/** Called when popup is closed */
	'onClose'?: (source: OnCloseSource, event: unknown) => void;
	/** Whether to inherit field context from parent */
	'inheritFieldContext'?: boolean;
	/** Allow content to overflow (e.g., for focus rings that extend beyond bounds) */
	'allowOverflow'?: boolean;
	/**
	 * CSS selector to focus when the popup opens. When `trapFocus` is true,
	 * the first element matching this selector will be focused when the popup opens.
	 */
	'initialFocusSelector'?: string;
	/** Popup content - use usePopupContext() hook to access close function and popup id */
	'children'?: ReactNode;
	/** Additional class name */
	'className'?: string;
	/** ARIA role for the popup container */
	'role'?: string;
	/** Data attributes */
	'data-qid-component-type'?: string;
	'data-testid'?: string;
}

/**
 * Shows a popup positioned relative to another element.
 * The popup closes if the user presses ESC or clicks outside of it.
 */
export function Popup({
	ref,
	id: providedId,
	variant = 'light',
	relativeTo,
	placement = 'bottom',
	strategy = 'absolute',
	trapFocus = false,
	enableArrow = false,
	spacing,
	xSpacing = 0,
	ySpacing,
	show = false,
	disableAnimation = false,
	willClose,
	onClose,
	allowOverflow = false,
	initialFocusSelector,
	children,
	className,
	...props
}: PopupProps) {
	const generatedId = useId();
	const id = providedId ?? generatedId;
	const isClosingRef = useRef(false);

	// Calculate ySpacing with defaults
	const effectiveYSpacing = ySpacing ?? spacing ?? (enableArrow ? 8 : 2);

	// Resolve the anchor - can be an Element, a CSS selector string, or a ref
	const getAnchor = useCallback((): Element | null => {
		if (!relativeTo) return null;

		if (typeof relativeTo === 'string') {
			return document.querySelector(relativeTo);
		}

		if ('current' in relativeTo) {
			return relativeTo.current;
		}

		return relativeTo;
	}, [relativeTo]);

	// Handle close with willClose callback
	const handleClose = useCallback(
		async (source: OnCloseSource = 'manual', event?: Event) => {
			if (isClosingRef.current) return;

			const shouldClose = await willClose?.(source, event);
			if (shouldClose === false) return;

			isClosingRef.current = true;

			// Small delay for animation (in production)
			await new Promise((resolve) => setTimeout(resolve, disableAnimation ? 0 : 100));

			onClose?.(source, event);
			isClosingRef.current = false;
		},
		[willClose, onClose, disableAnimation],
	);

	// Handle open state changes from Base UI
	const handleOpenChange = useCallback(
		(open: boolean, eventDetails: Popover.Root.ChangeEventDetails) => {
			if (!open) {
				// Determine the source of the close
				const details = eventDetails as unknown as { reason?: string; event?: Event };
				if (details.reason === 'escape-key') {
					handleClose('esc', details.event);
				} else {
					handleClose('click-outside', details.event);
				}
			}
		},
		[handleClose],
	);

	// Convert placement to Base UI side
	const getSide = (): 'top' | 'bottom' | 'left' | 'right' => {
		if (placement === 'overlay') return 'bottom';
		const side = placement.split('-')[0] as 'top' | 'bottom' | 'left' | 'right';
		return side;
	};

	// Convert placement to Base UI align
	const getAlign = (): 'start' | 'center' | 'end' => {
		if (placement === 'overlay') return 'center';
		const parts = placement.split('-');
		if (parts.length > 1) {
			return parts[1] as 'start' | 'end';
		}
		return 'center';
	};

	const contextValue: PopupContextValue = {
		close: handleClose,
		id,
	};

	// Ref for the popup element to query within
	const popupRef = useRef<HTMLDivElement>(null);

	// Merge refs for the popup element
	const setPopupRef = useCallback(
		(element: HTMLDivElement | null) => {
			popupRef.current = element;
			if (typeof ref === 'function') {
				ref(element);
			} else if (ref) {
				(ref as MutableRefObject<HTMLDivElement | null>).current = element;
			}
		},
		[ref],
	);

	// Convert selector to function for Base UI's initialFocus prop
	const initialFocus = initialFocusSelector
		? () => popupRef.current?.querySelector<HTMLElement>(initialFocusSelector) ?? null
		: undefined;

	// Viewport-aware resizing: mirrors Ember implementation
	// Sets max-height and CSS variable when popup would overflow viewport
	useEffect(() => {
		if (!show || !popupRef.current) return;

		const popupElement = popupRef.current;
		let rafId: number | null = null;

		const checkAndApplyConstraints = () => {
			const rect = popupElement.getBoundingClientRect();
			const viewportHeight = window.innerHeight;
			const availableHeight = viewportHeight - rect.top;

			// Handle overlay mode sizing
			if (placement === 'overlay' && relativeTo) {
				const anchor =
					typeof relativeTo === 'string'
						? document.querySelector(relativeTo)
						: 'current' in relativeTo
							? relativeTo.current
							: relativeTo;

				if (anchor) {
					const anchorRect = anchor.getBoundingClientRect();
					popupElement.style.minWidth = `${anchorRect.width}px`;
					popupElement.style.height = `${anchorRect.height}px`;
				}
			}

			// Viewport-aware max-height (matches Ember behavior)
			if (availableHeight < rect.height) {
				popupElement.style.maxHeight = `${availableHeight}px`;

				// Set CSS variable for children to reference (skip in tests)
				// Subtract 5px for visual breathing room (shadows, borders, etc.)
				if (typeof import.meta === 'undefined' || (import.meta as { env?: { MODE?: string } }).env?.MODE !== 'test') {
					popupElement.style.setProperty('--luna-popup-max-height', `${availableHeight - 5}px`);
				}
			} else {
				// Clear constraints when not needed
				popupElement.style.maxHeight = '';
				popupElement.style.removeProperty('--luna-popup-max-height');
			}
		};

		const scheduleCheckAndApplyConstraints = () => {
			if (rafId !== null) return;
			rafId = window.requestAnimationFrame(() => {
				rafId = null;
				checkAndApplyConstraints();
			});
		};

		// Check constraints initially and on resize
		checkAndApplyConstraints();

		const resizeObserver = new ResizeObserver(scheduleCheckAndApplyConstraints);
		resizeObserver.observe(popupElement);

		// Also listen for viewport resize
		window.addEventListener('resize', scheduleCheckAndApplyConstraints);

		return () => {
			resizeObserver.disconnect();
			window.removeEventListener('resize', scheduleCheckAndApplyConstraints);
			if (rafId !== null) {
				window.cancelAnimationFrame(rafId);
				rafId = null;
			}
			// Clean up styles
			if (popupElement) {
				popupElement.style.maxHeight = '';
				popupElement.style.minWidth = '';
				popupElement.style.height = '';
				popupElement.style.removeProperty('--luna-popup-max-height');
			}
		};
	}, [show, placement, relativeTo]);

	return (
		<Popover.Root open={show} onOpenChange={handleOpenChange} modal={trapFocus ? 'trap-focus' : false}>
			<Popover.Portal>
				<Popover.Positioner
					side={getSide()}
					sideOffset={effectiveYSpacing}
					alignOffset={xSpacing}
					align={getAlign()}
					anchor={getAnchor}
					positionMethod={strategy}
					className={styles.positioner}
				>
					<Popover.Popup
						ref={setPopupRef}
						initialFocus={initialFocus}
						className={cn(styles.popup, className)}
						data-variant={variant}
						data-qid-component-type="Luna::Popup"
						data-animation-disabled={disableAnimation ? 'true' : undefined}
						{...props}
					>
						{enableArrow && (
							<Popover.Arrow className={styles.arrowWrapper}>
								<div className={styles.arrow} />
							</Popover.Arrow>
						)}
						<PopupContext.Provider value={contextValue}>
							<div className={allowOverflow ? undefined : styles.overflow}>{children}</div>
						</PopupContext.Provider>
					</Popover.Popup>
				</Popover.Positioner>
			</Popover.Portal>
		</Popover.Root>
	);
}

export default Popup;
