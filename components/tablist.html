type TablistOptionArg<V> = TablistTab<V> | TablistGroup<V>;

/**

type TabYieldType<V, O extends TablistOptionArg<V>> = O extends TablistGroup<V> ? TablistTab<V> : O;

/**
 * A component for creating a tabbed interface.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/tablist.md)
 */
export default class LunaTablist<V extends string, O extends TablistOptionArg<V>> extends Component<
	LunaTablistSignature<V, O>
> {
	@service declare router: RouterService;
	@service declare intl: IntlService;
	@service('luna/announcer') declare announcer: AnnouncerService;

	/**
	 * The element containing all tab groups. Registered on render;
	 * used in querySelector calls and for measuring the space available for tabs.
	 */
	@tracked tablistElement: HTMLElement | null = null;

	/**
	 * The before-block element. Registered and measured on render; remeasured on resize.
	 * Used for measuring the space available for tabs.
	 */
	@tracked beforeBlockElement: HTMLElement | null = null;

	/**
	 * The element containing the overflow menu button. Used in focus calls
	 * when navigating using the keyboard.
	 */
	@tracked tabOverflowButton: HTMLElement | null = null;

	/**
	 * Whether the user is reordering tabs using the keyboard. True when the
	 * user presses space on a focused tab. When true, determines how keyboard
	 * navigation behaves.
	 */
	@tracked isReorderingViaKeyboard = false;

	/**
	 * Width of the container holding all the tabs in px. This will get
	 * updated if the element changes size for any reason.
	 */
	@tracked tablistWidth?: number;

	/**
	 * Width of the before block in px. Used to determine how many tabs can fit
	 * on the screen before truncation. Captured on render; updated on resize.
	 */
	@tracked beforeBlockWidth?: number;

	/**
	 * The API for the tab menu, registered on button render.
	 * Used to programmatically toggle the menu in cases where the mnu
	 * is not focusable, such as in a tab's action menu.
	 */
	@tracked
	tablistMenuApi: {
		menuIsShown: boolean;
		showMenu: (_event: Event) => void;
		hideMenu: (_event: Event) => void;
	} | null = null;

	/**
	 * For the purpose of calculating how many tabs we can show and how many should go
	 * into the overflow menu, we store the width of each tab.
	 */
	tabWidths = new TrackedMap<V, number>();

	/**
	 * Keeps track of which of the tabs is focusable. When the user tabs off and back onto
	 * Tablist the same tab should receive focus. It is not necessarily the same tab
	 * as the one that is selected.
	 */
	// Allowing nullish because this can actually be a string
	// eslint-disable-next-line @typescript-eslint/prefer-nullish-coalescing
	@tracked _focusableTabValue: V | null = (this.activeTab?.value || this.sortedAllTabs[0]?.value) ?? null;

	/**
	 * The key of the tab that is being closed. Set on close; used by
	 * `getTabTooltipIsDisabled` to disable the tooltip before its anchor is removed
	 * from the DOM, preventing the tooltip from getting stuck in the top left corner.
	 */
	@tracked keyOfTabBeingClosed: string | null = null;

	/**
	 * The key of the tab being opened. Set on open; used by `getTabTooltipIsDisabled`
	 * to disable the tooltip when a tab is activated by the keyboard.
	 * (The tooltip modifier handles this automatically on click.)
	 */
	@tracked keyOfTabBeingOpened: string | null = null;

	/**
	 * The option format to save when the onReorder action is called.
	 * Determined by the format of the options passed in.
	 */
	get optionTypeToSave(): 'group' | 'tab' {
		if (this.args.options[0] && 'tabs' in this.args.options[0]) {
			return 'group';
		}
		return 'tab';
	}

	get pinnedTabs() {
		return this.allTabs.filter((tab) => tab.isPinned);
	}

	get unpinnedTabs() {
		return this.allTabs.filter((tab) => !tab.isPinned);
	}

	/**
	 * Whether to use the `tablist`/`tab` semantics which require the use of
	 * corresponding `tabpanel`s. True for simple, button-based tablists unless
	 * explicitly overridden by the `roleIsNav` argument.
	 */
	get roleIsTablist() {
		return (
			!this.args.onDelete &&
			!this.sortedAllTabs.some((tab) => tab.menuItems?.length) &&
			this.args.roleIsNav !== true &&
			!!this.sortedAllTabs.length &&
			this.sortedAllTabs.every((tab) => tab.href === undefined)
		);
	}

	/**
	 * Whether the role of the tablist is essentially a nav.
	 * When true, the `tablist`/`tab` roles are omitted, and the aria attribute
	 * designating the active tab will be slightly different.
	 */
	get roleIsNav() {
		return this.roleIsTablist === false;
	}

	/**
	 * A TablistTab-shaped representation of the overflow tab.
	 * Used for overflow-button attributes and in focusedTab targeting.
	 */
	get overflowTab(): TablistTab<V> {
		return {
			value: `tablist-overflow-button-${guidFor(this)}` as V,
			label: this.intl.t('input.tablist.overflow.button'),
		};
	}

	/**
	 * Whether the overflow menu button is focused. Used to determine
	 * where focus should move when navigating using the keyboard.
	 */
	get tabOverflowButtonIsFocused() {
		return this._focusableTabValue === this.overflowTab.value;
	}

	/**
	 * The tab groups to loop through. If the options are groups, we use them.
	 * Otherwise, we assign a default group, with pinned tabs in a separate "pinned" group.
	 * Pinned tabs are moved to the front of their respective groups.
	 */
	get tabGroups() {
		const { options } = this.args;

		if (this.optionTypeToSave === 'group') {
			const groups = options as Array<TablistGroup<V>>;
			return groups.map((group) => ({
				...group,
				tabs: this.sortTabsWithPinnedFirst(group.tabs),
			}));
		}

		const groups: Array<TablistGroup<V>> = [];

		// If there are pinned tabs, create a separate "pinned" group
		if (this.pinnedTabs.length > 0) {
			groups.push({
				id: 'pinned',
				tabs: this.pinnedTabs,
			});
		}

		// Create the default group for unpinned tabs
		if (this.unpinnedTabs.length > 0) {
			groups.push({
				id: 'default',
				tabs: this.unpinnedTabs,
			});
		}

		return groups;
	}

	/**
	 * Sorts tabs to put pinned tabs first, maintaining the original order within each category.
	 */
	sortTabsWithPinnedFirst(tabs: Array<TablistTab<V>>): Array<TablistTab<V>> {
		return [...tabs].sort((a, b) => {
			if (a.isPinned === b.isPinned) {
				return 0;
			}

			return a.isPinned ? -1 : 1;
		});
	}

	set focusableTab(tab: TablistTab<V> | undefined) {
		this._focusableTabValue = tab?.value ?? null;
	}

	@cached
	get focusableTab(): TablistTab<V> | undefined {
		if (this.tabOverflowButtonIsFocused) {
			return this.overflowTab;
		}

		let focusableTab = this.sortedAllTabs.find((tab) => tab.value === this._focusableTabValue);

		if (!focusableTab) {
			const activeTabValue = this.activeTab?.value;
			if (activeTabValue) {
				focusableTab = this.sortedAllTabs.find((tab) => tab.value === activeTabValue);
			}
		}

		return focusableTab;
	}

	/**
	 * A flatted array of all tabs passed in via the `options` arg.
	 * Used by the pinnedTabs and unpinnedTabs getters.
	 */
	get allTabs(): Array<TablistTab<V>> {
		const { options } = this.args;

		let tabs: Array<TablistTab<V>> = [];

		for (const option of options) {
			if ('tabs' in option) {
				// It's a group; push its tabs into the array.
				tabs = tabs.concat(option.tabs);
			} else {
				// It's a tab; push it into the array.
				tabs.push(option);
			}
		}
		return tabs;
	}
	/**
	 * A flatted array of all tabs, ordered by group and pinned status.
	 * Looped through or filtered when determining overflowTabs and keyboard navigation.
	 */
	get sortedAllTabs(): Array<TablistTab<V>> {
		return this.tabGroups.flatMap((group) => group.tabs);
	}

	/**
	 * Tabs that live within the overflow menu. Looped through by ButtonWithMenu,
	 * and used in some keyboard-navigation scenarios.
	 */
	@cached
	get overflowTabs() {
		return this.sortedAllTabs.filter((tab) => !this.visibleTabs.includes(tab));
	}

	/**
	 * The visible tabs as determined by available space in the tablistElement.
	 * Any tabs that don't fit are moved to the overflow menu.
	 * Pinned tabs are always visible and never moved to overflow.
	 */
	@cached
	get visibleTabs() {
		const { activeTab, focusableTab, tabWidths } = this;
		const visibleTabs: Array<TablistTab<V>> = [];

		let widthAvailable = this.tablistWidth
			? this.tablistWidth - OVERFLOW_MENU_BUTTON_WIDTH - (this.beforeBlockWidth ?? 0)
			: 0;

		visibleTabs.push(...this.pinnedTabs);

		// Reserve space for selected tab if it's not pinned and not already counted
		if (activeTab && !activeTab.isPinned) {
			visibleTabs.push(activeTab);
			widthAvailable -= this.args.shouldBalanceTabs ? MIN_BALANCED_TAB_WIDTH : (tabWidths.get(activeTab.value) ?? 0);
		}

		// Reserve space for focusable tab if it's different from selected and not pinned
		if (
			focusableTab &&
			focusableTab !== activeTab &&
			!focusableTab.isPinned &&
			focusableTab.value !== this.overflowTab.value
		) {
			visibleTabs.push(focusableTab);
			widthAvailable -= this.args.shouldBalanceTabs ? MIN_BALANCED_TAB_WIDTH : (tabWidths.get(focusableTab.value) ?? 0);
		}

		const notYetVisibleUnpinnedTabs = this.unpinnedTabs.filter((tab) => !visibleTabs.includes(tab));

		if (this.args.shouldBalanceTabs) {
			widthAvailable -=
				(this.args.shouldUseCompactPinnedTabs ? COMPACT_PINNED_TAB_WIDTH : MIN_BALANCED_TAB_WIDTH) *
				this.pinnedTabs.length;

			notYetVisibleUnpinnedTabs.forEach((tab) => {
				if (widthAvailable >= MIN_BALANCED_TAB_WIDTH) {
					widthAvailable -= MIN_BALANCED_TAB_WIDTH;
					visibleTabs.push(tab);
				}
			});
		} else {
			// Calculate reserved width for pinned tabs
			const pinnedTabsWidth = this.pinnedTabs.reduce((total, tab) => {
				return (
					total + (tabWidths.get(tab.value) ?? (this.args.shouldUseCompactPinnedTabs ? COMPACT_PINNED_TAB_WIDTH : 0))
				);
			}, 0);
			// Calculate width available for non-pinned tabs
			// Reserve space for overflow button and selected/focusable tabs if they're not pinned
			widthAvailable -= pinnedTabsWidth;

			this.unpinnedTabs.forEach((tab) => {
				// Skip tabs that are already guaranteed to be visible (selected/focusable)
				if (tab === activeTab || tab === focusableTab) {
					if (!visibleTabs.includes(tab)) {
						visibleTabs.push(tab);
					}
					return;
				}

				// Check if this tab fits in the available space
				const tabWidth = tabWidths.get(tab.value) ?? 0;
				if (widthAvailable >= tabWidth) {
					widthAvailable -= tabWidth;
					if (!visibleTabs.includes(tab)) {
						visibleTabs.push(tab);
					}
				}
			});
		}

		/**
		 * Use `allTabs` as the basis since it preserves the order
		 * Pinned tabs will always be included since they're added first and never removed
		 */
		return this.sortedAllTabs.filter((tab) => visibleTabs.includes(tab));
	}

	/**
	 * Whether reordering is enabled for tablist groups. True if an `onReorder`
	 * function was passed in, unless the group is specifically marked `isLocked`.
	 */
	get reorderingIsEnabled() {
		return Boolean(this.args.onReorder);
	}

	/**
	 * The current URL, used for determining the selected tab. When the url matches
	 * the href of the tab, that tab will be shown as active.
	 */
	@cached
	get currentUrl(): string {
		const { context } = this.args;

		if (context) {
			return context.fullUrlPathName || '';
		}

		if (!this.router.currentURL) {
			return '';
		}

		const url = new URL(this.router.currentURL, window.location.origin);
		return url.pathname || '';
	}

	/**
	 * The currently selected tab. Determined by the activeTab argument if provided,
	 * otherwise by matching the current URL to the href of a tab. Always visible.
	 */
	get activeTab(): TablistTab<V> | undefined {
		if (this.args.activeTab) return this.args.activeTab;

		const { currentUrl } = this;

		return this.sortedAllTabs?.find((tab) => {
			if (tab.currentWhenUrls) {
				return tab.currentWhenUrls.includes(currentUrl);
			} else if (tab.href) {
				// Check both exact match and pathname-only match for tabs with query params
				if (currentUrl === tab.href) {
					return true;
				}

				// Extract pathname from tab.href to compare with currentUrl (which is pathname-only)
				try {
					const tabUrl = new URL(tab.href, window.location.origin);
					return currentUrl === tabUrl.pathname;
				} catch {
					// If tab.href is already a pathname, compare directly
					const tabPathname = tab.href.split('?')[0];
					return currentUrl === tabPathname;
				}
			}
			return tab.value === this.args.value;
		});
	}

	/**
	 * Custom properties passed to CSS via the style attribute of the container.
	 */
	get cssCustomProperties() {
		const styles: string[] = [`--tablist-overflow-menu-button-width: ${OVERFLOW_MENU_BUTTON_WIDTH}px;`];

		if (this.args.shouldUseCompactPinnedTabs) {
			styles.push(`--tablist-compact-pinned-tab-width: ${COMPACT_PINNED_TAB_WIDTH}px;`);
		}

		if (this.args.shouldBalanceTabs) {
			styles.push(`--tablist-balanced-tab-width: ${this.balancedTabWidth}px;`);
		}

		return styles.join(' ');
	}

	/**
	 * Moves focus to the tab associated with the key pressed.
	 * If the user is reordering tabs via keyboard,
	 * moves the tab within its group.
	 */
	moveTabOrChangeFocus(event: KeyboardEvent) {
		const acceptedKeys: string[] = [
			KEYBOARD_KEYS.LEFT_ARROW,
			KEYBOARD_KEYS.RIGHT_ARROW,
			KEYBOARD_KEYS.HOME,
			KEYBOARD_KEYS.END,
		];

		if (!acceptedKeys.includes(event.key)) {
			return;
		}

		event.stopPropagation();
		event.preventDefault();

		const { visibleTabs } = this;

		switch (event.key) {
			case KEYBOARD_KEYS.LEFT_ARROW:
				this.onLeftArrow();
				break;

			case KEYBOARD_KEYS.RIGHT_ARROW:
				this.onRightArrow();
				break;

			case KEYBOARD_KEYS.HOME:
				this.focusableTab = visibleTabs[0];
				break;

			case KEYBOARD_KEYS.END:
				if (this.overflowTabs.length) {
					this.focusTabOverflowButton();
				} else {
					this.focusableTab = visibleTabs[visibleTabs.length - 1];
				}
				break;
		}

		if (this.tabOverflowButtonIsFocused) {
			assert('tabOverflowButton must exist', this.tabOverflowButton);
			this.tabOverflowButton.focus();
		} else {
			this.focusFocusableTab(event.key);
		}
	}

	/**
	 * The action used when moving a tab via keyboard arrows.
	 * Constrains tab movement within its group and respects pinned tab boundaries.
	 */
	moveTabWithinGroup(direction: 'left' | 'right') {
		if (!this.isReorderingViaKeyboard || !this.focusableTab) {
			return;
		}

		// Find which group contains the focusable tab
		const focusableTab = this.focusableTab;
		const groupContainingTab = this.getGroupContainingTab(focusableTab);

		if (!groupContainingTab) {
			return;
		}

		const groupTabs = groupContainingTab.tabs;
		const currentIndex = groupTabs.indexOf(focusableTab);
		const isMovingLeft = direction === 'left';
		const targetIndex = isMovingLeft ? currentIndex - 1 : currentIndex + 1;

		// Check bounds
		if (targetIndex < 0 || targetIndex >= groupTabs.length) {
			return;
		}

		const targetTab = groupTabs[targetIndex];

		// Check if target tab exists and prevent moving pinned tabs past unpinned tabs and vice versa
		if (!targetTab || focusableTab.isPinned !== targetTab.isPinned) {
			return;
		}

		// Create new array with the tab moved
		const newGroupTabs = [...groupTabs];
		const [movedTab] = newGroupTabs.splice(currentIndex, 1);

		if (movedTab) {
			newGroupTabs.splice(targetIndex, 0, movedTab);

			this.announcer.announce(
				this.intl.t('tablist.moved', {
					tabLabel: movedTab.label || movedTab.value,
					position: targetIndex + 1,
					total: newGroupTabs.length,
				}),
				{ urgency: 'polite' },
			); // Call onReorder with the group's tabs as both old and new order
			// The onReorder method will handle updating the group and calling the callback
			this.onReorder(groupTabs, newGroupTabs, focusableTab);
		}
	}

	/**
	 * The action to move focus or a grabbed tab left.
	 */
	onLeftArrow() {
		const { visibleTabs } = this;
		if (this.isReorderingViaKeyboard && this.focusableTab) {
			this.moveTabWithinGroup('left');
		} else if (this.tabOverflowButtonIsFocused) {
			this.focusableTab = visibleTabs[visibleTabs.length - 1];
		} else {
			const indexOfFocusableTab = this.focusableTab ? visibleTabs.indexOf(this.focusableTab) : -1;

			if (indexOfFocusableTab > 0) {
				this.focusableTab = visibleTabs[indexOfFocusableTab - 1];
			}
		}
	}

	/**
	 * The action to move focus or a grabbed tab right.
	 */
	onRightArrow() {
		const { visibleTabs } = this;
		if (this.isReorderingViaKeyboard && this.focusableTab) {
			this.moveTabWithinGroup('right');
		} else {
			if (this.tabOverflowButtonIsFocused) {
				return;
			}

			const indexOfFocusableTab = this.focusableTab ? visibleTabs.indexOf(this.focusableTab) : -1;

			if (indexOfFocusableTab === visibleTabs.length - 1 && this.overflowTabs.length) {
				this.focusTabOverflowButton();
			} else if (indexOfFocusableTab < visibleTabs.length - 1) {
				this.focusableTab = visibleTabs[indexOfFocusableTab + 1];
			}
		}
	}

	/**
	 * Returns the group that contains a given tab. Used as a utility when determining
	 * reorder boundaries and when moving an overflow tab into the visible tablist.
	 */
	getGroupContainingTab = (tab: TablistTab<V> | undefined) => {
		if (!tab) return;
		return this.tabGroups.find((group) => group.tabs.includes(tab));
	};

	/**
	 * The action to focus the "focusable" tab. Finds the tab by its
	 * value-associated element and focuses it. Called after a selection or reorder action.
	 */
	focusFocusableTab(key?: string) {
		const { tablistElement, focusableTab } = this;

		if (focusableTab && tablistElement) {
			const selector = `[data-tab="${focusableTab.value}"]`;
			const tabElement = tablistElement.querySelector(selector);
			if (tabElement) {
				assert('Expected selected tab to be an HTMLElement', tabElement instanceof HTMLElement);
				tabElement.focus();

				// If we're focusing due to keyboard navigation, dispatch a synthetic keyup event
				// to trigger tooltip behavior, since the tooltip modifier expects a keyup event
				// on the newly focused element to determine if it should show a tooltip
				if (key) {
					const keyupEvent = new KeyboardEvent('keyup', {
						key,
						bubbles: true,
						cancelable: true,
					});
					tabElement.dispatchEvent(keyupEvent);
				}
			}
		}
	}

	/**
	 * The action to close a tab and move focus to the next logical tab.
	 * Called when the close button is clicked or on delete/backspace on a focused tab.
	 */
	closeTab = (tab: TablistTab<V>, event?: Event) => {
		assert('[BUG] Expected onDelete to be set', this.args.onDelete);

		if (this.getGroupContainingTab(tab)?.isLocked) return;

		this.keyOfTabBeingClosed = tab.value;
		schedule('afterRender', () => (this.keyOfTabBeingClosed = null));

		if (event) {
			event.stopPropagation();
			event.preventDefault();
		}

		const { sortedAllTabs } = this;
		const currentTabIndex = sortedAllTabs.indexOf(tab);

		// Determine which tab should receive focus after deletion
		let nextFocusableTab: TablistTab<V> | undefined;

		if (sortedAllTabs.length === 1) {
			// If we're closing the absolutely last tab, there are no tabs to focus
			nextFocusableTab = undefined;
		} else if (currentTabIndex !== 0) {
			// Focus the previous tab if one exists
			nextFocusableTab = sortedAllTabs[currentTabIndex - 1];
		} else {
			// Otherwise, focus the next tab
			nextFocusableTab = sortedAllTabs[currentTabIndex + 1];
		}

		// Update selection if we're deleting the selected tab
		if (tab === this.activeTab) {
			this.onTabClick(nextFocusableTab, 'close');
		}

		this.focusableTab = nextFocusableTab;

		// Focus the next tab before deleting the current one to avoid focus jumping to body
		if (nextFocusableTab && this.tablistElement) {
			const selector = `[data-tab="${nextFocusableTab.value}"]`;
			const tabElement = this.tablistElement.querySelector(selector);

			if (tabElement instanceof HTMLElement) {
				tabElement.focus();
			} else if (this.overflowTabs.length) {
				assert('tabOverflowButton must exist', this.tabOverflowButton);
				this.tabOverflowButton.focus();
			}
		}

		// Now delete the tab
		this.args.onDelete?.(tab.value);
	};

	/**
	 * Event that triggers this option selection and updates the focusable tab.
	 */
	onTabClick = (tab?: TablistTab<V>, event?: Event | 'close') => {
		if (!tab) return;

		this.keyOfTabBeingOpened = tab.value;
		schedule('afterRender', () => (this.keyOfTabBeingOpened = null));

		if (event instanceof Event) {
			event.preventDefault();
			event.stopPropagation();

			if (
				event.target instanceof Element &&
				event.target.closest('[data-qid-tablist-tab-wrapper]')?.classList.contains('is-dropping')
			) {
				// Prevent dragged tabs from being considered clicked
				return;
			}
		}

		const isCloseEvent = event === 'close';

		if (tab.href) {
			if (event && event instanceof PointerEvent && (event.metaKey || event.ctrlKey)) {
				// Unfortunately, keyboard PointerEvents don't capture metaKey/ctrLKey
				window.open(tab.href, '_blank');
				return;
			}
			this.handleTransition(tab, () => this.processTabClick(tab, { isCloseEvent }));
		} else {
			this.processTabClick(tab, { isCloseEvent });
		}
	};

	/**
	 * A transition handler used `onTabClick` to handle cases where route interruptions
	 * occur, such as when showing an "Are you sure you want to navigate away?" modal.
	 * Allows us to stay aware of which tab to focus after interruption concludes.
	 */
	handleTransition = (tab: TablistTab<V>, action: () => void) => {
		let transition: Transition;

		next(this, () => {
			if (this.args.context?.isQuickView) {
				transition = this.router.transitionTo(this.router.currentRouteName, { queryParams: { view: tab.href } });
			} else {
				transition = this.router.transitionTo(tab.href);
			}
			if (
				transition.isAborted &&
				'isCausedByVerifyingTransitionWithUser' in transition &&
				transition.isCausedByVerifyingTransitionWithUser
			) {
				const originalRetry = transition.retry;
				transition.retry = () => {
					action();
					return originalRetry.call(transition);
				};
				return;
			}

			action();
		});
	};

	/**
	 * A utility to call a tab's menu action while ensuring its menu closes.
	 * Handles quirks in the ButtonWithMenu component that sometimes cause the menu
	 * to remain open after an action is selected.
	 */
	handleTabItemAction = (tab: TablistTab<V>, item: TablistTabMenuItem, hideMenu: () => void) => {
		item.action?.(tab);
		hideMenu();
	};

	/**
	 * Processes a valid tab click after being triaged by `onTabClick`. Runs immediately
	 * on for tab buttons; called after the transition is handled on href-driven tabs.
	 */
	processTabClick = (tab: TablistTab<V>, { isCloseEvent = false } = {}) => {
		if (!isCloseEvent) {
			tab?.action?.();
			this.args.onItemClick?.(tab);
		}

		this.args.onChange?.(tab?.value ?? null);
		this.focusableTab = tab;
		this.focusFocusableTab();
	};

	/**
	 * The action that runs when an overflow tab is clicked. Differs from `onTabClick`
	 * in that it also closes the overflow menu and conditionally calls the onReorder
	 * action to preserve the tab order when an overflow tab moves into the tablist.
	 */
	onOverflowTabClick = (tab: TablistTab<V>, event: Event) => {
		this.args.onChange?.(tab?.value ?? null);
		this.focusableTab = tab;

		assert('tablistMenuApi must exist', this.tablistMenuApi);

		this.tablistMenuApi.hideMenu(event);

		schedule('afterRender', () => {
			this.focusFocusableTab();

			if (this.args.onReorder) {
				const groupContainingTab = this.getGroupContainingTab(tab);

				if (groupContainingTab) {
					const originalGroupTabs = groupContainingTab.tabs;
					const visibleTabsInGroup = this.visibleTabs.filter((t) => originalGroupTabs.includes(t));
					const lastVisibleTabInGroup = visibleTabsInGroup[visibleTabsInGroup.length - 1];

					// Insert the clicked tab after the last visible tab in its group
					const newGroupTabOrder = [...originalGroupTabs];
					const clickedTabIndex = newGroupTabOrder.indexOf(tab);

					if (lastVisibleTabInGroup && clickedTabIndex !== -1) {
						// Remove the clicked tab from its current position
						newGroupTabOrder.splice(clickedTabIndex, 1);

						// Find where to insert it (after the last visible tab)
						const insertionIndex = newGroupTabOrder.indexOf(lastVisibleTabInGroup) + 1;
						newGroupTabOrder.splice(insertionIndex, 0, tab);
					}

					// Call onReorder with the group's tabs in their new order
					this.onReorder(originalGroupTabs, newGroupTabOrder, tab);
				}
			}
		});
	};

	/**
	 * Triages key presses while the tablist or its overflow menu is focused.
	 */
	onKeydown = (tab: TablistTab<V> | null, event: KeyboardEvent) => {
		if (event.key === KEYBOARD_KEYS.ESC) {
			this.isReorderingViaKeyboard = false;

			if (this.tablistMenuApi?.menuIsShown) {
				this.tablistMenuApi.hideMenu(event);
				const asyncTestToken = testWaiter.beginAsync();
				requestAnimationFrame(() => {
					this.focusTabOverflowButton();
					testWaiter.endAsync(asyncTestToken);
				});
			}
			return;
		}

		if (event.key === KEYBOARD_KEYS.SPACE) {
			// For overflow menu button, allow default space behavior
			if (tab === null) {
				return;
			}

			// For tabs in locked groups, allow default space behavior
			const group = this.getGroupContainingTab(tab);
			if (group?.isLocked) {
				return;
			}

			// For unlocked tabs, preserve existing reordering behavior
			event.preventDefault();
			if (this.args.onReorder) {
				const wasReordering = this.isReorderingViaKeyboard;
				this.isReorderingViaKeyboard = !this.isReorderingViaKeyboard;

				if (this.isReorderingViaKeyboard) {
					this.announcer.announce(
						this.intl.t('tablist.grabbed', {
							tabLabel: tab.label || tab.value,
						}),
						{ urgency: 'assertive' },
					);
				} else if (wasReordering) {
					this.announcer.announce(this.intl.t('tablist.dropped', { tabLabel: tab.label || tab.value }), {
						urgency: 'assertive',
					});
				}
			}

			return;
		}

		if (event.key === KEYBOARD_KEYS.ENTER) {
			event.preventDefault();
			assert('Expected target to be an HTMLElement', event.target instanceof HTMLElement);
			event.target.click();
			return;
		}

		if (
			(event.key === KEYBOARD_KEYS.DELETE || event.key === KEYBOARD_KEYS.BACKSPACE) &&
			this.args.onDelete &&
			this.focusableTab
		) {
			// close the tooltip

			return this.closeTab(this.focusableTab, event);
		}

		const isOverflowMenu = tab === null;

		// Handle down/up arrow keys for overflow menu button and tab menus
		if (
			(event.key === KEYBOARD_KEYS.DOWN_ARROW || event.key === KEYBOARD_KEYS.UP_ARROW) &&
			((isOverflowMenu && this.overflowTabs.length) || tab?.menuItems?.length)
		) {
			event.preventDefault();

			let menuSelector: string;

			// Open the appropriate menu
			if (isOverflowMenu) {
				assert('tablistMenuApi must exist', this.tablistMenuApi);
				this.tablistMenuApi.showMenu(event);
				menuSelector = '[data-test-tablist-menu-option]:last-child';
			} else {
				const tabElement = event.target;
				assert('event.target must be an HTMLElement', tabElement instanceof HTMLElement);

				const overflowButton = tabElement.parentElement?.querySelector('[data-qid-tablist-tab-menu-button]');
				assert('overflowButton must be an HTMLElement', overflowButton instanceof HTMLElement);

				overflowButton.click();
				menuSelector = '[data-qid-tablist-tab-overflow-item]:last-child';
			}

			// Focus last item if up arrow was pressed
			if (event.key === KEYBOARD_KEYS.UP_ARROW) {
				later(
					this,
					() => {
						const lastMenuItem = document.querySelector(menuSelector);
						if (lastMenuItem instanceof HTMLElement) {
							lastMenuItem.focus();
						}
					},
					32,
				);
			}

			return;
		}

		return this.moveTabOrChangeFocus(event);
	};

	/**
	 * The action run on render and when the tablist resizes.
	 * Sets the tablistWidth, which recomputes which tabs are visible.
	 */
	onResize = (element: HTMLElement) => {
		this.tablistWidth = element.offsetWidth;
		this.beforeBlockWidth = this.beforeBlockElement?.offsetWidth;
	};

	/**
	 * A helper to avoid a glint error in the conditional sortableItem call.
	 */
	formatTabForSortableItemModifier = (tab: TablistTab<V>) => {
		// SAFETY: ModifierHelper thinks we're passing the wrong model type to sortableItem
		return tab as never;
	};

	/**
	 * Type assertion helper for yielding tabs with the correct type.
	 * When options are extended tabs, this preserves the extended type in the yield.
	 * When options are groups, this yields the base TablistTab type.
	 */
	asYieldable = (tab: TablistTab<V>): TabYieldType<V, O> => {
		return tab as TabYieldType<V, O>;
	};

	/**
	 * The reorder function passed to the sortable modifiers, which return
	 * information about tabs after a drag-and-drop operation.
	 * Ensures pinned tabs remain at the front of their group.
	 */
	onReorder = (_groupModel: unknown[], itemModels: unknown[], draggedModel: unknown) => {
		if (!this.args.onReorder) {
			return;
		}

		const draggedTab = draggedModel as TablistTab<V>;
		const tabsInNewOrder = itemModels as Array<TablistTab<V>>;

		const groupContainingDraggedTab = this.tabGroups.find((group) => group.tabs.includes(draggedTab));

		assert('group must exist', groupContainingDraggedTab);

		const updatedTabGroups = this.tabGroups.map((group) => {
			if (group.id === groupContainingDraggedTab.id) {
				return {
					...group,
					tabs: this.sortTabsWithPinnedFirst(tabsInNewOrder),
				};
			}

			return group;
		});

		const reorderedAllTabs =
			this.optionTypeToSave === 'group' ? updatedTabGroups : updatedTabGroups.flatMap((g) => g.tabs);

		this.args.onReorder(reorderedAllTabs);

		// Ensure focus happens after DOM updates
		const asyncTestToken = testWaiter.beginAsync();
		requestAnimationFrame(() => {
			this.focusFocusableTab();
			testWaiter.endAsync(asyncTestToken);
		});
	};

	/**
	 * Helper to get a group's visible tabs, if any. Allows us to initially render all
	 * tabs for the purpose of calculating their size, and then filter out overflow tabs
	 * once they've been computed.
	 */
	getGroupTabsToRender = (groupId: string) => {
		const group = this.tabGroups.find((g) => g.id === groupId);
		if (!group) {
			return [];
		}

		return group.tabs.filter((tab) => this.visibleTabs.includes(tab));
	};

	/**
	 * Whether to render a close button for a tab. Dependent on whether an onDelete
	 * function was passed in, and whether the rules of the tab or group allow closing.
	 */
	getCloseButtonIsShown = (group: TablistGroup<V>, tab: TablistTab<V>) => {
		return this.args.onDelete && !tab.isPinned && !group.isLocked;
	};

	/**
	 * Whether the tab is currently "grabbed" by the keyboard-reordering feature.
	 * When true, applies a style to the tab and disables its tooltip.
	 */
	getIsGrabbed = (tab: TablistTab<V>) => {
		const group = this.tabGroups.find((g) => g.tabs.includes(tab));
		if (group?.isLocked) {
			return false;
		}
		return this.isReorderingViaKeyboard && this.focusableTab === tab;
	};

	/**
	 * Whether a tab's menu is currently open. When true, the tooltip for that tab
	 * will be disabled to avoid two popups showing at once.
	 */
	getTabMenuIsOpen = (tab: TablistTab<V>) => {
		if (this.focusableTab !== tab) return false;
		if (!tab.menuItems?.length) return false;

		const tabElement = this.tablistElement?.querySelector(`[data-tab="${tab.value}"]`);
		if (tabElement) {
			const menuButton = tabElement.parentElement?.querySelector('[data-qid-tablist-tab-menu-button]');
			return menuButton?.getAttribute('aria-expanded') === 'true';
		}

		return false;
	};

	/**
	 * Whether the tooltip for a tab should be disabled.
	 * True when the user's interaction with a tab dictates tooltip closure.
	 */
	getTabTooltipIsDisabled = (tab: TablistTab<V>) => {
		return (
			this.keyOfTabBeingOpened === tab.value ||
			this.keyOfTabBeingClosed === tab.value ||
			this.getTabMenuIsOpen(tab) ||
			this.getIsGrabbed(tab)
		);
	};

	/**
	 * Returns the text for a pinned tab's tooltip and aria label.
	 * Will always return the specified `tooltip` if it exists.
	 * Compact pinned tabs fall back to the label text.
	 */
	getTabTooltip = (tab: TablistTab<V>) => {
		if (tab.tooltip) return decodeEntities(tab.tooltip);

		if (tab.isPinned && this.args.shouldUseCompactPinnedTabs) {
			return decodeEntities(tab.label);
		}
	};

	/**
	 * Function to focus the overflow button as a result of keyboard navigation.
	 */
	focusTabOverflowButton = () => {
		assert('tabOverflowButton must exist', this.tabOverflowButton);
		this.focusableTab = this.overflowTab;
		this.tabOverflowButton.focus();
	};

	/**
	 * The width of non-compact tabs when `shouldBalanceTabs` is true.
	 */
	get balancedTabWidth() {
		if (!this.tablistWidth) return;

		// For simplicity, always keep space for the overflow button
		let availableWidth = this.tablistWidth - OVERFLOW_MENU_BUTTON_WIDTH - (this.beforeBlockWidth ?? 0);

		// Subtract space used by compact pinned tabs
		if (this.args.shouldUseCompactPinnedTabs) {
			availableWidth -= this.pinnedTabs.length * COMPACT_PINNED_TAB_WIDTH;
		}

		// Count unpinned visible tabs (excluding those already in visibleTabs like activeTab/focusableTab)
		const unpinnedVisibleTabs = this.visibleTabs.filter((tab) => !tab.isPinned);
		const numTabsToBalance = this.args.shouldUseCompactPinnedTabs
			? unpinnedVisibleTabs.length
			: this.visibleTabs.length;

		if (numTabsToBalance > 0) {
			const optimalWidth = Math.floor(availableWidth / numTabsToBalance);
			return Math.max(MIN_BALANCED_TAB_WIDTH, Math.min(MAX_BALANCED_TAB_WIDTH, optimalWidth));
		}
		return MIN_BALANCED_TAB_WIDTH;
	}

	/**
	 * The action to internally focus the overflow button.
	 * Called on overflow-button focus.
	 */
	onTabOverflowButtonFocus = () => {
		this.focusableTab = this.overflowTab;
	};

	/**
	 * The action run when any tab is blurred. Resets the `isReorderingViaKeyboard` state.
	 */
	onTabBlur = (event: Event) => {
		if (
			event instanceof FocusEvent &&
			event.target instanceof HTMLElement &&
			event.target.getAttribute('data-tab') === this.focusableTab?.value &&
			event.relatedTarget === null
		) {
			// This most likely indicates the tab being moved by keyboard
			return;
		}
		this.isReorderingViaKeyboard = false;
	};

	/**
	 * Modifier to register the tablistElement
	 */
	registerTablistElement = templateModifier((element: HTMLElement) => {
		this.tablistElement = element;
		this.tablistWidth = element.offsetWidth;
	});

	/**
	 * Modifier to register the tablistMenuButton and its API.
	 */
	registerTablistMenu = templateModifier(
		(
			element: HTMLElement,
			positional: [
				{
					menuIsShown: boolean;
					showMenu: (_event: Event) => void;
					hideMenu: (_event: Event) => void;
				},

			this.tabOverflowButton = element;
			this.tablistMenuApi = positional[0];
		},
	);

	/**
	 * Modifier to register the before block for measuring purposes.
	 * The before block's width will be considered when determining which
	 * tabs are shown vs. collapsed into the overflow menu.
	 */
	registerBeforeBlock = templateModifier((element: HTMLElement) => {
		this.beforeBlockElement = element;
		this.beforeBlockWidth = this.beforeBlockElement.offsetWidth;
	});

	/**
	 * Modifier to register the tab width of every rendered tab.
	 * All tabs are initially rendered before overflow truncation.
	 */
	registerTabWidth = templateModifier((element: HTMLElement, positional: [tab: TablistTab<V>]) => {
		if (!this.args.shouldBalanceTabs) {
			schedule('afterRender', () => {
				this.tabWidths.set(positional[0].value, element.offsetWidth);
			});
		}
	});

	/**
	 * Generate a unique ID for the tab's accessibility description.
	 */
	getTabDescriptionId = (tab: TablistTab<V>): string => {
		return `tab-description-${tab.value}`;
	};

	/**
	 * Gets the screen-reader description for a tab based on its capabilities.
	 * Used by the visually hidden element linked to the tab via `aria-describedby`.
	 */
	getTabAccessibilityDescription = (tab: TablistTab<V>): string | undefined => {
		const group = this.getGroupContainingTab(tab);
		const capabilities = [];

		if (tab.menuItems?.length) {
			capabilities.push(this.intl.t('tablist.accessibility.hasMenu'));
		}

		if (this.args.onReorder && !group?.isLocked) {
			capabilities.push(this.intl.t('tablist.accessibility.movable'));
		}

		if (this.args.onDelete && !tab.isPinned && !group?.isLocked) {
			capabilities.push(this.intl.t('tablist.accessibility.closable'));
		}

		if (!capabilities.length) return undefined;

		return capabilities.join(this.intl.t('tablist.accessibility.separator'));
	};

// Helper function to determine if options are groups
function isTablistGroup<V extends string>(option: TablistTab<V> | TablistGroup<V>): option is TablistGroup<V> {
	return 'tabs' in option;
}

// Helper to flatten tabs from options
function getAllTabs<V extends string>(options: Array<TablistTab<V> | TablistGroup<V>>): Array<TablistTab<V>> {
	const tabs: Array<TablistTab<V>> = [];
	for (const option of options) {
		if (isTablistGroup(option)) {
			tabs.push(...option.tabs);
		} else {
			tabs.push(option);
		}
	}
	return tabs;
}

// Sort tabs with pinned first
function sortTabsWithPinnedFirst<V extends string>(tabs: Array<TablistTab<V>>): Array<TablistTab<V>> {
	return [...tabs].sort((a, b) => {
		if (a.isPinned === b.isPinned) return 0;
		return a.isPinned ? -1 : 1;
	});
}

// Get tab groups from options
function getTabGroups<V extends string>(options: Array<TablistTab<V> | TablistGroup<V>>): Array<TablistGroup<V>> {
	const firstOption = options[0];
	if (firstOption && isTablistGroup(firstOption)) {
		// Options are already groups
		return (options as Array<TablistGroup<V>>).map((group) => ({
			...group,
			tabs: sortTabsWithPinnedFirst(group.tabs),
		}));
	}

	// Convert flat tabs to groups
	const allTabs = options as Array<TablistTab<V>>;
	const pinnedTabs = allTabs.filter((tab) => tab.isPinned);
	const unpinnedTabs = allTabs.filter((tab) => !tab.isPinned);
	const groups: Array<TablistGroup<V>> = [];

	if (pinnedTabs.length > 0) {
		groups.push({ id: 'pinned', tabs: pinnedTabs });
	}
	if (unpinnedTabs.length > 0) {
		groups.push({ id: 'default', tabs: unpinnedTabs });
	}

	return groups;
}

export function Tablist<V extends string = string>(props: TablistProps<V>): React.ReactElement {
	const {
		ref,
		value,
		variant = 'default',
		size = 'md',
		options,
		'activeTab': activeTabProp,
		shouldBalanceTabs = false,
		shouldUseCompactPinnedTabs = false,
		onItemClick,
		onReorder,
		onChange,
		onDelete,
		context,
		before,
		children,
		menuOption,
		className,
		'aria-label': ariaLabel,
		'data-testid': testId,
		...restProps
	} = props;
	const generatedId = useId();

	// Track container width for overflow calculation
	const containerRef = useRef<HTMLDivElement>(null);
	const beforeBlockRef = useRef<HTMLDivElement>(null);
	const [containerWidth, setContainerWidth] = useState<number>(0);
	const [beforeBlockWidth, setBeforeBlockWidth] = useState<number>(0);
	const [tabWidths, setTabWidths] = useState<Map<V, number>>(new Map());

	// Track which tab should receive focus
	const [focusableTabValue, setFocusableTabValue] = useState<V | null>(null);

	// Track keyboard reordering state
	const [isReorderingViaKeyboard, setIsReorderingViaKeyboard] = useState(false);

	// Track overflow menu state
	const [overflowMenuOpen, setOverflowMenuOpen] = useState(false);
	const overflowButtonRef = useRef<HTMLButtonElement>(null);

	// Track tab menu state
	const [openTabMenuValue, setOpenTabMenuValue] = useState<V | null>(null);
	const tabMenuButtonRefs = useRef<Map<V, HTMLButtonElement | null>>(new Map());

	// Derive tab groups and all tabs
	const tabGroups = useMemo(() => getTabGroups(options), [options]);
	const allTabs = useMemo(() => getAllTabs(options), [options]);
	const sortedAllTabs = useMemo(() => tabGroups.flatMap((group) => group.tabs), [tabGroups]);
	const pinnedTabs = useMemo(() => allTabs.filter((tab) => tab.isPinned), [allTabs]);
	const unpinnedTabs = useMemo(() => allTabs.filter((tab) => !tab.isPinned), [allTabs]);

	// Determine if this is a tablist role (vs nav)
	const roleIsTablist = useMemo(() => {
		return (
			!onDelete &&
			!sortedAllTabs.some((tab) => tab.menuItems?.length) &&
			sortedAllTabs.length > 0 &&
			sortedAllTabs.every((tab) => tab.href === undefined)
		);
	}, [onDelete, sortedAllTabs]);

	// Get current URL for active tab detection
	const currentUrl = useMemo(() => {
		if (context) {
			return context.fullUrlPathName || '';
		}
		if (typeof window !== 'undefined') {
			return window.location.pathname || '';
		}
		return '';
	}, [context]);

	// Determine active tab
	const activeTab = useMemo<TablistTab<V> | undefined>(() => {
		if (activeTabProp) return activeTabProp;

		return sortedAllTabs.find((tab) => {
			if (tab.currentWhenUrls) {
				return tab.currentWhenUrls.includes(currentUrl);
			}
			if (tab.href) {
				if (currentUrl === tab.href) return true;
				try {
					const tabUrl = new URL(tab.href, window.location.origin);
					return currentUrl === tabUrl.pathname;
				} catch {
					const tabPathname = tab.href.split('?')[0];
					return currentUrl === tabPathname;
				}
			}
			return tab.value === value;
		});
	}, [activeTabProp, sortedAllTabs, currentUrl, value]);

	// Calculate visible tabs based on available width
	const visibleTabs = useMemo<Array<TablistTab<V>>>(() => {
		const visible: Array<TablistTab<V>> = [];
		let widthAvailable = containerWidth ? containerWidth - OVERFLOW_MENU_BUTTON_WIDTH - beforeBlockWidth : 0;

		// Pinned tabs are always visible
		visible.push(...pinnedTabs);

		// Reserve space for active tab if it's not pinned
		if (activeTab && !activeTab.isPinned && !visible.includes(activeTab)) {
			visible.push(activeTab);
			widthAvailable -= shouldBalanceTabs ? MIN_BALANCED_TAB_WIDTH : (tabWidths.get(activeTab.value) ?? 0);
		}

		// Reserve space for focusable tab if different from active
		const focusableTab = sortedAllTabs.find((t) => t.value === focusableTabValue);
		if (focusableTab && focusableTab !== activeTab && !focusableTab.isPinned && !visible.includes(focusableTab)) {
			visible.push(focusableTab);
			widthAvailable -= shouldBalanceTabs ? MIN_BALANCED_TAB_WIDTH : (tabWidths.get(focusableTab.value) ?? 0);
		}

		if (shouldBalanceTabs) {
			widthAvailable -=
				(shouldUseCompactPinnedTabs ? COMPACT_PINNED_TAB_WIDTH : MIN_BALANCED_TAB_WIDTH) * pinnedTabs.length;

			for (const tab of unpinnedTabs) {
				if (!visible.includes(tab) && widthAvailable >= MIN_BALANCED_TAB_WIDTH) {
					widthAvailable -= MIN_BALANCED_TAB_WIDTH;
					visible.push(tab);
				}
			}
		} else {
			const pinnedTabsWidth = pinnedTabs.reduce((total, tab) => {
				return total + (tabWidths.get(tab.value) ?? (shouldUseCompactPinnedTabs ? COMPACT_PINNED_TAB_WIDTH : 0));
			}, 0);
			widthAvailable -= pinnedTabsWidth;

			for (const tab of unpinnedTabs) {
				if (visible.includes(tab)) continue;
				const tabWidth = tabWidths.get(tab.value) ?? 0;
				if (widthAvailable >= tabWidth) {
					widthAvailable -= tabWidth;
					visible.push(tab);
				}
			}
		}

		// Return in sorted order
		return sortedAllTabs.filter((tab) => visible.includes(tab));
	}, [
		containerWidth,
		beforeBlockWidth,
		pinnedTabs,
		activeTab,
		focusableTabValue,
		sortedAllTabs,
		shouldBalanceTabs,
		shouldUseCompactPinnedTabs,
		tabWidths,
		unpinnedTabs,
	]);

	// Calculate overflow tabs
	const visibleTabValues = useMemo(() => {
		return new Set(visibleTabs.map((tab) => tab.value));
	}, [visibleTabs]);

	const overflowTabs = useMemo(() => {
		return sortedAllTabs.filter((tab) => !visibleTabValues.has(tab.value));
	}, [sortedAllTabs, visibleTabValues]);

	// Calculate balanced tab width
	const balancedTabWidth = useMemo(() => {
		if (!containerWidth) return MIN_BALANCED_TAB_WIDTH;

		let availableWidth = containerWidth - OVERFLOW_MENU_BUTTON_WIDTH - beforeBlockWidth;

		if (shouldUseCompactPinnedTabs) {
			availableWidth -= pinnedTabs.length * COMPACT_PINNED_TAB_WIDTH;
		}

		const unpinnedVisibleTabs = visibleTabs.filter((tab) => !tab.isPinned);
		const numTabsToBalance = shouldUseCompactPinnedTabs ? unpinnedVisibleTabs.length : visibleTabs.length;

		if (numTabsToBalance > 0) {
			const optimalWidth = Math.floor(availableWidth / numTabsToBalance);
			return Math.max(MIN_BALANCED_TAB_WIDTH, Math.min(MAX_BALANCED_TAB_WIDTH, optimalWidth));
		}
		return MIN_BALANCED_TAB_WIDTH;
	}, [containerWidth, beforeBlockWidth, shouldUseCompactPinnedTabs, pinnedTabs.length, visibleTabs]);

	// Initialize focusable tab
	useEffect(() => {
		if (focusableTabValue === null && sortedAllTabs.length > 0) {
			setFocusableTabValue((activeTab?.value ?? sortedAllTabs[0]?.value) as V);
		}
	}, [focusableTabValue, activeTab, sortedAllTabs]);

	// Handle resize observer
	useEffect(() => {
		const container = containerRef.current;
		if (!container) return;

		// Set initial width
		setContainerWidth(container.offsetWidth);

		// Use ResizeObserver if available (not in test environment)
		if (typeof ResizeObserver === 'undefined') return;

		const resizeObserver = new ResizeObserver((entries) => {
			for (const entry of entries) {
				setContainerWidth(entry.contentRect.width);
			}
		});

		resizeObserver.observe(container);

		return () => resizeObserver.disconnect();
	}, []);

	// Measure before block
	useEffect(() => {
		if (beforeBlockRef.current) {
			setBeforeBlockWidth(beforeBlockRef.current.offsetWidth);
		}
	}, [before]);

	// Handle tab click
	const handleTabClick = useCallback(
		(tab: TablistTab<V>, event?: React.MouseEvent) => {
			if (event) {
				event.preventDefault();
				event.stopPropagation();
			}

			tab.action?.();
			onItemClick?.(tab);
			onChange?.(tab.value);
			setFocusableTabValue(tab.value);
		},
		[onItemClick, onChange],
	);

	// Handle tab close
	const handleTabClose = useCallback(
		(tab: TablistTab<V>, event?: React.MouseEvent | React.KeyboardEvent) => {
			if (event) {
				event.preventDefault();
				event.stopPropagation();
			}

			if (!onDelete) return;

			const tabGroup = tabGroups.find((g) => g.tabs.includes(tab));
			if (tabGroup?.isLocked) return;

			const currentIndex = sortedAllTabs.indexOf(tab);
			let nextFocusableTab: TablistTab<V> | undefined;

			if (sortedAllTabs.length === 1) {
				nextFocusableTab = undefined;
			} else if (currentIndex !== 0) {
				nextFocusableTab = sortedAllTabs[currentIndex - 1];
			} else {
				nextFocusableTab = sortedAllTabs[currentIndex + 1];
			}

			if (tab === activeTab && nextFocusableTab) {
				onChange?.(nextFocusableTab.value);
			}

			setFocusableTabValue(nextFocusableTab?.value ?? null);
			onDelete(tab.value);
		},
		[onDelete, tabGroups, sortedAllTabs, activeTab, onChange],
	);

	// Handle keyboard navigation
	const handleKeyDown = useCallback(
		(tab: TablistTab<V> | null, event: React.KeyboardEvent) => {
			const { key } = event;

			if (key === 'Escape') {
				setIsReorderingViaKeyboard(false);
				if (overflowMenuOpen) {
					setOverflowMenuOpen(false);
					overflowButtonRef.current?.focus();
				}
				if (openTabMenuValue !== null) {
					setOpenTabMenuValue(null);
				}
				return;
			}

			if (key === ' ' && tab !== null) {
				const group = tabGroups.find((g) => g.tabs.some((t) => t.value === tab.value));
				if (group?.isLocked) return;

				event.preventDefault();
				if (onReorder) {
					setIsReorderingViaKeyboard(!isReorderingViaKeyboard);
				}
				return;
			}

			if (key === 'Enter') {
				event.preventDefault();
				if (event.target instanceof HTMLElement) {
					event.target.click();
				}
				return;
			}

			if ((key === 'Delete' || key === 'Backspace') && onDelete && focusableTabValue) {
				const focusableTab = sortedAllTabs.find((t) => t.value === focusableTabValue);
				if (focusableTab) {
					handleTabClose(focusableTab, event);
				}
				return;
			}

			// Arrow key navigation
			if (['ArrowLeft', 'ArrowRight', 'Home', 'End'].includes(key)) {
				event.preventDefault();
				event.stopPropagation();

				const currentIndex = visibleTabs.findIndex((t) => t.value === focusableTabValue);
				let newIndex = currentIndex;

				if (key === 'ArrowLeft') {
					if (isReorderingViaKeyboard && focusableTabValue) {
						// Move tab left within group
						moveTabInGroup(focusableTabValue, -1);
					} else if (focusableTabValue === `overflow-${generatedId}`) {
						newIndex = visibleTabs.length - 1;
					} else if (currentIndex > 0) {
						newIndex = currentIndex - 1;
					}
				} else if (key === 'ArrowRight') {
					if (isReorderingViaKeyboard && focusableTabValue) {
						// Move tab right within group
						moveTabInGroup(focusableTabValue, 1);
					} else if (currentIndex === visibleTabs.length - 1 && overflowTabs.length > 0) {
						setFocusableTabValue(`overflow-${generatedId}` as V);
						overflowButtonRef.current?.focus();
						return;
					} else if (currentIndex < visibleTabs.length - 1) {
						newIndex = currentIndex + 1;
					}
				} else if (key === 'Home') {
					newIndex = 0;
				} else if (key === 'End') {
					if (overflowTabs.length > 0) {
						setFocusableTabValue(`overflow-${generatedId}` as V);
						overflowButtonRef.current?.focus();
						return;
					}
					newIndex = visibleTabs.length - 1;
				}

				if (newIndex >= 0 && newIndex < visibleTabs.length) {
					const newTab = visibleTabs[newIndex];
					if (newTab) {
						setFocusableTabValue(newTab.value);
						// Focus the new tab element
						const tabElement = containerRef.current?.querySelector(`[data-tab="${newTab.value}"]`);
						if (tabElement instanceof HTMLElement) {

						}
					}
				}
			}

		[
			tabGroups,
			onReorder,
			isReorderingViaKeyboard,
			onDelete,
			focusableTabValue,
			sortedAllTabs,
			handleTabClose,
			visibleTabs,
			overflowTabs.length,
			generatedId,
			overflowMenuOpen,
		],

	const moveTabInGroup = useCallback(

			if (!onReorder) return;

			const tab = sortedAllTabs.find((t) => t.value === tabValue);
			if (!tab) return;

			const group = tabGroups.find((g) => g.tabs.includes(tab));
			if (!group || group.isLocked) return;

			const groupTabs = group.tabs;
			const currentIndex = groupTabs.indexOf(tab);
			const targetIndex = currentIndex + direction;

			if (targetIndex < 0 || targetIndex >= groupTabs.length) return;

			const targetTab = groupTabs[targetIndex];
			if (!targetTab || tab.isPinned !== targetTab.isPinned) return;

			const newGroupTabs = [...groupTabs];
			newGroupTabs.splice(currentIndex, 1);
			newGroupTabs.splice(targetIndex, 0, tab);

			// Update options based on format
			const firstOption = options[0];
			if (firstOption && isTablistGroup(firstOption)) {
				const updatedGroups = tabGroups.map((g) => {
					if (g.id === group.id) {
						return { ...g, tabs: sortTabsWithPinnedFirst(newGroupTabs) };
					}
					return g;
				});
				onReorder(updatedGroups as Array<TablistTab<V> | TablistGroup<V>>);
			} else {
				const updatedTabs = tabGroups.flatMap((g) => {
					if (g.id === group.id) {
						return sortTabsWithPinnedFirst(newGroupTabs);
					}
					return g.tabs;
				});
				onReorder(updatedTabs);
			}
		},
		[onReorder, sortedAllTabs, tabGroups, options],
	);

	// Handle tab menu toggle
	const handleTabMenuToggle = useCallback((tabValue: V, event?: React.MouseEvent) => {
		if (event) {
			event.preventDefault();
			event.stopPropagation();
		}
		setOpenTabMenuValue((current) => (current === tabValue ? null : tabValue));
	}, []);

	// Handle tab menu item click
	const handleTabMenuItemClick = useCallback((tab: TablistTab<V>, menuItem: TablistTabMenuItem) => {
		menuItem.action?.(tab);
		setOpenTabMenuValue(null);
	}, []);

	// Handle overflow tab click
	const handleOverflowTabClick = useCallback(
		(tab: TablistTab<V>) => {
			onChange?.(tab.value);
			setFocusableTabValue(tab.value);
			setOverflowMenuOpen(false);

			// If reordering is enabled, move the tab into visible area
			if (onReorder) {
				const group = tabGroups.find((g) => g.tabs.includes(tab));
				if (group) {
					const visibleTabsInGroup = visibleTabs.filter((t) => group.tabs.includes(t));
					const lastVisibleTab = visibleTabsInGroup[visibleTabsInGroup.length - 1];

					if (lastVisibleTab) {
						const newGroupTabs = [...group.tabs];
						const clickedIndex = newGroupTabs.indexOf(tab);
						if (clickedIndex !== -1) {
							newGroupTabs.splice(clickedIndex, 1);
							const insertIndex = newGroupTabs.indexOf(lastVisibleTab) + 1;
							newGroupTabs.splice(insertIndex, 0, tab);
						}

						const firstOption = options[0];
						if (firstOption && isTablistGroup(firstOption)) {
							const updatedGroups = tabGroups.map((g) => {
								if (g.id === group.id) {
									return { ...g, tabs: newGroupTabs };
								}
								return g;
							});
							onReorder(updatedGroups as Array<TablistTab<V> | TablistGroup<V>>);
						} else {
							const updatedTabs = tabGroups.flatMap((g) => {
								if (g.id === group.id) {
									return newGroupTabs;
								}
								return g.tabs;
							});
							onReorder(updatedTabs);
						}
					}
				}
			}
		},
		[onChange, onReorder, tabGroups, visibleTabs, options],
	);

	// Track tab widths using a ref to avoid infinite loops during render
	const tabWidthsRef = useRef<Map<V, number>>(new Map());
	const pendingWidthUpdate = useRef(false);

	// Effect to sync tabWidths state from ref after mount
	useEffect(() => {
		if (pendingWidthUpdate.current && tabWidthsRef.current.size > 0) {
			pendingWidthUpdate.current = false;
			setTabWidths(new Map(tabWidthsRef.current));
		}
	}, []);

	// Register tab width - using ref to avoid state updates during render
	const registerTabRef = useCallback(
		(tabValue: V) => {
			return (element: HTMLElement | null) => {
				if (element && !shouldBalanceTabs) {
					const newWidth = element.offsetWidth;
					if (tabWidthsRef.current.get(tabValue) !== newWidth) {
						tabWidthsRef.current.set(tabValue, newWidth);
						pendingWidthUpdate.current = true;
					}
				}
			};
		},
		[shouldBalanceTabs],
	);

	// Get accessibility description for tab
	const getTabAccessibilityDescription = useCallback(
		(tab: TablistTab<V>): string | undefined => {
			const group = tabGroups.find((g) => g.tabs.includes(tab));
			const capabilities: string[] = [];

			if (tab.menuItems?.length) {
				capabilities.push('Has menu');
			}
			if (onReorder && !group?.isLocked) {
				capabilities.push('Movable');
			}
			if (onDelete && !tab.isPinned && !group?.isLocked) {
				capabilities.push('Closable');
			}

			return capabilities.length > 0 ? capabilities.join('; ') : undefined;
		},
		[tabGroups, onReorder, onDelete],
	);

	// Check if tab should show close button
	const shouldShowCloseButton = useCallback(
		(tab: TablistTab<V>): boolean => {
			if (!onDelete || tab.isPinned) return false;
			const group = tabGroups.find((g) => g.tabs.includes(tab));
			return !group?.isLocked;
		},
		[onDelete, tabGroups],
	);

	// Get tab tooltip
	const getTabTooltip = useCallback(
		(tab: TablistTab<V>): string | undefined => {
			if (tab.tooltip) return tab.tooltip;
			if (tab.isPinned && shouldUseCompactPinnedTabs) {
				return tab.label;
			}
			return undefined;
		},
		[shouldUseCompactPinnedTabs],
	);

	// CSS custom properties
	const cssVars: CSSProperties = {
		'--tablist-overflow-menu-button-width': `${OVERFLOW_MENU_BUTTON_WIDTH}px`,
		...(shouldUseCompactPinnedTabs && {
			'--tablist-compact-pinned-tab-width': `${COMPACT_PINNED_TAB_WIDTH}px`,
		}),
		...(shouldBalanceTabs && {
			'--tablist-balanced-tab-width': `${balancedTabWidth}px`,
		}),
	} as CSSProperties;

	// Get tabs to render for a group
	const getGroupTabsToRender = (groupId: string) => {
		const group = tabGroups.find((g) => g.id === groupId);
		if (!group) return [];
		return group.tabs.filter((tab) => visibleTabs.includes(tab));
	};

	// Context value
	const contextValue: TablistContextValue = {
		activeTabValue: activeTab?.value ?? null,
	};

	// Render individual tab
	const renderTab = (tab: TablistTab<V>, index: number, group: TablistGroup<V>) => {
		const isActive = tab === activeTab;
		const isFocusable = tab.value === focusableTabValue;
		const isGrabbed = isReorderingViaKeyboard && isFocusable && !group.isLocked;
		const Tag = tab.href ? 'a' : 'button';
		const descriptionId = `tab-description-${tab.value}`;
		const description = getTabAccessibilityDescription(tab);

		return (
			<div
				key={tab.value}
				ref={registerTabRef(tab.value)}
				className={cn(
					styles.tabWrapper,
					tab.isPinned && shouldUseCompactPinnedTabs && styles.compact,
					shouldBalanceTabs && !tab.isPinned && styles.balanced,
				)}
				data-qid-tablist-tab-wrapper={tab.value}
			>
				<Tag
					className={styles.tab}
					role={roleIsTablist ? 'tab' : undefined}
					type={roleIsTablist ? 'button' : undefined}
					href={tab.href}
					tabIndex={isFocusable ? 0 : -1}
					data-tab={tab.value}
					data-qid-tablist-tab
					data-qid-index={index}
					data-qid={tab.qid ?? 'Tablist__Link'}
					data-qid-component-type="Luna::Tab"
					data-has-menu={tab.menuItems ? 'true' : 'false'}
					data-is-active={isActive}
					data-is-grabbed={isGrabbed || undefined}
					data-is-pinned={tab.isPinned ? 'true' : 'false'}
					aria-label={tab.isPinned ? getTabTooltip(tab) : undefined}
					aria-current={!roleIsTablist ? (isActive ? 'true' : 'false') : undefined}
					aria-selected={roleIsTablist ? (isActive ? 'true' : 'false') : undefined}
					aria-controls={roleIsTablist ? `luna-panel-${tab.value}` : undefined}
					aria-describedby={description ? descriptionId : undefined}
					onClick={(e) => handleTabClick(tab, e as React.MouseEvent)}
					onKeyDown={(e) => handleKeyDown(tab, e)}
					onBlur={() => setIsReorderingViaKeyboard(false)}
				>
					<div data-qid-tablist-tab-inner className={styles.inner}>
						<div data-qid-tablist-tab-inner-content className={styles.content}>
							{children ? children(tab) : tab.label}
						</div>
					</div>
				</Tag>

				{(tab.menuItems?.length || shouldShowCloseButton(tab)) && (
					<div data-qid-tablist-tab-controls className={styles.controls}>
						{tab.menuItems?.length && (
							<>
								<Button
									ref={(el: HTMLButtonElement | HTMLAnchorElement | null) => {
										tabMenuButtonRefs.current.set(tab.value, el as HTMLButtonElement | null);
									}}
									tabIndex={-1}
									variant="unstyled"
									size="xs"
									aria-hidden={roleIsTablist ? 'true' : undefined}
									aria-label={t('tablist.ariaActions')}
									aria-expanded={openTabMenuValue === tab.value}
									className={cn(styles.tabControl, styles.menuButton)}
									data-qid-tablist-tab-menu-button
									data-tablist-tab-control
									onClick={(e) => handleTabMenuToggle(tab.value, e)}
								>
									<Icon icon="more-vertical" size="xs" />
								</Button>
								<Popup
									show={openTabMenuValue === tab.value}
									relativeTo={{ current: tabMenuButtonRefs.current.get(tab.value) ?? null }}
									placement="bottom-end"
									variant="light"
									onClose={() => setOpenTabMenuValue(null)}
								>
									<div className={styles.menu}>
										{tab.menuItems.map((menuItem, menuIndex) => (
											<button
												key={menuIndex}
												className={styles.menuOption}
												data-test-tab-menu-option
												onClick={() => handleTabMenuItemClick(tab, menuItem)}
											>
												{menuItem.label}
											</button>
										))}
									</div>
								</Popup>
							</>
						)}
						{shouldShowCloseButton(tab) && (
							<Button
								variant="unstyled"
								data-qid-tablist-tab-close-button
								data-tablist-tab-control
								data-qid="NavTab__Close"
								className={cn(styles.tabControl, styles.deleteButton)}
								tabIndex={-1}
								size="xs"
								aria-hidden={roleIsTablist ? 'true' : undefined}
								aria-label={t('tablist.ariaCloseTab')}
								onClick={(e) => handleTabClose(tab, e)}
							>
								<Icon icon="close" size="xs" />
							</Button>
						)}
					</div>
				)}

				{description && (
					<div id={descriptionId} className={styles.visuallyHidden}>
						{description}
					</div>
				)}
			</div>
		);
	};

	return (
		<TablistContext.Provider value={contextValue}>
			<div
				ref={(el) => {
					if (typeof ref === 'function') ref(el);
					else if (ref) (ref as React.MutableRefObject<HTMLDivElement | null>).current = el;
					(containerRef as RefObject<HTMLDivElement | null>).current = el;
				}}
				className={cn(styles.Tablist, className)}
				data-variant={variant}
				data-size={size}
				data-qid-luna-tablist
				data-testid={testId}
				aria-label={ariaLabel}
				{...restProps}
			>
				<div style={cssVars} className={styles.tablistInner} role={roleIsTablist ? 'tablist' : undefined}>
					{before && (
						<div ref={beforeBlockRef} data-qid-luna-tablist-before className={styles.before}>
							{before}
						</div>
					)}

					{tabGroups.map((group) => (
						<div
							key={group.id}
							role={roleIsTablist ? 'none' : undefined}
							className={styles.tabGroup}
							data-tablist-group-id={group.id}
							data-test-tablist-tabs
							data-qid-component-type="Luna::Tablist"
						>
							{getGroupTabsToRender(group.id).map((tab, index) => renderTab(tab, index, group))}
						</div>
					))}
				</div>

				{overflowTabs.length > 0 && (
					<div data-qid-tablist-tab-wrapper className={styles.tabWrapper}>
						<Button
							ref={overflowButtonRef}
							tabIndex={focusableTabValue === `overflow-${generatedId}` ? 0 : -1}
							className={cn(styles.tab, styles.tablistOverflowButton)}
							variant="unstyled"
							aria-label={t('tablist.ariaMoreTabs')}
							data-test-tablist-overflow-menu
							data-tab={`overflow-${generatedId}`}
							data-qid-component-type="Luna::Tablist::OverflowMenu"
							onClick={() => setOverflowMenuOpen(!overflowMenuOpen)}
							onFocus={() => setFocusableTabValue(`overflow-${generatedId}` as V)}
							onBlur={() => setIsReorderingViaKeyboard(false)}
							onKeyDown={(e) => handleKeyDown(null, e)}
						>
							<Icon icon="more-horizontal" />
						</Button>
						<Popup
							show={overflowMenuOpen}
							relativeTo={overflowButtonRef}
							placement="bottom-end"
							variant="light"
							onClose={() => setOverflowMenuOpen(false)}
						>
							<div className={styles.menu}>
								{overflowTabs.map((tab) => (
									<button
										key={tab.value}
										className={styles.menuOption}
										data-test-tablist-menu-option
										data-tab={tab.value}
										onClick={() => handleOverflowTabClick(tab)}
									>
										{menuOption ? menuOption(tab) : children ? children(tab) : tab.label}
									</button>
								))}
							</div>
						</Popup>
					</div>
				)}
			</div>
		</TablistContext.Provider>
	);
}
