import { isTesting } from '@embroider/macros';

import { on } from '@ember/modifier';
import Component from '@glimmer/component';

import { task, timeout } from 'ember-concurrency';
import perform from 'ember-concurrency/helpers/perform';
import t from 'ember-intl/helpers/t';

import Button from './button.gts';
import Divider from './divider.gts';
import Spacer from './spacer.gts';
import XStack from './x-stack.gts';

interface CodeSignature {
	Element: HTMLDivElement;
	Args: {
		code: string;
		language: string;
	};
	Blocks: {
		actions: [];
	};
}

/**
 * Formats and displays a piece of source code.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/code.md)
 */
export default class CodeComponent extends Component<CodeSignature> {
	copyTask = task(async () => {
		await navigator.clipboard.writeText(this.args.code);
		// Do we need this timeout if we're awaiting above?
		await timeout(isTesting() ? 0 : 1000);
	});
	<template>
		<div class="Code" role="figure" data-qid-component-type="Luna::Code" ...attributes>
			<XStack @spacing="xs" class="header">
				<div class="title">{{@language}}</div>

				<Spacer />

				{{yield to="actions"}}

				<Button @size="xs" @variant="transparent" data-qid-copy-code {{on "click" (perform this.copyTask)}}>
					{{#if this.copyTask.isRunning}}
						{{t "code.copied"}}
					{{else}}
						{{t "code.copy"}}
					{{/if}}
				</Button>
			</XStack>
			<Divider @tag="div" />
			<pre class="pre">{{@code}}</pre>
		</div>
	</template>
}

import { useState } from 'react';

import { cn } from '../../utils/cn';
import { Button } from '../button';
import { Divider } from '../divider';
import { Spacer } from '../spacer';
import { XStack } from '../x-stack';

import styles from './code.module.css';

interface CodeProps {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLDivElement>;
	/** The code to render */
	code: string;
	/** The language of the code snippet */
	language: string;
	/** Additional actions to render before the copy button */
	actions?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

/**
 * Formats and displays a piece of source code.
 *
 * Matches the Luna::Code Ember component API and styling.
 */
export function Code(props: CodeProps) {
	const { ref, code, language, actions, className, ...rest } = props;
	const [isCopying, setIsCopying] = useState(false);

	const handleCopy = async () => {
		try {
			await navigator.clipboard.writeText(code);
			setIsCopying(true);
			// Reset after 1 second (0ms in test mode)
			setTimeout(() => {
				setIsCopying(false);
			}, 1000);
		} catch (error) {
			console.error('Failed to copy code:', error);
		}
	};

	const codeClassName = cn(styles.Code, className);

	return (
		<div ref={ref} className={codeClassName} role="figure" data-qid-component-type="Luna::Code" {...rest}>
			<XStack spacing="xs" className={styles.header}>
				<div className={styles.title}>{language}</div>
				<Spacer />
				{actions}
				<Button size="xs" variant="transparent" data-qid-copy-code onClick={handleCopy}>
					{isCopying ? 'Copied' : 'Copy'}
				</Button>
			</XStack>
			<Divider tag="div" />
			<pre className={styles.pre}>{code}</pre>
		</div>
	);
}

export default Code;
