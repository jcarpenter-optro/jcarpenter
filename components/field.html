export interface LunaFieldSignature {
	Element: HTMLDivElement;
	Args: {
		variant?: 'vertical' | 'horizontal';
		level?: 1 | 2;
		checkbox?: boolean;
		required?: boolean;
		id?: string;
	};
	Blocks: {
		'prefix'?: [];
		'label': [inputId: string];
		'action': [];
		'label-decoration'?: [];
		'description'?: [];
		'input': [
			{
				id: string;
			},
		];
	};
}

	get checkbox() {
		return String(this.args.checkbox ?? false);

field-context.tsx
import { createContext, useContext } from 'react';

/**
 * Value provided by Field to its descendants so that Label and form controls
 * (Input, Select, Checkbox, etc.) can share a single input id and label id
 * without prop drilling. Matches the Ember luna-core pattern where Field
 * provides 'luna-input-id' and controls consume it.
 */
export interface FieldContextValue {
	/** Id for the primary form control (used on the input and for label's htmlFor). */
	inputId: string;
	/** Id for the label (used on the label element and for control's aria-labelledby). */
	labelId: string;
}

const FieldContext = createContext<FieldContextValue | undefined>(undefined);

export function useFieldContext(): FieldContextValue | undefined {
	return useContext(FieldContext);
}

export { FieldContext };
field.tsx
import { useId } from 'react';

import { Field as BaseField } from '@base-ui-components/react/field';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { Label } from '../label';
import { Text } from '../text';

import { FieldContext } from './field-context';

import styles from './field.module.css';

export type FieldVariant = 'vertical' | 'horizontal';
export type FieldLevel = 1 | 2;

// Generate a label ID for a given input ID (used for aria-labelledby on Select)
export const labelIdForField = (inputId: string) => `${inputId}-label`;

interface FieldBaseProps {
	/** Layout variant */
	variant?: FieldVariant;
	/** Level (1 or 2) - affects label color for default variant */
	level?: FieldLevel;
	/** Whether this is a checkbox field (affects layout) */
	checkbox?: boolean;
	/** Whether the field is required */
	required?: boolean;
	/** ID for the input (auto-generated if not provided) */
	id?: string;
	/** Label content */
	label?: React.ReactNode;
	/** Prefix content (rendered before label) */
	prefix?: React.ReactNode;
	/** Label decoration (rendered after label, e.g., Help icon) */
	labelDecoration?: React.ReactNode;
	/** Action content (rendered on the right side of label layout) */
	action?: React.ReactNode;
	/** Description/helper text */
	description?: React.ReactNode;
	/** Input content (receives { id } as prop) */
	input?: (context: { id: string }) => React.ReactNode;
	/** Additional CSS class */
	className?: string;
	/** Children - can be used instead of slots for simple cases */
	children?: React.ReactNode;
}

type FieldProps = ComponentPropsWithRef<FieldBaseProps, HTMLDivElement>;

/**
 * Combines a label and an input together in different layouts.
 *
 * Matches the Luna::Field Ember component API and styling.
 * Uses Base UI Field.Root as the foundation.
 */
export function Field(props: FieldProps) {
	const {
		ref,
		variant = 'vertical',
		level,
		checkbox = false,
		required = false,
		id: providedId,
		label,
		prefix,
		labelDecoration,
		action,
		description,
		input,
		children,
		className,
		...rest
	} = props;

	// Generate IDs
	const generatedId = useId();
	const inputId = providedId || generatedId.replace(/:/g, '-');
	const labelId = labelIdForField(inputId);

	const fieldClassName = cn(styles.Field, className);
	const fieldContextValue = { inputId, labelId };

	// Checkbox fields have a different layout
	if (checkbox) {
		return (
			<FieldContext.Provider value={fieldContextValue}>
				<BaseField.Root
					className={fieldClassName}
					data-variant={variant}
					data-checkbox="true"
					data-qid-component-type="Luna::Field"
					ref={ref}
					{...rest}
				>
					{variant === 'horizontal' && <div />}
					<div>
						<div className={styles.checkboxInput}>
							{prefix}
							{input ? input({ id: inputId }) : children}
							{required && (
								<Text color="red-500" font="200" aria-label="Required" className={styles.asterisk}>
									*
								</Text>
							)}
							{labelDecoration}
						</div>
						{description && <div className={styles.description}>{description}</div>}
					</div>
				</BaseField.Root>
			</FieldContext.Provider>
		);
	}

	// Non-checkbox fields
	const hasLabelContent = label || labelDecoration;

	return (
		<FieldContext.Provider value={fieldContextValue}>
			<BaseField.Root
				className={fieldClassName}
				data-variant={variant}
				data-checkbox="false"
				data-qid-component-type="Luna::Field"
				ref={ref}
				{...rest}
			>
				{hasLabelContent || action ? (
					<div className={styles.labelLayout}>
						{hasLabelContent ? (
							<div className={styles.labelStart}>
								<Label
									for={inputId}
									variant={variant === 'horizontal' ? 'light' : 'default'}
									level={level}
									className={styles.label}
									id={labelId}
								>
									{prefix}
									{label}
									{required && (
										<Text className={styles.asterisk} color="red-500" font="200">
											*
										</Text>
									)}
								</Label>
								{labelDecoration}
							</div>
						) : (
							<div />
						)}
						{action && <div className={styles.action}>{action}</div>}
					</div>
				) : null}

				<div>
					<div>{input ? input({ id: inputId }) : children}</div>
					{description && <div className={styles.description}>{description}</div>}
				</div>
			</BaseField.Root>
		</FieldContext.Provider>
	);
}

export default Field;
