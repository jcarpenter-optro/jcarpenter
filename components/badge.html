import { cn } from '../../utils/cn';

import styles from './badge.module.css';

export type BadgeColor = 'gray' | 'blue' | 'red' | 'orange' | 'green' | 'purple' | 'yellow' | 'teal';
export type BadgeSize = 'sm' | 'md' | 'lg';
export type BadgeVariant = 'rounded' | 'round';
export type BadgeWeight = 'weak' | 'normal' | 'strong' | 'xstrong';

interface BadgeProps extends React.HTMLAttributes<HTMLSpanElement> {
	/** Optional ref for the underlying element (React 19 ref-as-prop) */
	ref?: React.Ref<HTMLSpanElement>;
	/** Badge color */
	color?: BadgeColor;
	/** Badge size */
	size?: BadgeSize;
	/** Badge variant (rounded or round) */
	variant?: BadgeVariant;
	/** Badge weight (visual emphasis) */
	weight?: BadgeWeight;
	/** Children content */
	children?: React.ReactNode;
}

/**
 * A small label, generally appearing inside or in close proximity to another larger interface component,
 * representing a status, property, or some other metadata.
 *
 * Matches the Luna::Badge Ember component API and styling.
 */
export function Badge(props: BadgeProps) {
	const {
		ref,
		color = 'gray',
		size = 'md',
		variant = 'rounded',
		weight = 'weak',
		children,
		className,
		...rest
	} = props;

	const badgeClassName = cn(styles.Badge, className);

	return (
		<span
			ref={ref}
			className={badgeClassName}
			data-color={color}
			data-size={size}
			data-variant={variant}
			data-weight={weight}
			data-qid-component-type="Luna::Badge"
			{...rest}
		>
			{children}
		</span>
	);
}

export default Badge;
