notice.tsx
import type { ReactNode } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { extractCompoundSlotsWithRest } from '../../utils/extract-compound-slots';
import { Frame } from '../frame';
import { Icon } from '../icon';

import styles from './notice.module.css';

const NOTICE_SLOT_MAP = { 'Notice.Title': 'title' } as const;

export type NoticeVariant = 'red' | 'green' | 'orange' | 'blue' | 'gray' | 'teal' | 'purple' | 'yellow' | 'grayA';
export type NoticeSize = 'sm' | 'lg';

// ============================================================================
// Title Subcomponent
// ============================================================================

export interface TitleProps {
	children: ReactNode;
}

/**
 * Optional title for the notice.
 */
export function Title({ children }: TitleProps) {
	return <>{children}</>;
}

Title.displayName = 'Notice.Title';

// ============================================================================
// Main Component
// ============================================================================

interface NoticeBaseProps extends Omit<React.HTMLAttributes<HTMLDivElement>, 'title'> {
	/** Icon name to display. If not provided, uses default based on variant */
	icon?: string;
	/** Notice variant/color. Defaults to 'blue' */
	variant?: NoticeVariant;
	/** Notice size. Defaults to 'lg' */
	size?: NoticeSize;
	/** Children components - use Notice.Title for optional title */
	children: ReactNode;
}

export type NoticeProps = ComponentPropsWithRef<NoticeBaseProps, HTMLDivElement>;

const DEFAULT_ICONS: Record<NoticeVariant, string> = {
	red: 'circle-important',
	green: 'module-controls',
	orange: 'module-issues',
	blue: 'circle-info',
	gray: 'circle-info',
	teal: 'circle-info',
	purple: '',
	yellow: '',
	grayA: '',
};

/**
 * Displays a brief, important message to the user in a way that attracts their attention without interrupting their task.
 *
 * Matches the Luna::Notice Ember component API and styling.
 *
 * @example
 * ```tsx
 * <Notice variant="blue">
 *   <Notice.Title>Important Update</Notice.Title>
 *   <Text>Your changes have been saved.</Text>
 * </Notice>
 * ```
 */
function NoticeComponent(props: NoticeProps) {
	const { ref, icon: iconProp, variant = 'blue', size = 'lg', children, className, ...rest } = props;

	const { slots, rest: restChildren } = extractCompoundSlotsWithRest(children, NOTICE_SLOT_MAP);
	const titleContent = slots.title;

	const icon = iconProp ?? DEFAULT_ICONS[variant];
	const highlight = `left-${variant}` as const;
	const textColor = `${variant}-500`;
	const padding = size === 'sm' ? 'sm' : 'md';

	const noticeClassName = cn(styles.Notice, className);

	return (
		<Frame
			ref={ref}
			highlight={highlight}
			padding={padding}
			className={noticeClassName}
			data-size={size}
			aria-live="polite"
			data-qid-component-type="Luna::Notice"
			{...rest}
		>
			<div className={styles.layout} data-qid-notice={icon}>
				<Icon icon={icon} color={textColor} size={size} />
				<div className={styles.content}>
					{titleContent && <div className={styles.title}>{titleContent}</div>}
					<div>{restChildren}</div>
				</div>
			</div>
		</Frame>
	);
}

// Attach subcomponents
const Notice = Object.assign(NoticeComponent, {
	Title,
});

export { Notice };

export default Notice;
