/**
 * Adds appropriate ARIA attributes to the SelectDate trigger element based on component configuration
 */
const triggerAttrs = modifier(
	(
		element: SelectDateTriggerAttrsSignature['Element'],
		_positional: unknown,
		named: SelectDateTriggerAttrsSignature['Args']['Named'],
	) => {
		if (!named.shouldRenderAsDiv) {
			element.setAttribute('role', 'combobox');
			element.setAttribute('aria-expanded', named.showPopup ? 'true' : 'false');
			element.setAttribute('aria-owns', named.popupId);
			element.setAttribute('aria-haspopup', 'dialog');
			element.setAttribute('aria-controls', named.showPopup ? named.popupId : '');
		}
		element.setAttribute(
			'aria-required',
			named.requirements && requirementArg(named.requirements, 'required') ? 'true' : 'false',
		);
		if (named.showErrorMessage) {
			element.setAttribute('aria-invalid', 'true');
		} else {
			element.setAttribute('aria-invalid', 'false');
		}
	},
);

/**
 * A date picker that lets people select a date or a range of dates.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/select-date.md)
 */
export default class LunaSelectDate<Value extends SingleValue | RangeValue = SingleValue> extends Component<
	LunaSelectDateSignature<Value>
> {
	@service declare intl: IntlService;

	selectionType: SelectionType = 'single';
	popupId = `${guidFor(this)}--popup`;

	@consume('luna-input-id') inputId!: string;
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;

	manager: InputManager;

	constructor(owner: Owner, args: LunaSelectDateSignature<Value>['Args']) {
		super(owner, args);

		assert('You cannot provide both ariaLabel and ariaLabelledBy', !args.ariaLabel || !args.ariaLabelledBy);
		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;
		const managerOptions: InputManagerOptions = {
			mode: 'change',
			name: args.name,
			didAutosave: this.didAutosave.bind(this),

			focus: this.focusInput.bind(this),

			ariaDescribedBy: this.args.ariaDescribedBy ?? null,

			get value() {
				return args.value;
			},

			get isDisabled(): boolean {
				return args.disabled ?? false;
			},

			get requirements() {
				return args.requirements;
			},

			get serverErrors() {
				// eslint-disable-next-line @typescript-eslint/no-deprecated
				return args.serverErrors;
			},

			get formContext() {
				return that.formContext;
			},

			get selectorIsVisible() {
				return that.showPopup;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	@tracked popup: LunaPopupBlockValue | null = null;

	@tracked showPopup = false;

	/**
	 * Tracks the user selection internally so we can reset it if the user cancels
	 * selection, and only autosave if a full selection is made.
	 */
	@localCopy('args.value') declare internalValue: Value;

	/**
	 * Determines if the user has made a new selection and thus the onChange should fire,
	 * vs a onCancel
	 */
	@dedupeTracked valueIsUnchanged = true;

	get id() {
		return this.args.id ?? this.inputId ?? `luna-select-date-${nextInputId++}`;
	}

	get isSingleSelection() {
		return this.selectionType === 'single';
	}

	get isRangeSelection() {
		return this.selectionType === 'range';
	}

	get popupAriaLabel() {
		return this.intl.t('input.select-date.placeholder', {
			kind: this.selectionType,
			textEntry: String(this.textEntryIsEnabled),
		});
	}

	/**
	 * Whether the text entry is enabled. When true, alters the popup aria-label.
	 * Redeclared and bound in the instance initializer of the consuming app.
	 * See `client/app/initializers/luna.ts` for example implementation.
	 */
	// eslint-disable-next-line @typescript-eslint/class-literal-property-style
	get textEntryIsEnabled() {
		return false;
	}

	get hasValue() {
		const { value } = this.args;

		if (value !== null && typeof value === 'object') {
			return Boolean(value?.start) || Boolean(value?.end);
		}

		return Boolean(value);
	}

	get triggerIsActive() {
		return this.args.active ?? this.showPopup;
	}

	get formattedPrimaryDate(): string | null {
		const value = this.internalValue;

		if (!value) {
			return null;
		}

		const primaryDate = typeof value === 'string' ? value : (value.start ?? null);

		if (!primaryDate) {
			return null;
		}

		const primaryDateObj = createDate(primaryDate);
		assert('primary date must be valid', primaryDateObj instanceof Date);

		return this.intl.formatDate(primaryDateObj, { day: 'numeric', month: 'short', year: 'numeric' });
	}

	get formattedSecondaryDate(): string | null {
		const value = this.internalValue;

		if (!value || typeof value === 'string') {
			return null;
		}

		const secondaryDate = value.end ?? null;

		if (!secondaryDate) {
			return null;
		}

		const secondaryDateObj = createDate(secondaryDate);
		assert('secondary date must be valid', secondaryDateObj instanceof Date);

		return this.intl.formatDate(secondaryDateObj, { day: 'numeric', month: 'short', year: 'numeric' });
	}

	get formattedSelection(): string | null {
		if (!this.formattedPrimaryDate) {
			return null;
		}

		if (this.isSingleSelection) {
			return this.formattedPrimaryDate;
		}

		// Range selection requires both dates
		if (!this.formattedSecondaryDate) {
			return null;
		}

		const value = this.internalValue;
		const primaryDate = typeof value === 'object' ? (value?.start ?? null) : null;
		const secondaryDate = typeof value === 'object' ? (value?.end ?? null) : null;
		const sameYear = primaryDate && secondaryDate && primaryDate.split('-')[0] === secondaryDate.split('-')[0];

		if (sameYear) {
			// Omit the year from the primary date when both dates are in the same year
			const primaryDateObj = createDate(primaryDate);
			assert('primary date must be valid', primaryDateObj instanceof Date);
			const formattedPrimaryDateWithoutYear = this.intl.formatDate(primaryDateObj, { day: 'numeric', month: 'short' });
			return `${formattedPrimaryDateWithoutYear} – ${this.formattedSecondaryDate}`;
		}

		return `${this.formattedPrimaryDate} – ${this.formattedSecondaryDate}`;
	}

	get labelId() {
		return this.inputId ? labelIdForField(this.inputId) : null;
	}

	onFocusIn = () => {
		if (this.args.shouldRenderAsDiv) {
			return;
		}
		this.manager.focusIn();
	};

	@action
	maybeOpenDateSelector() {
		if (this.args.shouldRenderAsDiv) {
			return;
		}
		this.internalValue = this.args.value;
		this.valueIsUnchanged = true;
		this.showPopup = true;
		this.args.onOpen?.();
	}

	@action
	async willCloseDateSelector(reason: string) {
		// User explicitly cancelled via escape key or clicking outside
		if (reason === 'esc' || reason === 'click-outside') {
			return this.args.onCancel?.(reason);
		}

		// SAFETY: Should be a RangeValue if it's a range selection
		if (this.valueIsUnchanged || (this.isRangeSelection && isIncomplete(this.internalValue as RangeValue))) {
			return this.args.onCancel?.(reason);
		}

		const value: Value = this.internalValue;
		assert('Expected a value', value !== undefined);
		// SAFETY: I'm not sure why we need this `as` given that the above should have verified it for us
		await this.args.onChange?.(value as Exclude<Value, undefined>);
		this.manager.onChange();
	}

	@action
	onCloseDateSelector() {
		if (this.valueIsUnchanged) {
			this.args.didCancel?.();
		}
		this.showPopup = false;
	}

	/**
	 * Handles value changes from the calendar component.
	 * Called when user selects a date from the calendar or submits via text entry.
	 */
	onSave = (value: Exclude<Value, undefined>, { shouldClose }: { shouldClose: boolean }) => {
		if (this.isRangeSelection) {
			this.valueIsUnchanged =
				typeof value === 'object' &&
				typeof this.internalValue === 'object' &&
				value?.start === this.internalValue?.start &&
				value?.end === this.internalValue?.end;
		} else {
			this.valueIsUnchanged = value === this.internalValue;
		}

		this.internalValue = value;

		if (shouldClose) {
			assert('popup must be registered', this.popup);
			void this.popup.closeTask.perform();
		}
	};

	@action
	didAutosave() {
		this.valueIsUnchanged = true;
		this.args.didAutosave?.();
	}

	focusInput = () => {
		focusById(this.id);
	};

	registerPopup = (popup: LunaPopupBlockValue) => {
		this.popup = popup;
	};

			class="SelectDate {{@class}}"
			data-size={{or @size "md"}}
			data-disabled={{toStringHelper @disabled}}
			data-qid-component-type="Luna::SelectDate"

/**
 * Checks if a range value is incomplete (has start but no end)
 */
function isIncomplete(value: SelectDateRangeValue): boolean {
	return typeof value === 'object' && value !== null && Boolean(value?.start) && !value?.end;
}

/**
 * Type guard to check if value is a range value
 */
function isRangeValue(value: SelectDateValue): value is SelectDateRangeValue {
	return typeof value === 'object' && value !== null && ('start' in value || 'end' in value);
}

/**
 * Formats a date string for display
 */
function formatDateForDisplay(dateStr: string | undefined): string | null {
	if (!dateStr) return null;
	try {
		const date = parseISO(dateStr);
		if (!isValid(date)) return null;
		return format(date, 'd MMM yyyy');
	} catch {
		return null;
	}
}

export function SelectDate({
	value,
	selectionType: selectionTypeProp,
	active,
	onCancel,
	onChange,
	onOpen,
	didCancel,
	didClose,
	className,
	placeholder,
	chevronIsHidden = false,
	triggerClass,
	triggerComponent: TriggerComponent,
	shouldRenderAsDiv = false,
	showIcon = true,
	id: providedId,
	requirements,
	name,
	disabled = false,
	min,
	max,
	serverErrors,
	ariaLabel,
	ariaLabelledBy,
	ariaDescribedBy,
	didAutosave,
	size = 'md',
	variant = 'white',
	align,
	placement = 'bottom',
	disableDate,
	firstDayOfWeek,
	showToday = true,
	testId,
	children,
	ref,
	...props
}: SelectDateProps): React.ReactElement {
	const fieldContext = useFieldContext();
	const id = providedId ?? fieldContext?.inputId ?? `luna-select-date-${nextInputId++}`;
	const popupId = `${id}--popup`;

	const formContext = useFormContext();
	const triggerRef = useRef<HTMLButtonElement>(null);

	// Track popup visibility
	const [showPopup, setShowPopup] = useState(false);

	// Track internal selection (for range, can be partial)
	const [internalValue, setInternalValue] = useState<SelectDateValue>(value);
	const valueIsUnchangedRef = useRef(true);

	// Sync internal value when external value changes
	useEffect(() => {
		setInternalValue(value);
	}, [value]);

	// Determine selection type - prop takes precedence, then infer from value
	const selectionType: SelectionType = useMemo(() => {
		if (selectionTypeProp) {
			return selectionTypeProp;
		}
		if (isRangeValue(value)) {
			return 'range';
		}
		return 'single';
	}, [selectionTypeProp, value]);

	const isRangeSelection = selectionType === 'range';

	// Input manager for form integration
	const manager = useInputManager({
		mode: 'change',
		name,
		valueIsUnchanged: valueIsUnchangedRef.current,
		didAutosave: useCallback(() => {
			valueIsUnchangedRef.current = true;
			didAutosave?.();
		}, [didAutosave]),
		ariaDescribedBy: ariaDescribedBy ?? null,
		value,
		isDisabled: disabled,
		requirements,
		serverErrors,
		formContext,
		selectorIsVisible: showPopup,
	} as InputManagerOptions);

	// Computed properties
	const hasValue = useMemo(() => {
		if (isRangeValue(value)) {
			return Boolean(value?.start) || Boolean(value?.end);
		}
		return Boolean(value);
	}, [value]);

	// Show clear button when there's a saved value AND field is not required
	const clearButtonIsShown = useMemo(() => {
		const canBeCleared = (!isRangeSelection && value) || (isRangeValue(value) && value?.end);
		return Boolean(canBeCleared) && !requirements?.required;
	}, [isRangeSelection, value, requirements?.required]);

	const triggerIsActive = active ?? showPopup;

	// Format the selection for display
	const formattedSelection = useMemo(() => {
		if (!internalValue) return null;

		if (typeof internalValue === 'string') {
			return formatDateForDisplay(internalValue);
		}

		if (!isRangeValue(internalValue)) return null;

		// Range value
		const primary = formatDateForDisplay(internalValue?.start);
		const secondary = formatDateForDisplay(internalValue?.end);

		if (!primary) return null;
		if (!secondary) return null;

		// Check if same year to format more compactly
		if (internalValue?.start && internalValue?.end) {
			const startYear = internalValue.start.split('-')[0];
			const endYear = internalValue.end.split('-')[0];
			if (startYear === endYear) {
				// Omit year from start date
				const startDate = parseISO(internalValue.start);
				const formattedStart = format(startDate, 'd MMM');
				return `${formattedStart} – ${secondary}`;
			}
		}

		return `${primary} – ${secondary}`;
	}, [internalValue]);

	// Handlers
	const handleOpen = useCallback(() => {
		if (shouldRenderAsDiv || disabled) return;

		setInternalValue(value);
		valueIsUnchangedRef.current = true;
		setShowPopup(true);
		onOpen?.();
	}, [shouldRenderAsDiv, disabled, value, onOpen]);

	const handleWillClose = useCallback(
		async (source: 'esc' | 'enter' | 'click-outside' | 'manual') => {
			// User explicitly cancelled
			if (source === 'esc' || source === 'click-outside') {
				onCancel?.(source);
				return;
			}

			// Range selection incomplete - treat as cancel
			if (
				valueIsUnchangedRef.current ||
				(isRangeSelection && isRangeValue(internalValue) && isIncomplete(internalValue))
			) {
				onCancel?.(source);
				return;
			}

			// Apply the change
			if (internalValue !== undefined) {
				if (typeof internalValue === 'string') {
					await onChange?.(internalValue);
				} else if (isRangeValue(internalValue) && internalValue?.start) {
					await onChange?.({
						start: internalValue.start,
						end: internalValue.end || '',
					});
				}
				manager.onChange();
			}
		},
		[isRangeSelection, internalValue, onCancel, onChange, manager],
	);

	const handleClose = useCallback(
		(_source?: OnCloseSource, _event?: unknown) => {
			if (valueIsUnchangedRef.current) {
				didCancel?.();
			}
			setShowPopup(false);
			didClose?.();
		},
		[didCancel, didClose],
	);

	const handleCalendarChange = useCallback(
		(newValue: string | { start: string; end: string } | null) => {
			// Update internal value
			setInternalValue(newValue);

			// Check if value actually changed
			if (isRangeSelection) {
				const oldVal = isRangeValue(internalValue) ? internalValue : null;
				const newVal = typeof newValue === 'object' ? newValue : null;
				valueIsUnchangedRef.current = oldVal?.start === newVal?.start && oldVal?.end === newVal?.end;
			} else {
				valueIsUnchangedRef.current = newValue === internalValue;
			}

			// For single selection, close immediately after selection
			if (!isRangeSelection && newValue && typeof newValue === 'string') {
				// Trigger close with the new value
				setTimeout(() => {
					// Apply the change
					onChange?.(newValue);
					manager.setValueIsUnchanged(false);
					manager.onChange();
					setShowPopup(false);
					didClose?.();
				}, 0);
			}

			// For range selection, close when both dates are selected
			if (isRangeSelection && newValue && typeof newValue === 'object' && newValue.start && newValue.end) {
				setTimeout(() => {
					onChange?.(newValue);
					manager.setValueIsUnchanged(false);
					manager.onChange();
					setShowPopup(false);
					didClose?.();
				}, 0);
			}
		},
		[isRangeSelection, internalValue, onChange, manager, didClose],
	);

	const handleFocusIn = useCallback(() => {
		if (!shouldRenderAsDiv) {
			manager.focusIn();
		}
	}, [shouldRenderAsDiv, manager]);

	const handleClear = useCallback(() => {
		onChange?.(null);
		manager.setValueIsUnchanged(false);
		manager.onChange();
		setShowPopup(false);
		didClose?.();
	}, [onChange, manager, didClose]);

	// Common ARIA props for custom triggers
	const customTriggerAriaProps = useMemo(() => {
		const ariaProps: Record<string, string | undefined> = {
			'aria-required': requirements?.required ? 'true' : 'false',
			'aria-invalid': manager.showErrorMessage ? 'true' : 'false',
		};

		if (!shouldRenderAsDiv) {
			ariaProps['role'] = 'combobox';
			ariaProps['aria-expanded'] = showPopup ? 'true' : 'false';
			ariaProps['aria-owns'] = popupId;
			ariaProps['aria-haspopup'] = 'dialog';
			ariaProps['aria-controls'] = showPopup ? popupId : '';
		}

		if (ariaLabel) {
			ariaProps['aria-label'] = ariaLabel;
		}
		if (ariaLabelledBy) {
			ariaProps['aria-labelledby'] = ariaLabelledBy;
		}
		if (manager.ariaDescribedBy) {
			ariaProps['aria-describedby'] = manager.ariaDescribedBy;
		}

		return ariaProps;
	}, [
		requirements?.required,
		manager.showErrorMessage,
		manager.ariaDescribedBy,
		shouldRenderAsDiv,
		showPopup,
		popupId,
		ariaLabel,
		ariaLabelledBy,
	]);

	// Determine if we should show the date icon
	const shouldShowIcon = showIcon && variant !== 'transparent';

	// Common popup component (used by both custom and default triggers)
	const datePickerPopup = (
		<DatePickerPopup
			showPopup={showPopup}
			popupId={popupId}
			triggerRef={triggerRef}
			placement={placement}
			selectionType={selectionType}
			internalValue={internalValue}
			min={min}
			max={max}
			disableDate={disableDate}
			firstDayOfWeek={firstDayOfWeek}
			showToday={showToday}
			clearButtonIsShown={clearButtonIsShown}
			onCalendarChange={handleCalendarChange}
			onClear={handleClear}
			onWillClose={handleWillClose}
			onClose={handleClose}
		/>
	);

	// Common wrapper props
	const wrapperProps = {
		ref,
		'className': cn(styles.SelectDate, className),
		'data-size': size,
		'data-disabled': disabled ? 'true' : undefined,
		'data-qid-component-type': 'Luna::SelectDate',
		'data-focused': manager.inputHasFocus ? 'true' : undefined,
		'data-errored': manager.showErrorMessage ? 'true' : undefined,
		'data-luna-feature': 'meld',
		'data-testid': testId,
		...props,
	};

	// Render custom trigger
	if (TriggerComponent) {
		return (
			<div {...wrapperProps}>
				<TriggerComponent
					id={id}
					data-qid-select-date-trigger="custom"
					onFocus={handleFocusIn}
					onClick={handleOpen}
					{...customTriggerAriaProps}
				/>
				<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
					{manager.errorMessageToDisplay}
				</InputError>
				{datePickerPopup}
			</div>
		);
	}

	return (
		<div {...wrapperProps}>
			<Button
				ref={triggerRef}
				id={id}
				variant={variant}
				size={size}
				align={align}
				width="full"
				errored={manager.showErrorMessage}
				disabled={disabled}
				active={triggerIsActive}
				className={cn(styles.trigger, triggerClass)}
				data-qid-select-date-trigger="default"
				onFocus={handleFocusIn}
				onClick={handleOpen}
				aria-required={requirements?.required || false}
				aria-invalid={manager.showErrorMessage}
				aria-expanded={!shouldRenderAsDiv ? showPopup : undefined}
				aria-haspopup={!shouldRenderAsDiv ? 'dialog' : undefined}
				aria-controls={!shouldRenderAsDiv && showPopup ? popupId : undefined}
				aria-label={ariaLabel}
				aria-labelledby={ariaLabelledBy}
				aria-describedby={manager.ariaDescribedBy}
			>
				{shouldShowIcon && <Icon className={styles.preIcon} icon="date-range" size="sm" />}
				{hasValue ? (
					<div className={styles.label}>{children ? children(value) : formattedSelection}</div>
				) : (
					<div className={styles.label} data-placeholder>
						{placeholder}
					</div>
				)}
				{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
					<Spinner size="sm" transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined} />
				)}
				{!chevronIsHidden && <Icon className={styles.postIcon} icon="chevron-down" size="sm" />}
			</Button>
			<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
				{manager.errorMessageToDisplay}
			</InputError>
			{datePickerPopup}
		</div>
	);
}

export default SelectDate;
