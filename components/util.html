import { hash } from '@ember/helper';
import { action } from '@ember/object';
import { service } from '@ember/service';
import Component from '@glimmer/component';

import type LunaStateManager from '../../services/state-manager.ts';

interface CacheStateSignature {
	Args: {
		key: string;
		defaultValue: unknown;
	};
	Blocks: {
		default: [
			{
				value: unknown;
				update: <T>(value: T) => T;
			},
		];
	};
}

export default class CacheState extends Component<CacheStateSignature> {
	@service('luna/state-manager') declare stateManager: LunaStateManager;

	get value() {
		return this.stateManager.getState(this.args.key, this.args.defaultValue);
	}

	@action
	update<T>(value: T): T {
		return this.stateManager.setState(this.args.key, value);
	}

	<template>{{yield (hash value=this.value update=this.update)}}</template>
}
tablist-manager.gts
import { assert } from '@ember/debug';
import { hash } from '@ember/helper';
import { action } from '@ember/object';
import { service } from '@ember/service';
import Component from '@glimmer/component';

import type LunaSessionStorageService from '../../services/session-storage.ts';

interface Tab {
	/** Any value except null or undefined */
	value: unknown;
}

function isTab(value: unknown): value is Tab {
	if (value !== null && typeof value === 'object' && 'value' in value) {
		const innerValue = (value as Tab).value;
		return innerValue !== undefined && innerValue !== null;
	}
	return false;
}

function isTabs(value: unknown): value is Tab[] {
	return Array.isArray(value) && value.every(isTab);
}

interface TablistManagerSignature {
	Args: {
		key: string;
	};

	Blocks: {
		default: [
			{
				tabs: Tab[];
				hasTabs: boolean;
				activeTab: unknown;
				onReorder: (tabs: Tab[]) => void;
				onChange: (tabValue: Tab) => void;
				closeTab: (value: unknown) => void;
				closeAllTabs: () => void;
				addTab: (tab: Tab) => void;
				addTabFromCmdClick: (tab: Tab, event: KeyboardEvent) => void;
			},
		];
	};
}

export default class TablistManager extends Component<TablistManagerSignature> {
	@service('luna/session-storage') declare sessionStorage: LunaSessionStorageService;

	get activeTabKey() {
		return `tablist-manager-${this.args.key}-active-tab`;
	}

	get tabsKey() {
		return `tablist-manager-${this.args.key}-tabs`;
	}

	get tabs() {
		const tabs = this.sessionStorage.getItem<Tab[]>(this.tabsKey) ?? [];
		assert(`Expected ${this.tabsKey} to be an array of tabs`, isTabs(tabs));
		return tabs;
	}

	get activeTab() {
		return this.sessionStorage.getItem(this.activeTabKey);
	}

	get hasTabs() {
		return Boolean(this.tabs.length);
	}

	@action
	onReorder(tabs: Tab[]) {
		this.sessionStorage.setItem(this.tabsKey, tabs);
	}

	@action
	onChange(tabValue: Tab) {
		this.sessionStorage.setItem(this.activeTabKey, tabValue);
	}

	@action
	closeAllTabs() {
		this.sessionStorage.removeItem(this.activeTabKey);
		this.sessionStorage.removeItem(this.tabsKey);
	}

	@action
	closeTab(value: unknown) {
		const tabs = this.tabs.filter((option) => option.value !== value);
		this.sessionStorage.setItem(this.tabsKey, tabs);
	}

	@action
	addTabFromCmdClick(tab: Tab, event: KeyboardEvent) {
		if (!(event.metaKey || event.ctrlKey)) {
			return;
		}

		event.preventDefault();
		event.stopPropagation();

		this.addTab(tab);
	}

	@action
	addTab(tab: Tab) {
		assert('You must provide a "value" key on the tab object.', isTab(tab));

		const splitViewTabs = this.sessionStorage.getItem<Tab[]>(this.tabsKey) ?? [];
		assert(`Expected ${this.tabsKey} to be an array of tabs`, isTabs(splitViewTabs));
		const tabs = [...splitViewTabs];

		if (!tabs.length || !tabs.find((t) => t.value === tab.value)) {
			// Convert proxy object to a plain object
			tabs.push(Object.assign({}, tab));
		}

		this.sessionStorage.setItem(this.tabsKey, tabs);
		this.sessionStorage.setItem(this.activeTabKey, tab.value);
	}

	<template>
		{{yield
			(hash
				tabs=this.tabs
				hasTabs=this.hasTabs
				activeTab=this.activeTab
				onReorder=this.onReorder
				onChange=this.onChange
				closeTab=this.closeTab
				closeAllTabs=this.closeAllTabs
				addTab=this.addTab
				addTabFromCmdClick=this.addTabFromCmdClick
			)
		}}
	</template>
}
