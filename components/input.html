export interface LunaInputSignature<
	V extends AttrValue = AttrValue,
	IsTextarea extends boolean = false,
	IsFake extends boolean = false,
> {
	Element: IsFake extends true ? HTMLDivElement : IsTextarea extends true ? HTMLTextAreaElement : HTMLInputElement;
	Args: {
		class?: string;
		size?: 'xs' | 'sm' | 'md' | 'lg';
		variant?: 'default' | 'transparent';
		onChange?: (value: string) => unknown;
		didAutosave?: () => void;
		rows?: IsTextarea extends true ? number : never;
		value: V;
		id?: string;
		requirements?: Requirements;
		ariaDescribedBy?: string;

		/**
		 * Whether the input should validate with every change.
		 * Used in scenarios where immediate feedback is non-punishing, such as when
		 * requiring a unique value, or when the form is already in its errored state
		 * and the user is trying to fix it.
		 */
		shouldValidateEagerly?: boolean;

		/**
		 * Whether the input should validate on render.
		 * Used in scenarios where a form might be re-rendered but we want to immediately display invalid state.
		 */
		shouldValidateOnRender?: boolean;

		form?: LunaFormBlockValue;
		name?: string;
		disabled?: boolean;
		/** @deprecated This can almost always be handled better with `externalErrorsFor` on the form or a custom validator. */
		serverErrors?: Array<{ message?: string } | string>;
	};
	Blocks: IsTextarea extends true
		? Record<string, never>
		: {
				icon?: [];
			};
}

/**
 * A basic input for entering text.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/input.md)
 */
export default class LunaInput<
	V extends AttrValue,
	IsTextarea extends boolean = false,
	IsFake extends boolean = false,
> extends Component<LunaInputSignature<V, IsTextarea, IsFake>> {
	@consume('luna-input-id') inputId!: string;
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;

	manager: InputManager;

	/**
	 * The internally tracked value of the input.
	 * Set by the local `onChange` function before saving calling the parent method.
	 * Used by `onFocusOut` to re-call the `onChange` method if the value has whitespace.
	 */
	@tracked private _value = '';

	constructor(owner: Owner, args: LunaInputSignature<V, IsTextarea, IsFake>['Args']) {
		super(owner, args);
		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;
		const managerOptions: InputManagerOptions = {
			mode: 'blur',
			focus: this.focusInput.bind(this),
			didAutosave: this.didAutosave.bind(this),
			shouldValidateEagerly: args.shouldValidateEagerly,
			name: args.name,

			ariaDescribedBy: args.ariaDescribedBy ?? null,

			get value(): V {
				return args.value;
			},

			get isDisabled(): boolean {
				return args.disabled ?? false;
			},

			get requirements() {
				return args.requirements;
			},

			get serverErrors() {
				// eslint-disable-next-line @typescript-eslint/no-deprecated
				return args.serverErrors;
			},

			get formContext() {
				return that.formContext;
			},

			get valueIsUnchanged() {
				return that.valueIsUnchanged;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		if (this.args.shouldValidateOnRender) {
			this.manager.validate();
		}

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	get id() {
		return this.args.id ?? this.inputId ?? `luna-input-${nextInputId++}`;
	}

	// Set true in Fake subclasses
	// eslint-disable-next-line @typescript-eslint/class-literal-property-style
	get shouldRenderAsDiv(): IsTextarea {
		// @ts-expect-error The way we're doing subclassing here is confusing for TS.
		return false;
	}

	// Set to true in Textarea subclass
	// eslint-disable-next-line @typescript-eslint/class-literal-property-style
	get textarea(): IsTextarea {
		// @ts-expect-error The way we're doing subclassing here is confusing for TS.
		return false;
	}

	@dedupeTracked valueIsUnchanged = true;

	@action
	async onChange(eventOrString: Event | string) {
		this.valueIsUnchanged = false;
		let value: string;
		if (typeof eventOrString === 'string') {
			value = eventOrString;
		} else {
			assert(
				'Expected event target',
				eventOrString.target instanceof (this.textarea ? HTMLTextAreaElement : HTMLInputElement),
			);
			value = eventOrString.target.value;
		}
		this._value = value;
		await this.args.onChange?.(value);
		this.manager.onChange();
		if (this.args.shouldValidateEagerly) {
			this.manager.validate();
		}
	}

	onFocusOut = async (event: FocusEvent) => {
		if (this.valueIsUnchanged || (event.target instanceof HTMLElement && event.target.dataset.isEsc)) {
			this.manager.focusOutAndTrySave(event);
			return;
		}

		if (typeof this.args.value === 'string') {
			assert('value must exist', event.target instanceof (this.textarea ? HTMLTextAreaElement : HTMLInputElement));

			const value = event.target.value.trim();

			if (value !== this._value) {
				await this.onChange(value);
			}
		}

		this.manager.focusOutAndTrySave(event);
	};

	@action
	didAutosave() {
		this.valueIsUnchanged = true;
		this.args.didAutosave?.();
	}

	focusInput = () => {
		focusById(this.id);
	};
	<template>
		{{! template-lint-disable data-qid-interactive }}
		<div
			class="TextInput {{@class}}"
			data-variant={{or @variant "default"}}
			data-size={{or @size "md"}}
			data-textarea={{toString0 this.textarea}}
			data-disabled={{toString0 @disabled}}
			data-luna-feature="meld"
			data-qid-component-type="Luna::Input"
			data-focused={{toString0 this.manager.inputHasFocus}}
			data-errored={{toString0 this.manager.showErrorState}}
		>

			{{#unless this.textarea}}
				{{#if (has-block "icon")}}
					<div class="icon" role="button" {{on "click" this.focusInput}}>
						{{! @glint-expect-error Glint can't determine that this only exists for textarea }}
						{{yield to="icon"}}
					</div>
				{{/if}}
			{{/unless}}

			{{#if this.shouldRenderAsDiv}}
				<div class="fake {{if this.textarea 'textarea' 'input'}}" ...attributes>
					{{@value}}
				</div>
			{{else}}
				{{#if this.textarea}}

					<textarea
						id={{this.id}}
						aria-describedby={{this.manager.ariaDescribedBy}}
						aria-invalid={{toString0 this.manager.showErrorMessage}}
						aria-required={{if (requirementArg @requirements "required") "true" "false"}}
						rows={{or @rows "5"}}
						{{on "input" this.onChange}}
						{{on "focusin" this.manager.focusIn}}
						{{on "focusout" this.onFocusOut}}
						...attributes
						value={{@value}}
						readonly={{@disabled}}
					></textarea>
				{{else}}

					<input
						id={{this.id}}
						aria-describedby={{this.manager.ariaDescribedBy}}
						aria-invalid={{toString0 this.manager.showErrorMessage}}
						aria-required={{if (requirementArg @requirements "required") "true" "false"}}
						min={{requirementArg @requirements "min"}}
						max={{requirementArg @requirements "max"}}
						{{on "input" this.onChange}}
						{{on "focusin" this.manager.focusIn}}
						{{on "focusout" this.onFocusOut}}
						...attributes
						value={{@value}}
						disabled={{@disabled}}
					/>
				{{/if}}

				{{#if (or this.manager.isAutosaving this.manager.didAutosaveSuccessfully)}}
					<div class="suffix">
						<Spinner @size="sm" @transitionTo={{if this.manager.didAutosaveSuccessfully "success"}} />
					</div>
				{{/if}}

				<InputError id={{this.manager.errorId}} @showWhen={{this.manager.showErrorMessage}}>
					{{this.manager.errorMessageToDisplay}}
				</InputError>
			{{/if}}
		</div>
	</template>

import { useCallback, useEffect, useId, useMemo, useRef, useState } from 'react';

import { Input as BaseInput } from '@base-ui-components/react/input';
import type { Requirements } from '@soxhub/luna-utils/validators';

import { cn } from '../../utils/cn';
import { useInputManager } from '../../utils/use-input-manager';
import { useOptionalClickToEditContext } from '../click-to-edit/click-to-edit';
import { useFieldContext } from '../field/field-context';
import { useFormContext } from '../form/form';
import { InputError } from '../input-error';
import { Spinner } from '../spinner';

import styles from './input.module.css';

export type InputSize = 'xs' | 'sm' | 'md' | 'lg';
export type InputVariant = 'default' | 'transparent';

interface InputBaseProps {
	/** Optional ref for the underlying input (React 19 ref-as-prop) */
	'ref'?: React.Ref<HTMLInputElement>;
	/** Input value */
	'value'?: string;
	/** Callback fired when value changes */
	'onChange'?: (value: string) => void;
	/** Input size */
	'size'?: InputSize;
	/** Visual variant */
	'variant'?: InputVariant;
	/** Whether input is disabled */
	'disabled'?: boolean;
	/** Input ID (auto-generated if not provided) */
	'id'?: string;
	/** Input name */
	'name'?: string;
	/** Placeholder text */
	'placeholder'?: string;
	/** Icon to display on the left */
	'icon'?: React.ReactNode;
	/** Whether input is in error state */
	'errored'?: boolean;
	/** Whether input is focused (controlled) */
	'focused'?: boolean;
	/** Additional CSS class */
	'className'?: string;
	/** ARIA described by */
	'aria-describedby'?: string;
	/** ARIA required */
	'aria-required'?: boolean | 'true' | 'false';
	/** ARIA invalid */
	'aria-invalid'?: boolean | 'true' | 'false';
	/** Validation requirements */
	'requirements'?: Requirements;
	/** Whether to validate eagerly on change */
	'shouldValidateEagerly'?: boolean;
	/** Whether to validate on render */
	'shouldValidateOnRender'?: boolean;
	/** @deprecated Server errors (use externalErrorsFor on Form instead) */
	'serverErrors'?: Array<{ message?: string } | string>;
	/** Callback fired when autosave completes successfully */
	'didAutosave'?: () => void;
}

export type InputProps = InputBaseProps &
	Omit<React.InputHTMLAttributes<HTMLInputElement>, keyof InputBaseProps | 'onChange' | 'value'>;

/**
 * A basic input for entering text.
 *
 * Matches the Luna::Input Ember component API and styling.
 * Uses Base UI Input as the foundation.
 */
export function Input(props: InputProps) {
	const {
		ref,
		'value': _value = '',
		onChange,
		size = 'md',
		variant = 'default',
		disabled = false,
		'id': providedId,
		name,
		placeholder,
		icon,
		errored = false,
		'focused': controlledFocused,
		className,
		'aria-describedby': providedAriaDescribedBy,
		'aria-required': ariaRequired,
		'aria-invalid': ariaInvalid,
		requirements,
		shouldValidateEagerly = false,
		shouldValidateOnRender = false,
		serverErrors,
		didAutosave,
		...rest
	} = props;

	const generatedId = useId();
	const fieldContext = useFieldContext();
	const formContextFromHook = useFormContext();
	const clickToEditContext = useOptionalClickToEditContext();
	const id = providedId ?? clickToEditContext?.id ?? fieldContext?.inputId ?? generatedId;
	const formContext = clickToEditContext?.form ?? formContextFromHook;
	const effectiveValue = clickToEditContext
		? typeof clickToEditContext.value === 'string'
			? clickToEditContext.value
			: String(clickToEditContext.value ?? '')
		: (_value ?? '');
	const effectiveOnChange: (value: string) => void = useMemo(
		() =>
			clickToEditContext
				? (v: string) => clickToEditContext.onChange(v as Parameters<typeof clickToEditContext.onChange>[0])
				: (onChange ?? (() => {})),
		[clickToEditContext, onChange],
	);
	const effectiveSize = clickToEditContext?.size ?? size;
	const effectiveVariant = clickToEditContext?.variant ?? variant;
	const effectivePlaceholder = clickToEditContext?.placeholder ?? placeholder;
	const effectiveDidAutosave = clickToEditContext?.didAutosave ?? didAutosave;
	const inputRef = useRef<HTMLInputElement>(null);
	const [_internalValue, setInternalValue] = useState(effectiveValue);
	const valueIsUnchangedRef = useRef(true);

	const focusInput = useCallback(() => {
		if (inputRef.current) {
			inputRef.current.focus();
		} else if (ref && typeof ref === 'object' && ref.current) {
			ref.current.focus();
		}
	}, [ref]);

	const manager = useInputManager({
		mode: 'blur',
		valueIsUnchanged: valueIsUnchangedRef.current,
		serverErrors,
		requirements,
		name,
		value: effectiveValue,
		isDisabled: disabled,
		shouldValidateEagerly,
		formContext,
		ariaDescribedBy: providedAriaDescribedBy ?? null,
		focus: focusInput,
		didAutosave: () => {
			valueIsUnchangedRef.current = true;
			effectiveDidAutosave?.();
		},
	});

	// Register/unregister with form
	// Use a ref to track the manager for registration
	const managerRef = useRef(manager);
	managerRef.current = manager;

	useEffect(() => {
		if (formContext) {
			formContext.actions.registerInputManager(managerRef.current);
			return () => {
				formContext.actions.unregisterInputManager(managerRef.current);
			};
		}
	}, [formContext]);

	// Validate on render if requested
	useEffect(() => {
		if (shouldValidateOnRender) {
			manager.validate();
		}
	}, [shouldValidateOnRender, manager]);

	// Sync internal value with prop
	useEffect(() => {
		setInternalValue(effectiveValue);
	}, [effectiveValue]);

	const [internalFocused, setInternalFocused] = useState(false);
	const focused = controlledFocused !== undefined ? controlledFocused : internalFocused || manager.inputHasFocus;

	const inputClassName = cn(styles.TextInput, className);

	const handleChange = useCallback(
		async (newValue: string) => {
			valueIsUnchangedRef.current = false;
			manager.setValueIsUnchanged(false);
			setInternalValue(newValue);
			await effectiveOnChange?.(newValue);
			manager.onChange();
		},
		[effectiveOnChange, manager],
	);

	const handleFocus = useCallback(
		(_event: React.FocusEvent<HTMLInputElement>) => {
			setInternalFocused(true);
			manager.focusIn();
		},
		[manager],
	);

	const handleBlur = useCallback(
		async (event: React.FocusEvent<HTMLInputElement>) => {
			setInternalFocused(false);

			// Check for Esc key first
			if (event.target instanceof HTMLElement && event.target.dataset.isEsc) {
				manager.focusOutAndTrySave(event.nativeEvent);
				return;
			}

			// Handle value trimming and updating
			if (typeof effectiveValue === 'string') {
				const trimmedValue = event.target.value.trim();

				// Compare against prop value to catch all changes
				if (trimmedValue !== effectiveValue) {
					await handleChange(trimmedValue);
				}
			}

			// Always try to save on blur (autosave logic will check if value changed)
			manager.focusOutAndTrySave(event.nativeEvent);
		},
		[manager, effectiveValue, handleChange],
	);

	const handleIconClick = useCallback(() => {
		focusInput();
	}, [focusInput]);

	// showErrorState controls the error ring (validation errors only)
	// showErrorMessage controls whether the error message is displayed (includes serverErrors)
	const showErrorRing = errored || manager.showErrorState;
	const ariaDescribedBy = manager.ariaDescribedBy ?? providedAriaDescribedBy;
	const ariaInvalidAttr = ariaInvalid ?? (manager.showErrorMessage ? 'true' : undefined);
	const ariaRequiredAttr =
		(ariaRequired ?? requirements?.required) ? (requirements?.required === true ? 'true' : 'false') : undefined;

	const { 'data-testid': dataTestId, ...inputRest } = rest as React.InputHTMLAttributes<HTMLInputElement> & {
		'data-testid'?: string;
	};

	return (
		<div
			className={inputClassName}
			data-variant={effectiveVariant}
			data-size={effectiveSize}
			data-textarea="false"
			data-disabled={disabled ? 'true' : 'false'}
			data-luna-feature="meld"
			data-qid-component-type="Luna::Input"
			data-focused={focused ? 'true' : 'false'}
			data-errored={showErrorRing ? 'true' : 'false'}
			data-testid={dataTestId}
		>
			{icon && (
				<div className={styles.icon} role="button" onClick={handleIconClick}>
					{icon}
				</div>
			)}

			<BaseInput
				ref={(node) => {
					inputRef.current = node;
					if (typeof ref === 'function') {
						ref(node);
					} else if (ref && typeof ref === 'object') {
						(ref as React.MutableRefObject<HTMLInputElement | null>).current = node;
					}
				}}
				id={id}
				name={name}
				value={effectiveValue}
				onValueChange={handleChange}
				onFocus={handleFocus}
				onBlur={handleBlur}
				placeholder={effectivePlaceholder}
				disabled={disabled}
				aria-describedby={ariaDescribedBy}
				aria-required={ariaRequiredAttr}
				aria-invalid={ariaInvalidAttr}
				render={(inputProps) => <input {...inputProps} className={styles.input} {...inputRest} />}
			/>

			{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
				<div className={styles.suffix}>
					<Spinner size="sm" transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined} />
				</div>
			)}

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage}>
				{manager.errorMessageToDisplay}
			</InputError>
		</div>
	);
}

export default Input;
