import { Children, Fragment, type ReactElement } from 'react';

import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';
import { IconButton } from '../icon-button';

import styles from './banner.module.css';

interface BannerSlots {
	heading?: React.ReactNode;
	description?: React.ReactNode;
	body?: React.ReactNode;
	actions?: React.ReactNode;
}

/**
 * Extracts slot content from children by matching component displayName.
 * This allows children to be passed in any order while ensuring they render
 * in the correct positions.
 */
function extractBannerSlots(children: React.ReactNode): BannerSlots {
	const slots: BannerSlots = {};

	const processChild = (child: React.ReactNode): void => {
		if (!child) return;

		if (Array.isArray(child)) {
			child.forEach(processChild);
			return;
		}

		if (typeof child === 'object' && 'type' in child) {
			const element = child as ReactElement<{ children?: React.ReactNode }>;
			const displayName = (element.type as { displayName?: string }).displayName;

			switch (displayName) {
				case 'Banner.Heading':
					slots.heading = element.props.children;
					break;
				case 'Banner.Description':
					slots.description = element.props.children;
					break;
				case 'Banner.Body':
					slots.body = element.props.children;
					break;
				case 'Banner.Actions':
					slots.actions = element.props.children;
					break;
				default:
					// Handle fragments by recursing into their children
					if (element.type === Fragment) {
						Children.forEach(element.props.children, processChild);
					}
			}
		}
	};

	Children.forEach(children, processChild);
	return slots;
}

interface BannerBaseProps {
	/** Whether to stack the content vertically (default: true) */
	stacked?: boolean;
	/** Callback when the close button is clicked */
	onClose?: () => void;
	/** Compound component children (Banner.Heading, Banner.Description, etc.) */
	children: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type BannerProps = ComponentPropsWithRef<BannerBaseProps, HTMLElement>;

interface BannerSubcomponentProps {
	children: React.ReactNode;
}

/**
 * An aside element for highlighting key information.
 *
 * Use compound components to compose the banner:
 * - `Banner.Heading` - Optional heading content
 * - `Banner.Description` - Optional description content
 * - `Banner.Body` - Optional body content
 * - `Banner.Actions` - Optional action buttons
 *
 * Children can be passed in any order - they will always render in the correct position.
 *
 * @example
 * ```tsx
 * <Banner stacked onClose={() => console.log('closed')}>
 *   <Banner.Heading>
 *     <Heading level={3} font="400">Important Notice</Heading>
 *   </Banner.Heading>
 *   <Banner.Description>
 *     <Text font="200" weight="weak">This is important information.</Text>
 *   </Banner.Description>
 * </Banner>
 * ```
 */
function BannerRoot(props: BannerProps) {
	const { ref, stacked = true, onClose, children, className, ...rest } = props;

	// Extract slot content from children (order-independent)
	const slots = extractBannerSlots(children);

	const hasHeading = slots.heading;
	const hasDescription = slots.description;

	return (
		<aside
			ref={ref}
			className={cn(styles.Banner, className)}
			data-stacked={stacked}
			data-qid-component-type="Luna::Banner"
			{...rest}
		>
			{onClose && (
				<div className={styles.CloseButton}>
					<IconButton
						icon="close"
						tooltip="Close"
						variant="transparent"
						size="sm"
						onClick={onClose}
						data-testid="banner-close-button"
					/>
				</div>
			)}

			{(hasHeading || hasDescription) && (
				<div className={styles.Text}>
					{slots.heading}
					{slots.description}
				</div>
			)}

			{slots.body && <div className={styles.Body}>{slots.body}</div>}

			{slots.actions && <div className={styles.Actions}>{slots.actions}</div>}
		</aside>
	);
}

BannerRoot.displayName = 'Banner';

// Subcomponents - these don't render themselves, they're extracted by the parent
const BannerHeading = ({ children }: BannerSubcomponentProps) => children;
BannerHeading.displayName = 'Banner.Heading';

const BannerDescription = ({ children }: BannerSubcomponentProps) => children;
BannerDescription.displayName = 'Banner.Description';

const BannerBody = ({ children }: BannerSubcomponentProps) => children;
BannerBody.displayName = 'Banner.Body';

const BannerActions = ({ children }: BannerSubcomponentProps) => children;
BannerActions.displayName = 'Banner.Actions';

/**
 * Banner compound component with namespaced subcomponents.
 *
 * Subcomponents can be passed in any order - they will render in the correct position.
 */
export const Banner = Object.assign(BannerRoot, {
	Heading: BannerHeading,
	Description: BannerDescription,
	Body: BannerBody,
	Actions: BannerActions,
});

export default Banner;
