interface LunaToggleSignature<V> {
	Element: HTMLDivElement;
	Args: {
		value: V;
		disabled?: boolean;
		class?: string;
		size?: 'xs' | 'sm' | 'md' | 'lg';
		align?: 'right' | 'left' | 'justify';
		onChange?: (value: boolean) => unknown;
		id?: string;
		requirements?: Requirements;
		form?: LunaFormBlockValue;
		name?: string;
		/** @deprecated This can almost always be handled better with `externalErrorsFor` on the form or a custom validator. */
		serverErrors?: Array<{ message?: string } | string>;
		ariaDescribedBy?: string | undefined;
		didAutosave?: () => void;
	};
	Blocks: {
		default: [];
	};
}

let nextInputId = 0;

/**
 * A component for creating a toggle switch input.
 *
 * [Luna Docs](https://docs.auditboardteam.com/Luna/components/components/toggle.md)
 */
export default class LunaToggle<V> extends Component<LunaToggleSignature<V>> {
	@consume('luna-form') formContext: LunaFormBlockValue | undefined;
	@consume('luna-input-id') inputId!: string;

	labelId = `${guidFor(this)}--label`;

	manager: InputManager;

	constructor(owner: Owner, args: LunaToggleSignature<V>['Args']) {
		super(owner, args);
		// eslint-disable-next-line @typescript-eslint/no-this-alias
		const that = this;
		const managerOptions: InputManagerOptions = {
			mode: 'change',
			didAutosave: this.didAutosave.bind(this),
			name: args.name,

			focus: this.focusInput.bind(this),

			ariaDescribedBy: this.args.ariaDescribedBy ?? null,

			get value(): V {
				return args.value;
			},

			get isDisabled(): boolean {
				return args.disabled ?? false;
			},

			get requirements() {
				return args.requirements;
			},

			get serverErrors() {
				// eslint-disable-next-line @typescript-eslint/no-deprecated
				return args.serverErrors;
			},

			get formContext() {
				return that.formContext;
			},
		};

		this.manager = new InputManager(owner, managerOptions);
		void this.registerManager();
	}

	registerManager = async () => {
		await timeout(0);
		if (this.isDestroyed || this.isDestroying) {
			return;
		}
		this.formContext?.actions?.registerInputManager(this.manager);

		registerDestructor(this, () => {
			this.formContext?.actions?.unregisterInputManager(this.manager);
		});
	};

	get id() {
		return this.args.id ?? this.inputId ?? `luna-toggle-${nextInputId++}`;
	}

	@dedupeTracked valueIsUnchanged = true;

	@action
	async toggle() {
		if (this.args.disabled) {
			return;
		}

		// This ensures a falsy undefined value goes to true on click
		const value = typeof this.args.value === 'boolean' ? !this.args.value : true;

		this.valueIsUnchanged = false;
		await this.args.onChange?.(value);
		this.manager.onChange();
	}

	@action
	onKeyUp(event: KeyboardEvent) {
		if (event.key === KEYBOARD_KEYS.SPACE) {
			event.preventDefault();
			void this.toggle();
		}
	}

	@action
	didAutosave() {
		this.valueIsUnchanged = true;
		this.args.didAutosave?.();
	}

	get isChecked() {
		const isBooleanCompatible =
			typeof this.args.value === 'boolean' ||
			!this.args.value ||
			this.args.value === 'true' ||
			this.args.value === 'false';

		warn(
			`Value of ${String(this.args.value)} provided to Luna Toggle can not be evaluated as boolean`,
			isBooleanCompatible,
			{
				id: 'luna.toggle.invalid-value',
			},
		);

		// Values passed in across the product are inconsistent so we need to handle them
		if (!this.args.value || this.args.value === 'false') {
			return false;
		}
		return true;
	}

	focusInput = () => {
		focusById(this.id);
	};
	<template>
		{{! template-lint-disable require-presentational-children }}
		<div class="ToggleWrapper {{@class}}" data-qid-component-type="Luna::Toggle">
			<XStack
				@spacing="md"
				id={{this.id}}
				class="Toggle"
				tabindex="0"
				role="switch"
				aria-checked={{String this.isChecked}}
				aria-disabled={{if @disabled "true"}}
				aria-labelledby={{if (has-block) this.labelId}}
				aria-describedby={{this.manager.ariaDescribedBy}}
				aria-invalid={{if this.manager.showErrorMessage "true"}}
				data-focused={{String this.manager.inputHasFocus}}
				data-errored={{String this.manager.showErrorMessage}}
				data-size={{or @size "md"}}
				...attributes
				{{on "click" this.toggle}}
				{{on "keyup" this.onKeyUp}}
			>
				{{#let (or @align "left") as |align|}}
					{{#if (and (eq align "right") (has-block))}}
						<Spacer />
					{{/if}}

					{{#if (and (or (eq align "right") (eq align "justify")) (has-block))}}
						<div id={{this.labelId}}>
							{{yield}}
						</div>
					{{/if}}

					{{#if (and (eq align "justify") (has-block))}}
						<Spacer />
					{{/if}}

					<div class="toggle-element">
						<div class="spacer"></div>
						<div class="handle">
							{{#if (or this.manager.isAutosaving this.manager.didAutosaveSuccessfully)}}
								<Spinner @size="sm" @transitionTo={{if this.manager.didAutosaveSuccessfully "success"}} />
							{{/if}}
						</div>
					</div>

					{{#if (and (eq align "left") (has-block))}}
						<div id={{this.labelId}}>
							{{yield}}
						</div>
					{{/if}}
				{{/let}}
			</XStack>

			<InputError
				id={{this.manager.errorId}}
				@showWhen={{this.manager.showErrorMessage}}
				@align={{if (eq @align "left") "left"}}
			>
				{{this.manager.errorMessageToDisplay}}
			</InputError>
		</div>
	</template>
}

import { useCallback, useEffect, useId, useRef, useState } from 'react';

import { Switch as BaseSwitch } from '@base-ui-components/react/switch';
import type { Requirements } from '@soxhub/luna-utils/validators';

import { cn } from '../../utils/cn';
import { useInputManager } from '../../utils/use-input-manager';
import { useFieldContext } from '../field/field-context';
import { useFormContext } from '../form/form';
import { InputError } from '../input-error';
import { Spacer } from '../spacer';
import { Spinner } from '../spinner';
import { XStack } from '../x-stack';

import styles from './toggle.module.css';

export type ToggleSize = 'xs' | 'sm' | 'md' | 'lg';
export type ToggleAlign = 'left' | 'right' | 'justify';

interface ToggleBaseProps {
	/** Optional ref for the root element (React 19 ref-as-prop) */
	'ref'?: React.Ref<HTMLDivElement>;
	/** Toggle value (boolean or truthy/falsy) */
	'value': boolean | string | number | null | undefined;
	/** Callback fired when value changes */
	'onChange'?: (value: boolean) => void;
	/** Toggle size */
	'size'?: ToggleSize;
	/** Label alignment */
	'align'?: ToggleAlign;
	/** Whether toggle is disabled */
	'disabled'?: boolean;
	/** Toggle ID (auto-generated if not provided) */
	'id'?: string;
	/** Toggle name */
	'name'?: string;
	/** ARIA described by */
	'aria-describedby'?: string;
	/** Additional CSS class */
	'className'?: string;
	/** Children (label content) */
	'children'?: React.ReactNode;
	/** Validation requirements */
	'requirements'?: Requirements;
	/** @deprecated Server errors (use externalErrorsFor on Form instead) */
	'serverErrors'?: Array<{ message?: string } | string>;
	/** Callback fired when autosave completes successfully */
	'didAutosave'?: () => void;
}

type ToggleProps = ToggleBaseProps & Omit<React.HTMLAttributes<HTMLDivElement>, keyof ToggleBaseProps | 'onChange'>;

/**
 * A component for creating a toggle switch input.
 *
 * Matches the Luna::Toggle Ember component API and styling.
 * Uses Base UI Switch as the foundation.
 */
export function Toggle(props: ToggleProps) {
	const {
		ref,
		value,
		onChange,
		size = 'md',
		align = 'left',
		disabled = false,
		'id': providedId,
		name,
		'aria-describedby': providedAriaDescribedBy,
		className,
		children,
		requirements,
		serverErrors,
		didAutosave,
		...rest
	} = props;

	const generatedId = useId();
	const fieldContext = useFieldContext();
	const id = providedId ?? fieldContext?.inputId ?? generatedId;
	const labelId = fieldContext?.labelId ?? `${id}--label`;
	const formContext = useFormContext();
	const toggleRef = useRef<HTMLDivElement>(null);
	const valueIsUnchangedRef = useRef(true);

	const focusInput = useCallback(() => {
		if (toggleRef.current) {
			toggleRef.current.focus();
		} else if (ref && typeof ref === 'object' && ref.current) {
			ref.current.focus();
		}
	}, [ref]);

	const manager = useInputManager({
		mode: 'change',
		valueIsUnchanged: valueIsUnchangedRef.current,
		serverErrors,
		requirements,
		name,
		value,
		isDisabled: disabled,
		formContext,
		ariaDescribedBy: providedAriaDescribedBy ?? null,
		focus: focusInput,
		didAutosave: () => {
			valueIsUnchangedRef.current = true;
			didAutosave?.();
		},
	});

	// Register/unregister with form
	const managerRef = useRef(manager);
	managerRef.current = manager;

	useEffect(() => {
		if (formContext) {
			formContext.actions.registerInputManager(managerRef.current);
			return () => {
				formContext.actions.unregisterInputManager(managerRef.current);
			};
		}
	}, [formContext]);

	const [focused, setFocused] = useState(false);
	const focusedState = focused || manager.inputHasFocus;

	// Convert value to boolean (similar to Ember's isChecked getter)
	const isChecked = Boolean(
		typeof value === 'boolean'
			? value
			: value === 'true' || (value !== null && value !== undefined && value !== 'false'),
	);

	const handleCheckedChange = useCallback(
		(checked: boolean) => {
			valueIsUnchangedRef.current = false;
			manager.setValueIsUnchanged(false);
			// When value is falsy/undefined, clicking should go to true
			const newValue = typeof value === 'boolean' ? checked : checked ? true : false;
			onChange?.(newValue);
			manager.onChange();
		},
		[onChange, value, manager],
	);

	const handleKeyUp = useCallback(
		(event: React.KeyboardEvent) => {
			if (event.key === ' ') {
				event.preventDefault();
				handleCheckedChange(!isChecked);
			}
		},
		[handleCheckedChange, isChecked],
	);

	const handleFocus = useCallback(() => {
		setFocused(true);
		manager.focusIn();
	}, [manager]);

	const handleBlur = useCallback(() => {
		setFocused(false);
		manager.focusOutAndTrySave(new Event('blur'));
	}, [manager]);

	const wrapperClassName = cn(styles.ToggleWrapper, className);
	const toggleClassName = cn(styles.Toggle, manager.showErrorMessage && styles.errored);

	const hasLabel = Boolean(children);

	return (
		<div className={wrapperClassName} data-qid-component-type="Luna::Toggle" ref={ref} {...rest}>
			<BaseSwitch.Root
				checked={isChecked}
				disabled={disabled}
				onCheckedChange={handleCheckedChange}
				id={id}
				render={(switchProps) => (
					<XStack
						{...switchProps}
						spacing="md"
						id={id}
						className={toggleClassName}
						tabIndex={disabled ? -1 : 0}
						role="switch"
						aria-checked={isChecked}
						aria-disabled={disabled ? 'true' : undefined}
						aria-labelledby={hasLabel ? labelId : undefined}
						aria-describedby={manager.ariaDescribedBy ?? providedAriaDescribedBy}
						aria-invalid={manager.showErrorMessage ? 'true' : undefined}
						data-focused={focusedState ? 'true' : 'false'}
						data-errored={manager.showErrorMessage ? 'true' : 'false'}
						data-size={size}
						onClick={(e: React.MouseEvent) => {
							switchProps.onClick?.(e);
							if (!disabled) {
								handleCheckedChange(!isChecked);
							}
						}}
						onKeyUp={handleKeyUp}
						onFocus={handleFocus}
						onBlur={handleBlur}
					>
						{align === 'right' && hasLabel && <Spacer />}
						{(align === 'right' || align === 'justify') && hasLabel && (
							<div id={labelId}>
								{children}
								{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
									<Spinner
										size="sm"
										transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined}
										className={styles.spinner}
									/>
								)}
							</div>
						)}
						{align === 'justify' && hasLabel && <Spacer />}
						<div className={styles.toggleElement}>
							<div className={styles.spacer}></div>
							<div className={styles.handle} data-checked={isChecked ? 'true' : 'false'}></div>
						</div>
						{align === 'left' && hasLabel && (
							<div id={labelId}>
								{children}
								{(manager.isAutosaving || manager.didAutosaveSuccessfully) && (
									<Spinner
										size="sm"
										transitionTo={manager.didAutosaveSuccessfully ? 'success' : undefined}
										className={styles.spinner}
									/>
								)}
							</div>
						)}
					</XStack>
				)}
			/>

			<InputError id={manager.errorId} showWhen={manager.showErrorMessage} align="left">
				{manager.errorMessageToDisplay}
			</InputError>
		</div>
	);
}

export default Toggle;
