● React (TSX)
y-stack.tsx
import { cn } from '../../utils/cn';
import { type ComponentPropsWithRef } from '../../utils/component-types';

import styles from './y-stack.module.css';

export type YStackSpacing = 'none' | 'xs' | 'sm' | 'md' | 'lg' | 'xl' | '2xl';
export type YStackTag = keyof React.JSX.IntrinsicElements;

interface YStackBaseProps {
	/** Controls how much distance to place between each child element */
	spacing?: YStackSpacing;
	/** HTML tag to render as, defaults to 'div' */
	tag?: YStackTag;
	/** Children content */
	children?: React.ReactNode;
	/** Additional CSS class */
	className?: string;
}

export type YStackProps = ComponentPropsWithRef<YStackBaseProps, HTMLElement>;

/**
 * Enables the layout of elements along the Y axis using flexbox. Items will not wrap.
 *
 * Matches the Luna::YStack Ember component API and styling.
 */
export function YStack(props: YStackProps) {
	const { ref, spacing, tag = 'div', children, className, ...rest } = props;

	const Tag = tag as React.ElementType;

	const ystackClassName = cn(styles.YStack, className);

	return (
		<Tag
			ref={ref as React.Ref<HTMLElement>}
			className={ystackClassName}
			data-spacing={spacing}
			data-qid-component-type="Luna::YStack"
			{...rest}
		>
			{children}
		</Tag>
	);
}

export default YStack;
